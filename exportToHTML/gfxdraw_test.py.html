8    def intensity(c, i):
9        """Return color c changed by intensity i 
10    
11       For 0 <= i <= 127 the color is a shade, with 0 being black, 127 being the 
12       unaltered color. 
13    
14       For 128 <= i <= 255 the color is a tint, with 255 being white, 128 the 
15       unaltered color. 
16    
17       """
18       r, g, b = c[0:3]
19       if 0 <= i <= 127:
20           # Darken
21           return ((r * i) // 127, (g * i) // 127, (b * i) // 127)
22       # Lighten
23       return (
24           r + ((255 - r) * (255 - i)) // 127,
25           g + ((255 - g) * (255 - i)) // 127,
26           b + ((255 - b) * (255 - i)) // 127,
27       )

37       def make_palette(base_color):
38           """Return color palette that is various intensities of base_color"""
39           # Need this function for Python 3.x so the base_color
40           # is within the scope of the list comprehension.
41           return [intensity(base_color, i) for i in range(0, 256)]

47       def check_at(self, surf, posn, color):
48           sc = surf.get_at(posn)
49           fail_msg = "%s != %s at %s, bitsize: %i, flags: %i, masks: %s" % (
50               sc,
51               color,
52               posn,
53               surf.get_bitsize(),
54               surf.get_flags(),
55               surf.get_masks(),
56           )
57           self.assertEqual(sc, color, fail_msg)
58   
59       def check_not_at(self, surf, posn, color):
60           sc = surf.get_at(posn)
61           fail_msg = "%s != %s at %s, bitsize: %i, flags: %i, masks: %s" % (
62               sc,
63               color,
64               posn,
65               surf.get_bitsize(),
66               surf.get_flags(),
67               surf.get_masks(),
68           )
69           self.assertNotEqual(sc, color, fail_msg)

129      def test_gfxdraw__subclassed_surface(self):
130          """Ensure pygame.gfxdraw works on subclassed surfaces."""
131          surface = SurfaceSubclass((11, 13), SRCALPHA, 32)
132          surface.fill(pygame.Color("blue"))
133          expected_color = pygame.Color("red")
134          x, y = 1, 2
135  
136          pygame.gfxdraw.pixel(surface, x, y, expected_color)
137  
138          self.assertEqual(surface.get_at((x, y)), expected_color)

597          """Ensure aatrigon draws horizontal edges correctly. 
598   
599          This test creates 2 surfaces and draws an aatrigon on each. The pixels 
600          on each surface are compared to ensure they are the same. The only 
601          difference between the 2 aatrigons is the order the points are drawn. 
602          The order of the points should have no impact on the final drawing. 
603   
604          Related to issue #622. 
605          """
606          bg_color = pygame.Color("white")
607          line_color = pygame.Color("black")
608          width, height = 11, 10
609          expected_surface = pygame.Surface((width, height), 0, 32)
610          expected_surface.fill(bg_color)
611          surface = pygame.Surface((width, height), 0, 32)
612          surface.fill(bg_color)
613  
614          x1, y1 = width - 1, 0
615          x2, y2 = (width - 1) // 2, height - 1
616          x3, y3 = 0, 0
617  
618          # The points in this order draw as expected.
619          pygame.gfxdraw.aatrigon(expected_surface, x1, y1, x2, y2, x3, y3, line_color)
620  
621          # The points in reverse order fail to draw the horizontal edge along
622          # the top.
623          pygame.gfxdraw.aatrigon(surface, x3, y3, x2, y2, x1, y1, line_color)
624  
625          # The surfaces are locked for a possible speed up of pixel access.
626          expected_surface.lock()
627          surface.lock()
628          for x in range(width):
629              for y in range(height):
630                  self.assertEqual(
631                      expected_surface.get_at((x, y)),
632                      surface.get_at((x, y)),
633                      "pos=({}, {})".format(x, y),

636          surface.unlock()
637          expected_surface.unlock()
