8    import warnings

10   from pylint.extensions import docparams
11   
12   
13   def register(linter):
14       """Required method to auto register this checker. 
15    
16       :param linter: Main interface object for Pylint plugins 
17       :type linter: Pylint object 
18       """
19       warnings.warn(
20           "This plugin is deprecated, use pylint.extensions.docparams instead.",
21           DeprecationWarning,
22       )
23       linter.register_checker(docparams.DocstringParameterChecker(linter))
