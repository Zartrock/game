16   def test_magic(f, magic_hex):
17       """ tests a given file to see if the magic hex matches. 
18       """
19       data = f.read(len(magic_hex))
20   
21       if len(data) != len(magic_hex):
22           return 0
23   
24       for i in range(len(magic_hex)):
25           if magic_hex[i] != ord_(data[i]):
26               return 0
27   
28       return 1

94       def testSaveJPG(self):
95           """ JPG equivalent to issue #211 - color channel swapping 
96    
97           Make sure the SDL surface color masks represent the rgb memory format 
98           required by the JPG library. The masks are machine endian dependent 
99           """
100  
101          from pygame import Color, Rect
102  
103          # The source image is a 2 by 2 square of four colors. Since JPEG is
104          # lossy, there can be color bleed. Make each color square 16 by 16,
105          # to avoid the significantly color value distorts found at color
106          # boundaries due to the compression value set by Pygame.
107          square_len = 16
108          sz = 2 * square_len, 2 * square_len
109  
110          #  +---------------------------------+
111          #  | red            | green          |
112          #  |----------------+----------------|
113          #  | blue           | (255, 128, 64) |
114          #  +---------------------------------+
115          #
116          #  as (rect, color) pairs.
117          def as_rect(square_x, square_y):
118              return Rect(
119                  square_x * square_len, square_y * square_len, square_len, square_len
120              )
121  
122          squares = [
123              (as_rect(0, 0), Color("red")),
124              (as_rect(1, 0), Color("green")),
125              (as_rect(0, 1), Color("blue")),
126              (as_rect(1, 1), Color(255, 128, 64)),
127          ]
128  
129          # A surface format which is not directly usable with libjpeg.
130          surf = pygame.Surface(sz, 0, 32)
131          for rect, color in squares:
132              surf.fill(color, rect)
133  
134          # Assume pygame.image.Load works correctly as it is handled by the
135          # third party SDL_image library.
136          f_path = tempfile.mktemp(suffix=".jpg")
137          pygame.image.save(surf, f_path)
138          jpg_surf = pygame.image.load(f_path)
139  
140          # Allow for small differences in the restored colors.
141          def approx(c):
142              mask = 0xFC
143              return pygame.Color(c.r & mask, c.g & mask, c.b & mask)
144  
145          offset = square_len // 2
146          for rect, color in squares:
147              posn = rect.move((offset, offset)).topleft
148              self.assertEqual(approx(jpg_surf.get_at(posn)), approx(color))
149  
150          os.remove(f_path)

286          s = pygame.Surface((10, 10))
287          s.fill((23, 23, 23))
288          magic_hex = {}
289          magic_hex["jpg"] = [0xFF, 0xD8, 0xFF, 0xE0]
290          magic_hex["png"] = [0x89, 0x50, 0x4E, 0x47]
291          magic_hex["bmp"] = [0x42, 0x4D]
292  
293          formats = ["tga", "jpg", "bmp", "png"]
294          # uppercase too... JPG
295          formats = formats + [x.upper() for x in formats]
296  
297          SDL_Im_version = pygame.image.get_sdl_image_version()
298          # We assume here that minor version and patch level of SDL_Image
299          # never goes above 99
300          isAtLeastSDL_image_2_0_2 = ((SDL_Im_version is not None) and
301                  (SDL_Im_version[0] * 10000 +
302                   SDL_Im_version[1] * 100 +
303                   SDL_Im_version[2])
304                  >= 20002)
305          for fmt in formats:
306              tmp_file, tmp_filename = tempfile.mkstemp(suffix=".%s"%fmt)
307              if not isAtLeastSDL_image_2_0_2 and fmt.lower() == "jpg":
308                  with os.fdopen(tmp_file, 'wb') as handle:
309                      with self.assertRaises(pygame.error):
310                          pygame.image.save(s, handle, tmp_filename)
311              else:
312                  with os.fdopen(tmp_file, 'r+b') as handle:
313                      pygame.image.save(s, handle, tmp_filename)
314  
315                      if fmt.lower() in magic_hex:
316                          # Test the magic numbers at the start of the file to
317                          # ensure they are saved as the correct file type.
318                          handle.seek(0)
319                          self.assertEqual(
320                              (1, fmt),
321                              (test_magic(handle, magic_hex[fmt.lower()]), fmt)
322                          )
323                      # load the file to make sure it was saved correctly.
324                      handle.flush()
325                      handle.seek(0)
326                      s2 = pygame.image.load(handle, tmp_filename)
327                      self.assertEqual(s2.get_at((0, 0)), s.get_at((0, 0)))
328              os.remove(tmp_filename)

387      def assertPremultipliedAreEqual(self, string1, string2, source_string):
388          self.assertEqual(len(string1), len(string2))
389          block_size = 20
390          if string1 != string2:
391              for block_start in xrange_(0, len(string1), block_size):
392                  block_end = min(block_start + block_size, len(string1))
393                  block1 = string1[block_start:block_end]
394                  block2 = string2[block_start:block_end]
395                  if block1 != block2:
396                      source_block = source_string[block_start:block_end]
397                      msg = (
398                          "string difference in %d to %d of %d:\n%s\n%s\nsource:\n%s"
399                          % (
400                              block_start,
401                              block_end,
402                              len(string1),
403                              binascii.hexlify(block1),
404                              binascii.hexlify(block2),
405                              binascii.hexlify(source_block),
406                          )
407                      )
408                      self.fail(msg)

410      def test_to_string__premultiplied(self):
411          """ test to make sure we can export a surface to a premultiplied alpha string 
412          """
413  
414          def convertRGBAtoPremultiplied(surface_to_modify):
415              for x in xrange_(surface_to_modify.get_width()):
416                  for y in xrange_(surface_to_modify.get_height()):
417                      color = surface_to_modify.get_at((x, y))
418                      premult_color = (
419                          color[0] * color[3] / 255,
420                          color[1] * color[3] / 255,
421                          color[2] * color[3] / 255,
422                          color[3],
423                      )
424                      surface_to_modify.set_at((x, y), premult_color)
425  
426          test_surface = pygame.Surface((256, 256), pygame.SRCALPHA, 32)
427          for x in xrange_(test_surface.get_width()):
428              for y in xrange_(test_surface.get_height()):
429                  i = x + y * test_surface.get_width()
430                  test_surface.set_at(
431                      (x, y), ((i * 7) % 256, (i * 13) % 256, (i * 27) % 256, y)
432                  )
433          premultiplied_copy = test_surface.copy()
434          convertRGBAtoPremultiplied(premultiplied_copy)
435          self.assertPremultipliedAreEqual(
436              pygame.image.tostring(test_surface, "RGBA_PREMULT"),
437              pygame.image.tostring(premultiplied_copy, "RGBA"),
438              pygame.image.tostring(test_surface, "RGBA"),
439          )
440          self.assertPremultipliedAreEqual(
441              pygame.image.tostring(test_surface, "ARGB_PREMULT"),
442              pygame.image.tostring(premultiplied_copy, "ARGB"),
443              pygame.image.tostring(test_surface, "ARGB"),
444          )
445  
446          no_alpha_surface = pygame.Surface((256, 256), 0, 24)
447          self.assertRaises(
448              ValueError, pygame.image.tostring, no_alpha_surface, "RGBA_PREMULT"
449          )

476      def test_fromstring__and_tostring(self):
477          """Ensure methods tostring() and fromstring() are symmetric."""
478  
479          ####################################################################
480          def RotateRGBAtoARGB(str_buf):
481              byte_buf = array.array("B", str_buf)
482              num_quads = len(byte_buf) // 4
483              for i in xrange_(num_quads):
484                  alpha = byte_buf[i * 4 + 3]
485                  byte_buf[i * 4 + 3] = byte_buf[i * 4 + 2]
486                  byte_buf[i * 4 + 2] = byte_buf[i * 4 + 1]
487                  byte_buf[i * 4 + 1] = byte_buf[i * 4 + 0]
488                  byte_buf[i * 4 + 0] = alpha
489              return tostring(byte_buf)
490  
491          ####################################################################
492          def RotateARGBtoRGBA(str_buf):
493              byte_buf = array.array("B", str_buf)
494              num_quads = len(byte_buf) // 4
495              for i in xrange_(num_quads):
496                  alpha = byte_buf[i * 4 + 0]
497                  byte_buf[i * 4 + 0] = byte_buf[i * 4 + 1]
498                  byte_buf[i * 4 + 1] = byte_buf[i * 4 + 2]
499                  byte_buf[i * 4 + 2] = byte_buf[i * 4 + 3]
500                  byte_buf[i * 4 + 3] = alpha
501              return tostring(byte_buf)
502  
503          ####################################################################
504          test_surface = pygame.Surface((64, 256), flags=pygame.SRCALPHA, depth=32)
505          for i in xrange_(256):
506              for j in xrange_(16):
507                  intensity = j * 16 + 15
508                  test_surface.set_at((j + 0, i), (intensity, i, i, i))
509                  test_surface.set_at((j + 16, i), (i, intensity, i, i))
510                  test_surface.set_at((j + 32, i), (i, i, intensity, i))
511                  test_surface.set_at((j + 32, i), (i, i, i, intensity))
512  
513          self._assertSurfaceEqual(
514              test_surface, test_surface, "failing with identical surfaces"
515          )
516  
517          rgba_buf = pygame.image.tostring(test_surface, "RGBA")
518          rgba_buf = RotateARGBtoRGBA(RotateRGBAtoARGB(rgba_buf))
519          test_rotate_functions = pygame.image.fromstring(
520              rgba_buf, test_surface.get_size(), "RGBA"
521          )
522  
523          self._assertSurfaceEqual(
524              test_surface, test_rotate_functions, "rotate functions are not symmetric"
525          )
526  
527          rgba_buf = pygame.image.tostring(test_surface, "RGBA")
528          argb_buf = RotateRGBAtoARGB(rgba_buf)
529          test_from_argb_string = pygame.image.fromstring(
530              argb_buf, test_surface.get_size(), "ARGB"
531          )
532  
533          self._assertSurfaceEqual(
534              test_surface, test_from_argb_string, '"RGBA" rotated to "ARGB" failed'
535          )
536  
537          argb_buf = pygame.image.tostring(test_surface, "ARGB")
538          rgba_buf = RotateARGBtoRGBA(argb_buf)
539          test_to_argb_string = pygame.image.fromstring(
540              rgba_buf, test_surface.get_size(), "RGBA"
541          )
542  
543          self._assertSurfaceEqual(
544              test_surface, test_to_argb_string, '"ARGB" rotated to "RGBA" failed'
545          )
546  
547          for fmt in ("ARGB", "RGBA"):
548              fmt_buf = pygame.image.tostring(test_surface, fmt)
549              test_to_from_fmt_string = pygame.image.fromstring(
550                  fmt_buf, test_surface.get_size(), fmt
551              )
552  
553              self._assertSurfaceEqual(
554                  test_surface,
555                  test_to_from_fmt_string,
556                  "tostring/fromstring functions are not "
557                  'symmetric with "{}" format'.format(fmt),
558              )

809      def test_save_extended(self):
810          surf = pygame.Surface((5, 5))
811          surf.fill((23, 23, 23))
812  
813          passing_formats = ['jpg', 'png']
814          passing_formats += [fmt.upper() for fmt in passing_formats]
815  
816          magic_hex = {}
817          magic_hex["jpg"] = [0xFF, 0xD8, 0xFF, 0xE0]
818          magic_hex["png"] = [0x89, 0x50, 0x4E, 0x47]
819  
820          failing_formats = ['bmp', 'tga']
821          failing_formats += [fmt.upper() for fmt in failing_formats]
822  
823          # check that .jpg and .png save
824          for fmt in passing_formats:
825              temp_file_name = "temp_file.%s" % fmt
826              # save image as .jpg and .png
827              pygame.image.save_extended(surf, temp_file_name)
828              with open(temp_file_name, "rb") as file:
829                  # Test the magic numbers at the start of the file to ensure
830                  # they are saved as the correct file type.
831                  self.assertEqual(
832                      1, (test_magic(file, magic_hex[fmt.lower()]))
833                  )
834              # load the file to make sure it was saved correctly
835              loaded_file = pygame.image.load(temp_file_name)
836              self.assertEqual(loaded_file.get_at((0, 0)), surf.get_at((0, 0)))
837              # clean up the temp file
838              os.remove(temp_file_name)
839          # check that .bmp and .tga do not save
840          for fmt in failing_formats:
841              self.assertRaises(pygame.error, pygame.image.save_extended, surf, "temp_file.%s" % fmt)
