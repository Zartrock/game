11   """ 
12   Data object model, as per https://docs.python.org/3/reference/datamodel.html. 
13    
14   This module describes, at least partially, a data object model for some 
15   of astroid's nodes. The model contains special attributes that nodes such 
16   as functions, classes, modules etc have, such as __doc__, __class__, 
17   __module__ etc, being used when doing attribute lookups over nodes. 
18    
19   For instance, inferring `obj.__class__` will first trigger an inference 
20   of the `obj` variable. If it was successfully inferred, then an attribute 
21   `__class__ will be looked for in the inferred object. This is the part 
22   where the data model occurs. The model is attached to those nodes 
23   and the lookup mechanism will try to see if attributes such as 
24   `__class__` are defined by the model or not. If they are defined, 
25   the model will be requested to return the corresponding value of that 
26   attribute. Thus the model can be viewed as a special part of the lookup 
27   mechanism. 
28   """

45   def _dunder_dict(instance, attributes):
46       obj = node_classes.Dict(parent=instance)
47   
48       # Convert the keys to node strings
49       keys = [
50           node_classes.Const(value=value, parent=obj) for value in list(attributes.keys())
51       ]
52   
53       # The original attribute has a list of elements for each key,
54       # but that is not useful for retrieving the special attribute's value.
55       # In this case, we're picking the last value from each list.
56       values = [elem[-1] for elem in attributes.values()]
57   
58       obj.postinit(list(zip(keys, values)))
59       return obj

62   class ObjectModel:
63       def __init__(self):
64           self._instance = None
65   
66       def __repr__(self):
67           result = []
68           cname = type(self).__name__
69           string = "%(cname)s(%(fields)s)"
70           alignment = len(cname) + 1
71           for field in sorted(self.attributes()):
72               width = 80 - len(field) - alignment
73               lines = pprint.pformat(field, indent=2, width=width).splitlines(True)
74   
75               inner = [lines[0]]
76               for line in lines[1:]:
77                   inner.append(" " * alignment + line)
78               result.append(field)
79   
80           return string % {
81               "cname": cname,
82               "fields": (",\n" + " " * alignment).join(result),
83           }
84   
85       def __call__(self, instance):
86           self._instance = instance
87           return self
88   
89       def __get__(self, instance, cls=None):
90           # ObjectModel needs to be a descriptor so that just doing
91           # `special_attributes = SomeObjectModel` should be enough in the body of a node.
92           # But at the same time, node.special_attributes should return an object
93           # which can be used for manipulating the special attributes. That's the reason
94           # we pass the instance through which it got accessed to ObjectModel.__call__,
95           # returning itself afterwards, so we can still have access to the
96           # underlying data model and to the instance for which it got accessed.
97           return self(instance)
98   
99       def __contains__(self, name):
100          return name in self.attributes()
101  
102      @lru_cache(maxsize=None)
103      def attributes(self):
104          """Get the attributes which are exported by this object model."""
105          return [
106              obj[len(IMPL_PREFIX) :] for obj in dir(self) if obj.startswith(IMPL_PREFIX)
107          ]
108  
109      def lookup(self, name):
110          """Look up the given *name* in the current model 
111   
112          It should return an AST or an interpreter object, 
113          but if the name is not found, then an AttributeInferenceError will be raised. 
114          """
115  
116          if name in self.attributes():
117              return getattr(self, IMPL_PREFIX + name)
118          raise exceptions.AttributeInferenceError(target=self._instance, attribute=name)

210      @property
211      def attr___defaults__(self):
212          func = self._instance
213          if not func.args.defaults:
214              return node_classes.Const(value=None, parent=func)
215  
216          defaults_obj = node_classes.Tuple(parent=func)
217          defaults_obj.postinit(func.args.defaults)

220      @property
221      def attr___annotations__(self):
222          obj = node_classes.Dict(parent=self._instance)
223  
224          if not self._instance.returns:
225              returns = None
226          else:
227              returns = self._instance.returns
228  
229          args = self._instance.args
230          pair_annotations = itertools.chain(
231              zip(args.args or [], args.annotations),
232              zip(args.kwonlyargs, args.kwonlyargs_annotations),
233              zip(args.posonlyargs or [], args.posonlyargs_annotations),
234          )
235  
236          annotations = {
237              arg.name: annotation for (arg, annotation) in pair_annotations if annotation
238          }
239          if args.varargannotation:
240              annotations[args.vararg] = args.varargannotation
241          if args.kwargannotation:
242              annotations[args.kwarg] = args.kwargannotation
243          if returns:
244              annotations["return"] = returns
245  
246          items = [
247              (node_classes.Const(key, parent=obj), value)
248              for (key, value) in annotations.items()
249          ]
250  
251          obj.postinit(items)
252          return obj

260      @property
261      def attr___kwdefaults__(self):
262          def _default_args(args, parent):
263              for arg in args.kwonlyargs:
264                  try:
265                      default = args.default_value(arg.name)
266                  except exceptions.NoDefault:
267                      continue
268  
269                  name = node_classes.Const(arg.name, parent=parent)
270                  yield name, default
271  
272          args = self._instance.args
273          obj = node_classes.Dict(parent=self._instance)
274          defaults = dict(_default_args(args, obj))
275  
276          obj.postinit(list(defaults.items()))
277          return obj

279      @property
280      def attr___module__(self):
281          return node_classes.Const(self._instance.root().qname())
282  
283      @property
284      def attr___get__(self):
285          # pylint: disable=import-outside-toplevel; circular import
286          from astroid import bases
287  
288          func = self._instance
289  
290          class DescriptorBoundMethod(bases.BoundMethod):
291              """Bound method which knows how to understand calling descriptor binding."""
292  
293              def implicit_parameters(self):
294                  # Different than BoundMethod since the signature
295                  # is different.
296                  return 0
297  
298              def infer_call_result(self, caller, context=None):
299                  if len(caller.args) > 2 or len(caller.args) < 1:
300                      raise exceptions.InferenceError(
301                          "Invalid arguments for descriptor binding",
302                          target=self,
303                          context=context,
304                      )
305  
306                  context = contextmod.copy_context(context)
307                  cls = next(caller.args[0].infer(context=context))
308  
309                  if cls is astroid.Uninferable:
310                      raise exceptions.InferenceError(
311                          "Invalid class inferred", target=self, context=context
312                      )
313  
314                  # For some reason func is a Node that the below
315                  # code is not expecting
316                  if isinstance(func, bases.BoundMethod):
317                      yield func
318                      return
319  
320                  # Rebuild the original value, but with the parent set as the
321                  # class where it will be bound.
322                  new_func = func.__class__(
323                      name=func.name,
324                      doc=func.doc,
325                      lineno=func.lineno,
326                      col_offset=func.col_offset,
327                  )
328                  # pylint: disable=no-member
329                  new_func.postinit(func.args, func.body, func.decorators, func.returns)
330  
331                  # Build a proper bound method that points to our newly built function.
332                  proxy = bases.UnboundMethod(new_func)
333                  yield bases.BoundMethod(proxy=proxy, bound=cls)
334  
335              @property
336              def args(self):
337                  """Overwrite the underlying args to match those of the underlying func 
338   
339                  Usually the underlying *func* is a function/method, as in: 
340   
341                      def test(self): 
342                          pass 
343   
344                  This has only the *self* parameter but when we access test.__get__ 
345                  we get a new object which has two parameters, *self* and *type*. 
346                  """
347                  nonlocal func
348                  positional_or_keyword_params = func.args.args.copy()
349                  positional_or_keyword_params.append(astroid.AssignName(name="type"))
350  
351                  positional_only_params = func.args.posonlyargs.copy()
352  
353                  arguments = astroid.Arguments(parent=func.args.parent)
354                  arguments.postinit(
355                      args=positional_or_keyword_params,
356                      posonlyargs=positional_only_params,
357                      defaults=[],
358                      kwonlyargs=[],
359                      kw_defaults=[],
360                      annotations=[],
361                  )
362                  return arguments
363  
364          return DescriptorBoundMethod(proxy=self._instance, bound=self._instance)
365

413          if not self._instance.newstyle:

423      @property
424      def attr_mro(self):
425          if not self._instance.newstyle:
426              raise exceptions.AttributeInferenceError(
427                  target=self._instance, attribute="mro"
428              )
429  
430          # pylint: disable=import-outside-toplevel; circular import
431          from astroid import bases
432  
433          other_self = self
434  
435          # Cls.mro is a method and we need to return one in order to have a proper inference.
436          # The method we're returning is capable of inferring the underlying MRO though.
437          class MroBoundMethod(bases.BoundMethod):
438              def infer_call_result(self, caller, context=None):
439                  yield other_self.attr___mro__
440  
441          implicit_metaclass = self._instance.implicit_metaclass()
442          mro_method = implicit_metaclass.locals["mro"][0]
443          return MroBoundMethod(proxy=mro_method, bound=implicit_metaclass)
444

447          obj = node_classes.Tuple()
448          context = contextmod.InferenceContext()
449          elts = list(self._instance._inferred_bases(context))
450          obj.postinit(elts=elts)
451          return obj

460      @property
461      def attr___subclasses__(self):
462          """Get the subclasses of the underlying class 
463   
464          This looks only in the current module for retrieving the subclasses, 
465          thus it might miss a couple of them. 
466          """
467          # pylint: disable=import-outside-toplevel; circular import
468          from astroid import bases
469          from astroid import scoped_nodes
470  
471          if not self._instance.newstyle:
472              raise exceptions.AttributeInferenceError(
473                  target=self._instance, attribute="__subclasses__"
474              )
475  
476          qname = self._instance.qname()
477          root = self._instance.root()
478          classes = [
479              cls
480              for cls in root.nodes_of_class(scoped_nodes.ClassDef)
481              if cls != self._instance and cls.is_subtype_of(qname)
482          ]
483  
484          obj = node_classes.List(parent=self._instance)
485          obj.postinit(classes)
486  
487          class SubclassesBoundMethod(bases.BoundMethod):
488              def infer_call_result(self, caller, context=None):
489                  yield obj
490  
491          implicit_metaclass = self._instance.implicit_metaclass()
492          subclasses_method = implicit_metaclass.locals["__subclasses__"][0]
493          return SubclassesBoundMethod(proxy=subclasses_method, bound=implicit_metaclass)
494

549  class GeneratorModel(FunctionModel):
550      def __new__(cls, *args, **kwargs):
551          # Append the values from the GeneratorType unto this object.
552          ret = super(GeneratorModel, cls).__new__(cls, *args, **kwargs)
553          generator = astroid.MANAGER.builtins_module["generator"]
554          for name, values in generator.locals.items():
555              method = values[0]
556              patched = lambda cls, meth=method: meth
557  
558              setattr(type(ret), IMPL_PREFIX + name, property(patched))
559  
560          return ret

685  class DictModel(ObjectModel):
686      @property
687      def attr___class__(self):
688          return self._instance._proxied
689  
690      def _generic_dict_attribute(self, obj, name):
691          """Generate a bound method that can infer the given *obj*."""
692  
693          class DictMethodBoundMethod(astroid.BoundMethod):
694              def infer_call_result(self, caller, context=None):
695                  yield obj
696  
697          meth = next(self._instance._proxied.igetattr(name))
698          return DictMethodBoundMethod(proxy=meth, bound=self._instance)
699  
700      @property
701      def attr_items(self):
702          elems = []
703          obj = node_classes.List(parent=self._instance)
704          for key, value in self._instance.items:
705              elem = node_classes.Tuple(parent=obj)
706              elem.postinit((key, value))
707              elems.append(elem)
708          obj.postinit(elts=elems)
709  
710          # pylint: disable=import-outside-toplevel; circular import
711          from astroid import objects
712  
713          obj = objects.DictItems(obj)
714          return self._generic_dict_attribute(obj, "items")
715  
716      @property
717      def attr_keys(self):
718          keys = [key for (key, _) in self._instance.items]
719          obj = node_classes.List(parent=self._instance)
720          obj.postinit(elts=keys)
721  
722          # pylint: disable=import-outside-toplevel; circular import
723          from astroid import objects
724  
725          obj = objects.DictKeys(obj)
726          return self._generic_dict_attribute(obj, "keys")
727  
728      @property
729      def attr_values(self):
730  
731          values = [value for (_, value) in self._instance.items]
732          obj = node_classes.List(parent=self._instance)
733          obj.postinit(values)
734  
735          # pylint: disable=import-outside-toplevel; circular import
736          from astroid import objects
737  
738          obj = objects.DictValues(obj)
739          return self._generic_dict_attribute(obj, "values")

742  class PropertyModel(ObjectModel):
743      """Model for a builtin property"""
744  
745      # pylint: disable=import-outside-toplevel
746      def _init_function(self, name):
747          from astroid.node_classes import Arguments
748          from astroid.scoped_nodes import FunctionDef
749  
750          args = Arguments()
751          args.postinit(
752              args=[],
753              defaults=[],
754              kwonlyargs=[],
755              kw_defaults=[],
756              annotations=[],
757              posonlyargs=[],
758              posonlyargs_annotations=[],
759              kwonlyargs_annotations=[],
760          )
761  
762          function = FunctionDef(name=name, parent=self._instance)
763  
764          function.postinit(args=args, body=[])
765          return function
766  
767      @property
768      def attr_fget(self):
769          from astroid.scoped_nodes import FunctionDef
770  
771          func = self._instance
772  
773          class PropertyFuncAccessor(FunctionDef):
774              def infer_call_result(self, caller=None, context=None):
775                  nonlocal func
776                  if caller and len(caller.args) != 1:
777                      raise exceptions.InferenceError(
778                          "fget() needs a single argument", target=self, context=context
779                      )
780  
781                  yield from func.function.infer_call_result(
782                      caller=caller, context=context
783                  )
784  
785          property_accessor = PropertyFuncAccessor(name="fget", parent=self._instance)
786          property_accessor.postinit(args=func.args, body=func.body)
787          return property_accessor
788  
789      @property
790      def attr_setter(self):
791          return self._init_function("setter")
792  
793      @property
794      def attr_deleter(self):
795          return self._init_function("deleter")

799          return self._init_function("getter")
