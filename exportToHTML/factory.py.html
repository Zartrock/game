115      def _make_candidate_from_dist(
116          self,
117          dist,  # type: Distribution
118          extras,  # type: FrozenSet[str]
119          template,  # type: InstallRequirement
120      ):
121          # type: (...) -> Candidate
122          base = AlreadyInstalledCandidate(dist, template, factory=self)
123          if extras:
124              return ExtrasCandidate(base, extras)
125          return base

127      def _make_candidate_from_link(
128          self,
129          link,  # type: Link
130          extras,  # type: FrozenSet[str]
131          template,  # type: InstallRequirement
132          name,  # type: Optional[str]
133          version,  # type: Optional[_BaseVersion]
134      ):
135          # type: (...) -> Candidate
136          # TODO: Check already installed candidate, and use it if the link and
137          # editable flag match.
138          if template.editable:
139              if link not in self._editable_candidate_cache:
140                  self._editable_candidate_cache[link] = EditableCandidate(
141                      link, template, factory=self, name=name, version=version,
142                  )
143              base = self._editable_candidate_cache[link]  # type: BaseCandidate
144          else:
145              if link not in self._link_candidate_cache:
146                  self._link_candidate_cache[link] = LinkCandidate(
147                      link, template, factory=self, name=name, version=version,
148                  )
149              base = self._link_candidate_cache[link]
150          if extras:
151              return ExtrasCandidate(base, extras)
152          return base

154      def _iter_found_candidates(
155          self,
156          ireqs,  # type: Sequence[InstallRequirement]
157          specifier,  # type: SpecifierSet
158          hashes,  # type: Hashes
159          prefers_installed,  # type: bool
160      ):
161          # type: (...) -> Iterable[Candidate]
162          if not ireqs:
163              return ()
164  
165          # The InstallRequirement implementation requires us to give it a
166          # "template". Here we just choose the first requirement to represent
167          # all of them.
168          # Hopefully the Project model can correct this mismatch in the future.
169          template = ireqs[0]
170          name = canonicalize_name(template.req.name)
171  
172          extras = frozenset()  # type: FrozenSet[str]
173          for ireq in ireqs:
174              specifier &= ireq.req.specifier
175              hashes &= ireq.hashes(trust_internet=False)
176              extras |= frozenset(ireq.extras)
177  
178          # Get the installed version, if it matches, unless the user
179          # specified `--force-reinstall`, when we want the version from
180          # the index instead.
181          installed_candidate = None
182          if not self._force_reinstall and name in self._installed_dists:
183              installed_dist = self._installed_dists[name]
184              if specifier.contains(installed_dist.version, prereleases=True):
185                  installed_candidate = self._make_candidate_from_dist(
186                      dist=installed_dist,
187                      extras=extras,
188                      template=template,
189                  )
190  
191          def iter_index_candidates():
192              # type: () -> Iterator[Candidate]
193              result = self._finder.find_best_candidate(
194                  project_name=name,
195                  specifier=specifier,
196                  hashes=hashes,
197              )
198              # PackageFinder returns earlier versions first, so we reverse.
199              for ican in reversed(list(result.iter_applicable())):
200                  yield self._make_candidate_from_link(
201                      link=ican.link,
202                      extras=extras,
203                      template=template,
204                      name=name,
205                      version=ican.version,
206                  )
207  
208          return FoundCandidates(
209              iter_index_candidates,
210              installed_candidate,
211              prefers_installed,
212          )

214      def find_candidates(
215          self,
216          requirements,  # type: Sequence[Requirement]
217          constraint,  # type: Constraint
218          prefers_installed,  # type: bool
219      ):
220          # type: (...) -> Iterable[Candidate]
221          explicit_candidates = set()  # type: Set[Candidate]
222          ireqs = []  # type: List[InstallRequirement]
223          for req in requirements:
224              cand, ireq = req.get_candidate_lookup()
225              if cand is not None:
226                  explicit_candidates.add(cand)
227              if ireq is not None:
228                  ireqs.append(ireq)
229  
230          # If none of the requirements want an explicit candidate, we can ask
231          # the finder for candidates.
232          if not explicit_candidates:
233              return self._iter_found_candidates(
234                  ireqs,
235                  constraint.specifier,
236                  constraint.hashes,
237                  prefers_installed,
238              )
239  
240          if constraint:
241              name = explicit_candidates.pop().name
242              raise InstallationError(
243                  "Could not satisfy constraints for {!r}: installation from "
244                  "path or url cannot be constrained to a version".format(name)
245              )
246  
247          return (
248              c for c in explicit_candidates
249              if all(req.is_satisfied_by(c) for req in requirements)
250          )

252      def make_requirement_from_install_req(self, ireq, requested_extras):
253          # type: (InstallRequirement, Iterable[str]) -> Optional[Requirement]
254          if not ireq.match_markers(requested_extras):
255              logger.info(
256                  "Ignoring %s: markers '%s' don't match your environment",
257                  ireq.name, ireq.markers,
258              )
259              return None
260          if not ireq.link:
261              return SpecifierRequirement(ireq)
262          if ireq.link.is_wheel:
263              wheel = Wheel(ireq.link.filename)
264              if not wheel.supported(self._finder.target_python.get_tags()):
265                  msg = "{} is not a supported wheel on this platform.".format(
266                      wheel.filename,
267                  )
268                  raise UnsupportedWheel(msg)
269          cand = self._make_candidate_from_link(
270              ireq.link,
271              extras=frozenset(ireq.extras),
272              template=ireq,
273              name=canonicalize_name(ireq.name) if ireq.name else None,
274              version=None,
275          )
276          return self.make_requirement_from_candidate(cand)

293          # type: (Optional[SpecifierSet]) -> Optional[Requirement]

298      def get_wheel_cache_entry(self, link, name):
299          # type: (Link, Optional[str]) -> Optional[CacheEntry]
300          """Look up the link in the wheel cache. 
301   
302          If ``preparer.require_hashes`` is True, don't use the wheel cache, 
303          because cached wheels, always built locally, have different hashes 
304          than the files downloaded from the index server and thus throw false 
305          hash mismatches. Furthermore, cached wheels at present have 
306          nondeterministic contents due to file modification times. 
307          """
308          if self._wheel_cache is None or self.preparer.require_hashes:
309              return None
310          return self._wheel_cache.get_cache_entry(
311              link=link,
312              package_name=name,
313              supported_tags=get_supported(),
314          )

316      def get_dist_to_uninstall(self, candidate):
317          # type: (Candidate) -> Optional[Distribution]
318          # TODO: Are there more cases this needs to return True? Editable?
319          dist = self._installed_dists.get(candidate.name)
320          if dist is None:  # Not installed, no uninstallation required.
321              return None
322  
323          # We're installing into global site. The current installation must
324          # be uninstalled, no matter it's in global or user site, because the
325          # user site installation has precedence over global.
326          if not self._use_user_site:
327              return dist
328  
329          # We're installing into user site. Remove the user site installation.
330          if dist_in_usersite(dist):
331              return dist
332  
333          # We're installing into user site, but the installed incompatible
334          # package is in global site. We can't uninstall that, and would let
335          # the new user installation to "shadow" it. But shadowing won't work
336          # in virtual environments, so we error out.
337          if running_under_virtualenv() and dist_in_site_packages(dist):
338              raise InstallationError(
339                  "Will not install to the user site because it will "
340                  "lack sys.path precedence to {} in {}".format(
341                      dist.project_name, dist.location,
342                  )
343              )
344          return None

346      def _report_requires_python_error(
347          self,
348          requirement,  # type: RequiresPythonRequirement
349          template,  # type: Candidate
350      ):
351          # type: (...) -> UnsupportedPythonVersion
352          message_format = (
353              "Package {package!r} requires a different Python: "
354              "{version} not in {specifier!r}"
355          )
356          message = message_format.format(
357              package=template.name,
358              version=self._python_candidate.version,
359              specifier=str(requirement.specifier),
360          )
361          return UnsupportedPythonVersion(message)

363      def get_installation_error(self, e):
364          # type: (ResolutionImpossible) -> InstallationError
365  
366          assert e.causes, "Installation error reported with no cause"
367  
368          # If one of the things we can't solve is "we need Python X.Y",
369          # that is what we report.
370          for cause in e.causes:
371              if isinstance(cause.requirement, RequiresPythonRequirement):
372                  return self._report_requires_python_error(
373                      cause.requirement,
374                      cause.parent,
375                  )
376  
377          # Otherwise, we have a set of causes which can't all be satisfied
378          # at once.
379  
380          # The simplest case is when we have *one* cause that can't be
381          # satisfied. We just report that case.
382          if len(e.causes) == 1:
383              req, parent = e.causes[0]
384              if parent is None:
385                  req_disp = str(req)
386              else:
387                  req_disp = '{} (from {})'.format(req, parent.name)
388              logger.critical(
389                  "Could not find a version that satisfies the requirement %s",
390                  req_disp,
391              )
392              return DistributionNotFound(
393                  'No matching distribution found for {}'.format(req)
394              )
395  
396          # OK, we now have a list of requirements that can't all be
397          # satisfied at once.
398  
399          # A couple of formatting helpers
400          def text_join(parts):
401              # type: (List[str]) -> str
402              if len(parts) == 1:
403                  return parts[0]
404  
405              return ", ".join(parts[:-1]) + " and " + parts[-1]
406  
407          def readable_form(cand):
408              # type: (Candidate) -> str
409              return "{} {}".format(cand.name, cand.version)
410  
411          def describe_trigger(parent):
412              # type: (Candidate) -> str
413              ireq = parent.get_install_requirement()
414              if not ireq or not ireq.comes_from:
415                  return "{} {}".format(parent.name, parent.version)
416              if isinstance(ireq.comes_from, InstallRequirement):
417                  return str(ireq.comes_from.name)
418              return str(ireq.comes_from)
419  
420          triggers = []
421          for req, parent in e.causes:
422              if parent is None:
423                  # This is a root requirement, so we can report it directly
424                  trigger = req.format_for_error()
425              else:
426                  trigger = describe_trigger(parent)
427              triggers.append(trigger)
428  
429          if triggers:
430              info = text_join(triggers)
431          else:
432              info = "the requested packages"
433  
434          msg = "Cannot install {} because these package versions " \
435              "have conflicting dependencies.".format(info)
436          logger.critical(msg)
437          msg = "\nThe conflict is caused by:"
438          for req, parent in e.causes:
439              msg = msg + "\n    "
440              if parent:
441                  msg = msg + "{} {} depends on ".format(
442                      parent.name,
443                      parent.version
444                  )
445              else:
446                  msg = msg + "The user requested "
447              msg = msg + req.format_for_error()
448  
449          msg = msg + "\n\n" + \
450              "To fix this you could try to:\n" + \
451              "1. loosen the range of package versions you've specified\n" + \
452              "2. remove package versions to allow pip attempt to solve " + \
453              "the dependency conflict\n"
454  
455          logger.info(msg)
456  
457          return DistributionNotFound(
458              "ResolutionImpossible: for help visit "
459              "https://pip.pypa.io/en/latest/user_guide/"
460              "#fixing-conflicting-dependencies"
461          )
