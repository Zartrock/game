1    #!/usr/bin/env python
2    """ pygame.examples.aliens 
3     
4    Shows a mini game where you have to defend against aliens. 
5     
6    What does it show you about pygame? 
7     
8    * pg.sprite, the difference between Sprite and Group. 
9    * dirty rectangle optimization for processing for speed. 
10   * music with pg.mixer.music, including fadeout 
11   * sound effects with pg.Sound 
12   * event processing, keyboard handling, QUIT handling. 
13   * a main loop frame limited with a game clock from pg.time.Clock 
14   * fullscreen switching. 
15    
16    
17   Controls 
18   -------- 
19    
20   * Left and right arrows to move. 
21   * Space bar to shoot 
22   * f key to toggle between fullscreen. 
23    
24   """
25   
26   import random
27   import os
28   
29   # import basic pygame modules
30   import pygame as pg
31   
32   # see if we can load more than standard BMP
33   if not pg.image.get_extended():
34       raise SystemExit("Sorry, extended image module required")
35   
36   
37   # game constants
38   MAX_SHOTS = 2  # most player bullets onscreen
39   ALIEN_ODDS = 22  # chances a new alien appears
40   BOMB_ODDS = 60  # chances a new bomb will drop
41   ALIEN_RELOAD = 12  # frames between new aliens
42   SCREENRECT = pg.Rect(0, 0, 640, 480)
43   SCORE = 0
44   
45   main_dir = os.path.split(os.path.abspath(__file__))[0]
46   
47   
48   def load_image(file):
49       """ loads an image, prepares it for play 
50       """
51       file = os.path.join(main_dir, "data", file)
52       try:
53           surface = pg.image.load(file)
54       except pg.error:
55           raise SystemExit('Could not load image "%s" %s' % (file, pg.get_error()))
56       return surface.convert()
57   
58   
59   def load_sound(file):
60       """ because pygame can be be compiled without mixer. 
61       """
62       if not pg.mixer:
63           return None
64       file = os.path.join(main_dir, "data", file)
65       try:
66           sound = pg.mixer.Sound(file)
67           return sound
68       except pg.error:
69           print("Warning, unable to load, %s" % file)
70       return None

73   # Each type of game object gets an init and an update function.
74   # The update function is called once per frame, and it is when each object should
75   # change it's current position and state.
76   #
77   # The Player object actually gets a "move" function instead of update,
78   # since it is passed extra information about the keyboard.
79   
80   
81   class Player(pg.sprite.Sprite):
82       """ Representing the player as a moon buggy type car. 
83       """
84   
85       speed = 10
86       bounce = 24
87       gun_offset = -11
88       images = []
89   
90       def __init__(self):
91           pg.sprite.Sprite.__init__(self, self.containers)
92           self.image = self.images[0]
93           self.rect = self.image.get_rect(midbottom=SCREENRECT.midbottom)
94           self.reloading = 0
95           self.origtop = self.rect.top
96           self.facing = -1
97   
98       def move(self, direction):
99           if direction:
100              self.facing = direction
101          self.rect.move_ip(direction * self.speed, 0)
102          self.rect = self.rect.clamp(SCREENRECT)
103          if direction < 0:
104              self.image = self.images[0]
105          elif direction > 0:
106              self.image = self.images[1]
107          self.rect.top = self.origtop - (self.rect.left // self.bounce % 2)
108  
109      def gunpos(self):
110          pos = self.facing * self.gun_offset + self.rect.centerx
111          return pos, self.rect.top
112

114  class Alien(pg.sprite.Sprite):
115      """ An alien space ship. That slowly moves down the screen. 
116      """
117  
118      speed = 13
119      animcycle = 12
120      images = []
121  
122      def __init__(self):
123          pg.sprite.Sprite.__init__(self, self.containers)
124          self.image = self.images[0]
125          self.rect = self.image.get_rect()
126          self.facing = random.choice((-1, 1)) * Alien.speed
127          self.frame = 0
128          if self.facing < 0:
129              self.rect.right = SCREENRECT.right
130  
131      def update(self):
132          self.rect.move_ip(self.facing, 0)
133          if not SCREENRECT.contains(self.rect):
134              self.facing = -self.facing
135              self.rect.top = self.rect.bottom + 1
136              self.rect = self.rect.clamp(SCREENRECT)
137          self.frame = self.frame + 1
138          self.image = self.images[self.frame // self.animcycle % 3]
139

141  class Explosion(pg.sprite.Sprite):
142      """ An explosion. Hopefully the Alien and not the player! 
143      """
144  
145      defaultlife = 12
146      animcycle = 3
147      images = []
148  
149      def __init__(self, actor):
150          pg.sprite.Sprite.__init__(self, self.containers)
151          self.image = self.images[0]
152          self.rect = self.image.get_rect(center=actor.rect.center)
153          self.life = self.defaultlife
154  
155      def update(self):
156          """ called every time around the game loop. 
157   
158          Show the explosion surface for 'defaultlife'. 
159          Every game tick(update), we decrease the 'life'. 
160   
161          Also we animate the explosion. 
162          """
163          self.life = self.life - 1
164          self.image = self.images[self.life // self.animcycle % 2]
165          if self.life <= 0:
166              self.kill()

169  class Shot(pg.sprite.Sprite):
170      """ a bullet the Player sprite fires. 
171      """
172  
173      speed = -11
174      images = []
175  
176      def __init__(self, pos):
177          pg.sprite.Sprite.__init__(self, self.containers)
178          self.image = self.images[0]
179          self.rect = self.image.get_rect(midbottom=pos)
180  
181      def update(self):
182          """ called every time around the game loop. 
183   
184          Every tick we move the shot upwards. 
185          """
186          self.rect.move_ip(0, self.speed)
187          if self.rect.top <= 0:
188              self.kill()

191  class Bomb(pg.sprite.Sprite):
192      """ A bomb the aliens drop. 
193      """
194  
195      speed = 9
196      images = []
197  
198      def __init__(self, alien):
199          pg.sprite.Sprite.__init__(self, self.containers)
200          self.image = self.images[0]
201          self.rect = self.image.get_rect(midbottom=alien.rect.move(0, 5).midbottom)
202  
203      def update(self):
204          """ called every time around the game loop. 
205   
206          Every frame we move the sprite 'rect' down. 
207          When it reaches the bottom we: 
208   
209          - make an explosion. 
210          - remove the Bomb. 
211          """
212          self.rect.move_ip(0, self.speed)
213          if self.rect.bottom >= 470:
214              Explosion(self)
215              self.kill()

218  class Score(pg.sprite.Sprite):
219      """ to keep track of the score. 
220      """
221  
222      def __init__(self):
223          pg.sprite.Sprite.__init__(self)
224          self.font = pg.font.Font(None, 20)
225          self.font.set_italic(1)
226          self.color = pg.Color("white")
227          self.lastscore = -1
228          self.update()
229          self.rect = self.image.get_rect().move(10, 450)
230  
231      def update(self):
232          """ We only update the score in update() when it has changed. 
233          """
234          if SCORE != self.lastscore:
235              self.lastscore = SCORE
236              msg = "Score: %d" % SCORE
237              self.image = self.font.render(msg, 0, self.color)

240  def main(winstyle=0):
241      # Initialize pygame
242      if pg.get_sdl_version()[0] == 2:
243          pg.mixer.pre_init(44100, 32, 2, 1024)
244      pg.init()
245      if pg.mixer and not pg.mixer.get_init():
246          print("Warning, no sound")
247          pg.mixer = None
248  
249      fullscreen = False
250      # Set the display mode
251      winstyle = 0  # |FULLSCREEN
252      bestdepth = pg.display.mode_ok(SCREENRECT.size, winstyle, 32)
253      screen = pg.display.set_mode(SCREENRECT.size, winstyle, bestdepth)
254  
255      # Load images, assign to sprite classes
256      # (do this before the classes are used, after screen setup)
257      img = load_image("player1.gif")
258      Player.images = [img, pg.transform.flip(img, 1, 0)]
259      img = load_image("explosion1.gif")
260      Explosion.images = [img, pg.transform.flip(img, 1, 1)]
261      Alien.images = [load_image(im) for im in ("alien1.gif", "alien2.gif", "alien3.gif")]
262      Bomb.images = [load_image("bomb.gif")]
263      Shot.images = [load_image("shot.gif")]
264  
265      # decorate the game window
266      icon = pg.transform.scale(Alien.images[0], (32, 32))
267      pg.display.set_icon(icon)
268      pg.display.set_caption("Pygame Aliens")
269      pg.mouse.set_visible(0)
270  
271      # create the background, tile the bgd image
272      bgdtile = load_image("background.gif")
273      background = pg.Surface(SCREENRECT.size)
274      for x in range(0, SCREENRECT.width, bgdtile.get_width()):
275          background.blit(bgdtile, (x, 0))
276      screen.blit(background, (0, 0))
277      pg.display.flip()
278  
279      # load the sound effects
280      boom_sound = load_sound("boom.wav")
281      shoot_sound = load_sound("car_door.wav")
282      if pg.mixer:
283          music = os.path.join(main_dir, "data", "house_lo.wav")
284          pg.mixer.music.load(music)
285          pg.mixer.music.play(-1)
286  
287      # Initialize Game Groups
288      aliens = pg.sprite.Group()
289      shots = pg.sprite.Group()
290      bombs = pg.sprite.Group()
291      all = pg.sprite.RenderUpdates()
292      lastalien = pg.sprite.GroupSingle()
293  
294      # assign default groups to each sprite class
295      Player.containers = all
296      Alien.containers = aliens, all, lastalien
297      Shot.containers = shots, all
298      Bomb.containers = bombs, all
299      Explosion.containers = all
300      Score.containers = all
301  
302      # Create Some Starting Values
303      global score
304      alienreload = ALIEN_RELOAD
305      clock = pg.time.Clock()
306  
307      # initialize our starting sprites
308      global SCORE
309      player = Player()
310      Alien()  # note, this 'lives' because it goes into a sprite group
311      if pg.font:
312          all.add(Score())
313  
314      # Run our main loop whilst the player is alive.
315      while player.alive():
316  
317          # get input
318          for event in pg.event.get():
319              if event.type == pg.QUIT:
320                  return
321              if event.type == pg.KEYDOWN and event.key == pg.K_ESCAPE:
322                  return
323              elif event.type == pg.KEYDOWN:
324                  if event.key == pg.K_f:
325                      if not fullscreen:
326                          print("Changing to FULLSCREEN")
327                          screen_backup = screen.copy()
328                          screen = pg.display.set_mode(
329                              SCREENRECT.size, winstyle | pg.FULLSCREEN, bestdepth
330                          )
331                          screen.blit(screen_backup, (0, 0))
332                      else:
333                          print("Changing to windowed mode")
334                          screen_backup = screen.copy()
335                          screen = pg.display.set_mode(
336                              SCREENRECT.size, winstyle, bestdepth
337                          )
338                          screen.blit(screen_backup, (0, 0))
339                      pg.display.flip()
340                      fullscreen = not fullscreen
341  
342          keystate = pg.key.get_pressed()
343  
344          # clear/erase the last drawn sprites
345          all.clear(screen, background)
346  
347          # update all the sprites
348          all.update()
349  
350          # handle player input
351          direction = keystate[pg.K_RIGHT] - keystate[pg.K_LEFT]
352          player.move(direction)
353          firing = keystate[pg.K_SPACE]
354          if not player.reloading and firing and len(shots) < MAX_SHOTS:
355              Shot(player.gunpos())
356              if pg.mixer:
357                  shoot_sound.play()
358          player.reloading = firing
359  
360          # Create new alien
361          if alienreload:
362              alienreload = alienreload - 1
363          elif not int(random.random() * ALIEN_ODDS):
364              Alien()
365              alienreload = ALIEN_RELOAD
366  
367          # Drop bombs
368          if lastalien and not int(random.random() * BOMB_ODDS):
369              Bomb(lastalien.sprite)
370  
371          # Detect collisions between aliens and players.
372          for alien in pg.sprite.spritecollide(player, aliens, 1):
373              if pg.mixer:
374                  boom_sound.play()
375              Explosion(alien)
376              Explosion(player)
377              SCORE = SCORE + 1
378              player.kill()
379  
380          # See if shots hit the aliens.
381          for alien in pg.sprite.groupcollide(shots, aliens, 1, 1).keys():
382              if pg.mixer:
383                  boom_sound.play()
384              Explosion(alien)
385              SCORE = SCORE + 1
386  
387          # See if alien boms hit the player.
388          for bomb in pg.sprite.spritecollide(player, bombs, 1):
389              if pg.mixer:
390                  boom_sound.play()
391              Explosion(player)
392              Explosion(bomb)
393              player.kill()
394  
395          # draw the scene
396          dirty = all.draw(screen)
397          pg.display.update(dirty)
398  
399          # cap the framerate at 40fps. Also called 40HZ or 40 times per second.
400          clock.tick(40)
401  
402      if pg.mixer:
403          pg.mixer.music.fadeout(1000)
404      pg.time.wait(1000)
405      pg.quit()
406  
407  
408  # call the "main" function if running this script
409  if __name__ == "__main__":
410      main()
