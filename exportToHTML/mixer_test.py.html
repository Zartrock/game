240      def test_array_keyword(self):
241          try:
242              from numpy import (
243                  array,
244                  arange,
245                  zeros,
246                  int8,
247                  uint8,
248                  int16,
249                  uint16,
250                  int32,
251                  uint32,
252              )
253          except ImportError:
254              self.skipTest("requires numpy")
255  
256          freq = 22050
257          format_list = [-8, 8, -16, 16]
258          channels_list = [1, 2]
259  
260          a_lists = dict((f, []) for f in format_list)
261          a32u_mono = arange(0, 256, 1, uint32)
262          a16u_mono = a32u_mono.astype(uint16)
263          a8u_mono = a32u_mono.astype(uint8)
264          au_list_mono = [(1, a) for a in [a8u_mono, a16u_mono, a32u_mono]]
265          for format in format_list:
266              if format > 0:
267                  a_lists[format].extend(au_list_mono)
268          a32s_mono = arange(-128, 128, 1, int32)
269          a16s_mono = a32s_mono.astype(int16)
270          a8s_mono = a32s_mono.astype(int8)
271          as_list_mono = [(1, a) for a in [a8s_mono, a16s_mono, a32s_mono]]
272          for format in format_list:
273              if format < 0:
274                  a_lists[format].extend(as_list_mono)
275          a32u_stereo = zeros([a32u_mono.shape[0], 2], uint32)
276          a32u_stereo[:, 0] = a32u_mono
277          a32u_stereo[:, 1] = 255 - a32u_mono
278          a16u_stereo = a32u_stereo.astype(uint16)
279          a8u_stereo = a32u_stereo.astype(uint8)
280          au_list_stereo = [(2, a) for a in [a8u_stereo, a16u_stereo, a32u_stereo]]
281          for format in format_list:
282              if format > 0:
283                  a_lists[format].extend(au_list_stereo)
284          a32s_stereo = zeros([a32s_mono.shape[0], 2], int32)
285          a32s_stereo[:, 0] = a32s_mono
286          a32s_stereo[:, 1] = -1 - a32s_mono
287          a16s_stereo = a32s_stereo.astype(int16)
288          a8s_stereo = a32s_stereo.astype(int8)
289          as_list_stereo = [(2, a) for a in [a8s_stereo, a16s_stereo, a32s_stereo]]
290          for format in format_list:
291              if format < 0:
292                  a_lists[format].extend(as_list_stereo)
293  
294          for format in format_list:
295              for channels in channels_list:
296                  try:
297                      mixer.init(freq, format, channels)
298                  except pygame.error:
299                      # Some formats (e.g. 16) may not be supported.
300                      continue
301                  try:
302                      __, f, c = mixer.get_init()
303                      if f != format or c != channels:
304                          # Some formats (e.g. -8) may not be supported.
305                          continue
306                      for c, a in a_lists[format]:
307                          self._test_array_argument(format, a, c == channels)
308                  finally:
309                      mixer.quit()

311      def _test_array_argument(self, format, a, test_pass):
312          from numpy import array, all as all_
313  
314          try:
315              snd = mixer.Sound(array=a)
316          except ValueError:
317              if not test_pass:
318                  return
319              self.fail("Raised ValueError: Format %i, dtype %s" % (format, a.dtype))
320          if not test_pass:
321              self.fail(
322                  "Did not raise ValueError: Format %i, dtype %s" % (format, a.dtype)
323              )
324          a2 = array(snd)
325          a3 = a.astype(a2.dtype)
326          lshift = abs(format) - 8 * a.itemsize
327          if lshift >= 0:
328              # This is asymmetric with respect to downcasting.
329              a3 <<= lshift
330          self.assertTrue(all_(a2 == a3), "Format %i, dtype %s" % (format, a.dtype))

363          ndim = 1 if (channels == 1) else 2
364          itemsize = abs(fmt) // 8
365          formats = {
366              8: "B",
367              -8: "b",
368              16: "=H",
369              -16: "=h",
370              32: "=I",
371              -32: "=i",  # 32 and 64 for future consideration
372              64: "=Q",
373              -64: "=q",
374          }
375          format = formats[fmt]
376          from pygame.tests.test_utils import buftools
377  
378          Exporter = buftools.Exporter

380          is_lil_endian = pygame.get_sdl_byteorder() == pygame.LIL_ENDIAN
