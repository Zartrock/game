52   def infer_end(self, context=None):
53       """Inference's end for nodes that yield themselves on inference 
54    
55       These are objects for which inference does not have any semantic, 
56       such as Module or Consts. 
57       """
58       yield self

68   def _infer_sequence_helper(node, context=None):
69       """Infer all values based on _BaseContainer.elts"""
70       values = []
71   
72       for elt in node.elts:
73           if isinstance(elt, nodes.Starred):
74               starred = helpers.safe_infer(elt.value, context)
75               if not starred:
76                   raise exceptions.InferenceError(node=node, context=context)
77               if not hasattr(starred, "elts"):
78                   raise exceptions.InferenceError(node=node, context=context)
79               values.extend(_infer_sequence_helper(starred))
80           elif isinstance(elt, nodes.NamedExpr):
81               value = helpers.safe_infer(elt.value, context)
82               if not value:
83                   raise exceptions.InferenceError(node=node, context=context)
84               values.append(value)
85           else:
86               values.append(elt)
87       return values

90   @decorators.raise_if_nothing_inferred
91   def infer_sequence(self, context=None):
92       has_starred_named_expr = any(
93           isinstance(e, (nodes.Starred, nodes.NamedExpr)) for e in self.elts
94       )
95       if has_starred_named_expr:
96           values = _infer_sequence_helper(self, context)
97           new_seq = type(self)(
98               lineno=self.lineno, col_offset=self.col_offset, parent=self.parent
99           )
100          new_seq.postinit(values)
101  
102          yield new_seq
103      else:
104          yield self

112  def infer_map(self, context=None):
113      if not any(isinstance(k, nodes.DictUnpack) for k, _ in self.items):
114          yield self
115      else:
116          items = _infer_map(self, context)
117          new_seq = type(self)(self.lineno, self.col_offset, self.parent)
118          new_seq.postinit(list(items.items()))
119          yield new_seq

122  def _update_with_replacement(lhs_dict, rhs_dict):
123      """Delete nodes that equate to duplicate keys 
124   
125      Since an astroid node doesn't 'equal' another node with the same value, 
126      this function uses the as_string method to make sure duplicate keys 
127      don't get through 
128   
129      Note that both the key and the value are astroid nodes 
130   
131      Fixes issue with DictUnpack causing duplicte keys 
132      in inferred Dict items 
133   
134      :param dict(nodes.NodeNG, nodes.NodeNG) lhs_dict: Dictionary to 'merge' nodes into 
135      :param dict(nodes.NodeNG, nodes.NodeNG) rhs_dict: Dictionary with nodes to pull from 
136      :return dict(nodes.NodeNG, nodes.NodeNG): merged dictionary of nodes 
137      """
138      combined_dict = itertools.chain(lhs_dict.items(), rhs_dict.items())
139      # Overwrite keys which have the same string values
140      string_map = {key.as_string(): (key, value) for key, value in combined_dict}
141      # Return to dictionary
142      return dict(string_map.values())

145  def _infer_map(node, context):
146      """Infer all values based on Dict.items"""
147      values = {}
148      for name, value in node.items:
149          if isinstance(name, nodes.DictUnpack):
150              double_starred = helpers.safe_infer(value, context)
151              if not double_starred:
152                  raise exceptions.InferenceError
153              if not isinstance(double_starred, nodes.Dict):
154                  raise exceptions.InferenceError(node=node, context=context)
155              unpack_items = _infer_map(double_starred, context)
156              values = _update_with_replacement(values, unpack_items)
157          else:
158              key = helpers.safe_infer(name, context=context)
159              value = helpers.safe_infer(value, context=context)
160              if any(not elem for elem in (key, value)):
161                  raise exceptions.InferenceError(node=node, context=context)
162              values = _update_with_replacement(values, {key: value})
163      return values

169  def _higher_function_scope(node):
170      """ Search for the first function which encloses the given 
171      scope. This can be used for looking up in that function's 
172      scope, in case looking up in a lower scope for a particular 
173      name fails. 
174   
175      :param node: A scope node. 
176      :returns: 
177          ``None``, if no parent function scope was found, 
178          otherwise an instance of :class:`astroid.scoped_nodes.Function`, 
179          which encloses the given node. 
180      """
181      current = node
182      while current.parent and not isinstance(current.parent, nodes.FunctionDef):
183          current = current.parent
184      if current and current.parent:
185          return current.parent
186      return None

189  def infer_name(self, context=None):
190      """infer a Name: use name lookup rules"""
191      frame, stmts = self.lookup(self.name)
192      if not stmts:
193          # Try to see if the name is enclosed in a nested function
194          # and use the higher (first function) scope for searching.
195          parent_function = _higher_function_scope(self.scope())
196          if parent_function:
197              _, stmts = parent_function.lookup(self.name)
198  
199          if not stmts:
200              raise exceptions.NameInferenceError(
201                  name=self.name, scope=self.scope(), context=context
202              )
203      context = contextmod.copy_context(context)
204      context.lookupname = self.name
205      return bases._infer_stmts(stmts, context, frame)

215  @decorators.raise_if_nothing_inferred
216  @decorators.path_wrapper
217  def infer_call(self, context=None):
218      """infer a Call node by trying to guess what the function returns"""
219      callcontext = contextmod.copy_context(context)
220      callcontext.callcontext = contextmod.CallContext(
221          args=self.args, keywords=self.keywords
222      )
223      callcontext.boundnode = None
224      if context is not None:
225          callcontext.extra_context = _populate_context_lookup(self, context.clone())
226  
227      for callee in self.func.infer(context):
228          if callee is util.Uninferable:
229              yield callee
230              continue
231          try:
232              if hasattr(callee, "infer_call_result"):
233                  yield from callee.infer_call_result(caller=self, context=callcontext)
234          except exceptions.InferenceError:
235              continue
236      return dict(node=self, context=context)

243  @decorators.path_wrapper
244  def infer_import(self, context=None, asname=True):
245      """infer an Import node: return the imported module/object"""
246      name = context.lookupname
247      if name is None:
248          raise exceptions.InferenceError(node=self, context=context)
249  
250      try:
251          if asname:
252              yield self.do_import_module(self.real_name(name))
253          else:
254              yield self.do_import_module(name)
255      except exceptions.AstroidBuildingError as exc:
256          raise exceptions.InferenceError(node=self, context=context) from exc

262  @decorators.raise_if_nothing_inferred
263  @decorators.path_wrapper
264  def infer_import_from(self, context=None, asname=True):
265      """infer a ImportFrom node: return the imported module/object"""
266      name = context.lookupname
267      if name is None:
268          raise exceptions.InferenceError(node=self, context=context)
269      if asname:
270          name = self.real_name(name)
271  
272      try:
273          module = self.do_import_module()
274      except exceptions.AstroidBuildingError as exc:
275          raise exceptions.InferenceError(node=self, context=context) from exc
276  
277      try:
278          context = contextmod.copy_context(context)
279          context.lookupname = name
280          stmts = module.getattr(name, ignore_locals=module is self.root())
281          return bases._infer_stmts(stmts, context)
282      except exceptions.AttributeInferenceError as error:
283          raise exceptions.InferenceError(
284              error.message, target=self, attribute=name, context=context
285          ) from error

291  def infer_attribute(self, context=None):
292      """infer an Attribute node by using getattr on the associated object"""
293      for owner in self.expr.infer(context):
294          if owner is util.Uninferable:
295              yield owner
296              continue
297  
298          if context and context.boundnode:
299              # This handles the situation where the attribute is accessed through a subclass
300              # of a base class and the attribute is defined at the base class's level,
301              # by taking in consideration a redefinition in the subclass.
302              if isinstance(owner, bases.Instance) and isinstance(
303                  context.boundnode, bases.Instance
304              ):
305                  try:
306                      if helpers.is_subtype(
307                          helpers.object_type(context.boundnode),
308                          helpers.object_type(owner),
309                      ):
310                          owner = context.boundnode
311                  except exceptions._NonDeducibleTypeHierarchy:
312                      # Can't determine anything useful.
313                      pass
314          elif not context:
315              context = contextmod.InferenceContext()
316  
317          try:
318              context.boundnode = owner
319              yield from owner.igetattr(self.attrname, context)
320          except (
321              exceptions.AttributeInferenceError,
322              exceptions.InferenceError,
323              AttributeError,
324          ):
325              pass
326          finally:
327              context.boundnode = None
328      return dict(node=self, context=context)

357  def infer_subscript(self, context=None):
358      """Inference for subscripts 
359   
360      We're understanding if the index is a Const 
361      or a slice, passing the result of inference 
362      to the value's `getitem` method, which should 
363      handle each supported index type accordingly. 
364      """
365  
366      found_one = False
367      for value in self.value.infer(context):
368          if value is util.Uninferable:
369              yield util.Uninferable
370              return None
371          for index in self.slice.infer(context):
372              if index is util.Uninferable:
373                  yield util.Uninferable
374                  return None
375  
376              # Try to deduce the index value.
377              index_value = _SUBSCRIPT_SENTINEL
378              if value.__class__ == bases.Instance:
379                  index_value = index
380              elif index.__class__ == bases.Instance:
381                  instance_as_index = helpers.class_instance_as_index(index)
382                  if instance_as_index:
383                      index_value = instance_as_index
384              else:
385                  index_value = index
386  
387              if index_value is _SUBSCRIPT_SENTINEL:
388                  raise exceptions.InferenceError(node=self, context=context)
389  
390              try:
391                  assigned = value.getitem(index_value, context)
392              except (
393                  exceptions.AstroidTypeError,
394                  exceptions.AstroidIndexError,
395                  exceptions.AttributeInferenceError,
396                  AttributeError,
397              ) as exc:
398                  raise exceptions.InferenceError(node=self, context=context) from exc
399  
400              # Prevent inferring if the inferred subscript
401              # is the same as the original subscripted object.
402              if self is assigned or assigned is util.Uninferable:
403                  yield util.Uninferable
404                  return None
405              yield from assigned.infer(context)
406              found_one = True
407  
408      if found_one:
409          return dict(node=self, context=context)
410      return None

419  @decorators.raise_if_nothing_inferred
420  @decorators.path_wrapper
421  def _infer_boolop(self, context=None):
422      """Infer a boolean operation (and / or / not). 
423   
424      The function will calculate the boolean operation 
425      for all pairs generated through inference for each component 
426      node. 
427      """
428      values = self.values
429      if self.op == "or":
430          predicate = operator.truth
431      else:
432          predicate = operator.not_
433  
434      try:
435          values = [value.infer(context=context) for value in values]
436      except exceptions.InferenceError:
437          yield util.Uninferable
438          return None
439  
440      for pair in itertools.product(*values):
441          if any(item is util.Uninferable for item in pair):
442              # Can't infer the final result, just yield Uninferable.
443              yield util.Uninferable
444              continue
445  
446          bool_values = [item.bool_value() for item in pair]
447          if any(item is util.Uninferable for item in bool_values):
448              # Can't infer the final result, just yield Uninferable.
449              yield util.Uninferable
450              continue
451  
452          # Since the boolean operations are short circuited operations,
453          # this code yields the first value for which the predicate is True
454          # and if no value respected the predicate, then the last value will
455          # be returned (or Uninferable if there was no last value).
456          # This is conforming to the semantics of `and` and `or`:
457          #   1 and 0 -> 1
458          #   0 and 1 -> 0
459          #   1 or 0 -> 1
460          #   0 or 1 -> 1
461          value = util.Uninferable
462          for value, bool_value in zip(pair, bool_values):
463              if predicate(bool_value):
464                  yield value
465                  break
466          else:
467              yield value
468  
469      return dict(node=self, context=context)

478  def _filter_operation_errors(self, infer_callable, context, error):
479      for result in infer_callable(self, context):
480          if isinstance(result, error):
481              # For the sake of .infer(), we don't care about operation
482              # errors, which is the job of pylint. So return something
483              # which shows that we can't infer the result.
484              yield util.Uninferable
485          else:
486              yield result

489  def _infer_unaryop(self, context=None):
490      """Infer what an UnaryOp should return when evaluated."""
491      for operand in self.operand.infer(context):
492          try:
493              yield operand.infer_unary_op(self.op)
494          except TypeError as exc:
495              # The operand doesn't support this operation.
496              yield util.BadUnaryOperationMessage(operand, self.op, exc)
497          except AttributeError as exc:
498              meth = protocols.UNARY_OP_METHOD[self.op]
499              if meth is None:
500                  # `not node`. Determine node's boolean
501                  # value and negate its result, unless it is
502                  # Uninferable, which will be returned as is.
503                  bool_value = operand.bool_value()
504                  if bool_value is not util.Uninferable:
505                      yield nodes.const_factory(not bool_value)
506                  else:
507                      yield util.Uninferable
508              else:
509                  if not isinstance(operand, (bases.Instance, nodes.ClassDef)):
510                      # The operation was used on something which
511                      # doesn't support it.
512                      yield util.BadUnaryOperationMessage(operand, self.op, exc)
513                      continue
514  
515                  try:
516                      try:
517                          methods = dunder_lookup.lookup(operand, meth)
518                      except exceptions.AttributeInferenceError:
519                          yield util.BadUnaryOperationMessage(operand, self.op, exc)
520                          continue
521  
522                      meth = methods[0]
523                      inferred = next(meth.infer(context=context))
524                      if inferred is util.Uninferable or not inferred.callable():
525                          continue
526  
527                      context = contextmod.copy_context(context)
528                      context.callcontext = contextmod.CallContext(args=[operand])
529                      call_results = inferred.infer_call_result(self, context=context)
530                      result = next(call_results, None)
531                      if result is None:
532                          # Failed to infer, return the same type.
533                          yield operand
534                      else:
535                          yield result
536                  except exceptions.AttributeInferenceError as exc:
537                      # The unary operation special method was not found.
538                      yield util.BadUnaryOperationMessage(operand, self.op, exc)
539                  except exceptions.InferenceError:
540                      yield util.Uninferable

546      """Infer what an UnaryOp should return when evaluated."""
547      yield from _filter_operation_errors(
548          self, _infer_unaryop, context, util.BadUnaryOperationMessage

562  def _invoke_binop_inference(instance, opnode, op, other, context, method_name):
563      """Invoke binary operation inference on the given instance."""
564      methods = dunder_lookup.lookup(instance, method_name)
565      context = contextmod.bind_context_to_node(context, instance)
566      method = methods[0]
567      inferred = next(method.infer(context=context))
568      if inferred is util.Uninferable:
569          raise exceptions.InferenceError
570      return instance.infer_binary_op(opnode, op, other, context, inferred)

602          context=context,

607  def _get_binop_contexts(context, left, right):
608      """Get contexts for binary operations. 
609   
610      This will return two inference contexts, the first one 
611      for x.__op__(y), the other one for y.__rop__(x), where 
612      only the arguments are inversed. 
613      """
614      # The order is important, since the first one should be
615      # left.__op__(right).
616      for arg in (right, left):
617          new_context = context.clone()
618          new_context.callcontext = contextmod.CallContext(args=[arg])
619          new_context.boundnode = None
620          yield new_context

628  def _get_binop_flow(
629      left, left_type, binary_opnode, right, right_type, context, reverse_context
630  ):
631      """Get the flow for binary operations. 
632   
633      The rules are a bit messy: 
634   
635          * if left and right have the same type, then only one 
636            method will be called, left.__op__(right) 
637          * if left and right are unrelated typewise, then first 
638            left.__op__(right) is tried and if this does not exist 
639            or returns NotImplemented, then right.__rop__(left) is tried. 
640          * if left is a subtype of right, then only left.__op__(right) 
641            is tried. 
642          * if left is a supertype of right, then right.__rop__(left) 
643            is first tried and then left.__op__(right) 
644      """
645      op = binary_opnode.op
646      if _same_type(left_type, right_type):
647          methods = [_bin_op(left, binary_opnode, op, right, context)]
648      elif helpers.is_subtype(left_type, right_type):
649          methods = [_bin_op(left, binary_opnode, op, right, context)]
650      elif helpers.is_supertype(left_type, right_type):
651          methods = [
652              _bin_op(right, binary_opnode, op, left, reverse_context, reverse=True),
653              _bin_op(left, binary_opnode, op, right, context),
654          ]
655      else:
656          methods = [
657              _bin_op(left, binary_opnode, op, right, context),
658              _bin_op(right, binary_opnode, op, left, reverse_context, reverse=True),
659          ]
660      return methods

663  def _get_aug_flow(
664      left, left_type, aug_opnode, right, right_type, context, reverse_context
665  ):
666      """Get the flow for augmented binary operations. 
667   
668      The rules are a bit messy: 
669   
670          * if left and right have the same type, then left.__augop__(right) 
671            is first tried and then left.__op__(right). 
672          * if left and right are unrelated typewise, then 
673            left.__augop__(right) is tried, then left.__op__(right) 
674            is tried and then right.__rop__(left) is tried. 
675          * if left is a subtype of right, then left.__augop__(right) 
676            is tried and then left.__op__(right). 
677          * if left is a supertype of right, then left.__augop__(right) 
678            is tried, then right.__rop__(left) and then 
679            left.__op__(right) 
680      """
681      bin_op = aug_opnode.op.strip("=")
682      aug_op = aug_opnode.op
683      if _same_type(left_type, right_type):
684          methods = [
685              _aug_op(left, aug_opnode, aug_op, right, context),
686              _bin_op(left, aug_opnode, bin_op, right, context),
687          ]
688      elif helpers.is_subtype(left_type, right_type):
689          methods = [
690              _aug_op(left, aug_opnode, aug_op, right, context),
691              _bin_op(left, aug_opnode, bin_op, right, context),
692          ]
693      elif helpers.is_supertype(left_type, right_type):
694          methods = [
695              _aug_op(left, aug_opnode, aug_op, right, context),
696              _bin_op(right, aug_opnode, bin_op, left, reverse_context, reverse=True),
697              _bin_op(left, aug_opnode, bin_op, right, context),
698          ]
699      else:
700          methods = [
701              _aug_op(left, aug_opnode, aug_op, right, context),
702              _bin_op(left, aug_opnode, bin_op, right, context),
703              _bin_op(right, aug_opnode, bin_op, left, reverse_context, reverse=True),
704          ]
705      return methods

708  def _infer_binary_operation(left, right, binary_opnode, context, flow_factory):
709      """Infer a binary operation between a left operand and a right operand 
710   
711      This is used by both normal binary operations and augmented binary 
712      operations, the only difference is the flow factory used. 
713      """
714  
715      context, reverse_context = _get_binop_contexts(context, left, right)
716      left_type = helpers.object_type(left)
717      right_type = helpers.object_type(right)
718      methods = flow_factory(
719          left, left_type, binary_opnode, right, right_type, context, reverse_context
720      )
721      for method in methods:
722          try:
723              results = list(method())
724          except AttributeError:
725              continue
726          except exceptions.AttributeInferenceError:
727              continue
728          except exceptions.InferenceError:
729              yield util.Uninferable
730              return
731          else:
732              if any(result is util.Uninferable for result in results):
733                  yield util.Uninferable
734                  return
735  
736              if all(map(_is_not_implemented, results)):
737                  continue
738              not_implemented = sum(
739                  1 for result in results if _is_not_implemented(result)
740              )
741              if not_implemented and not_implemented != len(results):
742                  # Can't infer yet what this is.
743                  yield util.Uninferable
744                  return
745  
746              yield from results
747              return
748      # The operation doesn't seem to be supported so let the caller know about it
749      yield util.BadBinaryOperationMessage(left_type, binary_opnode.op, right_type)
750

752  def _infer_binop(self, context):
753      """Binary operation inference logic."""
754      left = self.left
755      right = self.right
756  
757      # we use two separate contexts for evaluating lhs and rhs because
758      # 1. evaluating lhs may leave some undesired entries in context.path
759      #    which may not let us infer right value of rhs
760      context = context or contextmod.InferenceContext()
761      lhs_context = contextmod.copy_context(context)
762      rhs_context = contextmod.copy_context(context)
763      lhs_iter = left.infer(context=lhs_context)
764      rhs_iter = right.infer(context=rhs_context)
765      for lhs, rhs in itertools.product(lhs_iter, rhs_iter):
766          if any(value is util.Uninferable for value in (rhs, lhs)):
767              # Don't know how to process this.
768              yield util.Uninferable
769              return
770  
771          try:
772              yield from _infer_binary_operation(lhs, rhs, self, context, _get_binop_flow)
773          except exceptions._NonDeducibleTypeHierarchy:
774              yield util.Uninferable

789  def _infer_augassign(self, context=None):
790      """Inference logic for augmented binary operations."""
791      if context is None:
792          context = contextmod.InferenceContext()
793  
794      rhs_context = context.clone()
795  
796      lhs_iter = self.target.infer_lhs(context=context)
797      rhs_iter = self.value.infer(context=rhs_context)
798      for lhs, rhs in itertools.product(lhs_iter, rhs_iter):
799          if any(value is util.Uninferable for value in (rhs, lhs)):
800              # Don't know how to process this.
801              yield util.Uninferable
802              return
803  
804          try:
805              yield from _infer_binary_operation(
806                  left=lhs,
807                  right=rhs,
808                  binary_opnode=self,
809                  context=context,
810                  flow_factory=_get_aug_flow,
811              )
812          except exceptions._NonDeducibleTypeHierarchy:
813              yield util.Uninferable

903  def _populate_context_lookup(call, context):
904      # Allows context to be saved for later
905      # for inference inside a function
906      context_lookup = {}
907      if context is None:
908          return context_lookup
909      for arg in call.args:
910          if isinstance(arg, nodes.Starred):
911              context_lookup[arg.value] = context
912          else:
913              context_lookup[arg] = context
914      keywords = call.keywords if call.keywords is not None else []
915      for keyword in keywords:
916          context_lookup[keyword.value] = context
917      return context_lookup

920  @decorators.raise_if_nothing_inferred
921  def infer_ifexp(self, context=None):
922      """Support IfExp inference 
923   
924      If we can't infer the truthiness of the condition, we default 
925      to inferring both branches. Otherwise, we infer either branch 
926      depending on the condition. 
927      """
928      both_branches = False
929      # We use two separate contexts for evaluating lhs and rhs because
930      # evaluating lhs may leave some undesired entries in context.path
931      # which may not let us infer right value of rhs.
932  
933      context = context or contextmod.InferenceContext()
934      lhs_context = contextmod.copy_context(context)
935      rhs_context = contextmod.copy_context(context)
936      try:
937          test = next(self.test.infer(context=context.clone()))
938      except exceptions.InferenceError:
939          both_branches = True
940      else:
941          if test is not util.Uninferable:
942              if test.bool_value():
943                  yield from self.body.infer(context=lhs_context)
944              else:
945                  yield from self.orelse.infer(context=rhs_context)
946          else:
947              both_branches = True
948      if both_branches:
949          yield from self.body.infer(context=lhs_context)
950          yield from self.orelse.infer(context=rhs_context)
951

958  def _cached_generator(func, instance, args, kwargs, _cache={}):
959      node = args[0]
960      try:
961          return iter(_cache[func, id(node)])
962      except KeyError:
963          result = func(*args, **kwargs)
964          # Need to keep an iterator around
965          original, copy = itertools.tee(result)
966          _cache[func, id(node)] = list(copy)
967          return original

970  # When inferring a property, we instantiate a new `objects.Property` object,
971  # which in turn, because it inherits from `FunctionDef`, sets itself in the locals
972  # of the wrapping frame. This means that everytime we infer a property, the locals
973  # are mutated with a new instance of the property. This is why we cache the result
974  # of the function's inference.
975  @_cached_generator
976  def infer_functiondef(self, context=None):
977      if not self.decorators or not bases._is_property(self):
978          yield self
979          return dict(node=self, context=context)
980  
981      prop_func = objects.Property(
982          function=self,
983          name=self.name,
984          doc=self.doc,
985          lineno=self.lineno,
986          parent=self.parent,
987          col_offset=self.col_offset,
988      )
989      prop_func.postinit(body=[], args=self.args)
990      yield prop_func
991      return dict(node=self, context=context)
