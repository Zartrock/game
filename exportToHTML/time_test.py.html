25           # Test fps correctness for 100, 60 and 30 fps

30       def _fps_test(self, clock, fps, delta):
31           """ticks fps times each second, hence get_fps() should return fps"""
32           delay_per_frame = 1.0/fps
33           for f in range(fps):  # For one second tick and sleep
34               clock.tick()
35               time.sleep(delay_per_frame)
36           # We should get around fps (+- fps*delta -- delta % of fps)
37           self.assertAlmostEqual(clock.get_fps(), fps, delta=fps*delta)

95       def test_tick(self):
96           """Tests time.Clock.tick()"""
97           """ 
98           Loops with a set delay a few times then checks what tick reports to 
99           verify its accuracy. Then calls tick with a desired frame-rate and 
100          verifies it is not faster than the desired frame-rate nor is it taking 
101          a dramatically long time to complete 
102          """
103  
104          # Adjust this value to increase the acceptable sleep jitter
105          epsilon = 1.5
106          # Adjust this value to increase the acceptable locked frame-rate jitter
107          epsilon2 = 0.3
108          # adjust this value to increase the acceptable frame-rate margin
109          epsilon3 = 20
110          testing_framerate = 60
111          milliseconds = 5.0
112  
113          collection = []
114          c = Clock()
115  
116          # verify time.Clock.tick() will measure the time correctly
117          c.tick()
118          for i in range(100):
119              time.sleep(milliseconds / 1000) # convert to seconds
120              collection.append(c.tick())
121  
122          # removes the first highest and lowest value
123          for outlier in [min(collection), max(collection)]:
124              if outlier != milliseconds:
125                  collection.remove(outlier)
126  
127          average_time = float(sum(collection)) / len(collection)
128  
129          # assert the deviation from the intended frame-rate is within the
130          # acceptable amount (the delay is not taking a dramatically long time)
131          self.assertAlmostEqual(average_time, milliseconds, delta=epsilon)
132  
133          # verify tick will control the frame-rate
134  
135          c = Clock()
136          collection = []
137  
138          start = time.time()
139  
140          for i in range(testing_framerate):
141              collection.append(c.tick(testing_framerate))
142  
143          # remove the highest and lowest outliers
144          for outlier in [min(collection), max(collection)]:
145              if outlier != round(1000/testing_framerate):
146                  collection.remove(outlier)
147  
148          end = time.time()
149  
150          # Since calling tick with a desired fps will prevent the program from
151          # running at greater than the given fps, 100 iterations at 100 fps
152          # should last no less than 1 second
153          self.assertAlmostEqual(end - start, 1, delta=epsilon2)
154  
155          average_tick_time = float(sum(collection)) / len(collection)
156          self.assertAlmostEqual(1000/average_tick_time,
157                                 testing_framerate, delta=epsilon3)

159      def test_tick_busy_loop(self):
160          """Test tick_busy_loop"""
161  
162          c = Clock()
163  
164          # Test whether the return value of tick_busy_loop is equal to
165          # (FPS is accurate) or greater than (slower than the set FPS)
166          # with a small margin for error based on differences in how this
167          # test runs in practise - it either sometimes runs slightly fast
168          # or seems to based on a rounding error.
169          second_length = 1000
170          shortfall_tolerance = 1  # (ms) The amount of time a tick is allowed to run short of, to account for underlying rounding errors
171          sample_fps = 40
172  
173          self.assertGreaterEqual(c.tick_busy_loop(sample_fps), (second_length/sample_fps) - shortfall_tolerance)
174          pygame.time.wait(10)  # incur delay between ticks that's faster than sample_fps
175          self.assertGreaterEqual(c.tick_busy_loop(sample_fps), (second_length/sample_fps) - shortfall_tolerance)
176          pygame.time.wait(200)  # incur delay between ticks that's slower than sample_fps
177          self.assertGreaterEqual(c.tick_busy_loop(sample_fps), (second_length/sample_fps) - shortfall_tolerance)
178  
179          high_fps = 500
180          self.assertGreaterEqual(c.tick_busy_loop(high_fps), (second_length/high_fps) - shortfall_tolerance)
181  
182          low_fps = 1
183          self.assertGreaterEqual(c.tick_busy_loop(low_fps), (second_length/low_fps) - shortfall_tolerance)
184  
185          low_non_factor_fps = 35  # 1000/35 makes 28.5714285714
186          frame_length_without_decimal_places = int(second_length/low_non_factor_fps)  # Same result as math.floor
187          self.assertGreaterEqual(c.tick_busy_loop(low_non_factor_fps), frame_length_without_decimal_places - shortfall_tolerance)
188  
189          high_non_factor_fps = 750  # 1000/750 makes 1.3333...
190          frame_length_without_decimal_places_2 = int(second_length/high_non_factor_fps)  # Same result as math.floor
191          self.assertGreaterEqual(c.tick_busy_loop(high_non_factor_fps), frame_length_without_decimal_places_2 - shortfall_tolerance)
192  
193          zero_fps = 0
194          self.assertEqual(c.tick_busy_loop(zero_fps), 0)
195  
196          # Check behaviour of unexpected values
197  
198          negative_fps = -1
199          self.assertEqual(c.tick_busy_loop(negative_fps), 0)
200  
201          fractional_fps = 32.75
202          frame_length_without_decimal_places_3 = int(second_length/fractional_fps)
203          self.assertGreaterEqual(c.tick_busy_loop(fractional_fps), frame_length_without_decimal_places_3 - shortfall_tolerance)
204  
205          bool_fps = True
206          self.assertGreaterEqual(c.tick_busy_loop(bool_fps), (second_length/bool_fps) - shortfall_tolerance)

222          """Tests time.get_ticks()"""
223          """ 
224           Iterates and delays for arbitrary amount of time for each iteration, 
225           check get_ticks to equal correct gap time 
226          """
227          iterations = 20
228          millis = 50
229          delta = 15  # Acceptable margin of error in ms
230          # Assert return type to be int
231          self.assertTrue(type(pygame.time.get_ticks()) == int)
232          for i in range(iterations):
233              curr_ticks = pygame.time.get_ticks()  # Save current tick count
234              curr_time = time.time()  # Save current time
235              pygame.time.delay(millis)  # Delay for millis
236              # Time and Ticks difference from start of the iteration
237              time_diff = round((time.time() - curr_time)*1000)
238              ticks_diff = pygame.time.get_ticks() - curr_ticks
239              # Assert almost equality of the ticking time and time difference
240              self.assertAlmostEqual(ticks_diff, time_diff, delta=delta)

242      def test_set_timer(self):
243          """Tests time.set_timer()"""
244          """ 
245          Tests if a timer will post the correct amount of eventid events in 
246          the specified delay. 
247          Also tests if setting milliseconds to 0 stops the timer and if 
248          the once argument works. 
249          """
250          pygame.display.init()
251          TIMER_EVENT_TYPE = pygame.event.custom_type()
252          timer_event = pygame.event.Event(TIMER_EVENT_TYPE, {'code': 0})
253          delta = 400
254          timer_delay = 250
255          test_number = 8  # Number of events to read for the test
256          events = 0  # Events read
257          # Get the events a few times. The time SDL_PumpEvents takes
258          # for the first 2-3 calls is longer and less stable...
259          for _ in range(5):
260              pygame.event.get()
261  
262          pygame.time.set_timer(TIMER_EVENT_TYPE, timer_delay)
263          # Test that 'test_number' events are posted in the right amount of time
264          t1 = pygame.time.get_ticks()
265          max_test_time = t1 + timer_delay * test_number + delta
266          while events < test_number:
267              for event in pygame.event.get():
268                  if event == timer_event:
269                      events += 1
270              # The test takes too much time
271              if pygame.time.get_ticks() > max_test_time:
272                  break
273          pygame.time.set_timer(TIMER_EVENT_TYPE, 0)
274          t2 = pygame.time.get_ticks()
275          # Is the number ef events and the timing right?
276          self.assertEqual(events, test_number)
277          self.assertAlmostEqual(timer_delay * test_number, t2-t1, delta=delta)
278  
279          # Test that the timer stopped when set with 0ms delay.
280          pygame.event.get()
281          pygame.time.delay(500)
282          self.assertNotIn(timer_event, pygame.event.get())
283  
284          # Test that the once argument works
285          pygame.time.set_timer(TIMER_EVENT_TYPE, 10, True)
286          pygame.time.delay(100)
287          self.assertEqual(pygame.event.get().count(timer_event), 1)

289      def test_wait(self):
290          """Tests time.wait() function."""
291          millis = 100  # millisecond to wait on each iteration
292          iterations = 10  # number of iterations
293          delta = 50  # Represents acceptable margin of error for wait in ms
294          # Call checking function
295          self._wait_delay_check(pygame.time.wait, millis, iterations, delta)
296          # After timing behaviour, check argument type exceptions
297          self._type_error_checks(pygame.time.wait)

299      def _wait_delay_check(self, func_to_check, millis, iterations, delta):
300          """" 
301           call func_to_check(millis) "iterations" times and check each time if 
302           function "waited" for given millisecond (+- delta). At the end, take 
303           average time for each call (whole_duration/iterations), which should 
304           be equal to millis (+- delta - acceptable margin of error). 
305           *Created to avoid code duplication during delay and wait tests 
306          """
307          # take starting time for duration calculation
308          start_time = time.time()
309          for i in range(iterations):
310              wait_time = func_to_check(millis)
311              # Check equality of wait_time and millis with margin of error delta
312              self.assertAlmostEqual(wait_time, millis, delta=delta)
313          stop_time = time.time()
314          # Cycle duration in millisecond
315          duration = round((stop_time-start_time)*1000)
316          # Duration/Iterations should be (almost) equal to predefined millis
317          self.assertAlmostEqual(duration/iterations, millis, delta=delta)

319      def _type_error_checks(self, func_to_check):
320          """Checks 3 TypeError (float, tuple, string) for the func_to_check"""
321          """Intended for time.delay and time.wait functions"""
322          # Those methods throw no exceptions on negative integers
323          self.assertRaises(TypeError, func_to_check, 0.1)  # check float
324          self.assertRaises(TypeError, pygame.time.delay, (0, 1))  # check tuple
325          self.assertRaises(TypeError, pygame.time.delay, "10")  # check string
