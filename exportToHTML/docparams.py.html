21   """Pylint plugin for checking in Sphinx, Google, or Numpy style docstrings 
22   """
23   import astroid

25   from pylint.checkers import BaseChecker
26   from pylint.checkers import utils as checker_utils
27   from pylint.extensions import _check_docs_utils as utils
28   from pylint.interfaces import IAstroidChecker

31   class DocstringParameterChecker(BaseChecker):
32       """Checker for Sphinx, Google, or Numpy style docstrings 
33    
34       * Check that all function, method and constructor parameters are mentioned 
35         in the params and types part of the docstring.  Constructor parameters 
36         can be documented in either the class docstring or ``__init__`` docstring, 
37         but not both. 
38       * Check that there are no naming inconsistencies between the signature and 
39         the documentation, i.e. also report documented parameters that are missing 
40         in the signature. This is important to find cases where parameters are 
41         renamed only in the code, not in the documentation. 
42       * Check that all explicitly raised exceptions in a function are documented 
43         in the function docstring. Caught exceptions are ignored. 
44    
45       Activate this checker by adding the line:: 
46    
47           load-plugins=pylint.extensions.docparams 
48    
49       to the ``MASTER`` section of your ``.pylintrc``. 
50    
51       :param linter: linter object 
52       :type linter: :class:`pylint.lint.PyLinter` 
53       """
54   
55       __implements__ = IAstroidChecker

190      def visit_functiondef(self, node):
191          """Called for function and method definitions (def). 
192   
193          :param node: Node for a function or method definition in the AST 
194          :type node: :class:`astroid.scoped_nodes.Function` 
195          """
196          node_doc = utils.docstringify(node.doc, self.config.default_docstring_type)
197          self.check_functiondef_params(node, node_doc)
198          self.check_functiondef_returns(node, node_doc)
199          self.check_functiondef_yields(node, node_doc)
200  
201      def check_functiondef_params(self, node, node_doc):
202          node_allow_no_param = None
203          if node.name in self.constructor_names:
204              class_node = checker_utils.node_frame_class(node)
205              if class_node is not None:
206                  class_doc = utils.docstringify(
207                      class_node.doc, self.config.default_docstring_type
208                  )
209                  self.check_single_constructor_params(class_doc, node_doc, class_node)
210  
211                  # __init__ or class docstrings can have no parameters documented
212                  # as long as the other documents them.
213                  node_allow_no_param = (
214                      class_doc.has_params()
215                      or class_doc.params_documented_elsewhere()
216                      or None
217                  )
218                  class_allow_no_param = (
219                      node_doc.has_params()
220                      or node_doc.params_documented_elsewhere()
221                      or None
222                  )
223  
224                  self.check_arguments_in_docstring(
225                      class_doc, node.args, class_node, class_allow_no_param
226                  )
227  
228          self.check_arguments_in_docstring(
229              node_doc, node.args, node, node_allow_no_param
230          )
231  
232      def check_functiondef_returns(self, node, node_doc):
233          if (not node_doc.supports_yields and node.is_generator()) or node.is_abstract():
234              return
235  
236          return_nodes = node.nodes_of_class(astroid.Return)
237          if (node_doc.has_returns() or node_doc.has_rtype()) and not any(
238              utils.returns_something(ret_node) for ret_node in return_nodes
239          ):
240              self.add_message("redundant-returns-doc", node=node)
241  
242      def check_functiondef_yields(self, node, node_doc):
243          if not node_doc.supports_yields or node.is_abstract():
244              return
245  
246          if (
247              node_doc.has_yields() or node_doc.has_yields_type()
248          ) and not node.is_generator():
249              self.add_message("redundant-yields-doc", node=node)
250  
251      def visit_raise(self, node):
252          func_node = node.frame()
253          if not isinstance(func_node, astroid.FunctionDef):
254              return
255  
256          expected_excs = utils.possible_exc_types(node)
257  
258          if not expected_excs:
259              return
260  
261          if not func_node.doc:
262              # If this is a property setter,
263              # the property should have the docstring instead.
264              property_ = utils.get_setters_property(func_node)
265              if property_:
266                  func_node = property_
267  
268          doc = utils.docstringify(func_node.doc, self.config.default_docstring_type)
269          if not doc.is_valid():
270              if doc.doc:
271                  self._handle_no_raise_doc(expected_excs, func_node)
272              return
273  
274          found_excs_full_names = doc.exceptions()
275  
276          # Extract just the class name, e.g. "error" from "re.error"
277          found_excs_class_names = {exc.split(".")[-1] for exc in found_excs_full_names}
278          missing_excs = expected_excs - found_excs_class_names
279          self._add_raise_message(missing_excs, func_node)
280  
281      def visit_return(self, node):
282          if not utils.returns_something(node):
283              return
284  
285          func_node = node.frame()
286          if not isinstance(func_node, astroid.FunctionDef):
287              return
288  
289          doc = utils.docstringify(func_node.doc, self.config.default_docstring_type)
290          if not doc.is_valid() and self.config.accept_no_return_doc:
291              return
292  
293          is_property = checker_utils.decorated_with_property(func_node)
294  
295          if not (doc.has_returns() or (doc.has_property_returns() and is_property)):
296              self.add_message("missing-return-doc", node=func_node)
297  
298          if func_node.returns:
299              return
300  
301          if not (doc.has_rtype() or (doc.has_property_type() and is_property)):
302              self.add_message("missing-return-type-doc", node=func_node)
303  
304      def visit_yield(self, node):
305          func_node = node.frame()
306          if not isinstance(func_node, astroid.FunctionDef):
307              return
308  
309          doc = utils.docstringify(func_node.doc, self.config.default_docstring_type)
310          if not doc.is_valid() and self.config.accept_no_yields_doc:
311              return
312  
313          if doc.supports_yields:
314              doc_has_yields = doc.has_yields()
315              doc_has_yields_type = doc.has_yields_type()
316          else:
317              doc_has_yields = doc.has_returns()
318              doc_has_yields_type = doc.has_rtype()
319  
320          if not doc_has_yields:
321              self.add_message("missing-yield-doc", node=func_node)
322  
323          if not (doc_has_yields_type or func_node.returns):
324              self.add_message("missing-yield-type-doc", node=func_node)
325  
326      def visit_yieldfrom(self, node):
327          self.visit_yield(node)

329      def _compare_missing_args(
330          self,
331          found_argument_names,
332          message_id,
333          not_needed_names,
334          expected_argument_names,
335          warning_node,
336      ):
337          """Compare the found argument names with the expected ones and 
338          generate a message if there are arguments missing. 
339   
340          :param set found_argument_names: argument names found in the 
341              docstring 
342   
343          :param str message_id: pylint message id 
344   
345          :param not_needed_names: names that may be omitted 
346          :type not_needed_names: set of str 
347   
348          :param set expected_argument_names: Expected argument names 
349          :param NodeNG warning_node: The node to be analyzed 
350          """
351          missing_argument_names = (
352              expected_argument_names - found_argument_names
353          ) - not_needed_names
354          if missing_argument_names:
355              self.add_message(
356                  message_id,
357                  args=(", ".join(sorted(missing_argument_names)),),
358                  node=warning_node,
359              )
360  
361      def _compare_different_args(
362          self,
363          found_argument_names,
364          message_id,
365          not_needed_names,
366          expected_argument_names,
367          warning_node,
368      ):
369          """Compare the found argument names with the expected ones and 
370          generate a message if there are extra arguments found. 
371   
372          :param set found_argument_names: argument names found in the 
373              docstring 
374   
375          :param str message_id: pylint message id 
376   
377          :param not_needed_names: names that may be omitted 
378          :type not_needed_names: set of str 
379   
380          :param set expected_argument_names: Expected argument names 
381          :param NodeNG warning_node: The node to be analyzed 
382          """
383          differing_argument_names = (
384              (expected_argument_names ^ found_argument_names)
385              - not_needed_names
386              - expected_argument_names
387          )
388  
389          if differing_argument_names:
390              self.add_message(
391                  message_id,
392                  args=(", ".join(sorted(differing_argument_names)),),
393                  node=warning_node,
394              )

396      def check_arguments_in_docstring(
397          self, doc, arguments_node, warning_node, accept_no_param_doc=None
398      ):
399          """Check that all parameters in a function, method or class constructor 
400          on the one hand and the parameters mentioned in the parameter 
401          documentation (e.g. the Sphinx tags 'param' and 'type') on the other 
402          hand are consistent with each other. 
403   
404          * Undocumented parameters except 'self' are noticed. 
405          * Undocumented parameter types except for 'self' and the ``*<args>`` 
406            and ``**<kwargs>`` parameters are noticed. 
407          * Parameters mentioned in the parameter documentation that don't or no 
408            longer exist in the function parameter list are noticed. 
409          * If the text "For the parameters, see" or "For the other parameters, 
410            see" (ignoring additional whitespace) is mentioned in the docstring, 
411            missing parameter documentation is tolerated. 
412          * If there's no Sphinx style, Google style or NumPy style parameter 
413            documentation at all, i.e. ``:param`` is never mentioned etc., the 
414            checker assumes that the parameters are documented in another format 
415            and the absence is tolerated. 
416   
417          :param doc: Docstring for the function, method or class. 
418          :type doc: :class:`Docstring` 
419   
420          :param arguments_node: Arguments node for the function, method or 
421              class constructor. 
422          :type arguments_node: :class:`astroid.scoped_nodes.Arguments` 
423   
424          :param warning_node: The node to assign the warnings to 
425          :type warning_node: :class:`astroid.scoped_nodes.Node` 
426   
427          :param accept_no_param_doc: Whether or not to allow no parameters 
428              to be documented. 
429              If None then this value is read from the configuration. 
430          :type accept_no_param_doc: bool or None 
431          """
432          # Tolerate missing param or type declarations if there is a link to
433          # another method carrying the same name.
434          if not doc.doc:
435              return
436  
437          if accept_no_param_doc is None:
438              accept_no_param_doc = self.config.accept_no_param_doc
439          tolerate_missing_params = doc.params_documented_elsewhere()
440  
441          # Collect the function arguments.
442          expected_argument_names = {arg.name for arg in arguments_node.args}
443          expected_argument_names.update(arg.name for arg in arguments_node.kwonlyargs)
444          not_needed_type_in_docstring = self.not_needed_param_in_docstring.copy()
445  
446          if arguments_node.vararg is not None:
447              expected_argument_names.add(arguments_node.vararg)
448              not_needed_type_in_docstring.add(arguments_node.vararg)
449          if arguments_node.kwarg is not None:
450              expected_argument_names.add(arguments_node.kwarg)
451              not_needed_type_in_docstring.add(arguments_node.kwarg)
452          params_with_doc, params_with_type = doc.match_param_docs()
453  
454          # Tolerate no parameter documentation at all.
455          if not params_with_doc and not params_with_type and accept_no_param_doc:
456              tolerate_missing_params = True
457  
458          if not tolerate_missing_params:
459              self._compare_missing_args(
460                  params_with_doc,
461                  "missing-param-doc",
462                  self.not_needed_param_in_docstring,
463                  expected_argument_names,
464                  warning_node,
465              )
466  
467          for index, arg_name in enumerate(arguments_node.args):
468              if arguments_node.annotations[index]:
469                  params_with_type.add(arg_name.name)
470          for index, arg_name in enumerate(arguments_node.kwonlyargs):
471              if arguments_node.kwonlyargs_annotations[index]:
472                  params_with_type.add(arg_name.name)
473  
474          if not tolerate_missing_params:
475              self._compare_missing_args(
476                  params_with_type,
477                  "missing-type-doc",
478                  not_needed_type_in_docstring,
479                  expected_argument_names,
480                  warning_node,
481              )
482  
483          self._compare_different_args(
484              params_with_doc,
485              "differing-param-doc",
486              self.not_needed_param_in_docstring,
487              expected_argument_names,
488              warning_node,
489          )
490          self._compare_different_args(
491              params_with_type,
492              "differing-type-doc",
493              not_needed_type_in_docstring,
494              expected_argument_names,
495              warning_node,
496          )

498      def check_single_constructor_params(self, class_doc, init_doc, class_node):
499          if class_doc.has_params() and init_doc.has_params():
500              self.add_message(
501                  "multiple-constructor-doc", args=(class_node.name,), node=class_node
502              )
503  
504      def _handle_no_raise_doc(self, excs, node):
505          if self.config.accept_no_raise_doc:
506              return
507  
508          self._add_raise_message(excs, node)
509  
510      def _add_raise_message(self, missing_excs, node):
511          """ 
512          Adds a message on :param:`node` for the missing exception type. 
513   
514          :param missing_excs: A list of missing exception types. 
515          :type missing_excs: set(str) 
516   
517          :param node: The node show the message on. 
518          :type node: astroid.node_classes.NodeNG 
519          """
520          if node.is_abstract():
521              try:
522                  missing_excs.remove("NotImplementedError")
523              except KeyError:
524                  pass
525  
526          if not missing_excs:
527              return
528  
529          self.add_message(
530              "missing-raises-doc", args=(", ".join(sorted(missing_excs)),), node=node
531          )

534  def register(linter):
535      """Required method to auto register this checker. 
536   
537      :param linter: Main interface object for Pylint plugins 
538      :type linter: Pylint object 
539      """
540      linter.register_checker(DocstringParameterChecker(linter))
