13               default_icon_data = None
14               try:
15                   with getResource('pygame_icon.tiff') as file_resource:
16                       default_icon_data = file_resource.read()
17               except (IOError, NotImplementedError):
18                   pass
19               sdlmain_osx.InstallNSApplication(default_icon_data)
20           if (os.getcwd() == '/') and len(sys.argv) > 1:
21               os.chdir(os.path.dirname(sys.argv[0]))
22       return True
