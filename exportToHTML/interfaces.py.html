21   HIGH = Confidence("HIGH", "No false positive possible.")

31   class Interface:
32       """Base class for interfaces."""
33   
34       @classmethod
35       def is_implemented_by(cls, instance):
36           return implements(instance, cls)
37   
38   
39   def implements(obj, interface):
40       """Return true if the give object (maybe an instance or class) implements 
41       the interface. 
42       """
43       kimplements = getattr(obj, "__implements__", ())
44       if not isinstance(kimplements, (list, tuple)):
45           kimplements = (kimplements,)
46       for implementedinterface in kimplements:
47           if issubclass(implementedinterface, interface):
48               return True
49       return False

52   class IChecker(Interface):
53       """This is a base interface, not designed to be used elsewhere than for 
54       sub interfaces definition. 
55       """
56   
57       def open(self):
58           """called before visiting project (i.e set of modules)"""
59   
60       def close(self):
61           """called after visiting project (i.e set of modules)"""
62   
63   
64   class IRawChecker(IChecker):
65       """interface for checker which need to parse the raw file 
66       """
67   
68       def process_module(self, astroid):
69           """ process a module 
70    
71           the module's content is accessible via astroid.stream 
72           """

75   class ITokenChecker(IChecker):
76       """Interface for checkers that need access to the token list."""
77   
78       def process_tokens(self, tokens):
79           """Process a module. 
80    
81           tokens is a list of all source code tokens in the file. 
82           """
