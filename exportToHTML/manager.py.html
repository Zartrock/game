18   """astroid manager: avoid multiple astroid build of a same module when 
19   possible by providing a class responsible to get astroid representation 
20   from various source and using a cache of built modules) 
21   """

32   ZIP_IMPORT_EXTS = (".zip", ".egg", ".whl")

35   def safe_repr(obj):
36       try:
37           return repr(obj)
38       except Exception:  # pylint: disable=broad-except
39           return "???"

42   class AstroidManager:
43       """the astroid manager, responsible to build astroid from files 
44        or modules. 
45    
46       Use the Borg pattern. 
47       """
48   
49       name = "astroid loader"
50       brain = {}
51   
52       def __init__(self):
53           self.__dict__ = AstroidManager.brain
54           if not self.__dict__:
55               # NOTE: cache entries are added by the [re]builder
56               self.astroid_cache = {}
57               self._mod_file_cache = {}
58               self._failed_import_hooks = []
59               self.always_load_extensions = False
60               self.optimize_ast = False
61               self.extension_package_whitelist = set()
62               self._transform = transforms.TransformVisitor()
63   
64               # Export these APIs for convenience
65               self.register_transform = self._transform.register_transform
66               self.unregister_transform = self._transform.unregister_transform
67               self.max_inferable_values = 100
68   
69       @property
70       def builtins_module(self):
71           return self.astroid_cache["builtins"]
72   
73       def visit_transforms(self, node):
74           """Visit the transforms and apply them to the given *node*."""
75           return self._transform.visit(node)
76   
77       def ast_from_file(self, filepath, modname=None, fallback=True, source=False):
78           """given a module name, return the astroid object"""
79           try:
80               filepath = modutils.get_source_file(filepath, include_no_ext=True)
81               source = True
82           except modutils.NoSourceFile:
83               pass
84           if modname is None:
85               try:
86                   modname = ".".join(modutils.modpath_from_file(filepath))
87               except ImportError:
88                   modname = filepath
89           if (
90               modname in self.astroid_cache
91               and self.astroid_cache[modname].file == filepath
92           ):
93               return self.astroid_cache[modname]
94           if source:
95               # pylint: disable=import-outside-toplevel; circular import
96               from astroid.builder import AstroidBuilder
97   
98               return AstroidBuilder(self).file_build(filepath, modname)
99           if fallback and modname:
100              return self.ast_from_module_name(modname)
101          raise exceptions.AstroidBuildingError(
102              "Unable to build an AST for {path}.", path=filepath
103          )
104  
105      def ast_from_string(self, data, modname="", filepath=None):
106          """ Given some source code as a string, return its corresponding astroid object"""
107          # pylint: disable=import-outside-toplevel; circular import
108          from astroid.builder import AstroidBuilder
109  
110          return AstroidBuilder(self).string_build(data, modname, filepath)

112      def _build_stub_module(self, modname):
113          # pylint: disable=import-outside-toplevel; circular import
114          from astroid.builder import AstroidBuilder
115  
116          return AstroidBuilder(self).string_build("", modname)
117  
118      def _build_namespace_module(self, modname, path):
119          # pylint: disable=import-outside-toplevel; circular import
120          from astroid.builder import build_namespace_package_module
121  
122          return build_namespace_package_module(modname, path)
123  
124      def _can_load_extension(self, modname):
125          if self.always_load_extensions:
126              return True
127          if modutils.is_standard_module(modname):
128              return True
129          parts = modname.split(".")
130          return any(
131              ".".join(parts[:x]) in self.extension_package_whitelist
132              for x in range(1, len(parts) + 1)
133          )
134  
135      def ast_from_module_name(self, modname, context_file=None):
136          """given a module name, return the astroid object"""
137          if modname in self.astroid_cache:
138              return self.astroid_cache[modname]
139          if modname == "__main__":
140              return self._build_stub_module(modname)
141          if context_file:
142              old_cwd = os.getcwd()
143              os.chdir(os.path.dirname(context_file))
144          try:
145              found_spec = self.file_from_module_name(modname, context_file)
146              if found_spec.type == spec.ModuleType.PY_ZIPMODULE:
147                  module = self.zip_import_data(found_spec.location)
148                  if module is not None:
149                      return module
150  
151              elif found_spec.type in (
152                  spec.ModuleType.C_BUILTIN,
153                  spec.ModuleType.C_EXTENSION,
154              ):
155                  if (
156                      found_spec.type == spec.ModuleType.C_EXTENSION
157                      and not self._can_load_extension(modname)
158                  ):
159                      return self._build_stub_module(modname)
160                  try:
161                      module = modutils.load_module_from_name(modname)
162                  except Exception as ex:
163                      raise exceptions.AstroidImportError(
164                          "Loading {modname} failed with:\n{error}",
165                          modname=modname,
166                          path=found_spec.location,
167                      ) from ex
168                  return self.ast_from_module(module, modname)
169  
170              elif found_spec.type == spec.ModuleType.PY_COMPILED:
171                  raise exceptions.AstroidImportError(
172                      "Unable to load compiled module {modname}.",
173                      modname=modname,
174                      path=found_spec.location,
175                  )
176  
177              elif found_spec.type == spec.ModuleType.PY_NAMESPACE:
178                  return self._build_namespace_module(
179                      modname, found_spec.submodule_search_locations
180                  )
181              elif found_spec.type == spec.ModuleType.PY_FROZEN:
182                  return self._build_stub_module(modname)
183  
184              if found_spec.location is None:
185                  raise exceptions.AstroidImportError(
186                      "Can't find a file for module {modname}.", modname=modname
187                  )
188  
189              return self.ast_from_file(found_spec.location, modname, fallback=False)
190          except exceptions.AstroidBuildingError as e:
191              for hook in self._failed_import_hooks:
192                  try:
193                      return hook(modname)
194                  except exceptions.AstroidBuildingError:
195                      pass
196              raise e
197          finally:
198              if context_file:
199                  os.chdir(old_cwd)
200  
201      def zip_import_data(self, filepath):
202          if zipimport is None:
203              return None
204  
205          # pylint: disable=import-outside-toplevel; circular import
206          from astroid.builder import AstroidBuilder
207  
208          builder = AstroidBuilder(self)
209          for ext in ZIP_IMPORT_EXTS:
210              try:
211                  eggpath, resource = filepath.rsplit(ext + os.path.sep, 1)
212              except ValueError:
213                  continue
214              try:
215                  importer = zipimport.zipimporter(eggpath + ext)
216                  zmodname = resource.replace(os.path.sep, ".")
217                  if importer.is_package(resource):
218                      zmodname = zmodname + ".__init__"
219                  module = builder.string_build(
220                      importer.get_source(resource), zmodname, filepath
221                  )
222                  return module
223              except Exception:  # pylint: disable=broad-except
224                  continue
225          return None

227      def file_from_module_name(self, modname, contextfile):
228          try:
229              value = self._mod_file_cache[(modname, contextfile)]
230          except KeyError:
231              try:
232                  value = modutils.file_info_from_modpath(
233                      modname.split("."), context_file=contextfile
234                  )
235              except ImportError as ex:
236                  value = exceptions.AstroidImportError(
237                      "Failed to import module {modname} with error:\n{error}.",
238                      modname=modname,
239                      error=ex,
240                  )
241              self._mod_file_cache[(modname, contextfile)] = value
242          if isinstance(value, exceptions.AstroidBuildingError):
243              raise value
244          return value

246      def ast_from_module(self, module, modname=None):
247          """given an imported module, return the astroid object"""
248          modname = modname or module.__name__
249          if modname in self.astroid_cache:
250              return self.astroid_cache[modname]
251          try:
252              # some builtin modules don't have __file__ attribute
253              filepath = module.__file__
254              if modutils.is_python_source(filepath):
255                  return self.ast_from_file(filepath, modname)
256          except AttributeError:
257              pass
258  
259          # pylint: disable=import-outside-toplevel; circular import
260          from astroid.builder import AstroidBuilder
261  
262          return AstroidBuilder(self).module_build(module, modname)
263  
264      def ast_from_class(self, klass, modname=None):
265          """get astroid for the given class"""
266          if modname is None:
267              try:
268                  modname = klass.__module__
269              except AttributeError as exc:
270                  raise exceptions.AstroidBuildingError(
271                      "Unable to get module for class {class_name}.",
272                      cls=klass,
273                      class_repr=safe_repr(klass),
274                      modname=modname,
275                  ) from exc
276          modastroid = self.ast_from_module_name(modname)
277          return modastroid.getattr(klass.__name__)[0]  # XXX

279      def infer_ast_from_something(self, obj, context=None):
280          """infer astroid for the given class"""
281          if hasattr(obj, "__class__") and not isinstance(obj, type):
282              klass = obj.__class__
283          else:
284              klass = obj
285          try:
286              modname = klass.__module__
287          except AttributeError as exc:
288              raise exceptions.AstroidBuildingError(
289                  "Unable to get module for {class_repr}.",
290                  cls=klass,
291                  class_repr=safe_repr(klass),
292              ) from exc
293          except Exception as exc:
294              raise exceptions.AstroidImportError(
295                  "Unexpected error while retrieving module for {class_repr}:\n"
296                  "{error}",
297                  cls=klass,
298                  class_repr=safe_repr(klass),
299              ) from exc
300          try:
301              name = klass.__name__
302          except AttributeError as exc:
303              raise exceptions.AstroidBuildingError(
304                  "Unable to get name for {class_repr}:\n",
305                  cls=klass,
306                  class_repr=safe_repr(klass),
307              ) from exc
308          except Exception as exc:
309              raise exceptions.AstroidImportError(
310                  "Unexpected error while retrieving name for {class_repr}:\n" "{error}",
311                  cls=klass,
312                  class_repr=safe_repr(klass),
313              ) from exc
314          # take care, on living object __module__ is regularly wrong :(
315          modastroid = self.ast_from_module_name(modname)
316          if klass is obj:
317              for inferred in modastroid.igetattr(name, context):
318                  yield inferred
319          else:
320              for inferred in modastroid.igetattr(name, context):
321                  yield inferred.instantiate_class()
322  
323      def register_failed_import_hook(self, hook):
324          """Registers a hook to resolve imports that cannot be found otherwise. 
325   
326          `hook` must be a function that accepts a single argument `modname` which 
327          contains the name of the module or package that could not be imported. 
328          If `hook` can resolve the import, must return a node of type `astroid.Module`, 
329          otherwise, it must raise `AstroidBuildingError`. 
330          """
331          self._failed_import_hooks.append(hook)
332  
333      def cache_module(self, module):
334          """Cache a module if no module with the same name is known yet."""
335          self.astroid_cache.setdefault(module.name, module)
336  
337      def bootstrap(self):
338          """Bootstrap the required AST modules needed for the manager to work 
339   
340          The bootstrap usually involves building the AST for the builtins 
341          module, which is required by the rest of astroid to work correctly. 
342          """
343          from astroid import raw_building  # pylint: disable=import-outside-toplevel
344  
345          raw_building._astroid_bootstrapping()
346  
347      def clear_cache(self):
348          """Clear the underlying cache. Also bootstraps the builtins module."""
349          self.astroid_cache.clear()
350          self.bootstrap()
