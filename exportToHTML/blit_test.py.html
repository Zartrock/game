69   
70           # no overflow in sub blend.

72           d.fill((30, 0, 255, 255))
73           s.blit(d, (0, 0), None, BLEND_SUB)
74           self.assertEqual(s.get_at((0, 0))[0], 0)

76       def make_blit_list(self, num_surfs):
77   
78           blit_list = []
79           for i in range(num_surfs):
80               dest = (i * 10, 0)
81               surf = pygame.Surface((10, 10), SRCALPHA, 32)
82               color = (i * 1, i * 1, i * 1)
83               surf.fill(color)
84               blit_list.append((surf, dest))
85           return blit_list

87       def test_blits(self):
88   
89           NUM_SURFS = 255
90           PRINT_TIMING = 0
91           dst = pygame.Surface((NUM_SURFS * 10, 10), SRCALPHA, 32)
92           dst.fill((230, 230, 230))
93           blit_list = self.make_blit_list(NUM_SURFS)
94   
95           def blits(blit_list):
96               for surface, dest in blit_list:
97                   dst.blit(surface, dest)
98   
99           from time import time
100  
101          t0 = time()
102          results = blits(blit_list)
103          t1 = time()
104          if PRINT_TIMING:
105              print("python blits: %s" % (t1 - t0))
106  
107          dst.fill((230, 230, 230))
108          t0 = time()
109          results = dst.blits(blit_list)
110          t1 = time()
111          if PRINT_TIMING:
112              print("Surface.blits :%s" % (t1 - t0))
113  
114          # check if we blit all the different colors in the correct spots.
115          for i in range(NUM_SURFS):
116              color = (i * 1, i * 1, i * 1)
117              self.assertEqual(dst.get_at((i * 10, 0)), color)
118              self.assertEqual(dst.get_at(((i * 10) + 5, 5)), color)
119  
120          self.assertEqual(len(results), NUM_SURFS)
121  
122          t0 = time()
123          results = dst.blits(blit_list, doreturn=0)
124          t1 = time()
125          if PRINT_TIMING:
126              print("Surface.blits doreturn=0: %s" % (t1 - t0))
127          self.assertEqual(results, None)
