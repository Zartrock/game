66   def _qualified_names(modname):
67       """Split the names of the given module into subparts 
68    
69       For example, 
70           _qualified_names('pylint.checkers.ImportsChecker') 
71       returns 
72           ['pylint', 'pylint.checkers', 'pylint.checkers.ImportsChecker'] 
73       """
74       names = modname.split(".")
75       return [".".join(names[0 : i + 1]) for i in range(len(names))]

78   def _get_import_name(importnode, modname):
79       """Get a prepared module name from the given import node 
80    
81       In the case of relative imports, this will return the 
82       absolute qualified module name, which might be useful 
83       for debugging. Otherwise, the initial module name 
84       is returned unchanged. 
85       """
86       if isinstance(importnode, astroid.ImportFrom):
87           if importnode.level:
88               root = importnode.root()
89               if isinstance(root, astroid.Module):
90                   modname = root.relative_to_absolute_name(
91                       modname, level=importnode.level
92                   )
93       return modname

96   def _get_first_import(node, context, name, base, level, alias):
97       """return the node where [base.]<name> is imported or None if not found 
98       """
99       fullname = "%s.%s" % (base, name) if base else name
100  
101      first = None
102      found = False
103      for first in context.body:
104          if first is node:
105              continue
106          if first.scope() is node.scope() and first.fromlineno > node.fromlineno:
107              continue
108          if isinstance(first, astroid.Import):
109              if any(fullname == iname[0] for iname in first.names):
110                  found = True
111                  break
112          elif isinstance(first, astroid.ImportFrom):
113              if level == first.level:
114                  for imported_name, imported_alias in first.names:
115                      if fullname == "%s.%s" % (first.modname, imported_name):
116                          found = True
117                          break
118                      if (
119                          name != "*"
120                          and name == imported_name
121                          and not (alias or imported_alias)
122                      ):
123                          found = True
124                          break
125                  if found:
126                      break
127      if found and not astroid.are_exclusive(first, node):
128          return first
129      return None

132  def _ignore_import_failure(node, modname, ignored_modules):
133      for submodule in _qualified_names(modname):
134          if submodule in ignored_modules:
135              return True
136  
137      return node_ignores_exception(node, ImportError)

143  def _make_tree_defs(mod_files_list):
144      """get a list of 2-uple (module, list_of_files_which_import_this_module), 
145      it will return a dictionary to represent this as a tree 
146      """
147      tree_defs = {}
148      for mod, files in mod_files_list:
149          node = (tree_defs, ())
150          for prefix in mod.split("."):
151              node = node[0].setdefault(prefix, [{}, []])
152          node[1] += files
153      return tree_defs

156  def _repr_tree_defs(data, indent_str=None):
157      """return a string which represents imports as a tree"""
158      lines = []
159      nodes = data.items()
160      for i, (mod, (sub, files)) in enumerate(sorted(nodes, key=lambda x: x[0])):
161          if not files:
162              files = ""
163          else:
164              files = "(%s)" % ",".join(sorted(files))
165          if indent_str is None:
166              lines.append("%s %s" % (mod, files))
167              sub_indent_str = "  "
168          else:
169              lines.append(r"%s\-%s %s" % (indent_str, mod, files))
170              if i == len(nodes) - 1:
171                  sub_indent_str = "%s  " % indent_str
172              else:
173                  sub_indent_str = "%s| " % indent_str
174          if sub:
175              lines.append(_repr_tree_defs(sub, sub_indent_str))
176      return "\n".join(lines)

179  def _dependencies_graph(filename, dep_info):
180      """write dependencies as a dot (graphviz) file 
181      """
182      done = {}
183      printer = DotBackend(filename[:-4], rankdir="LR")
184      printer.emit('URL="." node[shape="box"]')
185      for modname, dependencies in sorted(dep_info.items()):
186          done[modname] = 1
187          printer.emit_node(modname)
188          for depmodname in dependencies:
189              if depmodname not in done:
190                  done[depmodname] = 1
191                  printer.emit_node(depmodname)
192      for depmodname, dependencies in sorted(dep_info.items()):
193          for modname in dependencies:
194              printer.emit_edge(modname, depmodname)
195      printer.generate(filename)

199      """generate a dependencies graph and add some information about it in the 
200      report's section

202      _dependencies_graph(filename, dep_info)
203      sect.append(Paragraph("%simports graph has been written to %s" % (gtype, filename)))

437      @staticmethod
438      def _compute_site_packages():
439          def _normalized_path(path):
440              return os.path.normcase(os.path.abspath(path))
441  
442          paths = set()
443          real_prefix = getattr(sys, "real_prefix", None)
444          for prefix in filter(None, (real_prefix, sys.prefix)):
445              path = sysconfig.get_python_lib(prefix=prefix)
446              path = _normalized_path(path)
447              paths.add(path)
448  
449          # Handle Debian's derivatives /usr/local.
450          if os.path.isfile("/etc/debian_version"):
451              for prefix in filter(None, (real_prefix, sys.prefix)):
452                  libpython = os.path.join(
453                      prefix,
454                      "local",
455                      "lib",
456                      "python" + sysconfig.get_python_version(),
457                      "dist-packages",
458                  )
459                  paths.add(libpython)
460          return paths

462      def open(self):
463          """called before visiting project (i.e set of modules)"""
464          self.linter.add_stats(dependencies={})
465          self.linter.add_stats(cycles=[])
466          self.stats = self.linter.stats
467          self.import_graph = collections.defaultdict(set)
468          self._module_pkg = {}  # mapping of modules to the pkg they belong in
469          self._excluded_edges = collections.defaultdict(set)
470          self._ignored_modules = get_global_option(self, "ignored-modules", default=[])
471          # Build a mapping {'module': 'preferred-module'}
472          self.preferred_modules = dict(
473              module.split(":")
474              for module in self.config.preferred_modules
475              if ":" in module
476          )
477          self._allow_any_import_level = set(self.config.allow_any_import_level)

479      def _import_graph_without_ignored_edges(self):
480          filtered_graph = copy.deepcopy(self.import_graph)
481          for node in filtered_graph:
482              filtered_graph[node].difference_update(self._excluded_edges[node])
483          return filtered_graph

485      def close(self):
486          """called before visiting project (i.e set of modules)"""
487          if self.linter.is_message_enabled("cyclic-import"):
488              graph = self._import_graph_without_ignored_edges()
489              vertices = list(graph)
490              for cycle in get_cycles(graph, vertices=vertices):
491                  self.add_message("cyclic-import", args=" -> ".join(cycle))

493      @check_messages(*MSGS)
494      def visit_import(self, node):
495          """triggered when an import statement is seen"""
496          self._check_reimport(node)
497          self._check_import_as_rename(node)
498          self._check_toplevel(node)
499  
500          names = [name for name, _ in node.names]
501          if len(names) >= 2:
502              self.add_message("multiple-imports", args=", ".join(names), node=node)
503  
504          for name in names:
505              self._check_deprecated_module(node, name)
506              self._check_preferred_module(node, name)
507              imported_module = self._get_imported_module(node, name)
508              if isinstance(node.parent, astroid.Module):
509                  # Allow imports nested
510                  self._check_position(node)
511              if isinstance(node.scope(), astroid.Module):
512                  self._record_import(node, imported_module)
513  
514              if imported_module is None:
515                  continue
516  
517              self._add_imported_module(node, imported_module.name)
518  
519      @check_messages(*MSGS)
520      def visit_importfrom(self, node):
521          """triggered when a from statement is seen"""
522          basename = node.modname
523          imported_module = self._get_imported_module(node, basename)
524  
525          self._check_import_as_rename(node)
526          self._check_misplaced_future(node)
527          self._check_deprecated_module(node, basename)
528          self._check_preferred_module(node, basename)
529          self._check_wildcard_imports(node, imported_module)
530          self._check_same_line_imports(node)
531          self._check_reimport(node, basename=basename, level=node.level)
532          self._check_toplevel(node)
533  
534          if isinstance(node.parent, astroid.Module):
535              # Allow imports nested
536              self._check_position(node)
537          if isinstance(node.scope(), astroid.Module):
538              self._record_import(node, imported_module)
539          if imported_module is None:
540              return
541          for name, _ in node.names:
542              if name != "*":
543                  self._add_imported_module(node, "%s.%s" % (imported_module.name, name))
544              else:
545                  self._add_imported_module(node, imported_module.name)

547      @check_messages(*MSGS)
548      def leave_module(self, node):
549          # Check imports are grouped by category (standard, 3rd party, local)
550          std_imports, ext_imports, loc_imports = self._check_imports_order(node)
551  
552          # Check that imports are grouped by package within a given category
553          met_import = set()  #  set for 'import x' style
554          met_from = set()  #  set for 'from x import y' style
555          current_package = None
556          for import_node, import_name in std_imports + ext_imports + loc_imports:
557              if not self.linter.is_message_enabled(
558                  "ungrouped-imports", import_node.fromlineno
559              ):
560                  continue
561              if isinstance(import_node, astroid.node_classes.ImportFrom):
562                  met = met_from
563              else:
564                  met = met_import
565              package, _, _ = import_name.partition(".")
566              if current_package and current_package != package and package in met:
567                  self.add_message("ungrouped-imports", node=import_node, args=package)
568              current_package = package
569              met.add(package)
570  
571          self._imports_stack = []
572          self._first_non_import_node = None

574      def compute_first_non_import_node(self, node):
575          if not self.linter.is_message_enabled("wrong-import-position", node.fromlineno):
576              return
577          # if the node does not contain an import instruction, and if it is the
578          # first node of the module, keep a track of it (all the import positions
579          # of the module will be compared to the position of this first
580          # instruction)
581          if self._first_non_import_node:
582              return
583          if not isinstance(node.parent, astroid.Module):
584              return
585          nested_allowed = [astroid.TryExcept, astroid.TryFinally]
586          is_nested_allowed = [
587              allowed for allowed in nested_allowed if isinstance(node, allowed)
588          ]
589          if is_nested_allowed and any(
590              node.nodes_of_class((astroid.Import, astroid.ImportFrom))
591          ):
592              return
593          if isinstance(node, astroid.Assign):
594              # Add compatibility for module level dunder names
595              # https://www.python.org/dev/peps/pep-0008/#module-level-dunder-names
596              valid_targets = [
597                  isinstance(target, astroid.AssignName)
598                  and target.name.startswith("__")
599                  and target.name.endswith("__")
600                  for target in node.targets
601              ]
602              if all(valid_targets):
603                  return
604          self._first_non_import_node = node

616      def visit_functiondef(self, node):
617          if not self.linter.is_message_enabled("wrong-import-position", node.fromlineno):
618              return
619          # If it is the first non import instruction of the module, record it.
620          if self._first_non_import_node:
621              return
622  
623          # Check if the node belongs to an `If` or a `Try` block. If they
624          # contain imports, skip recording this node.
625          if not isinstance(node.parent.scope(), astroid.Module):
626              return
627  
628          root = node
629          while not isinstance(root.parent, astroid.Module):
630              root = root.parent
631  
632          if isinstance(root, (astroid.If, astroid.TryFinally, astroid.TryExcept)):
633              if any(root.nodes_of_class((astroid.Import, astroid.ImportFrom))):
634                  return
635  
636          self._first_non_import_node = node
637  
638      visit_classdef = visit_for = visit_while = visit_functiondef

640      def _check_misplaced_future(self, node):
641          basename = node.modname
642          if basename == "__future__":
643              # check if this is the first non-docstring statement in the module
644              prev = node.previous_sibling()
645              if prev:
646                  # consecutive future statements are possible
647                  if not (
648                      isinstance(prev, astroid.ImportFrom)
649                      and prev.modname == "__future__"
650                  ):
651                      self.add_message("misplaced-future", node=node)
652              return

654      def _check_same_line_imports(self, node):
655          # Detect duplicate imports on the same line.
656          names = (name for name, _ in node.names)
657          counter = collections.Counter(names)
658          for name, count in counter.items():
659              if count > 1:
660                  self.add_message("reimported", node=node, args=(name, node.fromlineno))

662      def _check_position(self, node):
663          """Check `node` import or importfrom node position is correct 
664   
665          Send a message  if `node` comes before another instruction 
666          """
667          # if a first non-import instruction has already been encountered,
668          # it means the import comes after it and therefore is not well placed
669          if self._first_non_import_node:
670              self.add_message("wrong-import-position", node=node, args=node.as_string())

672      def _record_import(self, node, importedmodnode):
673          """Record the package `node` imports from"""
674          if isinstance(node, astroid.ImportFrom):
675              importedname = node.modname
676          else:
677              importedname = importedmodnode.name if importedmodnode else None
678          if not importedname:
679              importedname = node.names[0][0].split(".")[0]
680  
681          if isinstance(node, astroid.ImportFrom) and (node.level or 0) >= 1:
682              # We need the importedname with first point to detect local package
683              # Example of node:
684              #  'from .my_package1 import MyClass1'
685              #  the output should be '.my_package1' instead of 'my_package1'
686              # Example of node:
687              #  'from . import my_package2'
688              #  the output should be '.my_package2' instead of '{pyfile}'
689              importedname = "." + importedname
690  
691          self._imports_stack.append((node, importedname))

693      @staticmethod
694      def _is_fallback_import(node, imports):
695          imports = [import_node for (import_node, _) in imports]
696          return any(astroid.are_exclusive(import_node, node) for import_node in imports)

698      def _check_imports_order(self, _module_node):
699          """Checks imports of module `node` are grouped by category 
700   
701          Imports must follow this order: standard, 3rd party, local 
702          """
703          std_imports = []
704          third_party_imports = []
705          first_party_imports = []
706          # need of a list that holds third or first party ordered import
707          external_imports = []
708          local_imports = []
709          third_party_not_ignored = []
710          first_party_not_ignored = []
711          local_not_ignored = []
712          isort_driver = IsortDriver(self.config)
713          for node, modname in self._imports_stack:
714              if modname.startswith("."):
715                  package = "." + modname.split(".")[1]
716              else:
717                  package = modname.split(".")[0]
718              nested = not isinstance(node.parent, astroid.Module)
719              ignore_for_import_order = not self.linter.is_message_enabled(
720                  "wrong-import-order", node.fromlineno
721              )
722              import_category = isort_driver.place_module(package)
723              node_and_package_import = (node, package)
724              if import_category in ("FUTURE", "STDLIB"):
725                  std_imports.append(node_and_package_import)
726                  wrong_import = (
727                      third_party_not_ignored
728                      or first_party_not_ignored
729                      or local_not_ignored
730                  )
731                  if self._is_fallback_import(node, wrong_import):
732                      continue
733                  if wrong_import and not nested:
734                      self.add_message(
735                          "wrong-import-order",
736                          node=node,
737                          args=(
738                              'standard import "%s"' % node.as_string(),
739                              '"%s"' % wrong_import[0][0].as_string(),
740                          ),
741                      )
742              elif import_category == "THIRDPARTY":
743                  third_party_imports.append(node_and_package_import)
744                  external_imports.append(node_and_package_import)
745                  if not nested and not ignore_for_import_order:
746                      third_party_not_ignored.append(node_and_package_import)
747                  wrong_import = first_party_not_ignored or local_not_ignored
748                  if wrong_import and not nested:
749                      self.add_message(
750                          "wrong-import-order",
751                          node=node,
752                          args=(
753                              'third party import "%s"' % node.as_string(),
754                              '"%s"' % wrong_import[0][0].as_string(),
755                          ),
756                      )
757              elif import_category == "FIRSTPARTY":
758                  first_party_imports.append(node_and_package_import)
759                  external_imports.append(node_and_package_import)
760                  if not nested and not ignore_for_import_order:
761                      first_party_not_ignored.append(node_and_package_import)
762                  wrong_import = local_not_ignored
763                  if wrong_import and not nested:
764                      self.add_message(
765                          "wrong-import-order",
766                          node=node,
767                          args=(
768                              'first party import "%s"' % node.as_string(),
769                              '"%s"' % wrong_import[0][0].as_string(),
770                          ),
771                      )
772              elif import_category == "LOCALFOLDER":
773                  local_imports.append((node, package))
774                  if not nested and not ignore_for_import_order:
775                      local_not_ignored.append((node, package))
776          return std_imports, external_imports, local_imports

778      def _get_imported_module(self, importnode, modname):
779          try:
780              return importnode.do_import_module(modname)
781          except astroid.TooManyLevelsError:
782              if _ignore_import_failure(importnode, modname, self._ignored_modules):
783                  return None
784              self.add_message("relative-beyond-top-level", node=importnode)
785          except astroid.AstroidSyntaxError as exc:
786              message = "Cannot import {!r} due to syntax error {!r}".format(
787                  modname, str(exc.error)  # pylint: disable=no-member; false positive
788              )
789              self.add_message("syntax-error", line=importnode.lineno, args=message)
790  
791          except astroid.AstroidBuildingException:
792              if not self.linter.is_message_enabled("import-error"):
793                  return None
794              if _ignore_import_failure(importnode, modname, self._ignored_modules):
795                  return None
796              if not self.config.analyse_fallback_blocks and is_from_fallback_block(
797                  importnode
798              ):
799                  return None
800  
801              dotted_modname = _get_import_name(importnode, modname)
802              self.add_message("import-error", args=repr(dotted_modname), node=importnode)
803  
804      def _add_imported_module(self, node, importedmodname):
805          """notify an imported module, used to analyze dependencies"""
806          module_file = node.root().file
807          context_name = node.root().name
808          base = os.path.splitext(os.path.basename(module_file))[0]
809  
810          try:
811              importedmodname = modutils.get_module_part(importedmodname, module_file)
812          except ImportError:
813              pass
814  
815          if context_name == importedmodname:
816              self.add_message("import-self", node=node)
817  
818          elif not modutils.is_standard_module(importedmodname):
819              # if this is not a package __init__ module
820              if base != "__init__" and context_name not in self._module_pkg:
821                  # record the module's parent, or the module itself if this is
822                  # a top level module, as the package it belongs to
823                  self._module_pkg[context_name] = context_name.rsplit(".", 1)[0]
824  
825              # handle dependencies
826              importedmodnames = self.stats["dependencies"].setdefault(
827                  importedmodname, set()
828              )
829              if context_name not in importedmodnames:
830                  importedmodnames.add(context_name)
831  
832              # update import graph
833              self.import_graph[context_name].add(importedmodname)
834              if not self.linter.is_message_enabled("cyclic-import", line=node.lineno):
835                  self._excluded_edges[context_name].add(importedmodname)
836

852      def _check_import_as_rename(self, node):
853          names = node.names
854          for name in names:
855              if not all(name):
856                  return
857  
858              real_name = name[0]
859              splitted_packages = real_name.rsplit(".")
860              real_name = splitted_packages[-1]
861              imported_name = name[1]
862              # consider only following cases
863              # import x as x
864              # and ignore following
865              # import x.y.z as z
866              if real_name == imported_name and len(splitted_packages) == 1:
867                  self.add_message("useless-import-alias", node=node)

869      def _check_reimport(self, node, basename=None, level=None):
870          """check if the import is necessary (i.e. not already done)"""
871          if not self.linter.is_message_enabled("reimported"):
872              return
873  
874          frame = node.frame()
875          root = node.root()
876          contexts = [(frame, level)]
877          if root is not frame:
878              contexts.append((root, None))
879  
880          for known_context, known_level in contexts:
881              for name, alias in node.names:
882                  first = _get_first_import(
883                      node, known_context, name, basename, known_level, alias
884                  )
885                  if first is not None:
886                      self.add_message(
887                          "reimported", node=node, args=(name, first.fromlineno)
888                      )

898      def _report_dependencies_graph(self, sect, _, _dummy):
899          """write dependencies as a dot (graphviz) file"""
900          dep_info = self.stats["dependencies"]
901          if not dep_info or not (
902              self.config.import_graph
903              or self.config.ext_import_graph
904              or self.config.int_import_graph
905          ):
906              raise EmptyReportError()
907          filename = self.config.import_graph
908          if filename:
909              _make_graph(filename, dep_info, sect, "")
910          filename = self.config.ext_import_graph
911          if filename:
912              _make_graph(filename, self._external_dependencies_info(), sect, "external ")
913          filename = self.config.int_import_graph
914          if filename:
915              _make_graph(filename, self._internal_dependencies_info(), sect, "internal ")

917      def _filter_dependencies_graph(self, internal):
918          """build the internal or the external dependency graph"""
919          graph = collections.defaultdict(set)
920          for importee, importers in self.stats["dependencies"].items():
921              for importer in importers:
922                  package = self._module_pkg.get(importer, importer)
923                  is_inside = importee.startswith(package)
924                  if is_inside and internal or not is_inside and not internal:
925                      graph[importee].add(importer)
926          return graph

942      def _check_wildcard_imports(self, node, imported_module):
943          if node.root().package:
944              # Skip the check if in __init__.py issue #2026
945              return
946  
947          wildcard_import_is_allowed = self._wildcard_import_is_allowed(imported_module)
948          for name, _ in node.names:
949              if name == "*" and not wildcard_import_is_allowed:
950                  self.add_message("wildcard-import", args=node.modname, node=node)

952      def _wildcard_import_is_allowed(self, imported_module):
953          return (
954              self.config.allow_wildcard_with_all
955              and imported_module is not None
956              and "__all__" in imported_module.locals
957          )

959      def _check_toplevel(self, node):
960          """Check whether the import is made outside the module toplevel. 
961          """
962          # If the scope of the import is a module, then obviously it is
963          # not outside the module toplevel.
964          if isinstance(node.scope(), astroid.Module):
965              return
966  
967          module_names = [
968              "{}.{}".format(node.modname, name[0])
969              if isinstance(node, astroid.ImportFrom)
970              else name[0]
971              for name in node.names
972          ]
973  
974          # Get the full names of all the imports that are not whitelisted.
975          scoped_imports = [
976              name for name in module_names if name not in self._allow_any_import_level
977          ]
978  
979          if scoped_imports:
980              self.add_message(
981                  "import-outside-toplevel", args=", ".join(scoped_imports), node=node
982              )
