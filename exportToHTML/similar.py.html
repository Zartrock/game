38   class Similar:
39       """finds copy-pasted lines of code in a project"""
40   
41       def __init__(
42           self,
43           min_lines=4,
44           ignore_comments=False,
45           ignore_docstrings=False,
46           ignore_imports=False,
47       ):
48           self.min_lines = min_lines
49           self.ignore_comments = ignore_comments
50           self.ignore_docstrings = ignore_docstrings
51           self.ignore_imports = ignore_imports
52           self.linesets = []
53   
54       def append_stream(self, streamid, stream, encoding=None):
55           """append a file to search for similarities"""
56           if encoding is None:
57               readlines = stream.readlines
58           else:
59               readlines = decoding_stream(stream, encoding).readlines
60           try:
61               self.linesets.append(
62                   LineSet(
63                       streamid,
64                       readlines(),
65                       self.ignore_comments,
66                       self.ignore_docstrings,
67                       self.ignore_imports,
68                   )
69               )
70           except UnicodeDecodeError:
71               pass
72   
73       def run(self):
74           """start looking for similarities and display results on stdout"""
75           self._display_sims(self._compute_sims())
76   
77       def _compute_sims(self):
78           """compute similarities in appended files"""
79           no_duplicates = defaultdict(list)
80           for num, lineset1, idx1, lineset2, idx2 in self._iter_sims():
81               duplicate = no_duplicates[num]
82               for couples in duplicate:
83                   if (lineset1, idx1) in couples or (lineset2, idx2) in couples:
84                       couples.add((lineset1, idx1))
85                       couples.add((lineset2, idx2))
86                       break
87               else:
88                   duplicate.append({(lineset1, idx1), (lineset2, idx2)})
89           sims = []
90           for num, ensembles in no_duplicates.items():
91               for couples in ensembles:
92                   sims.append((num, couples))
93           sims.sort()
94           sims.reverse()
95           return sims
96   
97       def _display_sims(self, sims):
98           """display computed similarities on stdout"""
99           nb_lignes_dupliquees = 0
100          for num, couples in sims:
101              print()
102              print(num, "similar lines in", len(couples), "files")
103              couples = sorted(couples)
104              lineset = idx = None
105              for lineset, idx in couples:
106                  print("==%s:%s" % (lineset.name, idx))
107              if lineset:
108                  for line in lineset._real_lines[idx : idx + num]:
109                      print("  ", line.rstrip())
110              nb_lignes_dupliquees += num * (len(couples) - 1)
111          nb_total_lignes = sum([len(lineset) for lineset in self.linesets])
112          print(
113              "TOTAL lines=%s duplicates=%s percent=%.2f"
114              % (
115                  nb_total_lignes,
116                  nb_lignes_dupliquees,
117                  nb_lignes_dupliquees * 100.0 / nb_total_lignes,
118              )
119          )
120  
121      def _find_common(self, lineset1, lineset2):
122          """find similarities in the two given linesets"""
123          lines1 = lineset1.enumerate_stripped
124          lines2 = lineset2.enumerate_stripped
125          find = lineset2.find
126          index1 = 0
127          min_lines = self.min_lines
128          while index1 < len(lineset1):
129              skip = 1
130              num = 0
131              for index2 in find(lineset1[index1]):
132                  non_blank = 0
133                  for num, ((_, line1), (_, line2)) in enumerate(
134                      zip(lines1(index1), lines2(index2))
135                  ):
136                      if line1 != line2:
137                          if non_blank > min_lines:
138                              yield num, lineset1, index1, lineset2, index2
139                          skip = max(skip, num)
140                          break
141                      if line1:
142                          non_blank += 1
143                  else:
144                      # we may have reach the end
145                      num += 1
146                      if non_blank > min_lines:
147                          yield num, lineset1, index1, lineset2, index2
148                      skip = max(skip, num)
149              index1 += skip
150  
151      def _iter_sims(self):
152          """iterate on similarities among all files, by making a cartesian 
153          product 
154          """
155          for idx, lineset in enumerate(self.linesets[:-1]):
156              for lineset2 in self.linesets[idx + 1 :]:
157                  yield from self._find_common(lineset, lineset2)

160  def stripped_lines(lines, ignore_comments, ignore_docstrings, ignore_imports):
161      """return lines with leading/trailing whitespace and any ignored code 
162      features removed 
163      """
164      if ignore_imports:
165          tree = astroid.parse("".join(lines))
166          node_is_import_by_lineno = (
167              (node.lineno, isinstance(node, (astroid.Import, astroid.ImportFrom)))
168              for node in tree.body
169          )
170          line_begins_import = {
171              lineno: all(is_import for _, is_import in node_is_import_group)
172              for lineno, node_is_import_group in groupby(
173                  node_is_import_by_lineno, key=lambda x: x[0]
174              )
175          }
176          current_line_is_import = False
177  
178      strippedlines = []
179      docstring = None
180      for lineno, line in enumerate(lines, start=1):
181          line = line.strip()
182          if ignore_docstrings:
183              if not docstring:
184                  if line.startswith('"""') or line.startswith("'''"):
185                      docstring = line[:3]
186                      line = line[3:]
187                  elif line.startswith('r"""') or line.startswith("r'''"):
188                      docstring = line[1:4]
189                      line = line[4:]
190              if docstring:
191                  if line.endswith(docstring):
192                      docstring = None
193                  line = ""
194          if ignore_imports:
195              current_line_is_import = line_begins_import.get(
196                  lineno, current_line_is_import
197              )
198              if current_line_is_import:
199                  line = ""
200          if ignore_comments:
201              line = line.split("#", 1)[0].strip()
202          strippedlines.append(line)
203      return strippedlines

206  class LineSet:
207      """Holds and indexes all the lines of a single source file"""
208  
209      def __init__(
210          self,
211          name,
212          lines,
213          ignore_comments=False,
214          ignore_docstrings=False,
215          ignore_imports=False,
216      ):
217          self.name = name
218          self._real_lines = lines
219          self._stripped_lines = stripped_lines(
220              lines, ignore_comments, ignore_docstrings, ignore_imports
221          )
222          self._index = self._mk_index()
223  
224      def __str__(self):
225          return "<Lineset for %s>" % self.name
226  
227      def __len__(self):
228          return len(self._real_lines)
229  
230      def __getitem__(self, index):
231          return self._stripped_lines[index]
232  
233      def __lt__(self, other):
234          return self.name < other.name
235  
236      def __hash__(self):
237          return id(self)
238  
239      def enumerate_stripped(self, start_at=0):
240          """return an iterator on stripped lines, starting from a given index 
241          if specified, else 0 
242          """
243          idx = start_at
244          if start_at:
245              lines = self._stripped_lines[start_at:]
246          else:
247              lines = self._stripped_lines
248          for line in lines:
249              # if line:
250              yield idx, line
251              idx += 1
252  
253      def find(self, stripped_line):
254          """return positions of the given stripped line in this set"""
255          return self._index.get(stripped_line, ())
256  
257      def _mk_index(self):
258          """create the index for this set"""
259          index = defaultdict(list)
260          for line_no, line in enumerate(self._stripped_lines):
261              if line:
262                  index[line].append(line_no)
263          return index

268          "Similar lines in %s files\n%s",

271          "among multiple file. This usually means that the code should "
272          "be refactored to avoid this duplication.",

277  def report_similarities(sect, stats, old_stats):
278      """make a layout with some stats about duplication"""
279      lines = ["", "now", "previous", "difference"]
280      lines += table_lines_from_stats(
281          stats, old_stats, ("nb_duplicated_lines", "percent_duplicated_lines")
282      )
283      sect.append(Table(children=lines, cols=4, rheaders=1, cheaders=1))

286  # wrapper to get a pylint checker from the similar class
287  class SimilarChecker(BaseChecker, Similar):
288      """checks for similarities and duplicated code. This computation may be 
289      memory / CPU intensive, so you should disable it if you experiment some 
290      problems. 
291      """
292  
293      __implements__ = (IRawChecker,)
294      # configuration section name
295      name = "similarities"
296      # messages
297      msgs = MSGS
298      # configuration options
299      # for available dict keys/values see the optik parser 'add_option' method
300      options = (
301          (
302              "min-similarity-lines",  # type: ignore
303              {
304                  "default": 4,
305                  "type": "int",
306                  "metavar": "<int>",
307                  "help": "Minimum lines number of a similarity.",
308              },
309          ),
310          (
311              "ignore-comments",
312              {
313                  "default": True,
314                  "type": "yn",
315                  "metavar": "<y or n>",
316                  "help": "Ignore comments when computing similarities.",
317              },
318          ),
319          (
320              "ignore-docstrings",
321              {
322                  "default": True,
323                  "type": "yn",
324                  "metavar": "<y or n>",
325                  "help": "Ignore docstrings when computing similarities.",
326              },
327          ),
328          (
329              "ignore-imports",
330              {
331                  "default": False,
332                  "type": "yn",
333                  "metavar": "<y or n>",
334                  "help": "Ignore imports when computing similarities.",
335              },
336          ),
337      )
338      # reports
339      reports = (("RP0801", "Duplication", report_similarities),)  # type: ignore
340  
341      def __init__(self, linter=None):
342          BaseChecker.__init__(self, linter)
343          Similar.__init__(
344              self, min_lines=4, ignore_comments=True, ignore_docstrings=True
345          )
346          self.stats = None
347  
348      def set_option(self, optname, value, action=None, optdict=None):
349          """method called to set an option (registered in the options list) 
350   
351          overridden to report options setting to Similar 
352          """
353          BaseChecker.set_option(self, optname, value, action, optdict)
354          if optname == "min-similarity-lines":
355              self.min_lines = self.config.min_similarity_lines
356          elif optname == "ignore-comments":
357              self.ignore_comments = self.config.ignore_comments
358          elif optname == "ignore-docstrings":
359              self.ignore_docstrings = self.config.ignore_docstrings
360          elif optname == "ignore-imports":
361              self.ignore_imports = self.config.ignore_imports
362  
363      def open(self):
364          """init the checkers: reset linesets and statistics information"""
365          self.linesets = []
366          self.stats = self.linter.add_stats(
367              nb_duplicated_lines=0, percent_duplicated_lines=0
368          )
369  
370      def process_module(self, node):
371          """process a module 
372   
373          the module's content is accessible via the stream object 
374   
375          stream must implement the readlines method 
376          """
377          with node.stream() as stream:
378              self.append_stream(self.linter.current_name, stream, node.file_encoding)
379  
380      def close(self):
381          """compute and display similarities on closing (i.e. end of parsing)"""
382          total = sum(len(lineset) for lineset in self.linesets)
383          duplicated = 0
384          stats = self.stats
385          for num, couples in self._compute_sims():
386              msg = []
387              lineset = idx = None
388              for lineset, idx in couples:
389                  msg.append("==%s:%s" % (lineset.name, idx))
390              msg.sort()
391  
392              if lineset:
393                  for line in lineset._real_lines[idx : idx + num]:
394                      msg.append(line.rstrip())
395  
396              self.add_message("R0801", args=(len(couples), "\n".join(msg)))
397              duplicated += num * (len(couples) - 1)
398          stats["nb_duplicated_lines"] = duplicated
399          stats["percent_duplicated_lines"] = total and duplicated * 100.0 / total
400

402  def register(linter):
403      """required method to auto register this checker """
404      linter.register_checker(SimilarChecker(linter))

407  def usage(status=0):
408      """display command line usage information"""
409      print("finds copy pasted blocks in a set of files")
410      print()
411      print(
412          "Usage: symilar [-d|--duplicates min_duplicated_lines] \ 
413  [-i|--ignore-comments] [--ignore-docstrings] [--ignore-imports] file1..."
414      )
415      sys.exit(status)

418  def Run(argv=None):
419      """standalone command line access point"""
420      if argv is None:
421          argv = sys.argv[1:]
422  
423      s_opts = "hdi"
424      l_opts = (
425          "help",
426          "duplicates=",
427          "ignore-comments",
428          "ignore-imports",
429          "ignore-docstrings",
430      )
431      min_lines = 4
432      ignore_comments = False
433      ignore_docstrings = False
434      ignore_imports = False
435      opts, args = getopt(argv, s_opts, l_opts)
436      for opt, val in opts:
437          if opt in ("-d", "--duplicates"):
438              min_lines = int(val)
439          elif opt in ("-h", "--help"):
440              usage()
441          elif opt in ("-i", "--ignore-comments"):
442              ignore_comments = True
443          elif opt in ("--ignore-docstrings",):
444              ignore_docstrings = True
445          elif opt in ("--ignore-imports",):
446              ignore_imports = True
447      if not args:
448          usage(1)
449      sim = Similar(min_lines, ignore_comments, ignore_docstrings, ignore_imports)
450      for filename in args:
451          with open(filename) as stream:
452              sim.append_stream(filename, stream)
453      sim.run()
454      sys.exit(0)
