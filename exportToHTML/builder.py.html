49   def open_source_file(filename):
50       with open(filename, "rb") as byte_stream:
51           encoding = detect_encoding(byte_stream.readline)[0]
52       stream = open(filename, "r", newline=None, encoding=encoding)
53       data = stream.read()
54       return stream, encoding, data

57   def _can_assign_attr(node, attrname):
58       try:
59           slots = node.slots()
60       except NotImplementedError:
61           pass
62       else:
63           if slots and attrname not in {slot.value for slot in slots}:
64               return False
65       return True

68   class AstroidBuilder(raw_building.InspectBuilder):
69       """Class for building an astroid tree from source code or from a live module. 
70    
71       The param *manager* specifies the manager class which should be used. 
72       If no manager is given, then the default one will be used. The 
73       param *apply_transforms* determines if the transforms should be 
74       applied after the tree was built from source or from a live object, 
75       by default being True. 
76       """
77   
78       # pylint: disable=redefined-outer-name
79       def __init__(self, manager=None, apply_transforms=True):
80           super().__init__()
81           self._manager = manager or MANAGER
82           self._apply_transforms = apply_transforms
83   
84       def module_build(self, module, modname=None):
85           """Build an astroid from a living module instance."""
86           node = None
87           path = getattr(module, "__file__", None)
88           if path is not None:
89               path_, ext = os.path.splitext(modutils._path_from_filename(path))
90               if ext in (".py", ".pyc", ".pyo") and os.path.exists(path_ + ".py"):
91                   node = self.file_build(path_ + ".py", modname)
92           if node is None:
93               # this is a built-in module
94               # get a partial representation by introspection
95               node = self.inspect_build(module, modname=modname, path=path)
96               if self._apply_transforms:
97                   # We have to handle transformation by ourselves since the
98                   # rebuilder isn't called for builtin nodes
99                   node = self._manager.visit_transforms(node)
100          return node
101  
102      def file_build(self, path, modname=None):
103          """Build astroid from a source code file (i.e. from an ast) 
104   
105          *path* is expected to be a python source file 
106          """
107          try:
108              stream, encoding, data = open_source_file(path)
109          except IOError as exc:
110              raise exceptions.AstroidBuildingError(
111                  "Unable to load file {path}:\n{error}",
112                  modname=modname,
113                  path=path,
114                  error=exc,
115              ) from exc
116          except (SyntaxError, LookupError) as exc:
117              raise exceptions.AstroidSyntaxError(
118                  "Python 3 encoding specification error or unknown encoding:\n"
119                  "{error}",
120                  modname=modname,
121                  path=path,
122                  error=exc,
123              ) from exc
124          except UnicodeError as exc:  # wrong encoding
125              # detect_encoding returns utf-8 if no encoding specified
126              raise exceptions.AstroidBuildingError(
127                  "Wrong or no encoding specified for {filename}.", filename=path
128              ) from exc
129          with stream:
130              # get module name if necessary
131              if modname is None:
132                  try:
133                      modname = ".".join(modutils.modpath_from_file(path))
134                  except ImportError:
135                      modname = os.path.splitext(os.path.basename(path))[0]
136              # build astroid representation
137              module = self._data_build(data, modname, path)
138              return self._post_build(module, encoding)
139  
140      def string_build(self, data, modname="", path=None):
141          """Build astroid from source code string."""
142          module = self._data_build(data, modname, path)
143          module.file_bytes = data.encode("utf-8")
144          return self._post_build(module, "utf-8")
145  
146      def _post_build(self, module, encoding):
147          """Handles encoding and delayed nodes after a module has been built"""
148          module.file_encoding = encoding
149          self._manager.cache_module(module)
150          # post tree building steps after we stored the module in the cache:
151          for from_node in module._import_from_nodes:
152              if from_node.modname == "__future__":
153                  for symbol, _ in from_node.names:
154                      module.future_imports.add(symbol)
155              self.add_from_names_to_locals(from_node)
156          # handle delayed assattr nodes
157          for delayed in module._delayed_assattr:
158              self.delayed_assattr(delayed)
159  
160          # Visit the transforms
161          if self._apply_transforms:
162              module = self._manager.visit_transforms(module)
163          return module
164  
165      def _data_build(self, data, modname, path):
166          """Build tree node from data and add some informations"""
167          try:
168              node, parser_module = _parse_string(data, type_comments=True)
169          except (TypeError, ValueError, SyntaxError) as exc:
170              raise exceptions.AstroidSyntaxError(
171                  "Parsing Python code failed:\n{error}",
172                  source=data,
173                  modname=modname,
174                  path=path,
175                  error=exc,
176              ) from exc
177  
178          if path is not None:
179              node_file = os.path.abspath(path)
180          else:
181              node_file = "<?>"
182          if modname.endswith(".__init__"):
183              modname = modname[:-9]
184              package = True
185          else:
186              package = (
187                  path is not None
188                  and os.path.splitext(os.path.basename(path))[0] == "__init__"
189              )
190          builder = rebuilder.TreeRebuilder(self._manager, parser_module)
191          module = builder.visit_module(node, modname, node_file, package)
192          module._import_from_nodes = builder._import_from_nodes
193          module._delayed_assattr = builder._delayed_assattr
194          return module

196      def add_from_names_to_locals(self, node):
197          """Store imported names to the locals 
198   
199          Resort the locals if coming from a delayed node 
200          """
201          _key_func = lambda node: node.fromlineno
202  
203          def sort_locals(my_list):
204              my_list.sort(key=_key_func)
205  
206          for (name, asname) in node.names:
207              if name == "*":
208                  try:
209                      imported = node.do_import_module()
210                  except exceptions.AstroidBuildingError:
211                      continue
212                  for name in imported.public_names():
213                      node.parent.set_local(name, node)
214                      sort_locals(node.parent.scope().locals[name])
215              else:
216                  node.parent.set_local(asname or name, node)
217                  sort_locals(node.parent.scope().locals[asname or name])
218  
219      def delayed_assattr(self, node):
220          """Visit a AssAttr node 
221   
222          This adds name to locals and handle members definition. 
223          """
224          try:
225              frame = node.frame()
226              for inferred in node.expr.infer():
227                  if inferred is util.Uninferable:
228                      continue
229                  try:
230                      cls = inferred.__class__
231                      if cls is bases.Instance or cls is objects.ExceptionInstance:
232                          inferred = inferred._proxied
233                          iattrs = inferred.instance_attrs
234                          if not _can_assign_attr(inferred, node.attrname):
235                              continue
236                      elif isinstance(inferred, bases.Instance):
237                          # Const, Tuple or other containers that inherit from
238                          # `Instance`
239                          continue
240                      elif inferred.is_function:
241                          iattrs = inferred.instance_attrs
242                      else:
243                          iattrs = inferred.locals
244                  except AttributeError:
245                      # XXX log error
246                      continue
247                  values = iattrs.setdefault(node.attrname, [])
248                  if node in values:
249                      continue
250                  # get assign in __init__ first XXX useful ?
251                  if (
252                      frame.name == "__init__"
253                      and values
254                      and values[0].frame().name != "__init__"
255                  ):
256                      values.insert(0, node)
257                  else:
258                      values.append(node)
259          except exceptions.InferenceError:
260              pass

267  def parse(code, module_name="", path=None, apply_transforms=True):
268      """Parses a source string in order to obtain an astroid AST from it 
269   
270      :param str code: The code for the module. 
271      :param str module_name: The name for the module, if any 
272      :param str path: The path for the module 
273      :param bool apply_transforms: 
274          Apply the transforms for the give code. Use it if you 
275          don't want the default transforms to be applied. 
276      """
277      code = textwrap.dedent(code)
278      builder = AstroidBuilder(manager=MANAGER, apply_transforms=apply_transforms)
279      return builder.string_build(code, modname=module_name, path=path)

282  def _extract_expressions(node):
283      """Find expressions in a call to _TRANSIENT_FUNCTION and extract them. 
284   
285      The function walks the AST recursively to search for expressions that 
286      are wrapped into a call to _TRANSIENT_FUNCTION. If it finds such an 
287      expression, it completely removes the function call node from the tree, 
288      replacing it by the wrapped expression inside the parent. 
289   
290      :param node: An astroid node. 
291      :type node:  astroid.bases.NodeNG 
292      :yields: The sequence of wrapped expressions on the modified tree 
293      expression can be found. 
294      """
295      if (
296          isinstance(node, nodes.Call)
297          and isinstance(node.func, nodes.Name)
298          and node.func.name == _TRANSIENT_FUNCTION
299      ):
300          real_expr = node.args[0]
301          real_expr.parent = node.parent
302          # Search for node in all _astng_fields (the fields checked when
303          # get_children is called) of its parent. Some of those fields may
304          # be lists or tuples, in which case the elements need to be checked.
305          # When we find it, replace it by real_expr, so that the AST looks
306          # like no call to _TRANSIENT_FUNCTION ever took place.
307          for name in node.parent._astroid_fields:
308              child = getattr(node.parent, name)
309              if isinstance(child, (list, tuple)):
310                  for idx, compound_child in enumerate(child):
311                      if compound_child is node:
312                          child[idx] = real_expr
313              elif child is node:
314                  setattr(node.parent, name, real_expr)
315          yield real_expr
316      else:
317          for child in node.get_children():
318              yield from _extract_expressions(child)

321  def _find_statement_by_line(node, line):
322      """Extracts the statement on a specific line from an AST. 
323   
324      If the line number of node matches line, it will be returned; 
325      otherwise its children are iterated and the function is called 
326      recursively. 
327   
328      :param node: An astroid node. 
329      :type node: astroid.bases.NodeNG 
330      :param line: The line number of the statement to extract. 
331      :type line: int 
332      :returns: The statement on the line, or None if no statement for the line 
333        can be found. 
334      :rtype:  astroid.bases.NodeNG or None 
335      """
336      if isinstance(node, (nodes.ClassDef, nodes.FunctionDef)):
337          # This is an inaccuracy in the AST: the nodes that can be
338          # decorated do not carry explicit information on which line
339          # the actual definition (class/def), but .fromline seems to
340          # be close enough.
341          node_line = node.fromlineno
342      else:
343          node_line = node.lineno
344  
345      if node_line == line:
346          return node
347  
348      for child in node.get_children():
349          result = _find_statement_by_line(child, line)
350          if result:
351              return result
352  
353      return None

356  def extract_node(code, module_name=""):
357      """Parses some Python code as a module and extracts a designated AST node. 
358   
359      Statements: 
360       To extract one or more statement nodes, append #@ to the end of the line 
361   
362       Examples: 
363         >>> def x(): 
364         >>>   def y(): 
365         >>>     return 1 #@ 
366   
367         The return statement will be extracted. 
368   
369         >>> class X(object): 
370         >>>   def meth(self): #@ 
371         >>>     pass 
372   
373        The function object 'meth' will be extracted. 
374   
375      Expressions: 
376       To extract arbitrary expressions, surround them with the fake 
377       function call __(...). After parsing, the surrounded expression 
378       will be returned and the whole AST (accessible via the returned 
379       node's parent attribute) will look like the function call was 
380       never there in the first place. 
381   
382       Examples: 
383         >>> a = __(1) 
384   
385         The const node will be extracted. 
386   
387         >>> def x(d=__(foo.bar)): pass 
388   
389         The node containing the default argument will be extracted. 
390   
391         >>> def foo(a, b): 
392         >>>   return 0 < __(len(a)) < b 
393   
394         The node containing the function call 'len' will be extracted. 
395   
396      If no statements or expressions are selected, the last toplevel 
397      statement will be returned. 
398   
399      If the selected statement is a discard statement, (i.e. an expression 
400      turned into a statement), the wrapped expression is returned instead. 
401   
402      For convenience, singleton lists are unpacked. 
403   
404      :param str code: A piece of Python code that is parsed as 
405      a module. Will be passed through textwrap.dedent first. 
406      :param str module_name: The name of the module. 
407      :returns: The designated node from the parse tree, or a list of nodes. 
408      :rtype: astroid.bases.NodeNG, or a list of nodes. 
409      """
410  
411      def _extract(node):
412          if isinstance(node, nodes.Expr):
413              return node.value
414  
415          return node
416  
417      requested_lines = []
418      for idx, line in enumerate(code.splitlines()):
419          if line.strip().endswith(_STATEMENT_SELECTOR):
420              requested_lines.append(idx + 1)
421  
422      tree = parse(code, module_name=module_name)
423      if not tree.body:
424          raise ValueError("Empty tree, cannot extract from it")
425  
426      extracted = []
427      if requested_lines:
428          extracted = [_find_statement_by_line(tree, line) for line in requested_lines]
429  
430      # Modifies the tree.
431      extracted.extend(_extract_expressions(tree))
432  
433      if not extracted:
434          extracted.append(tree.body[-1])
435  
436      extracted = [_extract(node) for node in extracted]
437      if len(extracted) == 1:
438          return extracted[0]
439      return extracted

442  def _parse_string(data, type_comments=True):
443      parser_module = get_parser_module(type_comments=type_comments)
444      try:
445          parsed = parser_module.parse(data + "\n", type_comments=type_comments)
446      except SyntaxError as exc:
447          # If the type annotations are misplaced for some reason, we do not want
448          # to fail the entire parsing of the file, so we need to retry the parsing without
449          # type comment support.
450          if exc.args[0] != MISPLACED_TYPE_ANNOTATION_ERROR or not type_comments:
451              raise
452  
453          parser_module = get_parser_module(type_comments=False)
454          parsed = parser_module.parse(data + "\n", type_comments=False)
455      return parsed, parser_module
