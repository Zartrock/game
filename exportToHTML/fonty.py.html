11   import pygame as pg
12   from pygame.compat import unicode_
13   import sys
14   import locale

17   if sys.version_info >= (3,):
18   
19       def print_unicode(s):
20           e = locale.getpreferredencoding()
21           print(s.encode(e, "backslashreplace").decode())
22   
23   
24   else:
25   
26       def print_unicode(s):
27           e = locale.getpreferredencoding()
28           print(s.encode(e, "backslashreplace"))

31   def main():
32       # initialize
33       pg.init()
34       resolution = 400, 200
35       screen = pg.display.set_mode(resolution)
36   
37       ##    pg.mouse.set_cursor(*pg.cursors.diamond)
38   
39       fg = 250, 240, 230
40       bg = 5, 5, 5
41       wincolor = 40, 40, 90
42   
43       # fill background
44       screen.fill(wincolor)
45   
46       # load font, prepare values
47       font = pg.font.Font(None, 80)
48       text = "Fonty"
49       size = font.size(text)
50   
51       # no AA, no transparancy, normal
52       ren = font.render(text, 0, fg, bg)
53       screen.blit(ren, (10, 10))
54   
55       # no AA, transparancy, underline
56       font.set_underline(1)
57       ren = font.render(text, 0, fg)
58       screen.blit(ren, (10, 40 + size[1]))
59       font.set_underline(0)
60   
61       a_sys_font = pg.font.SysFont("Arial", 60)
62   
63       # AA, no transparancy, bold
64       a_sys_font.set_bold(1)
65       ren = a_sys_font.render(text, 1, fg, bg)
66       screen.blit(ren, (30 + size[0], 10))
67       a_sys_font.set_bold(0)
68   
69       # AA, transparancy, italic
70       a_sys_font.set_italic(1)
71       ren = a_sys_font.render(text, 1, fg)
72       screen.blit(ren, (30 + size[0], 40 + size[1]))
73       a_sys_font.set_italic(0)
74   
75       # Get some metrics.
76       print("Font metrics for 'Fonty':  %s" % a_sys_font.metrics(text))
77       ch = unicode_("%c") % 0x3060
78       msg = unicode_("Font metrics for '%s':  %s") % (ch, a_sys_font.metrics(ch))
79       print_unicode(msg)
80   
81       ## #some_japanese_unicode = u"\u304b\u3070\u306b"
82       ##some_japanese_unicode = unicode_('%c%c%c') % (0x304b, 0x3070, 0x306b)
83   
84       # AA, transparancy, italic
85       ##ren = a_sys_font.render(some_japanese_unicode, 1, fg)
86       ##screen.blit(ren, (30 + size[0], 40 + size[1]))
87   
88       # show the surface and await user quit
89       pg.display.flip()
90       while 1:
91           # use event.wait to keep from polling 100% cpu
92           if pg.event.wait().type in (pg.QUIT, pg.KEYDOWN, pg.MOUSEBUTTONDOWN):
93               break
94   
95   
96   if __name__ == "__main__":
97       main()
