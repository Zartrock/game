16       def _assert_compatible(self, arr, size):
17           dtype = self.array_dtypes[size]
18           self.assertEqual(arr.dtype, dtype)

20       def test_array(self):
21           def check_array(size, channels, test_data):
22               try:
23                   pygame.mixer.init(22050, size, channels, allowedchanges=0)
24               except pygame.error:
25                   # Not all sizes are supported on all systems.
26                   return
27               try:
28                   __, sz, __ = pygame.mixer.get_init()
29                   if sz == size:
30                       srcarr = array(test_data, self.array_dtypes[size])
31                       snd = pygame.sndarray.make_sound(srcarr)
32                       arr = pygame.sndarray.array(snd)
33                       self._assert_compatible(arr, size)
34                       self.assertTrue(
35                           alltrue(arr == srcarr),
36                           "size: %i\n%s\n%s" % (size, arr, test_data),
37                       )
38               finally:
39                   pygame.mixer.quit()
40   
41           check_array(8, 1, [0, 0x0F, 0xF0, 0xFF])
42           check_array(8, 2, [[0, 0x80], [0x2D, 0x41], [0x64, 0xA1], [0xFF, 0x40]])
43           check_array(16, 1, [0, 0x00FF, 0xFF00, 0xFFFF])
44           check_array(
45               16, 2, [[0, 0xFFFF], [0xFFFF, 0], [0x00FF, 0xFF00], [0x0F0F, 0xF0F0]]
46           )
47           check_array(-8, 1, [0, -0x80, 0x7F, 0x64])
48           check_array(-8, 2, [[0, -0x80], [-0x64, 0x64], [0x25, -0x50], [0xFF, 0]])

98       def test_samples(self):
99   
100          null_byte = as_bytes("\x00")
101  
102          def check_sample(size, channels, test_data):
103              try:
104                  pygame.mixer.init(22050, size, channels, allowedchanges=0)
105              except pygame.error:
106                  # Not all sizes are supported on all systems.
107                  return
108              try:
109                  __, sz, __ = pygame.mixer.get_init()
110                  if sz == size:
111                      zeroed = null_byte * ((abs(size) // 8) * len(test_data) * channels)
112                      snd = pygame.mixer.Sound(buffer=zeroed)
113                      samples = pygame.sndarray.samples(snd)
114                      self._assert_compatible(samples, size)
115                      ##print ('X %s' % (samples.shape,))
116                      ##print ('Y %s' % (test_data,))
117                      samples[...] = test_data
118                      arr = pygame.sndarray.array(snd)
119                      self.assertTrue(
120                          alltrue(samples == arr),
121                          "size: %i\n%s\n%s" % (size, arr, test_data),
122                      )
123              finally:
124                  pygame.mixer.quit()
125  
126          check_sample(8, 1, [0, 0x0F, 0xF0, 0xFF])
127          check_sample(8, 2, [[0, 0x80], [0x2D, 0x41], [0x64, 0xA1], [0xFF, 0x40]])
128          check_sample(16, 1, [0, 0x00FF, 0xFF00, 0xFFFF])
129          check_sample(
130              16, 2, [[0, 0xFFFF], [0xFFFF, 0], [0x00FF, 0xFF00], [0x0F0F, 0xF0F0]]
131          )
132          check_sample(-8, 1, [0, -0x80, 0x7F, 0x64])
133          check_sample(-8, 2, [[0, -0x80], [-0x64, 0x64], [0x25, -0x50], [0xFF, 0]])
134          check_sample(-16, 1, [0, 0x7FFF, -0x7FFF, -1])
