475          checkers.initialize(self)
476          reporters.initialize(self)
477          # Make sure to load the default reporter, because
478          # the option has been set before the plugins had been loaded.
479          if not self.reporter:

482      def load_plugin_modules(self, modnames):
483          """take a list of module names which are pylint plugins and load 
484          and register them 
485          """
486          for modname in modnames:
487              if modname in self._dynamic_plugins:
488                  continue
489              self._dynamic_plugins.add(modname)
490              module = modutils.load_module_from_name(modname)
491              module.register(self)
492  
493      def load_plugin_configuration(self):
494          """Call the configuration hook for plugins 
495   
496          This walks through the list of plugins, grabs the "load_configuration" 
497          hook, if exposed, and calls it to allow plugins to configure specific 
498          settings. 
499          """
500          for modname in self._dynamic_plugins:
501              module = modutils.load_module_from_name(modname)
502              if hasattr(module, "load_configuration"):
503                  module.load_configuration(self)

505      def _load_reporter(self):
506          name = self._reporter_name.lower()
507          if name in self._reporters:
508              self.set_reporter(self._reporters[name]())
509          else:
510              try:
511                  reporter_class = self._load_reporter_class()
512              except (ImportError, AttributeError) as e:
513                  raise exceptions.InvalidReporterError(name) from e
514              else:
515                  self.set_reporter(reporter_class())
516  
517      def _load_reporter_class(self):
518          qname = self._reporter_name
519          module = modutils.load_module_from_name(modutils.get_module_part(qname))
520          class_name = qname.split(".")[-1]
521          reporter_class = getattr(module, class_name)
522          return reporter_class

529      def set_option(self, optname, value, action=None, optdict=None):
530          """overridden from config.OptionsProviderMixin to handle some 
531          special options 
532          """
533          if optname in self._options_methods or optname in self._bw_options_methods:
534              if value:
535                  try:
536                      meth = self._options_methods[optname]
537                  except KeyError:
538                      meth = self._bw_options_methods[optname]
539                      warnings.warn(
540                          "%s is deprecated, replace it by %s"
541                          % (optname, optname.split("-")[0]),
542                          DeprecationWarning,
543                      )
544                  value = utils._check_csv(value)
545                  if isinstance(value, (list, tuple)):
546                      for _id in value:
547                          meth(_id, ignore_unknown=True)
548                  else:
549                      meth(value)
550                  return  # no need to call set_option, disable/enable methods do it
551          elif optname == "output-format":
552              self._reporter_name = value
553              # If the reporters are already available, load
554              # the reporter class.
555              if self._reporters:
556                  self._load_reporter()
557  
558          try:
559              checkers.BaseTokenChecker.set_option(self, optname, value, action, optdict)
560          except config.UnsupportedAction:
561              print("option %s can't be read from config file" % optname, file=sys.stderr)

566      def report_order(self):
567          reports = sorted(self._reports, key=lambda x: getattr(x, "name", ""))
568          try:
569              # Remove the current reporter and add it
570              # at the end of the list.
571              reports.pop(reports.index(self))
572          except ValueError:
573              pass
574          else:
575              reports.append(self)
576          return reports

580      def register_checker(self, checker):
581          """register a new checker 
582   
583          checker is an object implementing IRawChecker or / and IAstroidChecker 
584          """
585          assert checker.priority <= 0, "checker priority can't be >= 0"
586          self._checkers[checker.name].append(checker)
587          for r_id, r_title, r_cb in checker.reports:
588              self.register_report(r_id, r_title, r_cb, checker)
589          self.register_options_provider(checker)
590          if hasattr(checker, "msgs"):
591              self.msgs_store.register_messages_from_checker(checker)
592          checker.load_defaults()
593  
594          # Register the checker, but disable all of its messages.
595          if not getattr(checker, "enabled", True):
596              self.disable(checker.name)

598      def disable_noerror_messages(self):
599          for msgcat, msgids in self.msgs_store._msgs_by_category.items():
600              # enable only messages with 'error' severity and above ('fatal')
601              if msgcat in ["E", "F"]:
602                  for msgid in msgids:
603                      self.enable(msgid)
604              else:
605                  for msgid in msgids:
606                      self.disable(msgid)

608      def disable_reporters(self):
609          """disable all reporters"""
610          for _reporters in self._reports.values():
611              for report_id, _, _ in _reporters:
612                  self.disable_report(report_id)

614      def error_mode(self):
615          """error mode: enable only errors; no reports, no persistent"""
616          self._error_mode = True
617          self.disable_noerror_messages()
618          self.disable("miscellaneous")
619          if self._python3_porting_mode:
620              self.disable("all")
621              for msg_id in self._checker_messages("python3"):
622                  if msg_id.startswith("E"):
623                      self.enable(msg_id)
624              config_parser = self.cfgfile_parser
625              if config_parser.has_option("MESSAGES CONTROL", "disable"):
626                  value = config_parser.get("MESSAGES CONTROL", "disable")
627                  self.global_set_option("disable", value)
628          else:
629              self.disable("python3")
630          self.set_option("reports", False)
631          self.set_option("persistent", False)
632          self.set_option("score", False)
633  
634      def python3_porting_mode(self):
635          """Disable all other checkers and enable Python 3 warnings."""
636          self.disable("all")
637          # re-enable some errors, or 'print', 'raise', 'async', 'await' will mistakenly lint fine
638          self.enable("fatal")  # F0001
639          self.enable("astroid-error")  # F0002
640          self.enable("parse-error")  # F0010
641          self.enable("syntax-error")  # E0001
642          self.enable("python3")
643          if self._error_mode:
644              # The error mode was activated, using the -E flag.
645              # So we'll need to enable only the errors from the
646              # Python 3 porting checker.
647              for msg_id in self._checker_messages("python3"):
648                  if msg_id.startswith("E"):
649                      self.enable(msg_id)
650                  else:
651                      self.disable(msg_id)
652          config_parser = self.cfgfile_parser
653          if config_parser.has_option("MESSAGES CONTROL", "disable"):
654              value = config_parser.get("MESSAGES CONTROL", "disable")
655              self.global_set_option("disable", value)
656          self._python3_porting_mode = True

658      def list_messages_enabled(self):
659          enabled = [
660              "  %s (%s)" % (message.symbol, message.msgid)
661              for message in self.msgs_store.messages
662              if self.is_message_enabled(message.msgid)
663          ]
664          disabled = [
665              "  %s (%s)" % (message.symbol, message.msgid)
666              for message in self.msgs_store.messages
667              if not self.is_message_enabled(message.msgid)
668          ]
669          print("Enabled messages:")
670          for msg in sorted(enabled):
671              print(msg)
672          print("\nDisabled messages:")
673          for msg in sorted(disabled):
674              print(msg)
675          print("")

677      # block level option handling #############################################
678      #
679      # see func_block_disable_msg.py test case for expected behaviour
680  
681      def process_tokens(self, tokens):
682          """process tokens from the current module to search for module/block 
683          level options 
684          """
685          control_pragmas = {"disable", "enable"}
686          prev_line = None
687          saw_newline = True
688          seen_newline = True
689          for (tok_type, content, start, _, _) in tokens:
690              if prev_line and prev_line != start[0]:
691                  saw_newline = seen_newline
692                  seen_newline = False
693  
694              prev_line = start[0]
695              if tok_type in (tokenize.NL, tokenize.NEWLINE):
696                  seen_newline = True
697  
698              if tok_type != tokenize.COMMENT:
699                  continue
700              match = OPTION_PO.search(content)
701              if match is None:
702                  continue
703  
704              try:
705                  for pragma_repr in parse_pragma(match.group(2)):
706                      if pragma_repr.action in ("disable-all", "skip-file"):
707                          if pragma_repr.action == "disable-all":
708                              self.add_message(
709                                  "deprecated-pragma",
710                                  line=start[0],
711                                  args=("disable-all", "skip-file"),
712                              )
713                          self.add_message("file-ignored", line=start[0])
714                          self._ignore_file = True
715                          return
716                      try:
717                          meth = self._options_methods[pragma_repr.action]
718                      except KeyError:
719                          meth = self._bw_options_methods[pragma_repr.action]
720                          # found a "(dis|en)able-msg" pragma deprecated suppression
721                          self.add_message(
722                              "deprecated-pragma",
723                              line=start[0],
724                              args=(
725                                  pragma_repr.action,
726                                  pragma_repr.action.replace("-msg", ""),
727                              ),
728                          )
729                      for msgid in pragma_repr.messages:
730                          # Add the line where a control pragma was encountered.
731                          if pragma_repr.action in control_pragmas:
732                              self._pragma_lineno[msgid] = start[0]
733  
734                          if (pragma_repr.action, msgid) == ("disable", "all"):
735                              self.add_message(
736                                  "deprecated-pragma",
737                                  line=start[0],
738                                  args=("disable=all", "skip-file"),
739                              )
740                              self.add_message("file-ignored", line=start[0])
741                              self._ignore_file = True
742                              return
743                              # If we did not see a newline between the previous line and now,
744                              # we saw a backslash so treat the two lines as one.
745                          l_start = start[0]
746                          if not saw_newline:
747                              l_start -= 1
748                          try:
749                              meth(msgid, "module", l_start)
750                          except exceptions.UnknownMessageError:
751                              self.add_message(
752                                  "bad-option-value", args=msgid, line=start[0]
753                              )
754              except UnRecognizedOptionError as err:
755                  self.add_message(
756                      "unrecognized-inline-option", args=err.token, line=start[0]
757                  )
758                  continue
759              except InvalidPragmaError as err:
760                  self.add_message("bad-inline-option", args=err.token, line=start[0])
761                  continue

766          """return all available checkers as a list"""
767          return [self] + [
768              c
769              for _checkers in self._checkers.values()
770              for c in _checkers

775          """Get all the checker names that this linter knows about."""
776          current_checkers = self.get_checkers()
777          return sorted(
778              {
779                  checker.name
780                  for checker in current_checkers
781                  if checker.name != MAIN_CHECKER_NAME
782              }

785      def prepare_checkers(self):
786          """return checkers needed for activated messages and reports"""
787          if not self.config.reports:
788              self.disable_reporters()
789          # get needed checkers
790          needed_checkers = [self]
791          for checker in self.get_checkers()[1:]:
792              messages = {msg for msg in checker.msgs if self.is_message_enabled(msg)}
793              if messages or any(self.report_is_enabled(r[0]) for r in checker.reports):
794                  needed_checkers.append(checker)
795          # Sort checkers by priority
796          needed_checkers = sorted(
797              needed_checkers, key=operator.attrgetter("priority"), reverse=True
798          )
799          return needed_checkers

803      def should_analyze_file(modname, path, is_argument=False):
804          """Returns whether or not a module should be checked. 
805   
806          This implementation returns True for all python source file, indicating 
807          that all files should be linted. 
808   
809          Subclasses may override this method to indicate that modules satisfying 
810          certain conditions should not be linted. 
811   
812          :param str modname: The name of the module to be checked. 
813          :param str path: The full path to the source code of the module. 
814          :param bool is_argument: Whetter the file is an argument to pylint or not. 
815                                   Files which respect this property are always 
816                                   checked, since the user requested it explicitly. 
817          :returns: True if the module should be checked. 
818          :rtype: bool 
819          """
820          if is_argument:
821              return True
822          return path.endswith(".py")

826      def initialize(self):
827          """Initialize linter for linting 
828   
829          This method is called before any linting is done. 
830          """
831          # initialize msgs_state now that all messages have been registered into
832          # the store
833          for msg in self.msgs_store.messages:
834              if not msg.may_be_emitted():
835                  self._msgs_state[msg.msgid] = False

837      def check(self, files_or_modules):
838          """main checking entry: check a list of files or modules from their name. 
839   
840          files_or_modules is either a string or list of strings presenting modules to check. 
841          """
842  
843          self.initialize()
844  
845          if not isinstance(files_or_modules, (list, tuple)):
846              files_or_modules = (files_or_modules,)
847  
848          if self.config.from_stdin:
849              if len(files_or_modules) != 1:
850                  raise exceptions.InvalidArgsError(
851                      "Missing filename required for --from-stdin"
852                  )
853  
854              filepath = files_or_modules[0]
855              with fix_import_path(files_or_modules):
856                  self._check_files(
857                      functools.partial(self.get_ast, data=_read_stdin()),
858                      [self._get_file_descr_from_stdin(filepath)],
859                  )
860          elif self.config.jobs == 1:
861              with fix_import_path(files_or_modules):
862                  self._check_files(
863                      self.get_ast, self._iterate_file_descrs(files_or_modules)
864                  )
865          else:
866              check_parallel(
867                  self,
868                  self.config.jobs,
869                  self._iterate_file_descrs(files_or_modules),
870                  files_or_modules,
871              )

873      def check_single_file(self, name, filepath, modname):
874          """Check single file 
875   
876          The arguments are the same that are documented in _check_files 
877   
878          The initialize() method should be called before calling this method 
879          """
880          with self._astroid_module_checker() as check_astroid_module:
881              self._check_file(
882                  self.get_ast, check_astroid_module, name, filepath, modname
883              )

885      def _check_files(self, get_ast, file_descrs):
886          """Check all files from file_descrs 
887   
888          The file_descrs should be iterable of tuple (name, filepath, modname) 
889          where 
890          - name: full name of the module 
891          - filepath: path of the file 
892          - modname: module name 
893          """
894          with self._astroid_module_checker() as check_astroid_module:
895              for name, filepath, modname in file_descrs:
896                  self._check_file(get_ast, check_astroid_module, name, filepath, modname)

898      def _check_file(self, get_ast, check_astroid_module, name, filepath, modname):
899          """Check a file using the passed utility functions (get_ast and check_astroid_module) 
900   
901          :param callable get_ast: callable returning AST from defined file taking the following arguments 
902          - filepath: path to the file to check 
903          - name: Python module name 
904          :param callable check_astroid_module: callable checking an AST taking the following arguments 
905          - ast: AST of the module 
906          :param str name: full name of the module 
907          :param str filepath: path to checked file 
908          :param str modname: name of the checked Python module 
909          """
910          self.set_current_module(name, filepath)
911          # get the module representation
912          ast_node = get_ast(filepath, name)
913          if ast_node is None:
914              return
915  
916          self._ignore_file = False
917  
918          self.file_state = FileState(modname)
919          # fix the current file (if the source file was not available or
920          # if it's actually a c extension)
921          self.current_file = ast_node.file  # pylint: disable=maybe-no-member
922          check_astroid_module(ast_node)
923          # warn about spurious inline messages handling
924          spurious_messages = self.file_state.iter_spurious_suppression_messages(
925              self.msgs_store
926          )
927          for msgid, line, args in spurious_messages:
928              self.add_message(msgid, line, None, args)

930      @staticmethod
931      def _get_file_descr_from_stdin(filepath):
932          """Return file description (tuple of module name, file path, base name) from given file path 
933   
934          This method is used for creating suitable file description for _check_files when the 
935          source is standard input. 
936          """
937          try:
938              # Note that this function does not really perform an
939              # __import__ but may raise an ImportError exception, which
940              # we want to catch here.
941              modname = ".".join(modutils.modpath_from_file(filepath))
942          except ImportError:
943              modname = os.path.splitext(os.path.basename(filepath))[0]
944  
945          return (modname, filepath, filepath)

947      def _iterate_file_descrs(self, files_or_modules):
948          """Return generator yielding file descriptions (tuples of module name, file path, base name) 
949   
950          The returned generator yield one item for each Python module that should be linted. 
951          """
952          for descr in self._expand_files(files_or_modules):
953              name, filepath, is_arg = descr["name"], descr["path"], descr["isarg"]
954              if self.should_analyze_file(name, filepath, is_argument=is_arg):
955                  yield (name, filepath, descr["basename"])
956  
957      def _expand_files(self, modules):
958          """get modules and errors from a list of modules and handle errors 
959          """
960          result, errors = utils.expand_modules(
961              modules, self.config.black_list, self.config.black_list_re
962          )
963          for error in errors:
964              message = modname = error["mod"]
965              key = error["key"]
966              self.set_current_module(modname)
967              if key == "fatal":
968                  message = str(error["ex"]).replace(os.getcwd() + os.sep, "")
969              self.add_message(key, args=message)
970          return result

972      def set_current_module(self, modname, filepath=None):
973          """set the name of the currently analyzed module and 
974          init statistics for it 
975          """
976          if not modname and filepath is None:
977              return
978          self.reporter.on_set_current_module(modname, filepath)
979          self.current_name = modname
980          self.current_file = filepath or modname
981          self.stats["by_module"][modname] = {}
982          self.stats["by_module"][modname]["statement"] = 0
983          for msg_cat in MSG_TYPES.values():
984              self.stats["by_module"][modname][msg_cat] = 0

986      @contextlib.contextmanager
987      def _astroid_module_checker(self):
988          """Context manager for checking ASTs 
989   
990          The value in the context is callable accepting AST as its only argument. 
991          """
992          walker = ASTWalker(self)
993          _checkers = self.prepare_checkers()
994          tokencheckers = [
995              c
996              for c in _checkers
997              if interfaces.implements(c, interfaces.ITokenChecker) and c is not self
998          ]
999          rawcheckers = [
1000             c for c in _checkers if interfaces.implements(c, interfaces.IRawChecker)
1001         ]
1002         # notify global begin
1003         for checker in _checkers:
1004             checker.open()
1005             if interfaces.implements(checker, interfaces.IAstroidChecker):
1006                 walker.add_checker(checker)
1007 
1008         yield functools.partial(
1009             self.check_astroid_module,
1010             walker=walker,
1011             tokencheckers=tokencheckers,
1012             rawcheckers=rawcheckers,
1013         )
1014 
1015         # notify global end
1016         self.stats["statement"] = walker.nbstatements
1017         for checker in reversed(_checkers):
1018             checker.close()

1020     def get_ast(self, filepath, modname, data=None):
1021         """Return an ast(roid) representation of a module or a string. 
1022  
1023         :param str filepath: path to checked file. 
1024         :param str modname: The name of the module to be checked. 
1025         :param str data: optional contents of the checked file. 
1026         :returns: the AST 
1027         :rtype: astroid.nodes.Module 
1028         """
1029         try:
1030             if data is None:
1031                 return MANAGER.ast_from_file(filepath, modname, source=True)
1032             return AstroidBuilder(MANAGER).string_build(data, modname, filepath)
1033         except astroid.AstroidSyntaxError as ex:
1034             # pylint: disable=no-member
1035             self.add_message(
1036                 "syntax-error",
1037                 line=getattr(ex.error, "lineno", 0),
1038                 col_offset=getattr(ex.error, "offset", None),
1039                 args=str(ex.error),
1040             )
1041         except astroid.AstroidBuildingException as ex:
1042             self.add_message("parse-error", args=ex)
1043         except Exception as ex:  # pylint: disable=broad-except
1044             traceback.print_exc()
1045             self.add_message("astroid-error", args=(ex.__class__, ex))
1046 
1047     def check_astroid_module(self, ast_node, walker, rawcheckers, tokencheckers):
1048         """Check a module from its astroid representation. 
1049  
1050         For return value see _check_astroid_module 
1051         """
1052         before_check_statements = walker.nbstatements
1053 
1054         retval = self._check_astroid_module(
1055             ast_node, walker, rawcheckers, tokencheckers
1056         )
1057 
1058         self.stats["by_module"][self.current_name]["statement"] = (
1059             walker.nbstatements - before_check_statements
1060         )
1061 
1062         return retval
1063 
1064     def _check_astroid_module(self, ast_node, walker, rawcheckers, tokencheckers):
1065         """Check given AST node with given walker and checkers 
1066  
1067         :param astroid.nodes.Module ast_node: AST node of the module to check 
1068         :param pylint.utils.ast_walker.ASTWalker walker: AST walker 
1069         :param list rawcheckers: List of token checkers to use 
1070         :param list tokencheckers: List of raw checkers to use 
1071  
1072         :returns: True if the module was checked, False if ignored, 
1073             None if the module contents could not be parsed 
1074         :rtype: bool 
1075         """
1076         try:
1077             tokens = utils.tokenize_module(ast_node)
1078         except tokenize.TokenError as ex:
1079             self.add_message("syntax-error", line=ex.args[1][0], args=ex.args[0])
1080             return None
1081 
1082         if not ast_node.pure_python:
1083             self.add_message("raw-checker-failed", args=ast_node.name)
1084         else:
1085             # assert astroid.file.endswith('.py')
1086             # invoke ITokenChecker interface on self to fetch module/block
1087             # level options
1088             self.process_tokens(tokens)
1089             if self._ignore_file:
1090                 return False
1091             # walk ast to collect line numbers
1092             self.file_state.collect_block_lines(self.msgs_store, ast_node)
1093             # run raw and tokens checkers
1094             for checker in rawcheckers:
1095                 checker.process_module(ast_node)
1096             for checker in tokencheckers:
1097                 checker.process_tokens(tokens)
1098         # generate events to astroid checkers
1099         walker.walk(ast_node)
1100         return True

1113     def generate_reports(self):
1114         """close the whole package /module, it's time to make reports ! 
1115  
1116         if persistent run, pickle results for later comparison 
1117         """
1118         # Display whatever messages are left on the reporter.
1119         self.reporter.display_messages(report_nodes.Section())
1120 
1121         if self.file_state.base_name is not None:
1122             # load previous results if any
1123             previous_stats = config.load_results(self.file_state.base_name)
1124             self.reporter.on_close(self.stats, previous_stats)
1125             if self.config.reports:
1126                 sect = self.make_reports(self.stats, previous_stats)
1127             else:
1128                 sect = report_nodes.Section()
1129 
1130             if self.config.reports:
1131                 self.reporter.display_reports(sect)
1132             score_value = self._report_evaluation()
1133             # save results if persistent run
1134             if self.config.persistent:
1135                 config.save_results(self.stats, self.file_state.base_name)
1136         else:
1137             self.reporter.on_close(self.stats, {})
1138             score_value = None
1139         return score_value

1141     def _report_evaluation(self):
1142         """make the global evaluation report"""
1143         # check with at least check 1 statements (usually 0 when there is a
1144         # syntax error preventing pylint from further processing)
1145         note = None
1146         previous_stats = config.load_results(self.file_state.base_name)
1147         if self.stats["statement"] == 0:
1148             return note
1149 
1150         # get a global note for the code
1151         evaluation = self.config.evaluation
1152         try:
1153             note = eval(evaluation, {}, self.stats)  # pylint: disable=eval-used
1154         except Exception as ex:  # pylint: disable=broad-except
1155             msg = "An exception occurred while rating: %s" % ex
1156         else:
1157             self.stats["global_note"] = note
1158             msg = "Your code has been rated at %.2f/10" % note
1159             pnote = previous_stats.get("global_note")
1160             if pnote is not None:
1161                 msg += " (previous run: %.2f/10, %+.2f)" % (pnote, note - pnote)
1162 
1163         if self.config.score:
1164             sect = report_nodes.EvaluationSection(msg)
1165             self.reporter.display_reports(sect)
1166         return note
