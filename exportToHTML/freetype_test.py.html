43   def surf_same_image(a, b):
44       """Return True if a's pixel buffer is identical to b's"""
45   
46       a_sz = a.get_height() * a.get_pitch()
47       b_sz = b.get_height() * b.get_pitch()
48       if a_sz != b_sz:
49           return False
50       a_bytes = ctypes.string_at(a._pixels_address, a_sz)
51       b_bytes = ctypes.string_at(b._pixels_address, b_sz)
52       return a_bytes == b_bytes
