42   def run(*args, **kwds):
43       """Run the Pygame unit test suite and return (total tests run, fails dict) 
44    
45       Positional arguments (optional): 
46       The names of tests to include. If omitted then all tests are run. Test 
47       names need not include the trailing '_test'. 
48    
49       Keyword arguments: 
50       incomplete - fail incomplete tests (default False) 
51       usesubprocess - run all test suites in the current process 
52                      (default False, use separate subprocesses) 
53       dump - dump failures/errors as dict ready to eval (default False) 
54       file - if provided, the name of a file into which to dump failures/errors 
55       timings - if provided, the number of times to run each individual test to 
56                 get an average run time (default is run each test once) 
57       exclude - A list of TAG names to exclude from the run. The items may be 
58                 comma or space separated. 
59       show_output - show silenced stderr/stdout on errors (default False) 
60       all - dump all results, not just errors (default False) 
61       randomize - randomize order of tests (default False) 
62       seed - if provided, a seed randomizer integer 
63       multi_thread - if provided, the number of THREADS in which to run 
64                      subprocessed tests 
65       time_out - if subprocess is True then the time limit in seconds before 
66                  killing a test (default 30) 
67       fake - if provided, the name of the fake tests package in the 
68              run_tests__tests subpackage to run instead of the normal 
69              Pygame tests 
70       python - the path to a python executable to run subprocessed tests 
71                (default sys.executable) 
72       interative - allow tests tagged 'interative'. 
73    
74       Return value: 
75       A tuple of total number of tests run, dictionary of error information. The 
76       dictionary is empty if no errors were recorded. 
77    
78       By default individual test modules are run in separate subprocesses. This 
79       recreates normal Pygame usage where pygame.init() and pygame.quit() are 
80       called only once per program execution, and avoids unfortunate 
81       interactions between test modules. Also, a time limit is placed on test 
82       execution, so frozen tests are killed when there time allotment expired. 
83       Use the single process option if threading is not working properly or if 
84       tests are taking too long. It is not guaranteed that all tests will pass 
85       in single process mode. 
86    
87       Tests are run in a randomized order if the randomize argument is True or a 
88       seed argument is provided. If no seed integer is provided then the system 
89       time is used. 
90    
91       Individual test modules may have a corresponding *_tags.py module, 
92       defining a __tags__ attribute, a list of tag strings used to selectively 
93       omit modules from a run. By default only the 'interactive', 'ignore', and 
94       'subprocess_ignore' tags are ignored. 'interactive' is for modules that 
95       take user input, like cdrom_test.py. 'ignore' and 'subprocess_ignore' for 
96       for disabling modules for foreground and subprocess modes respectively. 
97       These are for disabling tests on optional modules or for experimental 
98       modules with known problems. These modules can be run from the console as 
99       a Python program. 
100   
101      This function can only be called once per Python session. It is not 
102      reentrant. 
103   
104      """
105  
106      global was_run
107  
108      if was_run:
109          raise RuntimeError("run() was already called this session")
110      was_run = True
111  
112      options = kwds.copy()
113      option_usesubprocess = options.get("usesubprocess", False)
114      option_dump = options.pop("dump", False)
115      option_file = options.pop("file", None)
116      option_randomize = options.get("randomize", False)
117      option_seed = options.get("seed", None)
118      option_multi_thread = options.pop("multi_thread", 1)
119      option_time_out = options.pop("time_out", 120)
120      option_fake = options.pop("fake", None)
121      option_python = options.pop("python", sys.executable)
122      option_exclude = options.pop("exclude", ())
123      option_interactive = options.pop("interactive", False)
124  
125      if not option_interactive and "interactive" not in option_exclude:
126          option_exclude += ("interactive",)
127      if option_usesubprocess and "subprocess_ignore" not in option_exclude:
128          option_exclude += ("subprocess_ignore",)
129      elif "ignore" not in option_exclude:
130          option_exclude += ("ignore",)
131      if sys.version_info < (3, 0, 0):
132          option_exclude += ("python2_ignore",)
133      else:
134          option_exclude += ("python3_ignore",)
135  
136      if pygame.get_sdl_version() < (2, 0, 0):
137          option_exclude += ("SDL1_ignore",)
138      else:
139          option_exclude += ("SDL2_ignore",)
140      main_dir, test_subdir, fake_test_subdir = prepare_test_env()
141  
142      ###########################################################################
143      # Compile a list of test modules. If fake, then compile list of fake
144      # xxxx_test.py from run_tests__tests
145  
146      TEST_MODULE_RE = re.compile(r"^(.+_test)\.py$")
147  
148      test_mods_pkg_name = test_pkg_name
149  
150      working_dir_temp = tempfile.mkdtemp()
151  
152      if option_fake is not None:
153          test_mods_pkg_name = ".".join(
154              [test_mods_pkg_name, "run_tests__tests", option_fake]
155          )
156          test_subdir = os.path.join(fake_test_subdir, option_fake)
157          working_dir = test_subdir
158      else:
159          working_dir = working_dir_temp
160  
161      # Added in because some machines will need os.environ else there will be
162      # false failures in subprocess mode. Same issue as python2.6. Needs some
163      # env vars.
164  
165      test_env = os.environ
166  
167      fmt1 = "%s.%%s" % test_mods_pkg_name
168      fmt2 = "%s.%%s_test" % test_mods_pkg_name
169      if args:
170          test_modules = [m.endswith("_test") and (fmt1 % m) or (fmt2 % m) for m in args]
171      else:
172          test_modules = []
173          for f in sorted(os.listdir(test_subdir)):
174              for match in TEST_MODULE_RE.findall(f):
175                  test_modules.append(fmt1 % (match,))
176  
177      ###########################################################################
178      # Remove modules to be excluded.
179  
180      tmp = test_modules
181      test_modules = []
182      for name in tmp:
183          tag_module_name = "%s_tags" % (name[0:-5],)
184          try:
185              tag_module = import_submodule(tag_module_name)
186          except ImportError:
187              test_modules.append(name)
188          else:
189              try:
190                  tags = tag_module.__tags__
191              except AttributeError:
192                  print("%s has no tags: ignoring" % (tag_module_name,))
193                  test_modules.append(name)
194              else:
195                  for tag in tags:
196                      if tag in option_exclude:
197                          print("skipping %s (tag '%s')" % (name, tag))
198                          break
199                  else:
200                      test_modules.append(name)
201      del tmp, tag_module_name, name
202  
203      ###########################################################################
204      # Meta results
205  
206      results = {}
207      meta_results = {"__meta__": {}}
208      meta = meta_results["__meta__"]
209  
210      ###########################################################################
211      # Randomization
212  
213      if option_randomize or option_seed is not None:
214          if option_seed is None:
215              option_seed = time.time()
216          meta["random_seed"] = option_seed
217          print("\nRANDOM SEED USED: %s\n" % option_seed)
218          random.seed(option_seed)
219          random.shuffle(test_modules)
220  
221      ###########################################################################
222      # Single process mode
223  
224      if not option_usesubprocess:
225          options["exclude"] = option_exclude
226          t = time.time()
227          for module in test_modules:
228              results.update(run_test(module, **options))
229          t = time.time() - t
230  
231      ###########################################################################
232      # Subprocess mode
233      #
234  
235      else:
236          if is_pygame_pkg:
237              from pygame.tests.test_utils.async_sub import proc_in_time_or_kill
238          else:
239              from test.test_utils.async_sub import proc_in_time_or_kill
240  
241          pass_on_args = ["--exclude", ",".join(option_exclude)]
242          for field in ["randomize", "incomplete", "unbuffered", "verbosity"]:
243              if kwds.get(field, False):
244                  pass_on_args.append("--" + field)
245  
246          def sub_test(module):
247              print("loading %s" % module)
248  
249              cmd = [option_python, "-m", test_runner_mod, module] + pass_on_args
250  
251              return (
252                  module,
253                  (cmd, test_env, working_dir),
254                  proc_in_time_or_kill(
255                      cmd, option_time_out, env=test_env, wd=working_dir
256                  ),
257              )
258  
259          if option_multi_thread > 1:
260  
261              def tmap(f, args):
262                  return pygame.threads.tmap(
263                      f, args, stop_on_error=False, num_workers=option_multi_thread
264                  )
265  
266          else:
267              tmap = map
268  
269          t = time.time()
270  
271          for module, cmd, (return_code, raw_return) in tmap(sub_test, test_modules):
272              test_file = "%s.py" % os.path.join(test_subdir, module)
273              cmd, test_env, working_dir = cmd
274  
275              test_results = get_test_results(raw_return)
276              if test_results:
277                  results.update(test_results)
278              else:
279                  results[module] = {}
280  
281              results[module].update(
282                  dict(
283                      return_code=return_code,
284                      raw_return=raw_return,
285                      cmd=cmd,
286                      test_file=test_file,
287                      test_env=test_env,
288                      working_dir=working_dir,
289                      module=module,
290                  )
291              )
292  
293          t = time.time() - t
294  
295      ###########################################################################
296      # Output Results
297      #
298  
299      untrusty_total, combined = combine_results(results, t)
300      total, n_errors, n_failures = count_results(results)
301  
302      meta["total_tests"] = total
303      meta["combined"] = combined
304      meta["total_errors"] = n_errors
305      meta["total_failures"] = n_failures
306      results.update(meta_results)
307  
308      if not option_usesubprocess and total != untrusty_total:
309          raise AssertionError(
310              "Something went wrong in the Test Machinery:\n"
311              "total: %d != untrusty_total: %d" % (total, untrusty_total)
312          )
313  
314      if not option_dump:
315          print(combined)
316      else:
317          print(TEST_RESULTS_START)
318          print(pformat(results))
319  
320      if option_file is not None:
321          results_file = open(option_file, "w")
322          try:
323              results_file.write(pformat(results))
324          finally:
325              results_file.close()
326  
327      shutil.rmtree(working_dir_temp)
328  
329      return total, n_errors + n_failures

332  def count_results(results):
333      total = errors = failures = 0
334      for result in results.values():
335          if result.get("return_code", 0):
336              total += 1
337              errors += 1
338          else:
339              total += result["num_tests"]
340              errors += result["num_errors"]
341              failures += result["num_failures"]
342  
343      return total, errors, failures

346  def run_and_exit(*args, **kwargs):
347      """Run the tests, and if there are failures, exit with a return code of 1. 
348   
349      This is needed for various buildbots to recognise that the tests have 
350      failed. 
351      """
352      total, fails = run(*args, **kwargs)
353      if fails:
354          sys.exit(1)
355      sys.exit(0)
