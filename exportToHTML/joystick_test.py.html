72       def test_init(self):
73           """ 
74           This unit test is for joystick.init() 
75           It was written to help reduce maintenance costs 
76           and to help test against changes to the code or 
77           different platforms. 
78           """
79           pygame.quit()
80           #test that pygame.init automatically calls joystick.init
81           pygame.init()
82           self.assertEqual(pygame.joystick.get_init(), True)
83   
84           #test that get_count doesn't work w/o joystick init
85           #this is done before and after an init to test
86           #that init activates the joystick functions
87           pygame.joystick.quit()
88           with self.assertRaises(pygame.error):
89               pygame.joystick.get_count()
90   
91           #test explicit call(s) to joystick.init.
92           #Also test that get_count works once init is called
93           iterations = 20
94           for i in range(iterations):
95               pygame.joystick.init()
96           self.assertEqual(pygame.joystick.get_init(), True)
97           self.assertIsNotNone(pygame.joystick.get_count())

122  class JoystickInteractiveTest(unittest.TestCase):
123  
124      __tags__ = ["interactive"]
125  
126      def test_get_count_interactive(self):
127          # Test get_count correctly identifies number of connected joysticks
128          prompt(("Please connect any joysticks/controllers now before starting the "
129                  "joystick.get_count() test."))
130  
131          pygame.joystick.init()
132          # pygame.joystick.get_count(): return count
133          # number of joysticks on the system, 0 means no joysticks connected
134          count = pygame.joystick.get_count()
135  
136          response = question(
137              ("NOTE: Having Steam open may add an extra virtual controller for "
138               "each joystick/controller physically plugged in.\n"
139               "joystick.get_count() thinks there is [{}] joystick(s)/controller(s)"
140               "connected to this system. Is this correct?"
141               .format(count))
142          )
143  
144          self.assertTrue(response)
145  
146          # When you create Joystick objects using Joystick(id), you pass an
147          # integer that must be lower than this count.
148          # Test Joystick(id) for each connected joystick
149          if count != 0:
150              for x in range(count):
151                  pygame.joystick.Joystick(x)
152              with self.assertRaises(pygame.error):
153                  pygame.joystick.Joystick(count)
154  
155          pygame.joystick.quit()
