100  def _unflatten(iterable):
101      for index, elem in enumerate(iterable):
102          if isinstance(elem, Sequence) and not isinstance(elem, str):
103              yield from _unflatten(elem)
104          elif elem and not index:
105              # We're interested only in the first element.
106              yield elem

118  def _is_owner_ignored(owner, attrname, ignored_classes, ignored_modules):
119      """Check if the given owner should be ignored 
120   
121      This will verify if the owner's module is in *ignored_modules* 
122      or the owner's module fully qualified name is in *ignored_modules* 
123      or if the *ignored_modules* contains a pattern which catches 
124      the fully qualified name of the module. 
125   
126      Also, similar checks are done for the owner itself, if its name 
127      matches any name from the *ignored_classes* or if its qualified 
128      name can be found in *ignored_classes*. 
129      """
130      ignored_modules = set(ignored_modules)
131      module_name = owner.root().name
132      module_qname = owner.root().qname()
133  
134      for ignore in ignored_modules:
135          # Try to match the module name / fully qualified name directly
136          if module_qname in ignored_modules or module_name in ignored_modules:
137              return True
138  
139          # Try to see if the ignores pattern match against the module name.
140          if fnmatch.fnmatch(module_qname, ignore):
141              return True
142  
143          # Otherwise we might have a root module name being ignored,
144          # and the qualified owner has more levels of depth.
145          parts = deque(module_name.split("."))
146          current_module = ""
147  
148          while parts:
149              part = parts.popleft()
150              if not current_module:
151                  current_module = part
152              else:
153                  current_module += ".{}".format(part)
154              if current_module in ignored_modules:
155                  return True
156  
157      # Match against ignored classes.
158      ignored_classes = set(ignored_classes)
159      if hasattr(owner, "qname"):
160          qname = owner.qname()
161      else:
162          qname = ""
163      return any(ignore in (attrname, qname) for ignore in ignored_classes)

187  def _string_distance(seq1, seq2):
188      seq2_length = len(seq2)
189  
190      row = list(range(1, seq2_length + 1)) + [0]
191      for seq1_index, seq1_char in enumerate(seq1):
192          last_row = row
193          row = [0] * seq2_length + [seq1_index + 1]
194  
195          for seq2_index, seq2_char in enumerate(seq2):
196              row[seq2_index] = min(
197                  last_row[seq2_index] + 1,
198                  row[seq2_index - 1] + 1,
199                  last_row[seq2_index - 1] + (seq1_char != seq2_char),
200              )
201  
202      return row[seq2_length - 1]

205  def _similar_names(owner, attrname, distance_threshold, max_choices):
206      """Given an owner and a name, try to find similar names 
207   
208      The similar names are searched given a distance metric and only 
209      a given number of choices will be returned. 
210      """
211      possible_names = []
212      names = _node_names(owner)
213  
214      for name in names:
215          if name == attrname:
216              continue
217  
218          distance = _string_distance(attrname, name)
219          if distance <= distance_threshold:
220              possible_names.append((name, distance))
221  
222      # Now get back the values with a minimum, up to the given
223      # limit or choices.
224      picked = [
225          name
226          for (name, _) in heapq.nsmallest(
227              max_choices, possible_names, key=operator.itemgetter(1)
228          )
229      ]
230      return sorted(picked)

233  def _missing_member_hint(owner, attrname, distance_threshold, max_choices):
234      names = _similar_names(owner, attrname, distance_threshold, max_choices)
235      if not names:
236          # No similar name.
237          return ""
238  
239      names = list(map(repr, names))
240      if len(names) == 1:
241          names = ", ".join(names)
242      else:
243          names = "one of {} or {}".format(", ".join(names[:-1]), names[-1])
244  
245      return "; maybe {}?".format(names)

432  def _emit_no_member(node, owner, owner_name, ignored_mixins=True, ignored_none=True):
433      """Try to see if no-member should be emitted for the given owner. 
434   
435      The following cases are ignored: 
436   
437          * the owner is a function and it has decorators. 
438          * the owner is an instance and it has __getattr__, __getattribute__ implemented 
439          * the module is explicitly ignored from no-member checks 
440          * the owner is a class and the name can be found in its metaclass. 
441          * The access node is protected by an except handler, which handles 
442            AttributeError, Exception or bare except. 
443      """
444      # pylint: disable=too-many-return-statements
445      if node_ignores_exception(node, AttributeError):
446          return False
447      if ignored_none and isinstance(owner, astroid.Const) and owner.value is None:
448          return False
449      if is_super(owner) or getattr(owner, "type", None) == "metaclass":
450          return False
451      if owner_name and ignored_mixins and owner_name[-5:].lower() == "mixin":
452          return False
453      if isinstance(owner, astroid.FunctionDef) and owner.decorators:
454          return False
455      if isinstance(owner, (astroid.Instance, astroid.ClassDef)):
456          if owner.has_dynamic_getattr():
457              # Issue #2565: Don't ignore enums, as they have a `__getattr__` but it's not
458              # invoked at this point.
459              try:
460                  metaclass = owner.metaclass()
461              except exceptions.MroError:
462                  return False
463              if metaclass:
464                  return metaclass.qname() == "enum.EnumMeta"
465              return False
466          if not has_known_bases(owner):
467              return False
468  
469          # Exclude typed annotations, since these might actually exist
470          # at some point during the runtime of the program.
471          attribute = owner.locals.get(node.attrname, [None])[0]
472          if (
473              attribute
474              and isinstance(attribute, astroid.AssignName)
475              and isinstance(attribute.parent, astroid.AnnAssign)
476          ):
477              return False
478      if isinstance(owner, objects.Super):
479          # Verify if we are dealing with an invalid Super object.
480          # If it is invalid, then there's no point in checking that
481          # it has the required attribute. Also, don't fail if the
482          # MRO is invalid.
483          try:
484              owner.super_mro()
485          except (exceptions.MroError, exceptions.SuperError):
486              return False
487          if not all(map(has_known_bases, owner.type.mro())):
488              return False
489      if isinstance(owner, astroid.Module):
490          try:
491              owner.getattr("__getattr__")
492              return False
493          except astroid.NotFoundError:
494              pass
495      if owner_name and node.attrname.startswith("_" + owner_name):
496          # Test if an attribute has been mangled ('private' attribute)
497          unmangled_name = node.attrname.split("_" + owner_name)[-1]
498          try:
499              if owner.getattr(unmangled_name, context=None) is not None:
500                  return False
501          except astroid.NotFoundError:
502              return True
503      return True

506  def _determine_callable(callable_obj):
507      # Ordering is important, since BoundMethod is a subclass of UnboundMethod,
508      # and Function inherits Lambda.
509      parameters = 0
510      if hasattr(callable_obj, "implicit_parameters"):
511          parameters = callable_obj.implicit_parameters()
512      if isinstance(callable_obj, astroid.BoundMethod):
513          # Bound methods have an extra implicit 'self' argument.
514          return callable_obj, parameters, callable_obj.type
515      if isinstance(callable_obj, astroid.UnboundMethod):
516          return callable_obj, parameters, "unbound method"
517      if isinstance(callable_obj, astroid.FunctionDef):
518          return callable_obj, parameters, callable_obj.type
519      if isinstance(callable_obj, astroid.Lambda):
520          return callable_obj, parameters, "lambda"
521      if isinstance(callable_obj, astroid.ClassDef):
522          # Class instantiation, lookup __new__ instead.
523          # If we only find object.__new__, we can safely check __init__
524          # instead. If __new__ belongs to builtins, then we look
525          # again for __init__ in the locals, since we won't have
526          # argument information for the builtin __new__ function.
527          try:
528              # Use the last definition of __new__.
529              new = callable_obj.local_attr("__new__")[-1]
530          except exceptions.NotFoundError:
531              new = None
532  
533          from_object = new and new.parent.scope().name == "object"
534          from_builtins = new and new.root().name in sys.builtin_module_names
535  
536          if not new or from_object or from_builtins:
537              try:
538                  # Use the last definition of __init__.
539                  callable_obj = callable_obj.local_attr("__init__")[-1]
540              except exceptions.NotFoundError as e:
541                  # do nothing, covered by no-init.
542                  raise ValueError from e
543          else:
544              callable_obj = new
545  
546          if not isinstance(callable_obj, astroid.FunctionDef):
547              raise ValueError
548          # both have an extra implicit 'cls'/'self' argument.
549          return callable_obj, parameters, "constructor"
550  
551      raise ValueError

592  def _no_context_variadic(node, variadic_name, variadic_type, variadics):
593      """Verify if the given call node has variadic nodes without context 
594   
595      This is a workaround for handling cases of nested call functions 
596      which don't have the specific call context at hand. 
597      Variadic arguments (variable positional arguments and variable 
598      keyword arguments) are inferred, inherently wrong, by astroid 
599      as a Tuple, respectively a Dict with empty elements. 
600      This can lead pylint to believe that a function call receives 
601      too few arguments. 
602      """
603      scope = node.scope()
604      is_in_lambda_scope = not isinstance(scope, astroid.FunctionDef) and isinstance(
605          scope, astroid.Lambda
606      )
607      statement = node.statement()
608      for name in statement.nodes_of_class(astroid.Name):
609          if name.name != variadic_name:
610              continue
611  
612          inferred = safe_infer(name)
613          if isinstance(inferred, (astroid.List, astroid.Tuple)):
614              length = len(inferred.elts)
615          elif isinstance(inferred, astroid.Dict):
616              length = len(inferred.items)
617          else:
618              continue
619  
620          if is_in_lambda_scope and isinstance(inferred.parent, astroid.Arguments):

650  def _infer_from_metaclass_constructor(cls, func):
651      """Try to infer what the given *func* constructor is building 
652   
653      :param astroid.FunctionDef func: 
654          A metaclass constructor. Metaclass definitions can be 
655          functions, which should accept three arguments, the name of 
656          the class, the bases of the class and the attributes. 
657          The function could return anything, but usually it should 
658          be a proper metaclass. 
659      :param astroid.ClassDef cls: 
660          The class for which the *func* parameter should generate 
661          a metaclass. 
662      :returns: 
663          The class generated by the function or None, 
664          if we couldn't infer it. 
665      :rtype: astroid.ClassDef 
666      """
667      context = astroid.context.InferenceContext()
668  
669      class_bases = astroid.List()
670      class_bases.postinit(elts=cls.bases)
671  
672      attrs = astroid.Dict()
673      local_names = [(name, values[-1]) for name, values in cls.locals.items()]
674      attrs.postinit(local_names)
675  
676      builder_args = astroid.Tuple()
677      builder_args.postinit([cls.name, class_bases, attrs])
678  
679      context.callcontext = astroid.context.CallContext(builder_args)
680      try:
681          inferred = next(func.infer_call_result(func, context), None)
682      except astroid.InferenceError:
683          return None
684      return inferred or None

911      @check_messages("no-member", "c-extension-no-member")
912      def visit_attribute(self, node):
913          """check that the accessed attribute exists 
914   
915          to avoid too much false positives for now, we'll consider the code as 
916          correct if a single of the inferred nodes has the accessed attribute. 
917   
918          function/method, super call and metaclasses are ignored 
919          """
920          for pattern in self.config.generated_members:
921              # attribute is marked as generated, stop here
922              if re.match(pattern, node.attrname):
923                  return
924              if re.match(pattern, node.as_string()):
925                  return
926  
927          try:
928              inferred = list(node.expr.infer())
929          except exceptions.InferenceError:
930              return
931  
932          # list of (node, nodename) which are missing the attribute
933          missingattr = set()
934  
935          non_opaque_inference_results = [
936              owner
937              for owner in inferred
938              if owner is not astroid.Uninferable
939              and not isinstance(owner, astroid.nodes.Unknown)
940          ]
941          if (
942              len(non_opaque_inference_results) != len(inferred)
943              and self.config.ignore_on_opaque_inference
944          ):
945              # There is an ambiguity in the inference. Since we can't
946              # make sure that we won't emit a false positive, we just stop
947              # whenever the inference returns an opaque inference object.
948              return
949          for owner in non_opaque_inference_results:
950              name = getattr(owner, "name", None)
951              if _is_owner_ignored(
952                  owner, name, self.config.ignored_classes, self.config.ignored_modules
953              ):
954                  continue
955  
956              try:
957                  if not [
958                      n
959                      for n in owner.getattr(node.attrname)
960                      if not isinstance(n.statement(), astroid.AugAssign)
961                  ]:
962                      missingattr.add((owner, name))
963                      continue
964              except AttributeError:
965                  continue
966              except exceptions.NotFoundError:
967                  # This can't be moved before the actual .getattr call,
968                  # because there can be more values inferred and we are
969                  # stopping after the first one which has the attribute in question.
970                  # The problem is that if the first one has the attribute,
971                  # but we continue to the next values which doesn't have the
972                  # attribute, then we'll have a false positive.
973                  # So call this only after the call has been made.
974                  if not _emit_no_member(
975                      node,
976                      owner,
977                      name,
978                      ignored_mixins=self.config.ignore_mixin_members,
979                      ignored_none=self.config.ignore_none,
980                  ):
981                      continue
982                  missingattr.add((owner, name))
983                  continue
984              # stop on the first found
985              break
986          else:
987              # we have not found any node with the attributes, display the
988              # message for inferred nodes
989              done = set()
990              for owner, name in missingattr:
991                  if isinstance(owner, astroid.Instance):
992                      actual = owner._proxied
993                  else:
994                      actual = owner
995                  if actual in done:
996                      continue
997                  done.add(actual)
998  
999                  msg, hint = self._get_nomember_msgid_hint(node, owner)
1000                 self.add_message(
1001                     msg,
1002                     node=node,
1003                     args=(owner.display_type(), name, node.attrname, hint),
1004                     confidence=INFERENCE,
1005                 )

1041         """check that if assigning to a function call, the function is

1043         """

1045             return

1048         funcs = (astroid.FunctionDef, astroid.UnboundMethod, astroid.BoundMethod)
1049         if not isinstance(function_node, funcs):

1054             function_node._proxied, astroid.UnboundMethod

1062             not function_node.is_function

1113     def _check_uninferable_call(self, node):

1115         Check that the given uninferable Call node does not 
1116         call an actual function.

1118         if not isinstance(node.func, astroid.Attribute):

1122         # the lhs of the Attribute node and search the attribute

1164     def _check_argument_order(self, node, call_site, called, called_param_names):
1165         """Match the supplied argument names against the function parameters. 
1166         Warn if some argument names are not in the same order as they are in 
1167         the function signature. 
1168         """
1169         # Check for called function being an object instance function
1170         # If so, ignore the initial 'self' argument in the signature
1171         try:
1172             is_classdef = isinstance(called.parent, astroid.scoped_nodes.ClassDef)
1173             if is_classdef and called_param_names[0] == "self":
1174                 called_param_names = called_param_names[1:]
1175         except IndexError:
1176             return
1177 
1178         try:
1179             # extract argument names, if they have names
1180             calling_parg_names = [p.name for p in call_site.positional_arguments]
1181 
1182             # Additionally get names of keyword arguments to use in a full match
1183             # against parameters
1184             calling_kwarg_names = [
1185                 arg.name for arg in call_site.keyword_arguments.values()
1186             ]
1187         except AttributeError:
1188             # the type of arg does not provide a `.name`. In this case we
1189             # stop checking for out-of-order arguments because it is only relevant
1190             # for named variables.
1191             return
1192 
1193         # Don't check for ordering if there is an unmatched arg or param
1194         arg_set = set(calling_parg_names) | set(calling_kwarg_names)
1195         param_set = set(called_param_names)
1196         if arg_set != param_set:
1197             return
1198 
1199         # Warn based on the equality of argument ordering
1200         if calling_parg_names != called_param_names[: len(calling_parg_names)]:
1201             self.add_message("arguments-out-of-order", node=node, args=())

1212     # pylint: disable=too-many-branches,too-many-locals
1213     @check_messages(*(list(MSGS.keys())))
1214     def visit_call(self, node):
1215         """check that called functions/methods are inferred to callable objects, 
1216         and that the arguments passed to the function match the parameters in 
1217         the inferred function's definition 
1218         """
1219         called = safe_infer(node.func)
1220         # only function, generator and object defining __call__ are allowed
1221         # Ignore instances of descriptors since astroid cannot properly handle them
1222         # yet
1223         if called and not called.callable():
1224             if isinstance(called, astroid.Instance) and (
1225                 not has_known_bases(called)
1226                 or (
1227                     called.parent is not None
1228                     and isinstance(called.scope(), astroid.ClassDef)
1229                     and "__get__" in called.locals
1230                 )
1231             ):
1232                 # Don't emit if we can't make sure this object is callable.
1233                 pass
1234             else:
1235                 self.add_message("not-callable", node=node, args=node.func.as_string())
1236 
1237         self._check_uninferable_call(node)
1238         try:
1239             called, implicit_args, callable_name = _determine_callable(called)
1240         except ValueError:
1241             # Any error occurred during determining the function type, most of
1242             # those errors are handled by different warnings.
1243             return
1244 
1245         if called.args.args is None:
1246             if called.name == "isinstance":
1247                 # Verify whether second argument of isinstance is a valid type
1248                 self._check_isinstance_args(node)
1249             # Built-in functions have no argument information.
1250             return
1251 
1252         if len(called.argnames()) != len(set(called.argnames())):
1253             # Duplicate parameter name (see duplicate-argument).  We can't really
1254             # make sense of the function call in this case, so just return.
1255             return
1256 
1257         # Build the set of keyword arguments, checking for duplicate keywords,
1258         # and count the positional arguments.
1259         call_site = astroid.arguments.CallSite.from_call(node)
1260 
1261         # Warn about duplicated keyword arguments, such as `f=24, **{'f': 24}`
1262         for keyword in call_site.duplicated_keywords:
1263             self.add_message("repeated-keyword", node=node, args=(keyword,))
1264 
1265         if call_site.has_invalid_arguments() or call_site.has_invalid_keywords():
1266             # Can't make sense of this.
1267             return
1268 
1269         # Has the function signature changed in ways we cannot reliably detect?
1270         if hasattr(called, "decorators") and decorated_with(
1271             called, self.config.signature_mutators
1272         ):
1273             return
1274 
1275         num_positional_args = len(call_site.positional_arguments)
1276         keyword_args = list(call_site.keyword_arguments.keys())
1277         overload_function = is_overload_stub(called)
1278 
1279         # Determine if we don't have a context for our call and we use variadics.
1280         node_scope = node.scope()
1281         if isinstance(node_scope, (astroid.Lambda, astroid.FunctionDef)):
1282             has_no_context_positional_variadic = _no_context_variadic_positional(
1283                 node, node_scope
1284             )
1285             has_no_context_keywords_variadic = _no_context_variadic_keywords(
1286                 node, node_scope
1287             )
1288         else:
1289             has_no_context_positional_variadic = (
1290                 has_no_context_keywords_variadic
1291             ) = False
1292 
1293         # These are coming from the functools.partial implementation in astroid
1294         already_filled_positionals = getattr(called, "filled_positionals", 0)
1295         already_filled_keywords = getattr(called, "filled_keywords", {})
1296 
1297         keyword_args += list(already_filled_keywords)
1298         num_positional_args += implicit_args + already_filled_positionals
1299 
1300         # Analyze the list of formal parameters.
1301         args = list(itertools.chain(called.args.posonlyargs or (), called.args.args))
1302         num_mandatory_parameters = len(args) - len(called.args.defaults)
1303         parameters = []
1304         parameter_name_to_index = {}
1305         for i, arg in enumerate(args):
1306             if isinstance(arg, astroid.Tuple):
1307                 name = None
1308                 # Don't store any parameter names within the tuple, since those
1309                 # are not assignable from keyword arguments.
1310             else:
1311                 assert isinstance(arg, astroid.AssignName)
1312                 # This occurs with:
1313                 #    def f( (a), (b) ): pass
1314                 name = arg.name
1315                 parameter_name_to_index[name] = i
1316             if i >= num_mandatory_parameters:
1317                 defval = called.args.defaults[i - num_mandatory_parameters]
1318             else:
1319                 defval = None
1320             parameters.append([(name, defval), False])
1321 
1322         kwparams = {}
1323         for i, arg in enumerate(called.args.kwonlyargs):
1324             if isinstance(arg, astroid.Keyword):
1325                 name = arg.arg
1326             else:
1327                 assert isinstance(arg, astroid.AssignName)
1328                 name = arg.name
1329             kwparams[name] = [called.args.kw_defaults[i], False]
1330 
1331         self._check_argument_order(
1332             node, call_site, called, [p[0][0] for p in parameters]
1333         )
1334 
1335         # 1. Match the positional arguments.
1336         for i in range(num_positional_args):
1337             if i < len(parameters):
1338                 parameters[i][1] = True
1339             elif called.args.vararg is not None:
1340                 # The remaining positional arguments get assigned to the *args
1341                 # parameter.
1342                 break
1343             elif not overload_function:
1344                 # Too many positional arguments.
1345                 self.add_message(
1346                     "too-many-function-args", node=node, args=(callable_name,)
1347                 )
1348                 break
1349 
1350         # 2. Match the keyword arguments.
1351         for keyword in keyword_args:
1352             if keyword in parameter_name_to_index:
1353                 i = parameter_name_to_index[keyword]
1354                 if parameters[i][1]:
1355                     # Duplicate definition of function parameter.
1356 
1357                     # Might be too hardcoded, but this can actually
1358                     # happen when using str.format and `self` is passed
1359                     # by keyword argument, as in `.format(self=self)`.
1360                     # It's perfectly valid to so, so we're just skipping
1361                     # it if that's the case.
1362                     if not (keyword == "self" and called.qname() in STR_FORMAT):
1363                         self.add_message(
1364                             "redundant-keyword-arg",
1365                             node=node,
1366                             args=(keyword, callable_name),
1367                         )
1368                 else:
1369                     parameters[i][1] = True
1370             elif keyword in kwparams:
1371                 if kwparams[keyword][1]:
1372                     # Duplicate definition of function parameter.
1373                     self.add_message(
1374                         "redundant-keyword-arg",
1375                         node=node,
1376                         args=(keyword, callable_name),
1377                     )
1378                 else:
1379                     kwparams[keyword][1] = True
1380             elif called.args.kwarg is not None:
1381                 # The keyword argument gets assigned to the **kwargs parameter.
1382                 pass
1383             elif not overload_function:
1384                 # Unexpected keyword argument.
1385                 self.add_message(
1386                     "unexpected-keyword-arg", node=node, args=(keyword, callable_name)
1387                 )
1388 
1389         # 3. Match the **kwargs, if any.
1390         if node.kwargs:
1391             for i, [(name, defval), assigned] in enumerate(parameters):
1392                 # Assume that *kwargs provides values for all remaining
1393                 # unassigned named parameters.
1394                 if name is not None:
1395                     parameters[i][1] = True
1396                 else:
1397                     # **kwargs can't assign to tuples.
1398                     pass
1399 
1400         # Check that any parameters without a default have been assigned
1401         # values.
1402         for [(name, defval), assigned] in parameters:
1403             if (defval is None) and not assigned:
1404                 if name is None:
1405                     display_name = "<tuple>"
1406                 else:
1407                     display_name = repr(name)
1408                 if not has_no_context_positional_variadic and not overload_function:
1409                     self.add_message(
1410                         "no-value-for-parameter",
1411                         node=node,
1412                         args=(display_name, callable_name),
1413                     )
1414 
1415         for name in kwparams:
1416             defval, assigned = kwparams[name]
1417             if (
1418                 defval is None
1419                 and not assigned
1420                 and not has_no_context_keywords_variadic
1421                 and not overload_function
1422             ):
1423                 self.add_message("missing-kwoa", node=node, args=(name, callable_name))

1425     def _check_invalid_sequence_index(self, subscript: astroid.Subscript):
1426         # Look for index operations where the parent is a sequence type.
1427         # If the types can be determined, only allow indices to be int,
1428         # slice or instances with __index__.
1429         parent_type = safe_infer(subscript.value)
1430         if not isinstance(
1431             parent_type, (astroid.ClassDef, astroid.Instance)
1432         ) or not has_known_bases(parent_type):
1433             return None
1434 
1435         # Determine what method on the parent this index will use
1436         # The parent of this node will be a Subscript, and the parent of that
1437         # node determines if the Subscript is a get, set, or delete operation.
1438         if subscript.ctx is astroid.Store:
1439             methodname = "__setitem__"
1440         elif subscript.ctx is astroid.Del:
1441             methodname = "__delitem__"
1442         else:
1443             methodname = "__getitem__"
1444 
1445         # Check if this instance's __getitem__, __setitem__, or __delitem__, as
1446         # appropriate to the statement, is implemented in a builtin sequence
1447         # type. This way we catch subclasses of sequence types but skip classes
1448         # that override __getitem__ and which may allow non-integer indices.
1449         try:
1450             methods = dunder_lookup.lookup(parent_type, methodname)
1451             if methods is astroid.Uninferable:
1452                 return None
1453             itemmethod = methods[0]
1454         except (
1455             exceptions.NotFoundError,
1456             exceptions.AttributeInferenceError,
1457             IndexError,
1458         ):
1459             return None
1460 
1461         if (
1462             not isinstance(itemmethod, astroid.FunctionDef)
1463             or itemmethod.root().name != BUILTINS
1464             or not itemmethod.parent
1465             or itemmethod.parent.name not in SEQUENCE_TYPES
1466         ):
1467             return None
1468 
1469         # For ExtSlice objects coming from visit_extslice, no further
1470         # inference is necessary, since if we got this far the ExtSlice
1471         # is an error.
1472         if isinstance(subscript.value, astroid.ExtSlice):
1473             index_type = subscript.value
1474         else:
1475             index_type = safe_infer(subscript.slice)
1476         if index_type is None or index_type is astroid.Uninferable:
1477             return None
1478         # Constants must be of type int
1479         if isinstance(index_type, astroid.Const):
1480             if isinstance(index_type.value, int):
1481                 return None
1482         # Instance values must be int, slice, or have an __index__ method
1483         elif isinstance(index_type, astroid.Instance):
1484             if index_type.pytype() in (BUILTINS + ".int", BUILTINS + ".slice"):
1485                 return None
1486             try:
1487                 index_type.getattr("__index__")
1488                 return None
1489             except exceptions.NotFoundError:
1490                 pass
1491         elif isinstance(index_type, astroid.Slice):
1492             # A slice can be present
1493             # here after inferring the index node, which could
1494             # be a `slice(...)` call for instance.
1495             return self._check_invalid_slice_index(index_type)
1496 
1497         # Anything else is an error
1498         self.add_message("invalid-sequence-index", node=subscript)
1499         return None

1509     def _check_invalid_slice_index(self, node):
1510         # Check the type of each part of the slice
1511         invalid_slices = 0
1512         for index in (node.lower, node.upper, node.step):
1513             if index is None:
1514                 continue
1515 
1516             index_type = safe_infer(index)
1517             if index_type is None or index_type is astroid.Uninferable:
1518                 continue
1519 
1520             # Constants must of type int or None
1521             if isinstance(index_type, astroid.Const):
1522                 if isinstance(index_type.value, (int, type(None))):
1523                     continue
1524             # Instance values must be of type int, None or an object
1525             # with __index__
1526             elif isinstance(index_type, astroid.Instance):
1527                 if index_type.pytype() in (BUILTINS + ".int", BUILTINS + ".NoneType"):
1528                     continue
1529 
1530                 try:
1531                     index_type.getattr("__index__")
1532                     return
1533                 except exceptions.NotFoundError:
1534                     pass
1535             invalid_slices += 1
1536 
1537         if not invalid_slices:
1538             return
1539 
1540         # Anything else is an error, unless the object that is indexed
1541         # is a custom object, which knows how to handle this kind of slices
1542         parent = node.parent
1543         if isinstance(parent, astroid.ExtSlice):
1544             parent = parent.parent
1545         if isinstance(parent, astroid.Subscript):
1546             inferred = safe_infer(parent.value)
1547             if inferred is None or inferred is astroid.Uninferable:
1548                 # Don't know what this is
1549                 return
1550             known_objects = (
1551                 astroid.List,
1552                 astroid.Dict,
1553                 astroid.Tuple,
1554                 astroid.objects.FrozenSet,
1555                 astroid.Set,
1556             )
1557             if not isinstance(inferred, known_objects):
1558                 # Might be an instance that knows how to handle this slice object
1559                 return
1560         for _ in range(invalid_slices):
1561             self.add_message("invalid-slice-index", node=node)

1564     def visit_with(self, node):
1565         for ctx_mgr, _ in node.items:
1566             context = astroid.context.InferenceContext()
1567             inferred = safe_infer(ctx_mgr, context=context)
1568             if inferred is None or inferred is astroid.Uninferable:
1569                 continue
1570 
1571             if isinstance(inferred, bases.Generator):
1572                 # Check if we are dealing with a function decorated
1573                 # with contextlib.contextmanager.
1574                 if decorated_with(
1575                     inferred.parent, self.config.contextmanager_decorators
1576                 ):
1577                     continue
1578                 # If the parent of the generator is not the context manager itself,
1579                 # that means that it could have been returned from another
1580                 # function which was the real context manager.
1581                 # The following approach is more of a hack rather than a real
1582                 # solution: walk all the inferred statements for the
1583                 # given *ctx_mgr* and if you find one function scope
1584                 # which is decorated, consider it to be the real
1585                 # manager and give up, otherwise emit not-context-manager.
1586                 # See the test file for not_context_manager for a couple
1587                 # of self explaining tests.
1588 
1589                 # Retrieve node from all previusly visited nodes in the the inference history
1590                 context_path_names = filter(None, _unflatten(context.path))
1591                 inferred_paths = _flatten_container(
1592                     safe_infer(path) for path in context_path_names
1593                 )
1594                 for inferred_path in inferred_paths:
1595                     if not inferred_path:
1596                         continue
1597                     scope = inferred_path.scope()
1598                     if not isinstance(scope, astroid.FunctionDef):
1599                         continue
1600                     if decorated_with(scope, self.config.contextmanager_decorators):
1601                         break
1602                 else:
1603                     self.add_message(
1604                         "not-context-manager", node=node, args=(inferred.name,)
1605                     )
1606             else:
1607                 try:
1608                     inferred.getattr("__enter__")
1609                     inferred.getattr("__exit__")
1610                 except exceptions.NotFoundError:
1611                     if isinstance(inferred, astroid.Instance):
1612                         # If we do not know the bases of this class,
1613                         # just skip it.
1614                         if not has_known_bases(inferred):
1615                             continue
1616                         # Just ignore mixin classes.
1617                         if self.config.ignore_mixin_members:
1618                             if inferred.name[-5:].lower() == "mixin":
1619                                 continue
1620 
1621                     self.add_message(
1622                         "not-context-manager", node=node, args=(inferred.name,)
1623                     )
