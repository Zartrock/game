43   def input_main(device_id=None):
44       pg.init()
45       pg.fastevent.init()
46       event_get = pg.fastevent.get
47       event_post = pg.fastevent.post
48   
49       pygame.midi.init()
50   
51       _print_device_info()
52   
53       if device_id is None:
54           input_id = pygame.midi.get_default_input_id()
55       else:
56           input_id = device_id
57   
58       print("using input_id :%s:" % input_id)
59       i = pygame.midi.Input(input_id)
60   
61       pg.display.set_mode((1, 1))
62   
63       going = True
64       while going:
65           events = event_get()
66           for e in events:
67               if e.type in [pg.QUIT]:
68                   going = False
69               if e.type in [pg.KEYDOWN]:
70                   going = False
71               if e.type in [pygame.midi.MIDIIN]:
72                   print(e)
73   
74           if i.poll():
75               midi_events = i.read(10)
76               # convert them into pygame events.
77               midi_evs = pygame.midi.midis2events(midi_events, i.device_id)
78   
79               for m_e in midi_evs:
80                   event_post(m_e)
81   
82       del i
83       pygame.midi.quit()

86   def output_main(device_id=None):
87       """Execute a musical keyboard example for the Church Organ instrument 
88    
89       This is a piano keyboard example, with a two octave keyboard, starting at 
90       note F3. Left mouse down over a key starts a note, left up stops it. The 
91       notes are also mapped to the computer keyboard keys, assuming an American 
92       English PC keyboard (sorry everyone else, but I don't know if I can map to 
93       absolute key position instead of value.) The white keys are on the second 
94       row, TAB to BACKSLASH, starting with note F3. The black keys map to the top 
95       row, '1' to BACKSPACE, starting with F#3. 'r' is middle C. Close the 
96       window or press ESCAPE to quit the program. Key velocity (note 
97       amplitude) varies vertically on the keyboard image, with minimum velocity 
98       at the top of a key and maximum velocity at bottom. 
99    
100      Default Midi output, no device_id given, is to the default output device 
101      for the computer. 
102   
103      """
104  
105      # A note to new pygamers:
106      #
107      # All the midi module stuff is in this function. It is unnecessary to
108      # understand how the keyboard display works to appreciate how midi
109      # messages are sent.
110  
111      # The keyboard is drawn by a Keyboard instance. This instance maps Midi
112      # notes to musical keyboard keys. A regions surface maps window position
113      # to (Midi note, velocity) pairs. A key_mapping dictionary does the same
114      # for computer keyboard keys. Midi sound is controlled with direct method
115      # calls to a pygame.midi.Output instance.
116      #
117      # Things to consider when using pygame.midi:
118      #
119      # 1) Initialize the midi module with a to pygame.midi.init().
120      # 2) Create a midi.Output instance for the desired output device port.
121      # 3) Select instruments with set_instrument() method calls.
122      # 4) Play notes with note_on() and note_off() method calls.
123      # 5) Call pygame.midi.Quit() when finished. Though the midi module tries
124      #    to ensure that midi is properly shut down, it is best to do it
125      #    explicitly. A try/finally statement is the safest way to do this.
126      #
127  
128      # GRAND_PIANO = 0
129      CHURCH_ORGAN = 19
130  
131      instrument = CHURCH_ORGAN
132      # instrument = GRAND_PIANO
133      start_note = 53  # F3 (white key note), start_note != 0
134      n_notes = 24  # Two octaves (14 white keys)
135  
136      bg_color = pg.Color("slategray")
137  
138      key_mapping = make_key_mapping(
139          [
140              pg.K_TAB,
141              pg.K_1,
142              pg.K_q,
143              pg.K_2,
144              pg.K_w,
145              pg.K_3,
146              pg.K_e,
147              pg.K_r,
148              pg.K_5,
149              pg.K_t,
150              pg.K_6,
151              pg.K_y,
152              pg.K_u,
153              pg.K_8,
154              pg.K_i,
155              pg.K_9,
156              pg.K_o,
157              pg.K_0,
158              pg.K_p,
159              pg.K_LEFTBRACKET,
160              pg.K_EQUALS,
161              pg.K_RIGHTBRACKET,
162              pg.K_BACKSPACE,
163              pg.K_BACKSLASH,
164          ],
165          start_note,
166      )
167  
168      pg.init()
169      pygame.midi.init()
170  
171      _print_device_info()
172  
173      if device_id is None:
174          port = pygame.midi.get_default_output_id()
175      else:
176          port = device_id
177  
178      print("using output_id :%s:" % port)
179  
180      midi_out = pygame.midi.Output(port, 0)
181      try:
182          midi_out.set_instrument(instrument)
183          keyboard = Keyboard(start_note, n_notes)
184  
185          screen = pg.display.set_mode(keyboard.rect.size)
186          screen.fill(bg_color)
187          pg.display.flip()
188  
189          background = pg.Surface(screen.get_size())
190          background.fill(bg_color)
191          dirty_rects = []
192          keyboard.draw(screen, background, dirty_rects)
193          pg.display.update(dirty_rects)
194  
195          regions = pg.Surface(screen.get_size())  # initial color (0,0,0)
196          keyboard.map_regions(regions)
197  
198          pg.event.set_blocked(pg.MOUSEMOTION)
199          mouse_note = 0
200          on_notes = set()
201          while 1:
202              e = pg.event.wait()
203              if e.type == pg.MOUSEBUTTONDOWN:
204                  mouse_note, velocity, __, __ = regions.get_at(e.pos)
205                  if mouse_note and mouse_note not in on_notes:
206                      keyboard.key_down(mouse_note)
207                      midi_out.note_on(mouse_note, velocity)
208                      on_notes.add(mouse_note)
209                  else:
210                      mouse_note = 0
211              elif e.type == pg.MOUSEBUTTONUP:
212                  if mouse_note:
213                      midi_out.note_off(mouse_note)
214                      keyboard.key_up(mouse_note)
215                      on_notes.remove(mouse_note)
216                      mouse_note = 0
217              elif e.type == pg.QUIT:
218                  break
219              elif e.type == pg.KEYDOWN:
220                  if e.key == pg.K_ESCAPE:
221                      break
222                  try:
223                      note, velocity = key_mapping[e.key]
224                  except KeyError:
225                      pass
226                  else:
227                      if note not in on_notes:
228                          keyboard.key_down(note)
229                          midi_out.note_on(note, velocity)
230                          on_notes.add(note)
231              elif e.type == pg.KEYUP:
232                  try:
233                      note, __ = key_mapping[e.key]
234                  except KeyError:
235                      pass
236                  else:
237                      if note in on_notes and note != mouse_note:
238                          keyboard.key_up(note)
239                          midi_out.note_off(note, 0)
240                          on_notes.remove(note)
241  
242              dirty_rects = []
243              keyboard.draw(screen, background, dirty_rects)
244              pg.display.update(dirty_rects)
245      finally:
246          del midi_out
247          pygame.midi.quit()

250  def make_key_mapping(key_list, start_note):
251      """Return a dictionary of (note, velocity) by computer keyboard key code"""
252  
253      mapping = {}
254      for i in range(len(key_list)):
255          mapping[key_list[i]] = (start_note + i, 127)
256      return mapping

283  def key_class(updates, image_strip, image_rects, is_white_key=True):
284      """Return a keyboard key widget class 
285   
286      Arguments: 
287      updates - a set into which a key instance adds itself if it needs 
288          redrawing. 
289      image_strip - The surface containing the images of all key states. 
290      image_rects - A list of Rects giving the regions within image_strip that 
291          are relevant to this key class. 
292      is_white_key (default True) - Set false if this is a black key. 
293   
294      This function automates the creation of a key widget class for the 
295      three basic key types. A key has two basic states, up or down ( 
296      depressed). Corresponding up and down images are drawn for each 
297      of these two states. But to give the illusion of depth, a key 
298      may have shadows cast upon it by the adjacent keys to its right. 
299      These shadows change depending on the up/down state of the key and 
300      its neighbors. So a key may support multiple images and states 
301      depending on the shadows. A key type is determined by the length 
302      of image_rects and the value of is_white. 
303   
304      """
305  
306      # Naming convention: Variables used by the Key class as part of a
307      # closure start with 'c_'.
308  
309      # State logic and shadows:
310      #
311      # A key may cast a shadow upon the key to its left. A black key casts a
312      # shadow on an adjacent white key. The shadow changes depending of whether
313      # the black or white key is depressed. A white key casts a shadow on the
314      # white key to its left if it is up and the left key is down. Therefore
315      # a keys state, and image it will draw, is determined entirely by its
316      # itself and the key immediately adjacent to it on the right. A white key
317      # is always assumed to have an adjacent white key.
318      #
319      # There can be up to eight key states, representing all permutations
320      # of the three fundamental states of self up/down, adjacent white
321      # right up/down, adjacent black up/down.
322      #
323      down_state_none = 0
324      down_state_self = 1
325      down_state_white = down_state_self << 1
326      down_state_self_white = down_state_self | down_state_white
327      down_state_black = down_state_white << 1
328      down_state_self_black = down_state_self | down_state_black
329      down_state_white_black = down_state_white | down_state_black
330      down_state_all = down_state_self | down_state_white_black
331  
332      # Some values used in the class.
333      #
334      c_down_state_initial = down_state_none
335      c_down_state_rect_initial = image_rects[0]
336      c_updates = updates
337      c_image_strip = image_strip
338      c_width, c_height = image_rects[0].size
339  
340      # A key propagates its up/down state change to the adjacent white key on
341      # the left by calling the adjacent key's _right_black_down or
342      # _right_white_down method.
343      #
344      if is_white_key:
345          key_color = "white"
346      else:
347          key_color = "black"
348      c_notify_down_method = "_right_%s_down" % key_color
349      c_notify_up_method = "_right_%s_up" % key_color
350  
351      # Images:
352      #
353      # A black key only needs two images, for the up and down states. Its
354      # appearance is unaffected by the adjacent keys to its right, which cast no
355      # shadows upon it.
356      #
357      # A white key with a no adjacent black to its right only needs three
358      # images, for self up, self down, and both self and adjacent white down.
359      #
360      # A white key with both a black and white key to its right needs six
361      # images: self up, self up and adjacent black down, self down, self and
362      # adjacent white down, self and adjacent black down, and all three down.
363      #
364      # Each 'c_event' dictionary maps the current key state to a new key state,
365      # along with corresponding image, for the related event. If no redrawing
366      # is required for the state change then the image rect is simply None.
367      #
368      c_event_down = {down_state_none: (down_state_self, image_rects[1])}
369      c_event_up = {down_state_self: (down_state_none, image_rects[0])}
370      c_event_right_white_down = {
371          down_state_none: (down_state_none, None),
372          down_state_self: (down_state_self, None),
373      }
374      c_event_right_white_up = c_event_right_white_down.copy()
375      c_event_right_black_down = c_event_right_white_down.copy()
376      c_event_right_black_up = c_event_right_white_down.copy()
377      if len(image_rects) > 2:
378          c_event_down[down_state_white] = (down_state_self_white, image_rects[2])
379          c_event_up[down_state_self_white] = (down_state_white, image_rects[0])
380          c_event_right_white_down[down_state_none] = (down_state_white, None)
381          c_event_right_white_down[down_state_self] = (
382              down_state_self_white,
383              image_rects[2],
384          )
385          c_event_right_white_up[down_state_white] = (down_state_none, None)
386          c_event_right_white_up[down_state_self_white] = (
387              down_state_self,
388              image_rects[1],
389          )
390          c_event_right_black_down[down_state_white] = (down_state_white, None)
391          c_event_right_black_down[down_state_self_white] = (down_state_self_white, None)
392          c_event_right_black_up[down_state_white] = (down_state_white, None)
393          c_event_right_black_up[down_state_self_white] = (down_state_self_white, None)
394      if len(image_rects) > 3:
395          c_event_down[down_state_black] = (down_state_self_black, image_rects[4])
396          c_event_down[down_state_white_black] = (down_state_all, image_rects[5])
397          c_event_up[down_state_self_black] = (down_state_black, image_rects[3])
398          c_event_up[down_state_all] = (down_state_white_black, image_rects[3])
399          c_event_right_white_down[down_state_black] = (down_state_white_black, None)
400          c_event_right_white_down[down_state_self_black] = (
401              down_state_all,
402              image_rects[5],
403          )
404          c_event_right_white_up[down_state_white_black] = (down_state_black, None)
405          c_event_right_white_up[down_state_all] = (down_state_self_black, image_rects[4])
406          c_event_right_black_down[down_state_none] = (down_state_black, image_rects[3])
407          c_event_right_black_down[down_state_self] = (
408              down_state_self_black,
409              image_rects[4],
410          )
411          c_event_right_black_down[down_state_white] = (
412              down_state_white_black,
413              image_rects[3],
414          )
415          c_event_right_black_down[down_state_self_white] = (
416              down_state_all,
417              image_rects[5],
418          )
419          c_event_right_black_up[down_state_black] = (down_state_none, image_rects[0])
420          c_event_right_black_up[down_state_self_black] = (
421              down_state_self,
422              image_rects[1],
423          )
424          c_event_right_black_up[down_state_white_black] = (
425              down_state_white,
426              image_rects[0],
427          )
428          c_event_right_black_up[down_state_all] = (down_state_self_white, image_rects[2])

430      class Key(object):
431          """A key widget, maintains key state and draws the key's image 
432   
433          Constructor arguments: 
434          ident - A unique key identifier. Any immutable type suitable as a key. 
435          posn - The location of the key on the display surface. 
436          key_left - Optional, the adjacent white key to the left. Changes in 
437              up and down state are propagated to that key. 
438   
439          A key has an associated position and state. Related to state is the 
440          image drawn. State changes are managed with method calls, one method 
441          per event type. The up and down event methods are public. Other 
442          internal methods are for passing on state changes to the key_left 
443          key instance. 
444   
445          """
446  
447          is_white = is_white_key
448  
449          def __init__(self, ident, posn, key_left=None):
450              """Return a new Key instance 
451   
452              The initial state is up, with all adjacent keys to the right also 
453              up. 
454   
455              """
456              if key_left is None:
457                  key_left = null_key
458              rect = pg.Rect(posn[0], posn[1], c_width, c_height)
459              self.rect = rect
460              self._state = c_down_state_initial
461              self._source_rect = c_down_state_rect_initial
462              self._ident = ident
463              self._hash = hash(ident)
464              self._notify_down = getattr(key_left, c_notify_down_method)
465              self._notify_up = getattr(key_left, c_notify_up_method)
466              self._key_left = key_left
467              self._background_rect = pg.Rect(rect.left, rect.bottom - 10, c_width, 10)
468              c_updates.add(self)
469  
470          def down(self):
471              """Signal that this key has been depressed (is down)"""
472  
473              self._state, source_rect = c_event_down[self._state]
474              if source_rect is not None:
475                  self._source_rect = source_rect
476                  c_updates.add(self)
477                  self._notify_down()
478  
479          def up(self):
480              """Signal that this key has been released (is up)"""
481  
482              self._state, source_rect = c_event_up[self._state]
483              if source_rect is not None:
484                  self._source_rect = source_rect
485                  c_updates.add(self)
486                  self._notify_up()
487  
488          def _right_white_down(self):
489              """Signal that the adjacent white key has been depressed 
490   
491              This method is for internal propagation of events between 
492              key instances. 
493   
494              """
495  
496              self._state, source_rect = c_event_right_white_down[self._state]
497              if source_rect is not None:
498                  self._source_rect = source_rect
499                  c_updates.add(self)
500  
501          def _right_white_up(self):
502              """Signal that the adjacent white key has been released 
503   
504              This method is for internal propagation of events between 
505              key instances. 
506   
507              """
508  
509              self._state, source_rect = c_event_right_white_up[self._state]
510              if source_rect is not None:
511                  self._source_rect = source_rect
512                  c_updates.add(self)
513  
514          def _right_black_down(self):
515              """Signal that the adjacent black key has been depressed 
516   
517              This method is for internal propagation of events between 
518              key instances. 
519   
520              """
521  
522              self._state, source_rect = c_event_right_black_down[self._state]
523              if source_rect is not None:
524                  self._source_rect = source_rect
525                  c_updates.add(self)
526  
527          def _right_black_up(self):
528              """Signal that the adjacent black key has been released 
529   
530              This method is for internal propagation of events between 
531              key instances. 
532   
533              """
534  
535              self._state, source_rect = c_event_right_black_up[self._state]
536              if source_rect is not None:
537                  self._source_rect = source_rect
538                  c_updates.add(self)
539  
540          def __eq__(self, other):
541              """True if same identifiers"""
542  
543              return self._ident == other._ident
544  
545          def __hash__(self):
546              """Return the immutable hash value"""
547  
548              return self._hash
549  
550          def __str__(self):
551              """Return the key's identifier and position as a string"""
552  
553              return "<Key %s at (%d, %d)>" % (self._ident, self.rect.top, self.rect.left)
554  
555          def draw(self, surf, background, dirty_rects):
556              """Redraw the key on the surface surf 
557   
558              The background is redrawn. The altered region is added to the 
559              dirty_rects list. 
560   
561              """
562  
563              surf.blit(background, self._background_rect, self._background_rect)
564              surf.blit(c_image_strip, self.rect, self._source_rect)
565              dirty_rects.append(self.rect)
566  
567      return Key

570  def key_images():
571      """Return a keyboard keys image strip and a mapping of image locations 
572   
573      The return tuple is a surface and a dictionary of rects mapped to key 
574      type. 
575   
576      This function encapsulates the constants relevant to the keyboard image 
577      file. There are five key types. One is the black key. The other four 
578      white keys are determined by the proximity of the black keys. The plain 
579      white key has no black key adjacent to it. A white-left and white-right 
580      key has a black key to the left or right of it respectively. A white-center 
581      key has a black key on both sides. A key may have up to six related 
582      images depending on the state of adjacent keys to its right. 
583   
584      """
585  
586      my_dir = os.path.split(os.path.abspath(__file__))[0]
587      strip_file = os.path.join(my_dir, "data", "midikeys.png")
588      white_key_width = 42
589      white_key_height = 160
590      black_key_width = 22
591      black_key_height = 94
592      strip = pg.image.load(strip_file)
593      names = [
594          "black none",
595          "black self",
596          "white none",
597          "white self",
598          "white self-white",
599          "white-left none",
600          "white-left self",
601          "white-left black",
602          "white-left self-black",
603          "white-left self-white",
604          "white-left all",
605          "white-center none",
606          "white-center self",
607          "white-center black",
608          "white-center self-black",
609          "white-center self-white",
610          "white-center all",
611          "white-right none",
612          "white-right self",
613          "white-right self-white",
614      ]
615      rects = {}
616      for i in range(2):
617          rects[names[i]] = pg.Rect(
618              i * white_key_width, 0, black_key_width, black_key_height
619          )
620      for i in range(2, len(names)):
621          rects[names[i]] = pg.Rect(
622              i * white_key_width, 0, white_key_width, white_key_height
623          )
624      return strip, rects

627  class Keyboard(object):
628      """Musical keyboard widget 
629   
630      Constructor arguments: 
631      start_note: midi note value of the starting note on the keyboard. 
632      n_notes: number of notes (keys) on the keyboard. 
633   
634      A Keyboard instance draws the musical keyboard and maintains the state of 
635      all the keyboard keys. Individual keys can be in a down (depressed) or 
636      up (released) state. 
637   
638      """
639  
640      _image_strip, _rects = key_images()
641  
642      white_key_width, white_key_height = _rects["white none"].size
643      black_key_width, black_key_height = _rects["black none"].size
644  
645      _updates = set()
646  
647      # There are five key classes, representing key shape:
648      # black key (BlackKey), plain white key (WhiteKey), white key to the left
649      # of a black key (WhiteKeyLeft), white key between two black keys
650      # (WhiteKeyCenter), and white key to the right of a black key
651      # (WhiteKeyRight).
652      BlackKey = key_class(
653          _updates, _image_strip, [_rects["black none"], _rects["black self"]], False
654      )
655      WhiteKey = key_class(
656          _updates,
657          _image_strip,
658          [_rects["white none"], _rects["white self"], _rects["white self-white"]],
659      )
660      WhiteKeyLeft = key_class(
661          _updates,
662          _image_strip,
663          [
664              _rects["white-left none"],
665              _rects["white-left self"],
666              _rects["white-left self-white"],
667              _rects["white-left black"],
668              _rects["white-left self-black"],
669              _rects["white-left all"],
670          ],
671      )
672      WhiteKeyCenter = key_class(
673          _updates,
674          _image_strip,
675          [
676              _rects["white-center none"],
677              _rects["white-center self"],
678              _rects["white-center self-white"],
679              _rects["white-center black"],
680              _rects["white-center self-black"],
681              _rects["white-center all"],
682          ],
683      )
684      WhiteKeyRight = key_class(
685          _updates,
686          _image_strip,
687          [
688              _rects["white-right none"],
689              _rects["white-right self"],
690              _rects["white-right self-white"],
691          ],
692      )

694      def __init__(self, start_note, n_notes):
695          """Return a new Keyboard instance with n_note keys"""
696  
697          self._start_note = start_note
698          self._end_note = start_note + n_notes - 1
699          self._add_keys()
700  
701      def _add_keys(self):
702          """Populate the keyboard with key instances 
703   
704          Set the _keys and rect attributes. 
705   
706          """
707  
708          # Keys are entered in a list, where index is Midi note. Since there are
709          # only 128 possible Midi notes the list length is managable. Unassigned
710          # note positions should never be accessed, so are set None to ensure
711          # the bug is quickly detected.
712          #
713          key_map = [None] * 128
714  
715          start_note = self._start_note
716          end_note = self._end_note
717          black_offset = self.black_key_width // 2
718          prev_white_key = None
719          x = y = 0
720          if is_white_key(start_note):
721              is_prev_white = True
722          else:
723              x += black_offset
724              is_prev_white = False
725          for note in range(start_note, end_note + 1):
726              ident = note  # For now notes uniquely identify keyboard keys.
727              if is_white_key(note):
728                  if is_prev_white:
729                      if note == end_note or is_white_key(note + 1):
730                          key = self.WhiteKey(ident, (x, y), prev_white_key)
731                      else:
732                          key = self.WhiteKeyLeft(ident, (x, y), prev_white_key)
733                  else:
734                      if note == end_note or is_white_key(note + 1):
735                          key = self.WhiteKeyRight(ident, (x, y), prev_white_key)
736                      else:
737                          key = self.WhiteKeyCenter(ident, (x, y), prev_white_key)
738                  is_prev_white = True
739                  x += self.white_key_width
740                  prev_white_key = key
741              else:
742                  key = self.BlackKey(ident, (x - black_offset, y), prev_white_key)
743                  is_prev_white = False
744              key_map[note] = key
745          self._keys = key_map
746  
747          kb_width = key_map[self._end_note].rect.right
748          kb_height = self.white_key_height
749          self.rect = pg.Rect(0, 0, kb_width, kb_height)
750  
751      def map_regions(self, regions):
752          """Draw the key regions onto surface regions. 
753   
754          Regions must have at least 3 byte pixels. Each pixel of the keyboard 
755          rectangle is set to the color (note, velocity, 0). The regions surface 
756          must be at least as large as (0, 0, self.rect.left, self.rect.bottom) 
757   
758          """
759  
760          # First draw the white key regions. Then add the overlapping
761          # black key regions.
762          #
763          cutoff = self.black_key_height
764          black_keys = []
765          for note in range(self._start_note, self._end_note + 1):
766              key = self._keys[note]
767              if key.is_white:
768                  fill_region(regions, note, key.rect, cutoff)
769              else:
770                  black_keys.append((note, key))
771          for note, key in black_keys:
772              fill_region(regions, note, key.rect, cutoff)
773  
774      def draw(self, surf, background, dirty_rects):
775          """Redraw all altered keyboard keys"""
776  
777          changed_keys = self._updates
778          while changed_keys:
779              changed_keys.pop().draw(surf, background, dirty_rects)

781      def key_down(self, note):
782          """Signal a key down event for note"""
783  
784          self._keys[note].down()
785  
786      def key_up(self, note):
787          """Signal a key up event for note"""
788  
789          self._keys[note].up()

792  def fill_region(regions, note, rect, cutoff):
793      """Fill the region defined by rect with a (note, velocity, 0) color 
794   
795      The velocity varies from a small value at the top of the region to 
796      127 at the bottom. The vertical region 0 to cutoff is split into 
797      three parts, with velocities 42, 84 and 127. Everything below cutoff 
798      has velocity 127. 
799   
800      """
801  
802      x, y, width, height = rect
803      if cutoff is None:
804          cutoff = height
805      delta_height = cutoff // 3
806      regions.fill((note, 42, 0), (x, y, width, delta_height))
807      regions.fill((note, 84, 0), (x, y + delta_height, width, delta_height))
808      regions.fill(
809          (note, 127, 0), (x, y + 2 * delta_height, width, height - 2 * delta_height)
810      )

813  def is_white_key(note):
814      """True if note is represented by a white key"""
815  
816      key_pattern = [
817          True,
818          False,
819          True,
820          True,
821          False,
822          True,
823          False,
824          True,
825          True,
826          False,
827          True,
828          False,
829      ]
830      return key_pattern[(note - 21) % len(key_pattern)]
