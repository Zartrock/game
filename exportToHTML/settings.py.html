206      def __post_init__(self):
207          py_version = self.py_version
208          if py_version == "auto":  # pragma: no cover
209              if sys.version_info.major == 2 and sys.version_info.minor <= 6:
210                  py_version = "2"
211              elif sys.version_info.major == 3 and (
212                  sys.version_info.minor <= 5 or sys.version_info.minor >= 9
213              ):
214                  py_version = "3"
215              else:
216                  py_version = f"{sys.version_info.major}{sys.version_info.minor}"
217  
218          if py_version not in VALID_PY_TARGETS:
219              raise ValueError(
220                  f"The python version {py_version} is not supported. "
221                  "You can set a python version with the -py or --python-version flag. "
222                  f"The following versions are supported: {VALID_PY_TARGETS}"
223              )
224  
225          if py_version != "all":
226              object.__setattr__(self, "py_version", f"py{py_version}")
227  
228          if not self.known_standard_library:
229              object.__setattr__(
230                  self, "known_standard_library", frozenset(getattr(stdlibs, self.py_version).stdlib)
231              )
232  
233          if self.force_alphabetical_sort:
234              object.__setattr__(self, "force_alphabetical_sort_within_sections", True)
235              object.__setattr__(self, "no_sections", True)
236              object.__setattr__(self, "lines_between_types", 1)
237              object.__setattr__(self, "from_first", True)
238          if self.wrap_length > self.line_length:
239              raise ValueError(
240                  "wrap_length must be set lower than or equal to line_length: "
241                  f"{self.wrap_length} > {self.line_length}."
242              )

251  class Config(_Config):
252      def __init__(
253          self,
254          settings_file: str = "",
255          settings_path: str = "",
256          config: Optional[_Config] = None,
257          **config_overrides,
258      ):
259          self._known_patterns: Optional[List[Tuple[Pattern[str], str]]] = None
260          self._section_comments: Optional[Tuple[str, ...]] = None
261  
262          if config:
263              config_vars = vars(config).copy()
264              config_vars.update(config_overrides)
265              config_vars["py_version"] = config_vars["py_version"].replace("py", "")
266              config_vars.pop("_known_patterns")
267              config_vars.pop("_section_comments")
268              super().__init__(**config_vars)  # type: ignore
269              return
270  
271          # We can't use self.quiet to conditionally show warnings before super.__init__() is called
272          # at the end of this method. _Config is also frozen so setting self.quiet isn't possible.
273          # Therefore we extract quiet early here in a variable and use that in warning conditions.
274          quiet = config_overrides.get("quiet", False)
275  
276          sources: List[Dict[str, Any]] = [_DEFAULT_SETTINGS]
277  
278          config_settings: Dict[str, Any]
279          project_root: str
280          if settings_file:
281              config_settings = _get_config_data(
282                  settings_file,
283                  CONFIG_SECTIONS.get(os.path.basename(settings_file), FALLBACK_CONFIG_SECTIONS),
284              )
285              project_root = os.path.dirname(settings_file)
286              if not config_settings and not quiet:
287                  warn(
288                      f"A custom settings file was specified: {settings_file} but no configuration "
289                      "was found inside. This can happen when [settings] is used as the config "
290                      "header instead of [isort]. "
291                      "See: https://pycqa.github.io/isort/docs/configuration/config_files"
292                      "/#custom_config_files for more information."
293                  )
294          elif settings_path:
295              if not os.path.exists(settings_path):
296                  raise InvalidSettingsPath(settings_path)
297  
298              settings_path = os.path.abspath(settings_path)
299              project_root, config_settings = _find_config(settings_path)
300          else:
301              config_settings = {}
302              project_root = os.getcwd()
303  
304          profile_name = config_overrides.get("profile", config_settings.get("profile", ""))
305          profile: Dict[str, Any] = {}
306          if profile_name:
307              if profile_name not in profiles:
308                  import pkg_resources
309  
310                  for plugin in pkg_resources.iter_entry_points("isort.profiles"):
311                      profiles.setdefault(plugin.name, plugin.load())
312  
313              if profile_name not in profiles:
314                  raise ProfileDoesNotExist(profile_name)
315  
316              profile = profiles[profile_name].copy()
317              profile["source"] = f"{profile_name} profile"
318              sources.append(profile)
319  
320          if config_settings:
321              sources.append(config_settings)
322          if config_overrides:
323              config_overrides["source"] = RUNTIME_SOURCE
324              sources.append(config_overrides)
325  
326          combined_config = {**profile, **config_settings, **config_overrides}
327          if "indent" in combined_config:
328              indent = str(combined_config["indent"])
329              if indent.isdigit():
330                  indent = " " * int(indent)
331              else:
332                  indent = indent.strip("'").strip('"')
333                  if indent.lower() == "tab":
334                      indent = "\t"
335              combined_config["indent"] = indent
336  
337          known_other = {}
338          import_headings = {}
339          for key, value in tuple(combined_config.items()):
340              # Collect all known sections beyond those that have direct entries
341              if key.startswith(KNOWN_PREFIX) and key not in (
342                  "known_standard_library",
343                  "known_future_library",
344                  "known_third_party",
345                  "known_first_party",
346                  "known_local_folder",
347              ):
348                  import_heading = key[len(KNOWN_PREFIX) :].lower()
349                  maps_to_section = import_heading.upper()
350                  combined_config.pop(key)
351                  if maps_to_section in KNOWN_SECTION_MAPPING:
352                      section_name = f"known_{KNOWN_SECTION_MAPPING[maps_to_section].lower()}"
353                      if section_name in combined_config and not quiet:
354                          warn(
355                              f"Can't set both {key} and {section_name} in the same config file.\n"
356                              f"Default to {section_name} if unsure."
357                              "\n\n"
358                              "See: https://pycqa.github.io/isort/"
359                              "#custom-sections-and-ordering."
360                          )
361                      else:
362                          combined_config[section_name] = frozenset(value)
363                  else:
364                      known_other[import_heading] = frozenset(value)
365                      if maps_to_section not in combined_config.get("sections", ()) and not quiet:
366                          warn(
367                              f"`{key}` setting is defined, but {maps_to_section} is not"
368                              " included in `sections` config option:"
369                              f" {combined_config.get('sections', SECTION_DEFAULTS)}.\n\n"
370                              "See: https://pycqa.github.io/isort/"
371                              "#custom-sections-and-ordering."
372                          )
373              if key.startswith(IMPORT_HEADING_PREFIX):
374                  import_headings[key[len(IMPORT_HEADING_PREFIX) :].lower()] = str(value)
375  
376              # Coerce all provided config values into their correct type
377              default_value = _DEFAULT_SETTINGS.get(key, None)
378              if default_value is None:
379                  continue
380  
381              combined_config[key] = type(default_value)(value)
382  
383          for section in combined_config.get("sections", ()):
384              if section in SECTION_DEFAULTS:
385                  continue
386              elif not section.lower() in known_other:
387                  config_keys = ", ".join(known_other.keys())
388                  warn(
389                      f"`sections` setting includes {section}, but no known_{section.lower()} "
390                      "is defined. "
391                      f"The following known_SECTION config options are defined: {config_keys}."
392                  )
393  
394          if "directory" not in combined_config:
395              combined_config["directory"] = (
396                  os.path.dirname(config_settings["source"])
397                  if config_settings.get("source", None)
398                  else os.getcwd()
399              )
400  
401          path_root = Path(combined_config.get("directory", project_root)).resolve()
402          path_root = path_root if path_root.is_dir() else path_root.parent
403          if "src_paths" not in combined_config:
404              combined_config["src_paths"] = (path_root / "src", path_root)
405          else:
406              src_paths: List[Path] = []
407              for src_path in combined_config.get("src_paths", ()):
408                  full_path = path_root / src_path
409                  if full_path not in src_paths:
410                      src_paths.append(full_path)
411  
412              combined_config["src_paths"] = tuple(src_paths)
413  
414          if "formatter" in combined_config:
415              import pkg_resources
416  
417              for plugin in pkg_resources.iter_entry_points("isort.formatters"):
418                  if plugin.name == combined_config["formatter"]:
419                      combined_config["formatting_function"] = plugin.load()
420                      break
421              else:
422                  raise FormattingPluginDoesNotExist(combined_config["formatter"])
423  
424          # Remove any config values that are used for creating config object but
425          # aren't defined in dataclass
426          combined_config.pop("source", None)
427          combined_config.pop("sources", None)
428          combined_config.pop("runtime_src_paths", None)
429  
430          deprecated_options_used = [
431              option for option in combined_config if option in DEPRECATED_SETTINGS
432          ]
433          if deprecated_options_used:
434              for deprecated_option in deprecated_options_used:
435                  combined_config.pop(deprecated_option)
436              if not quiet:
437                  warn(
438                      "W0503: Deprecated config options were used: "
439                      f"{', '.join(deprecated_options_used)}."
440                      "Please see the 5.0.0 upgrade guide: bit.ly/isortv5."
441                  )
442  
443          if known_other:
444              combined_config["known_other"] = known_other
445          if import_headings:
446              for import_heading_key in import_headings:
447                  combined_config.pop(f"{IMPORT_HEADING_PREFIX}{import_heading_key}")
448              combined_config["import_headings"] = import_headings
449  
450          unsupported_config_errors = {}
451          for option in set(combined_config.keys()).difference(
452              getattr(_Config, "__dataclass_fields__", {}).keys()
453          ):
454              for source in reversed(sources):
455                  if option in source:
456                      unsupported_config_errors[option] = {
457                          "value": source[option],
458                          "source": source["source"],
459                      }
460          if unsupported_config_errors:
461              raise UnsupportedSettings(unsupported_config_errors)
462  
463          super().__init__(sources=tuple(sources), **combined_config)  # type: ignore

465      def is_supported_filetype(self, file_name: str):
466          _root, ext = os.path.splitext(file_name)
467          ext = ext.lstrip(".")
468          if ext in self.supported_extensions:
469              return True
470          elif ext in self.blocked_extensions:
471              return False
472  
473          # Skip editor backup files.
474          if file_name.endswith("~"):
475              return False
476  
477          try:
478              if stat.S_ISFIFO(os.stat(file_name).st_mode):
479                  return False
480          except OSError:
481              pass
482  
483          try:
484              with open(file_name, "rb") as fp:
485                  line = fp.readline(100)
486          except OSError:
487              return False
488          else:
489              return bool(_SHEBANG_RE.match(line))

491      def is_skipped(self, file_path: Path) -> bool:
492          """Returns True if the file and/or folder should be skipped based on current settings."""
493          if self.directory and Path(self.directory) in file_path.resolve().parents:
494              file_name = os.path.relpath(file_path.resolve(), self.directory)
495          else:
496              file_name = str(file_path)
497  
498          os_path = str(file_path)
499  
500          if self.skip_gitignore:
501              if file_path.name == ".git":  # pragma: no cover
502                  return True
503  
504              result = subprocess.run(  # nosec
505                  ["git", "-C", str(file_path.parent), "check-ignore", "--quiet", os_path]
506              )
507              if result.returncode == 0:
508                  return True
509  
510          normalized_path = os_path.replace("\\", "/")
511          if normalized_path[1:2] == ":":
512              normalized_path = normalized_path[2:]
513  
514          for skip_path in self.skip:
515              if posixpath.abspath(normalized_path) == posixpath.abspath(
516                  skip_path.replace("\\", "/")
517              ):
518                  return True
519  
520          position = os.path.split(file_name)
521          while position[1]:
522              if position[1] in self.skip:
523                  return True
524              position = os.path.split(position[0])
525  
526          for glob in self.skip_glob:
527              if fnmatch.fnmatch(file_name, glob) or fnmatch.fnmatch("/" + file_name, glob):
528                  return True
529  
530          if not (os.path.isfile(os_path) or os.path.isdir(os_path) or os.path.islink(os_path)):
531              return True
532  
533          return False

535      @property
536      def known_patterns(self):
537          if self._known_patterns is not None:
538              return self._known_patterns
539  
540          self._known_patterns = []
541          pattern_sections = [STDLIB] + [section for section in self.sections if section != STDLIB]
542          for placement in reversed(pattern_sections):
543              known_placement = KNOWN_SECTION_MAPPING.get(placement, placement).lower()
544              config_key = f"{KNOWN_PREFIX}{known_placement}"
545              known_modules = getattr(self, config_key, self.known_other.get(known_placement, ()))
546              extra_modules = getattr(self, f"extra_{known_placement}", ())
547              all_modules = set(extra_modules).union(known_modules)
548              known_patterns = [
549                  pattern
550                  for known_pattern in all_modules
551                  for pattern in self._parse_known_pattern(known_pattern)
552              ]
553              for known_pattern in known_patterns:
554                  regexp = "^" + known_pattern.replace("*", ".*").replace("?", ".?") + "$"
555                  self._known_patterns.append((re.compile(regexp), placement))
556  
557          return self._known_patterns

559      @property
560      def section_comments(self) -> Tuple[str, ...]:
561          if self._section_comments is not None:
562              return self._section_comments
563  
564          self._section_comments = tuple(f"# {heading}" for heading in self.import_headings.values())
565          return self._section_comments

567      def _parse_known_pattern(self, pattern: str) -> List[str]:
568          """Expand pattern if identified as a directory and return found sub packages"""
569          if pattern.endswith(os.path.sep):
570              patterns = [
571                  filename
572                  for filename in os.listdir(os.path.join(self.directory, pattern))
573                  if os.path.isdir(os.path.join(self.directory, pattern, filename))
574              ]
575          else:
576              patterns = [pattern]
577  
578          return patterns

588  def _as_list(value: str) -> List[str]:
589      if isinstance(value, list):
590          return [item.strip() for item in value]
591      filtered = [item.strip() for item in value.replace("\n", ",").split(",") if item.strip()]
592      return filtered

595  def _abspaths(cwd: str, values: Iterable[str]) -> Set[str]:
596      paths = {
597          os.path.join(cwd, value)
598          if not value.startswith(os.path.sep) and value.endswith(os.path.sep)
599          else value
600          for value in values
601      }
602      return paths

605  @lru_cache()
606  def _find_config(path: str) -> Tuple[str, Dict[str, Any]]:
607      current_directory = path
608      tries = 0
609      while current_directory and tries < MAX_CONFIG_SEARCH_DEPTH:
610          for config_file_name in CONFIG_SOURCES:
611              potential_config_file = os.path.join(current_directory, config_file_name)
612              if os.path.isfile(potential_config_file):
613                  config_data: Dict[str, Any]
614                  try:
615                      config_data = _get_config_data(
616                          potential_config_file, CONFIG_SECTIONS[config_file_name]
617                      )
618                  except Exception:
619                      warn(f"Failed to pull configuration information from {potential_config_file}")
620                      config_data = {}
621                  if config_data:
622                      return (current_directory, config_data)
623  
624          for stop_dir in STOP_CONFIG_SEARCH_ON_DIRS:
625              if os.path.isdir(os.path.join(current_directory, stop_dir)):
626                  return (current_directory, {})
627  
628          new_directory = os.path.split(current_directory)[0]
629          if new_directory == current_directory:
630              break
631  
632          current_directory = new_directory
633          tries += 1
634  
635      return (path, {})

638  @lru_cache()
639  def _get_config_data(file_path: str, sections: Tuple[str]) -> Dict[str, Any]:
640      settings: Dict[str, Any] = {}
641  
642      with open(file_path) as config_file:
643          if file_path.endswith(".toml"):
644              config = toml.load(config_file)
645              for section in sections:
646                  config_section = config
647                  for key in section.split("."):
648                      config_section = config_section.get(key, {})
649                  settings.update(config_section)
650          else:
651              if file_path.endswith(".editorconfig"):
652                  line = "\n"
653                  last_position = config_file.tell()
654                  while line:
655                      line = config_file.readline()
656                      if "[" in line:
657                          config_file.seek(last_position)
658                          break
659                      last_position = config_file.tell()
660  
661              config = configparser.ConfigParser(strict=False)
662              config.read_file(config_file)
663              for section in sections:
664                  if section.startswith("*.{") and section.endswith("}"):
665                      extension = section[len("*.{") : -1]
666                      for config_key in config.keys():
667                          if config_key.startswith("*.{") and config_key.endswith("}"):
668                              if extension in map(
669                                  lambda text: text.strip(), config_key[len("*.{") : -1].split(",")
670                              ):
671                                  settings.update(config.items(config_key))
672  
673                  elif config.has_section(section):
674                      settings.update(config.items(section))
675  
676      if settings:
677          settings["source"] = file_path
678  
679          if file_path.endswith(".editorconfig"):
680              indent_style = settings.pop("indent_style", "").strip()
681              indent_size = settings.pop("indent_size", "").strip()
682              if indent_size == "tab":
683                  indent_size = settings.pop("tab_width", "").strip()
684  
685              if indent_style == "space":
686                  settings["indent"] = " " * (indent_size and int(indent_size) or 4)
687  
688              elif indent_style == "tab":
689                  settings["indent"] = "\t" * (indent_size and int(indent_size) or 1)
690  
691              max_line_length = settings.pop("max_line_length", "").strip()
692              if max_line_length and (max_line_length == "off" or max_line_length.isdigit()):
693                  settings["line_length"] = (
694                      float("inf") if max_line_length == "off" else int(max_line_length)
695                  )
696              settings = {
697                  key: value
698                  for key, value in settings.items()
699                  if key in _DEFAULT_SETTINGS.keys() or key.startswith(KNOWN_PREFIX)
700              }
701  
702          for key, value in settings.items():
703              existing_value_type = _get_str_to_type_converter(key)
704              if existing_value_type == tuple:
705                  settings[key] = tuple(_as_list(value))
706              elif existing_value_type == frozenset:
707                  settings[key] = frozenset(_as_list(settings.get(key)))  # type: ignore
708              elif existing_value_type == bool:
709                  # Only some configuration formats support native boolean values.
710                  if not isinstance(value, bool):
711                      value = _as_bool(value)
712                  settings[key] = value
713              elif key.startswith(KNOWN_PREFIX):
714                  settings[key] = _abspaths(os.path.dirname(file_path), _as_list(value))
715              elif key == "force_grid_wrap":
716                  try:
717                      result = existing_value_type(value)
718                  except ValueError:  # backwards compatibility for true / false force grid wrap
719                      result = 0 if value.lower().strip() == "false" else 2
720                  settings[key] = result
721              elif key == "comment_prefix":
722                  settings[key] = str(value).strip("'").strip('"')
723              else:
724                  settings[key] = existing_value_type(value)
725  
726      return settings

729  def _as_bool(value: str) -> bool:
730      """Given a string value that represents True or False, returns the Boolean equivalent. 
731      Heavily inspired from distutils strtobool. 
732      """
733      try:
734          return _STR_BOOLEAN_MAPPING[value.lower()]
735      except KeyError:
736          raise ValueError(f"invalid truth value {value}")
