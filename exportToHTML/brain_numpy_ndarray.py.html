145  def _looks_like_numpy_ndarray(node):
146      return isinstance(node, astroid.Attribute) and node.attrname == "ndarray"
147  
148  
149  astroid.MANAGER.register_transform(
150      astroid.Attribute,
151      astroid.inference_tip(infer_numpy_ndarray),
152      _looks_like_numpy_ndarray,
153  )
