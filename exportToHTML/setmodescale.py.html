setmodescale.py 
1    #!/usr/bin/env python
2    """ pygame.examples.setmodescale 
3     
4    On high resolution displays(4k, 1080p) and tiny graphics games (640x480) 
5    show up very small so that they are unplayable. SCALED scales up the window 
6    for you. The game thinks it's a 640x480 window, but really it can be bigger. 
7    Mouse events are scaled for you, so your game doesn't need to do it. 
8     
9    Passing SCALED to pygame.display.set_mode means the resolution depends 
10   on desktop size and the graphics are scaled. 
11   """
12   import pygame as pg
13   
14   pg.init()
15   
16   RES = (160, 120)
17   FPS = 30
18   clock = pg.time.Clock()
19   
20   print ("desktops", pg.display.get_desktop_sizes())
21   screen = pg.display.set_mode(RES, pg.SCALED | pg.RESIZABLE)
22   
23   # MAIN LOOP
24   
25   done = False
26   
27   i = 0
28   j = 0
29   
30   r_name, r_flags = pg.display._get_renderer_info()
31   print("renderer:", r_name, "flags:", bin(r_flags))
32   for flag, name in [(1, "software"), (2, "accelerated"), (4, "VSync"), (8, "render to texture")]:
33       if (flag & r_flags):
34           print(name)
35   
36   while not done:
37       for event in pg.event.get():
38           if event.type == pg.KEYDOWN and event.key == pg.K_q:
39               done = True
40           if event.type == pg.QUIT:
41               done = True
42           if event.type == pg.KEYDOWN and event.key == pg.K_f:
43               pg.display.toggle_fullscreen()
44           if event.type == pg.VIDEORESIZE:
45               pg.display._resize_event(event)
46   
47       i += 1
48       i = i % screen.get_width()
49       j += i % 2
50       j = j % screen.get_height()
51   
52       screen.fill((255, 0, 255))
53       pg.draw.circle(screen, (0, 0, 0), (100, 100), 20)
54       pg.draw.circle(screen, (0, 0, 200), (0, 0), 10)
55       pg.draw.circle(screen, (200, 0, 0), (160, 120), 30)
56       pg.draw.line(screen, (250, 250, 0), (0, 120), (160, 0))
57       pg.draw.circle(screen, (255, 255, 255), (i, j), 5)
58   
59       pg.display.flip()
60       clock.tick(FPS)
