57   def _simplename(name):
58       """create simple version of the font name"""
59       # return alphanumeric characters of a string (converted to lowercase)
60       return ''.join(c.lower() for c in name if c.isalnum())

63   def _addfont(name, bold, italic, font, fontdict):
64       """insert a font and style into the font dictionary"""
65       if name not in fontdict:
66           fontdict[name] = {}
67       fontdict[name][bold, italic] = font

70   def initsysfonts_win32():
71       """initialize fonts dictionary on Windows"""
72   
73       fontdir = join(os.environ.get('WINDIR', 'C:\\Windows'), 'Fonts')
74   
75       fonts = {}
76   
77       # add fonts entered in the registry
78   
79       # find valid registry keys containing font information.
80       # http://docs.python.org/lib/module-sys.html
81       # 0 (VER_PLATFORM_WIN32s)          Win32s on Windows 3.1
82       # 1 (VER_PLATFORM_WIN32_WINDOWS)   Windows 95/98/ME
83       # 2 (VER_PLATFORM_WIN32_NT)        Windows NT/2000/XP
84       # 3 (VER_PLATFORM_WIN32_CE)        Windows CE
85       if sys.getwindowsversion()[0] == 1:
86           key_name = "SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Fonts"
87       else:
88           key_name = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Fonts"
89       key = _winreg.OpenKey(_winreg.HKEY_LOCAL_MACHINE, key_name)
90   
91       for i in xrange_(_winreg.QueryInfoKey(key)[1]):
92           try:
93               # name is the font's name e.g. Times New Roman (TrueType)
94               # font is the font's filename e.g. times.ttf
95               name, font = _winreg.EnumValue(key, i)[0:2]
96           except EnvironmentError:
97               break
98   
99           # try to handle windows unicode strings for file names with
100          # international characters
101          if PY_MAJOR_VERSION < 3:
102              # here are two documents with some information about it:
103              # http://www.python.org/peps/pep-0277.html
104              # https://www.microsoft.com/technet/archive/interopmigration/linux/mvc/lintowin.mspx#ECAA
105              try:
106                  font = str(font)
107              except UnicodeEncodeError:
108                  # MBCS is the windows encoding for unicode file names.
109                  try:
110                      font = font.encode('MBCS')
111                  except UnicodeEncodeError:
112                      # no success with str or MBCS encoding... skip this font.
113                      continue
114  
115          if splitext(font)[1].lower() not in OpenType_extensions:
116              continue
117          if not dirname(font):
118              font = join(fontdir, font)
119  
120          _parse_font_entry_win(name, font, fonts)
121  
122      return fonts

125  def _parse_font_entry_win(name, font, fonts):
126      """ 
127      Parse out a simpler name and the font style from the initial file name. 
128   
129      :param name: The font name 
130      :param font: The font file path 
131      :param fonts: The pygame font dictionary 
132   
133      :return: Tuple of (bold, italic, name) 
134      """
135      true_type_suffix = '(TrueType)'
136      mods = ('demibold', 'narrow', 'light', 'unicode', 'bt', 'mt')
137      if name.endswith(true_type_suffix):
138          name = name.rstrip(true_type_suffix).rstrip()
139      name = name.lower().split()
140      bold = italic = 0
141      for mod in mods:
142          if mod in name:
143              name.remove(mod)
144      if 'bold' in name:
145          name.remove('bold')
146          bold = 1
147      if 'italic' in name:
148          name.remove('italic')
149          italic = 1
150      name = ''.join(name)
151      name = _simplename(name)
152  
153      _addfont(name, bold, italic, font, fonts)

156  def _add_font_paths(sub_elements, fonts):
157      """ Gets each element, checks its tag content, 
158          if wanted fetches the next value in the iterable 
159      """
160      font_name = None
161      bold = False
162      italic = False
163      for tag in sub_elements:
164          if tag.text == "_name":
165              font_file_name = next(sub_elements).text
166              font_name = splitext(font_file_name)[0]
167              if splitext(font_file_name)[1] not in OpenType_extensions:
168                  break
169              bold = "bold" in font_name
170              italic = "italic" in font_name
171          if tag.text == "path" and font_name is not None:
172              font_path = next(sub_elements).text
173              _addfont(_simplename(font_name), bold, italic, font_path, fonts)
174              break

177  def _system_profiler_darwin():
178      fonts = {}
179      flout, _ = subprocess.Popen(
180          ' '.join(['system_profiler', '-xml', 'SPFontsDataType']),
181          shell=True,
182          stdout=subprocess.PIPE,
183          stderr=subprocess.PIPE,
184          close_fds=True
185      ).communicate()
186  
187      for font_node in ET.fromstring(flout).iterfind('./array/dict/array/dict'):
188          _add_font_paths(font_node.iter("*"), fonts)
189  
190      return fonts

193  def initsysfonts_darwin():
194      """ Read the fonts on MacOS, and OS X. 
195      """
196      # if the X11 binary exists... try and use that.
197      #  Not likely to be there on pre 10.4.x ... or MacOS 10.10+
198      if exists('/usr/X11/bin/fc-list'):
199          fonts = initsysfonts_unix('/usr/X11/bin/fc-list')
200      # This fc-list path will work with the X11 from the OS X 10.3 installation
201      # disc
202      elif exists('/usr/X11R6/bin/fc-list'):
203          fonts = initsysfonts_unix('/usr/X11R6/bin/fc-list')
204      elif exists('/usr/sbin/system_profiler'):
205          try:
206              fonts = _system_profiler_darwin()
207          except (OSError, ValueError):
208              fonts = {}
209      else:
210          fonts = {}
211  
212      return fonts

216  def initsysfonts_unix(path="fc-list"):
217      """use the fc-list from fontconfig to get a list of fonts"""
218      fonts = {}
219  
220      try:
221          # note, we capture stderr so if fc-list isn't there to stop stderr
222          # printing.
223          flout, _ = subprocess.Popen('%s : file family style' % path,
224                                      shell=True,
225                                      stdout=subprocess.PIPE,
226                                      stderr=subprocess.PIPE,
227                                      close_fds=True).communicate()
228      except (OSError, ValueError):
229          return fonts
230  
231      entries = toascii(flout)
232      try:
233          for entry in entries.split('\n'):
234  
235              try:
236                  _parse_font_entry_unix(entry, fonts)
237              except ValueError:
238                  # try the next one.
239                  pass
240  
241      except ValueError:
242          pass
243  
244      return fonts

247  def _parse_font_entry_unix(entry, fonts):
248      """ 
249      Parses an entry in the unix font data to add to the pygame font 
250      dictionary. 
251   
252      :param entry: A entry from the unix font list. 
253      :param fonts: The pygame font dictionary to add the parsed font data to. 
254   
255      """
256      filename, family, style = entry.split(':', 2)
257      if splitext(filename)[1].lower() in OpenType_extensions:
258          bold = 'Bold' in style
259          italic = 'Italic' in style
260          oblique = 'Oblique' in style
261          for name in family.strip().split(','):
262              if name:
263                  break
264          else:
265              name = splitext(basename(filename))[0]
266  
267          _addfont(_simplename(name), bold, italic or oblique,
268                   filename, fonts)

271  def create_aliases():
272      """ Map common fonts that are absent from the system to similar fonts 
273          that are installed in the system 
274      """
275      alias_groups = (
276          ('monospace', 'misc-fixed', 'courier', 'couriernew', 'console',
277           'fixed', 'mono', 'freemono', 'bitstreamverasansmono',
278           'verasansmono', 'monotype', 'lucidaconsole', 'consolas',
279           'dejavusansmono', 'liberationmono'),
280          ('sans', 'arial', 'helvetica', 'swiss', 'freesans',
281           'bitstreamverasans', 'verasans', 'verdana', 'tahoma',
282           'calibri', 'gillsans', 'segoeui', 'trebuchetms', 'ubuntu',
283           'dejavusans', 'liberationsans'),
284          ('serif', 'times', 'freeserif', 'bitstreamveraserif', 'roman',
285           'timesroman', 'timesnewroman', 'dutch', 'veraserif',
286           'georgia', 'cambria', 'constantia', 'dejavuserif',
287           'liberationserif'),
288          ('wingdings', 'wingbats'),
289      )
290      for alias_set in alias_groups:
291          for name in alias_set:
292              if name in Sysfonts:
293                  found = Sysfonts[name]
294                  break
295          else:
296              continue
297          for name in alias_set:
298              if name not in Sysfonts:
299                  Sysalias[name] = found

302  def initsysfonts():
303      """ 
304      Initialise the sysfont module, called once. Locates the installed fonts 
305      and creates some aliases for common font categories. 
306   
307      Has different initialisation functions for different platforms. 
308      """
309      if sys.platform == 'win32':
310          fonts = initsysfonts_win32()
311      elif sys.platform == 'darwin':
312          fonts = initsysfonts_darwin()
313      else:
314          fonts = initsysfonts_unix()
315      Sysfonts.update(fonts)
316      create_aliases()
317      if not Sysfonts:  # dummy so we don't try to reinit
318          Sysfonts[None] = None

321  def font_constructor(fontpath, size, bold, italic):
322      """ 
323      pygame.font specific declarations 
324   
325      :param fontpath: path to a font. 
326      :param size: size of a font. 
327      :param bold: bold style, True or False. 
328      :param italic: italic style, True or False. 
329   
330      :return: A font.Font object. 
331      """
332  
333      font = Font(fontpath, size)
334      if bold:
335          font.set_bold(True)
336      if italic:
337          font.set_italic(True)
338  
339      return font

344  def SysFont(name, size, bold=False, italic=False, constructor=None):
345      """pygame.font.SysFont(name, size, bold=False, italic=False, constructor=None) -> Font 
346         create a pygame Font from system font resources 
347   
348         This will search the system fonts for the given font 
349         name. You can also enable bold or italic styles, and 
350         the appropriate system font will be selected if available. 
351   
352         This will always return a valid Font object, and will 
353         fallback on the builtin pygame font if the given font 
354         is not found. 
355   
356         Name can also be a comma separated list of names, in 
357         which case set of names will be searched in order. Pygame 
358         uses a small set of common font aliases, if the specific 
359         font you ask for is not available, a reasonable alternative 
360         may be used. 
361   
362         if optional constructor is provided, it must be a function with 
363         signature constructor(fontpath, size, bold, italic) which returns 
364         a Font instance. If None, a pygame.font.Font object is created. 
365      """
366      if constructor is None:
367          constructor = font_constructor
368  
369      if not Sysfonts:
370          initsysfonts()
371  
372      gotbold = gotitalic = False
373      fontname = None
374      if name:
375          for single_name in name.split(','):
376              single_name = _simplename(single_name)
377              styles = Sysfonts.get(single_name)
378              if not styles:
379                  styles = Sysalias.get(single_name)
380              if styles:
381                  plainname = styles.get((False, False))
382                  fontname = styles.get((bold, italic))
383                  if not (fontname or plainname):
384                      # Neither requested style, nor plain font exists, so
385                      # return a font with the name requested, but an
386                      # arbitrary style.
387                      (style, fontname) = list(styles.items())[0]
388                      # Attempt to style it as requested. This can't
389                      # unbold or unitalicize anything, but it can
390                      # fake bold and/or fake italicize.
391                      if bold and style[0]:
392                          gotbold = True
393                      if italic and style[1]:
394                          gotitalic = True
395                  elif not fontname:
396                      fontname = plainname
397                  elif plainname != fontname:
398                      gotbold = bold
399                      gotitalic = italic
400              if fontname:
401                  break
402  
403      set_bold = set_italic = False
404      if bold and not gotbold:
405          set_bold = True
406      if italic and not gotitalic:
407          set_italic = True
408  
409      return constructor(fontname, size, set_bold, set_italic)

412  def get_fonts():
413      """pygame.font.get_fonts() -> list 
414         get a list of system font names 
415   
416         Returns the list of all found system fonts. Note that 
417         the names of the fonts will be all lowercase with spaces 
418         removed. This is how pygame internally stores the font 
419         names for matching. 
420      """
421      if not Sysfonts:
422          initsysfonts()
423      return list(Sysfonts)

426  def match_font(name, bold=0, italic=0):
427      """pygame.font.match_font(name, bold=0, italic=0) -> name 
428         find the filename for the named system font 
429   
430         This performs the same font search as the SysFont() 
431         function, only it returns the path to the TTF file 
432         that would be loaded. The font name can be a comma 
433         separated list of font names to try. 
434   
435         If no match is found, None is returned. 
436      """
437      if not Sysfonts:
438          initsysfonts()
439  
440      fontname = None
441      for single_name in name.split(','):
442          single_name = _simplename(single_name)
443          styles = Sysfonts.get(single_name)
444          if not styles:
445              styles = Sysalias.get(single_name)
446          if styles:
447              while not fontname:
448                  fontname = styles.get((bold, italic))
449                  if italic:
450                      italic = 0
451                  elif bold:
452                      bold = 0
453                  elif not fontname:
454                      fontname = list(styles.values())[0]
455          if fontname:
456              break
457      return fontname
