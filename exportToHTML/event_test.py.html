105  class EventTypeTest(unittest.TestCase):
106      def test_Event(self):
107          """Ensure an Event object can be created."""
108          e = pygame.event.Event(pygame.USEREVENT, some_attr=1, other_attr="1")
109  
110          self.assertEqual(e.some_attr, 1)
111          self.assertEqual(e.other_attr, "1")
112  
113          # Event now uses tp_dictoffset and tp_members: request 62
114          # on Motherhamster Bugzilla.
115          self.assertEqual(e.type, pygame.USEREVENT)
116          self.assertIs(e.dict, e.__dict__)
117  
118          e.some_attr = 12
119  
120          self.assertEqual(e.some_attr, 12)
121  
122          e.new_attr = 15
123  
124          self.assertEqual(e.new_attr, 15)
125  
126          # For Python 2.x a TypeError is raised for a readonly member;
127          # for Python 3.x it is an AttributeError.
128          self.assertRaises((TypeError, AttributeError), setattr, e, "type", 0)
129          self.assertRaises((TypeError, AttributeError), setattr, e, "dict", None)
130  
131          # Ensure attributes are visible to dir(), part of the original
132          # posted request.
133          d = dir(e)
134          attrs = ("type", "dict", "__dict__", "some_attr", "other_attr", "new_attr")
135  
136          for attr in attrs:
137              self.assertIn(attr, d)

224          """Ensure custom_type() raises error when no more custom types.

226          The last allowed custom type number should be (pygame.NUMEVENTS - 1). 
227          """
228          start = pygame.event.custom_type() + 1
229          for i in range(start, pygame.NUMEVENTS):

231          self.assertEqual(last, pygame.NUMEVENTS - 1)
232          with self.assertRaises(pygame.error):
233              pygame.event.custom_type()

248          # Handle method name differences between Python versions.
249          if PY3:
250              self.assertCountEqual(*args, **kwargs)
251          else:
252              self.assertItemsEqual(*args, **kwargs)

254      def _assertExpectedEvents(self, expected, got):
255          """Find events like expected events, raise on unexpected or missing, 
256          ignore additional event properties if expected properties are present."""
257  
258          # This does greedy matching, don't encode an NP-hard problem
259          # into your input data, *please*
260          items_left=got[:]
261          for expected_element in expected:
262              for item in items_left:
263                  for key in expected_element.__dict__:
264                      if item.__dict__[key]!=expected_element.__dict__[key]:
265                          break
266                  else:
267                      #found item!
268                      items_left.remove(item)
269                      break
270              else:
271                  raise AssertionError("Expected "+str(expected_element)+" among remaining events "+str(items_left)+" out of "+str(got))
272          if len(items_left)>0:
273              raise AssertionError("Unexpected Events: "+str(items_left))
