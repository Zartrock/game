36   # Conditions for adding methods.  The boxes indicate what action the
37   # dataclass decorator takes.  For all of these tables, when I talk
38   # about init=, repr=, eq=, order=, unsafe_hash=, or frozen=, I'm
39   # referring to the arguments to the @dataclass decorator.  When
40   # checking if a dunder method already exists, I mean check for an
41   # entry in the class's __dict__.  I never check to see if an attribute
42   # is defined in a base class.
43   
44   # Key:
45   # +=========+=========================================+
46   # + Value   | Meaning                                 |
47   # +=========+=========================================+
48   # | <blank> | No action: no method is added.          |
49   # +---------+-----------------------------------------+
50   # | add     | Generated method is added.              |
51   # +---------+-----------------------------------------+
52   # | raise   | TypeError is raised.                    |
53   # +---------+-----------------------------------------+
54   # | None    | Attribute is set to None.               |
55   # +=========+=========================================+
56   
57   # __init__
58   #
59   #   +--- init= parameter
60   #   |
61   #   v     |       |       |
62   #         |  no   |  yes  |  <--- class has __init__ in __dict__?
63   # +=======+=======+=======+
64   # | False |       |       |
65   # +-------+-------+-------+
66   # | True  | add   |       |  <- the default
67   # +=======+=======+=======+
68   
69   # __repr__
70   #
71   #    +--- repr= parameter
72   #    |
73   #    v    |       |       |
74   #         |  no   |  yes  |  <--- class has __repr__ in __dict__?
75   # +=======+=======+=======+
76   # | False |       |       |
77   # +-------+-------+-------+
78   # | True  | add   |       |  <- the default
79   # +=======+=======+=======+
80   
81   
82   # __setattr__
83   # __delattr__
84   #
85   #    +--- frozen= parameter
86   #    |
87   #    v    |       |       |
88   #         |  no   |  yes  |  <--- class has __setattr__ or __delattr__ in __dict__?
89   # +=======+=======+=======+
90   # | False |       |       |  <- the default
91   # +-------+-------+-------+
92   # | True  | add   | raise |
93   # +=======+=======+=======+
94   # Raise because not adding these methods would break the "frozen-ness"
95   # of the class.
96   
97   # __eq__
98   #
99   #    +--- eq= parameter
100  #    |
101  #    v    |       |       |
102  #         |  no   |  yes  |  <--- class has __eq__ in __dict__?
103  # +=======+=======+=======+
104  # | False |       |       |
105  # +-------+-------+-------+
106  # | True  | add   |       |  <- the default
107  # +=======+=======+=======+
108  
109  # __lt__
110  # __le__
111  # __gt__
112  # __ge__
113  #
114  #    +--- order= parameter
115  #    |
116  #    v    |       |       |
117  #         |  no   |  yes  |  <--- class has any comparison method in __dict__?
118  # +=======+=======+=======+
119  # | False |       |       |  <- the default
120  # +-------+-------+-------+
121  # | True  | add   | raise |
122  # +=======+=======+=======+
123  # Raise because to allow this case would interfere with using
124  # functools.total_ordering.
125  
126  # __hash__
127  
128  #    +------------------- unsafe_hash= parameter
129  #    |       +----------- eq= parameter
130  #    |       |       +--- frozen= parameter
131  #    |       |       |
132  #    v       v       v    |        |        |
133  #                         |   no   |  yes   |  <--- class has explicitly defined __hash__
134  # +=======+=======+=======+========+========+
135  # | False | False | False |        |        | No __eq__, use the base class __hash__
136  # +-------+-------+-------+--------+--------+
137  # | False | False | True  |        |        | No __eq__, use the base class __hash__
138  # +-------+-------+-------+--------+--------+
139  # | False | True  | False | None   |        | <-- the default, not hashable
140  # +-------+-------+-------+--------+--------+
141  # | False | True  | True  | add    |        | Frozen, so hashable, allows override
142  # +-------+-------+-------+--------+--------+
143  # | True  | False | False | add    | raise  | Has no __eq__, but hashable
144  # +-------+-------+-------+--------+--------+
145  # | True  | False | True  | add    | raise  | Has no __eq__, but hashable
146  # +-------+-------+-------+--------+--------+
147  # | True  | True  | False | add    | raise  | Not frozen, but hashable
148  # +-------+-------+-------+--------+--------+
149  # | True  | True  | True  | add    | raise  | Frozen, so hashable
150  # +=======+=======+=======+========+========+
151  # For boxes that are blank, __hash__ is untouched and therefore
152  # inherited from the base class.  If the base is object, then
153  # id-based hashing is used.
154  #
155  # Note that a class may already have __hash__=None if it specified an
156  # __eq__ method in the class body (not one that was created by
157  # @dataclass).
158  #
159  # See _hash_action (below) for a coded version of this table.

229  # Instances of Field are only ever created from within this module,
230  # and only from the field() function, although Field instances are
231  # exposed externally as (conceptually) read-only objects.
232  #
233  # name and type are filled in after the fact, not in __init__.
234  # They're not known at the time this class is instantiated, but it's
235  # convenient if they're available later.
236  #
237  # When cls._FIELDS is filled in with a list of Field objects, the name
238  # and type fields will have been populated.
239  class Field:
240      __slots__ = (
241          "name",
242          "type",
243          "default",
244          "default_factory",
245          "repr",
246          "hash",
247          "init",
248          "compare",
249          "metadata",
250          "_field_type",  # Private: not to be used by user code.
251      )
252  
253      def __init__(self, default, default_factory, init, repr, hash, compare, metadata):
254          self.name = None
255          self.type = None
256          self.default = default
257          self.default_factory = default_factory
258          self.init = init
259          self.repr = repr
260          self.hash = hash
261          self.compare = compare
262          self.metadata = (
263              _EMPTY_METADATA
264              if metadata is None or len(metadata) == 0
265              else types.MappingProxyType(metadata)
266          )
267          self._field_type = None
268  
269      def __repr__(self):
270          return (
271              "Field("
272              f"name={self.name!r},"
273              f"type={self.type!r},"
274              f"default={self.default!r},"
275              f"default_factory={self.default_factory!r},"
276              f"init={self.init!r},"
277              f"repr={self.repr!r},"
278              f"hash={self.hash!r},"
279              f"compare={self.compare!r},"
280              f"metadata={self.metadata!r},"
281              f"_field_type={self._field_type}"
282              ")"
283          )
284  
285      # This is used to support the PEP 487 __set_name__ protocol in the
286      # case where we're using a field that contains a descriptor as a
287      # defaul value.  For details on __set_name__, see
288      # https://www.python.org/dev/peps/pep-0487/#implementation-details.
289      #
290      # Note that in _process_class, this Field object is overwritten
291      # with the default value, so the end result is a descriptor that
292      # had __set_name__ called on it at the right time.
293      def __set_name__(self, owner, name):
294          func = getattr(type(self.default), "__set_name__", None)
295          if func:
296              # There is a __set_name__ method on the descriptor, call
297              # it.
298              func(self.default, owner, name)

325  # This function is used instead of exposing Field creation directly,
326  # so that a type checker can be told (via overloads) that this is a
327  # function whose type depends on its parameters.
328  def field(
329      *,
330      default=MISSING,
331      default_factory=MISSING,
332      init=True,
333      repr=True,
334      hash=None,
335      compare=True,
336      metadata=None,
337  ):
338      """Return an object to identify dataclass fields. 
339      default is the default value of the field.  default_factory is a 
340      0-argument function called to initialize a field's value.  If init 
341      is True, the field will be a parameter to the class's __init__() 
342      function.  If repr is True, the field will be included in the 
343      object's repr().  If hash is True, the field will be included in 
344      the object's hash().  If compare is True, the field will be used 
345      in comparison functions.  metadata, if specified, must be a 
346      mapping which is stored but not otherwise examined by dataclass. 
347      It is an error to specify both default and default_factory. 
348      """
349  
350      if default is not MISSING and default_factory is not MISSING:
351          raise ValueError("cannot specify both default and default_factory")
352      return Field(default, default_factory, init, repr, hash, compare, metadata)

355  def _tuple_str(obj_name, fields):
356      # Return a string representing each field of obj_name as a tuple
357      # member.  So, if fields is ['x', 'y'] and obj_name is "self",
358      # return "(self.x,self.y)".
359  
360      # Special case for the 0-tuple.
361      if not fields:
362          return "()"
363      # Note the trailing comma, needed if this turns out to be a 1-tuple.
364      return f'({",".join(f"{obj_name}.{f.name}" for f in fields)},)'

367  def _create_fn(name, args, body, *, globals=None, locals=None, return_type=MISSING):
368      # Note that we mutate locals when exec() is called.  Caller
369      # beware!  The only callers are internal to this module, so no
370      # worries about external callers.
371      if locals is None:
372          locals = {}
373      return_annotation = ""
374      if return_type is not MISSING:
375          locals["_return_type"] = return_type
376          return_annotation = "->_return_type"
377      args = ",".join(args)
378      body = "\n".join(f" {b}" for b in body)
379  
380      # Compute the text of the entire function.
381      txt = f"def {name}({args}){return_annotation}:\n{body}"
382  
383      exec(txt, globals, locals)  # nosec
384      return locals[name]

387  def _field_assign(frozen, name, value, self_name):
388      # If we're a frozen class, then assign to our fields in __init__
389      # via object.__setattr__.  Otherwise, just use a simple
390      # assignment.
391      #
392      # self_name is what "self" is called in this function: don't
393      # hard-code "self", since that might be a field name.
394      if frozen:
395          return f"object.__setattr__({self_name},{name!r},{value})"
396      return f"{self_name}.{name}={value}"
397

399  def _field_init(f, frozen, globals, self_name):
400      # Return the text of the line in the body of __init__ that will
401      # initialize this field.
402  
403      default_name = f"_dflt_{f.name}"
404      if f.default_factory is not MISSING:
405          if f.init:
406              # This field has a default factory.  If a parameter is
407              # given, use it.  If not, call the factory.
408              globals[default_name] = f.default_factory
409              value = f"{default_name}() " f"if {f.name} is _HAS_DEFAULT_FACTORY " f"else {f.name}"
410          else:
411              # This is a field that's not in the __init__ params, but
412              # has a default factory function.  It needs to be
413              # initialized here by calling the factory function,
414              # because there's no other way to initialize it.
415  
416              # For a field initialized with a default=defaultvalue, the
417              # class dict just has the default value
418              # (cls.fieldname=defaultvalue).  But that won't work for a
419              # default factory, the factory must be called in __init__
420              # and we must assign that to self.fieldname.  We can't
421              # fall back to the class dict's value, both because it's
422              # not set, and because it might be different per-class
423              # (which, after all, is why we have a factory function!).
424  
425              globals[default_name] = f.default_factory
426              value = f"{default_name}()"
427      else:
428          # No default factory.
429          if f.init:
430              if f.default is MISSING:
431                  # There's no default, just do an assignment.
432                  value = f.name
433              elif f.default is not MISSING:
434                  globals[default_name] = f.default
435                  value = f.name
436          else:
437              # This field does not need initialization.  Signify that
438              # to the caller by returning None.
439              return None
440  
441      # Only test this now, so that we can create variables for the
442      # default.  However, return None to signify that we're not going
443      # to actually do the assignment statement for InitVars.
444      if f._field_type == _FIELD_INITVAR:
445          return None
446  
447      # Now, actually generate the field assignment.
448      return _field_assign(frozen, f.name, value, self_name)

451  def _init_param(f):
452      # Return the __init__ parameter string for this field.  For
453      # example, the equivalent of 'x:int=3' (except instead of 'int',
454      # reference a variable set to int, and instead of '3', reference a
455      # variable set to 3).
456      if f.default is MISSING and f.default_factory is MISSING:
457          # There's no default, and no default_factory, just output the
458          # variable name and type.
459          default = ""
460      elif f.default is not MISSING:
461          # There's a default, this will be the name that's used to look
462          # it up.
463          default = f"=_dflt_{f.name}"
464      elif f.default_factory is not MISSING:
465          # There's a factory function.  Set a marker.
466          default = "=_HAS_DEFAULT_FACTORY"
467      return f"{f.name}:_type_{f.name}{default}"

470  def _init_fn(fields, frozen, has_post_init, self_name):
471      # fields contains both real fields and InitVar pseudo-fields.
472  
473      # Make sure we don't have fields without defaults following fields
474      # with defaults.  This actually would be caught when exec-ing the
475      # function source code, but catching it here gives a better error
476      # message, and future-proofs us in case we build up the function
477      # using ast.
478      seen_default = False
479      for f in fields:
480          # Only consider fields in the __init__ call.
481          if f.init:
482              if not (f.default is MISSING and f.default_factory is MISSING):
483                  seen_default = True
484              elif seen_default:
485                  raise TypeError(f"non-default argument {f.name!r} " "follows default argument")
486  
487      globals = {"MISSING": MISSING, "_HAS_DEFAULT_FACTORY": _HAS_DEFAULT_FACTORY}
488  
489      body_lines = []
490      for f in fields:
491          line = _field_init(f, frozen, globals, self_name)
492          # line is None means that this field doesn't require
493          # initialization (it's a pseudo-field).  Just skip it.
494          if line:
495              body_lines.append(line)
496  
497      # Does this class have a post-init function?
498      if has_post_init:
499          params_str = ",".join(f.name for f in fields if f._field_type is _FIELD_INITVAR)
500          body_lines.append(f"{self_name}.{_POST_INIT_NAME}({params_str})")
501  
502      # If no body lines, use 'pass'.
503      if not body_lines:
504          body_lines = ["pass"]
505  
506      locals = {f"_type_{f.name}": f.type for f in fields}
507      return _create_fn(
508          "__init__",
509          [self_name] + [_init_param(f) for f in fields if f.init],
510          body_lines,
511          locals=locals,
512          globals=globals,
513          return_type=None,
514      )

529  def _frozen_get_del_attr(cls, fields):
530      # XXX: globals is modified on the first call to _create_fn, then
531      # the modified version is used in the second call.  Is this okay?
532      globals = {"cls": cls, "FrozenInstanceError": FrozenInstanceError}
533      if fields:
534          fields_str = "(" + ",".join(repr(f.name) for f in fields) + ",)"
535      else:
536          # Special case for the zero-length tuple.
537          fields_str = "()"
538      return (
539          _create_fn(
540              "__setattr__",
541              ("self", "name", "value"),
542              (
543                  f"if type(self) is cls or name in {fields_str}:",
544                  ' raise FrozenInstanceError(f"cannot assign to field {name!r}")',
545                  f"super(cls, self).__setattr__(name, value)",
546              ),
547              globals=globals,
548          ),
549          _create_fn(
550              "__delattr__",
551              ("self", "name"),
552              (
553                  f"if type(self) is cls or name in {fields_str}:",
554                  ' raise FrozenInstanceError(f"cannot delete field {name!r}")',
555                  f"super(cls, self).__delattr__(name)",
556              ),
557              globals=globals,
558          ),
559      )

562  def _cmp_fn(name, op, self_tuple, other_tuple):
563      # Create a comparison function.  If the fields in the object are
564      # named 'x' and 'y', then self_tuple is the string
565      # '(self.x,self.y)' and other_tuple is the string
566      # '(other.x,other.y)'.
567  
568      return _create_fn(
569          name,
570          ("self", "other"),
571          [
572              "if other.__class__ is self.__class__:",
573              f" return {self_tuple}{op}{other_tuple}",
574              "return NotImplemented",
575          ],
576      )

580      self_tuple = _tuple_str("self", fields)
581      return _create_fn("__hash__", ("self",), [f"return hash({self_tuple})"])

596  def _is_type(annotation, cls, a_module, a_type, is_type_predicate):
597      # Given a type annotation string, does it refer to a_type in
598      # a_module?  For example, when checking that annotation denotes a
599      # ClassVar, then a_module is typing, and a_type is
600      # typing.ClassVar.
601  
602      # It's possible to look up a_module given a_type, but it involves
603      # looking in sys.modules (again!), and seems like a waste since
604      # the caller already knows a_module.
605  
606      # - annotation is a string type annotation
607      # - cls is the class that this annotation was found in
608      # - a_module is the module we want to match
609      # - a_type is the type in that module we want to match
610      # - is_type_predicate is a function called with (obj, a_module)
611      #   that determines if obj is of the desired type.
612  
613      # Since this test does not do a local namespace lookup (and
614      # instead only a module (global) lookup), there are some things it
615      # gets wrong.
616  
617      # With string annotations, cv0 will be detected as a ClassVar:
618      #   CV = ClassVar
619      #   @dataclass
620      #   class C0:
621      #     cv0: CV
622  
623      # But in this example cv1 will not be detected as a ClassVar:
624      #   @dataclass
625      #   class C1:
626      #     CV = ClassVar
627      #     cv1: CV
628  
629      # In C1, the code in this function (_is_type) will look up "CV" in
630      # the module and not find it, so it will not consider cv1 as a
631      # ClassVar.  This is a fairly obscure corner case, and the best
632      # way to fix it would be to eval() the string "CV" with the
633      # correct global and local namespaces.  However that would involve
634      # a eval() penalty for every single field of every dataclass
635      # that's defined.  It was judged not worth it.
636  
637      match = _MODULE_IDENTIFIER_RE.match(annotation)
638      if match:
639          ns = None
640          module_name = match.group(1)
641          if not module_name:
642              # No module name, assume the class's module did
643              # "from dataclasses import InitVar".
644              ns = sys.modules.get(cls.__module__).__dict__
645          else:
646              # Look up module_name in the class's module.
647              module = sys.modules.get(cls.__module__)
648              if module and module.__dict__.get(module_name) is a_module:
649                  ns = sys.modules.get(a_type.__module__).__dict__
650          if ns and is_type_predicate(ns.get(match.group(2)), a_module):
651              return True
652      return False

655  def _get_field(cls, a_name, a_type):
656      # Return a Field object for this field name and type.  ClassVars
657      # and InitVars are also returned, but marked as such (see
658      # f._field_type).
659  
660      # If the default value isn't derived from Field, then it's only a
661      # normal default value.  Convert it to a Field().
662      default = getattr(cls, a_name, MISSING)
663      if isinstance(default, Field):
664          f = default
665      else:
666          if isinstance(default, types.MemberDescriptorType):
667              # This is a field in __slots__, so it has no default value.
668              default = MISSING
669          f = field(default=default)
670  
671      # Only at this point do we know the name and the type.  Set them.
672      f.name = a_name
673      f.type = a_type
674  
675      # Assume it's a normal field until proven otherwise.  We're next
676      # going to decide if it's a ClassVar or InitVar, everything else
677      # is just a normal field.
678      f._field_type = _FIELD
679  
680      # In addition to checking for actual types here, also check for
681      # string annotations.  get_type_hints() won't always work for us
682      # (see https://github.com/python/typing/issues/508 for example),
683      # plus it's expensive and would require an eval for every stirng
684      # annotation.  So, make a best effort to see if this is a ClassVar
685      # or InitVar using regex's and checking that the thing referenced
686      # is actually of the correct type.
687  
688      # For the complete discussion, see https://bugs.python.org/issue33453
689  
690      # If typing has not been imported, then it's impossible for any
691      # annotation to be a ClassVar.  So, only look for ClassVar if
692      # typing has been imported by any module (not necessarily cls's
693      # module).
694      typing = sys.modules.get("typing")
695      if typing:
696          if _is_classvar(a_type, typing) or (
697              isinstance(f.type, str) and _is_type(f.type, cls, typing, typing.ClassVar, _is_classvar)
698          ):
699              f._field_type = _FIELD_CLASSVAR
700  
701      # If the type is InitVar, or if it's a matching string annotation,
702      # then it's an InitVar.
703      if f._field_type is _FIELD:
704          # The module we're checking against is the module we're
705          # currently in (dataclasses.py).
706          dataclasses = sys.modules[__name__]
707          if _is_initvar(a_type, dataclasses) or (
708              isinstance(f.type, str)
709              and _is_type(f.type, cls, dataclasses, dataclasses.InitVar, _is_initvar)
710          ):
711              f._field_type = _FIELD_INITVAR
712  
713      # Validations for individual fields.  This is delayed until now,
714      # instead of in the Field() constructor, since only here do we
715      # know the field name, which allows for better error reporting.
716  
717      # Special restrictions for ClassVar and InitVar.
718      if f._field_type in (_FIELD_CLASSVAR, _FIELD_INITVAR):
719          if f.default_factory is not MISSING:
720              raise TypeError(f"field {f.name} cannot have a " "default factory")
721          # Should I check for other field settings? default_factory
722          # seems the most serious to check for.  Maybe add others.  For
723          # example, how about init=False (or really,
724          # init=<not-the-default-init-value>)?  It makes no sense for
725          # ClassVar and InitVar to specify init=<anything>.
726  
727      # For real fields, disallow mutable defaults for known types.
728      if f._field_type is _FIELD and isinstance(f.default, (list, dict, set)):
729          raise ValueError(
730              f"mutable default {type(f.default)} for field "
731              f"{f.name} is not allowed: use default_factory"
732          )
733  
734      return f

797  def _process_class(cls, init, repr, eq, order, unsafe_hash, frozen):
798      # Now that dicts retain insertion order, there's no reason to use
799      # an ordered dict.  I am leveraging that ordering here, because
800      # derived class fields overwrite base class fields, but the order
801      # is defined by the base class, which is found first.
802      fields = {}
803  
804      setattr(cls, _PARAMS, _DataclassParams(init, repr, eq, order, unsafe_hash, frozen))
805  
806      # Find our base classes in reverse MRO order, and exclude
807      # ourselves.  In reversed order so that more derived classes
808      # override earlier field definitions in base classes.  As long as
809      # we're iterating over them, see if any are frozen.
810      any_frozen_base = False
811      has_dataclass_bases = False
812      for b in cls.__mro__[-1:0:-1]:
813          # Only process classes that have been processed by our
814          # decorator.  That is, they have a _FIELDS attribute.
815          base_fields = getattr(b, _FIELDS, None)
816          if base_fields:
817              has_dataclass_bases = True
818              for f in base_fields.values():
819                  fields[f.name] = f
820              if getattr(b, _PARAMS).frozen:
821                  any_frozen_base = True
822  
823      # Annotations that are defined in this class (not in base
824      # classes).  If __annotations__ isn't present, then this class
825      # adds no new annotations.  We use this to compute fields that are
826      # added by this class.
827      #
828      # Fields are found from cls_annotations, which is guaranteed to be
829      # ordered.  Default values are from class attributes, if a field
830      # has a default.  If the default value is a Field(), then it
831      # contains additional info beyond (and possibly including) the
832      # actual default value.  Pseudo-fields ClassVars and InitVars are
833      # included, despite the fact that they're not real fields.  That's
834      # dealt with later.
835      cls_annotations = cls.__dict__.get("__annotations__", {})
836  
837      # Now find fields in our class.  While doing so, validate some
838      # things, and set the default values (as class attributes) where
839      # we can.
840      cls_fields = [_get_field(cls, name, type) for name, type in cls_annotations.items()]
841      for f in cls_fields:
842          fields[f.name] = f
843  
844          # If the class attribute (which is the default value for this
845          # field) exists and is of type 'Field', replace it with the
846          # real default.  This is so that normal class introspection
847          # sees a real default value, not a Field.
848          if isinstance(getattr(cls, f.name, None), Field):
849              if f.default is MISSING:
850                  # If there's no default, delete the class attribute.
851                  # This happens if we specify field(repr=False), for
852                  # example (that is, we specified a field object, but
853                  # no default value).  Also if we're using a default
854                  # factory.  The class attribute should not be set at
855                  # all in the post-processed class.
856                  delattr(cls, f.name)
857              else:
858                  setattr(cls, f.name, f.default)
859  
860      # Do we have any Field members that don't also have annotations?
861      for name, value in cls.__dict__.items():
862          if isinstance(value, Field) and not name in cls_annotations:
863              raise TypeError(f"{name!r} is a field but has no type annotation")
864  
865      # Check rules that apply if we are derived from any dataclasses.
866      if has_dataclass_bases:
867          # Raise an exception if any of our bases are frozen, but we're not.
868          if any_frozen_base and not frozen:
869              raise TypeError("cannot inherit non-frozen dataclass from a " "frozen one")
870  
871          # Raise an exception if we're frozen, but none of our bases are.
872          if not any_frozen_base and frozen:
873              raise TypeError("cannot inherit frozen dataclass from a " "non-frozen one")
874  
875      # Remember all of the fields on our class (including bases).  This
876      # also marks this class as being a dataclass.
877      setattr(cls, _FIELDS, fields)
878  
879      # Was this class defined with an explicit __hash__?  Note that if
880      # __eq__ is defined in this class, then python will automatically
881      # set __hash__ to None.  This is a heuristic, as it's possible
882      # that such a __hash__ == None was not auto-generated, but it
883      # close enough.
884      class_hash = cls.__dict__.get("__hash__", MISSING)
885      has_explicit_hash = not (
886          class_hash is MISSING or (class_hash is None and "__eq__" in cls.__dict__)
887      )
888  
889      # If we're generating ordering methods, we must be generating the
890      # eq methods.
891      if order and not eq:
892          raise ValueError("eq must be true if order is true")
893  
894      if init:
895          # Does this class have a post-init function?
896          has_post_init = hasattr(cls, _POST_INIT_NAME)
897  
898          # Include InitVars and regular fields (so, not ClassVars).
899          flds = [f for f in fields.values() if f._field_type in (_FIELD, _FIELD_INITVAR)]
900          _set_new_attribute(
901              cls,
902              "__init__",
903              _init_fn(
904                  flds,
905                  frozen,
906                  has_post_init,
907                  # The name to use for the "self"
908                  # param in __init__.  Use "self"
909                  # if possible.
910                  "__dataclass_self__" if "self" in fields else "self",
911              ),
912          )
913  
914      # Get the fields as a list, and include only real fields.  This is
915      # used in all of the following methods.
916      field_list = [f for f in fields.values() if f._field_type is _FIELD]
917  
918      if repr:
919          flds = [f for f in field_list if f.repr]
920          _set_new_attribute(cls, "__repr__", _repr_fn(flds))
921  
922      if eq:
923          # Create _eq__ method.  There's no need for a __ne__ method,
924          # since python will call __eq__ and negate it.
925          flds = [f for f in field_list if f.compare]
926          self_tuple = _tuple_str("self", flds)
927          other_tuple = _tuple_str("other", flds)
928          _set_new_attribute(cls, "__eq__", _cmp_fn("__eq__", "==", self_tuple, other_tuple))
929  
930      if order:
931          # Create and set the ordering methods.
932          flds = [f for f in field_list if f.compare]
933          self_tuple = _tuple_str("self", flds)
934          other_tuple = _tuple_str("other", flds)
935          for name, op in [("__lt__", "<"), ("__le__", "<="), ("__gt__", ">"), ("__ge__", ">=")]:
936              if _set_new_attribute(cls, name, _cmp_fn(name, op, self_tuple, other_tuple)):
937                  raise TypeError(
938                      f"Cannot overwrite attribute {name} "
939                      f"in class {cls.__name__}. Consider using "
940                      "functools.total_ordering"
941                  )
942  
943      if frozen:
944          for fn in _frozen_get_del_attr(cls, field_list):
945              if _set_new_attribute(cls, fn.__name__, fn):
946                  raise TypeError(
947                      f"Cannot overwrite attribute {fn.__name__} " f"in class {cls.__name__}"
948                  )
949  
950      # Decide if/how we're going to create a hash function.
951      hash_action = _hash_action[bool(unsafe_hash), bool(eq), bool(frozen), has_explicit_hash]
952      if hash_action:
953          # No need to call _set_new_attribute here, since by the time
954          # we're here the overwriting is unconditional.
955          cls.__hash__ = hash_action(cls, field_list)
956  
957      if not getattr(cls, "__doc__"):
958          # Create a class doc-string.
959          cls.__doc__ = cls.__name__ + str(inspect.signature(cls)).replace(" -> None", "")
960  
961      return cls
962

964  # _cls should never be specified by keyword, so start it with an
965  # underscore.  The presence of _cls is used to detect if this
966  # decorator is being called with parameters or not.
967  def dataclass(
968      _cls=None, *, init=True, repr=True, eq=True, order=False, unsafe_hash=False, frozen=False
969  ):
970      """Returns the same class as was passed in, with dunder methods 
971      added based on the fields defined in the class. 
972      Examines PEP 526 __annotations__ to determine fields. 
973      If init is true, an __init__() method is added to the class. If 
974      repr is true, a __repr__() method is added. If order is true, rich 
975      comparison dunder methods are added. If unsafe_hash is true, a 
976      __hash__() method function is added. If frozen is true, fields may 
977      not be assigned to after instance creation. 
978      """
979  
980      def wrap(cls):
981          return _process_class(cls, init, repr, eq, order, unsafe_hash, frozen)
982  
983      # See if we're being called as @dataclass or @dataclass().
984      if _cls is None:
985          # We're called with parens.
986          return wrap
987  
988      # We're called as @dataclass without parens.
989      return wrap(_cls)

992  def fields(class_or_instance):
993      """Return a tuple describing the fields of this dataclass. 
994      Accepts a dataclass or an instance of one. Tuple elements are of 
995      type Field. 
996      """
997  
998      # Might it be worth caching this, per class?
999      try:
1000         fields = getattr(class_or_instance, _FIELDS)
1001     except AttributeError:
1002         raise TypeError("must be called with a dataclass type or instance")
1003 
1004     # Exclude pseudo-fields.  Note that fields is sorted by insertion
1005     # order, so the order of the tuple is as the fields were defined.
1006     return tuple(f for f in fields.values() if f._field_type is _FIELD)

1020 def asdict(obj, *, dict_factory=dict):
1021     """Return the fields of a dataclass instance as a new dictionary mapping 
1022     field names to field values. 
1023     Example usage: 
1024       @dataclass 
1025       class C: 
1026           x: int 
1027           y: int 
1028       c = C(1, 2) 
1029       assert asdict(c) == {'x': 1, 'y': 2} 
1030     If given, 'dict_factory' will be used instead of built-in dict. 
1031     The function applies recursively to field values that are 
1032     dataclass instances. This will also look into built-in containers: 
1033     tuples, lists, and dicts. 
1034     """
1035     if not _is_dataclass_instance(obj):
1036         raise TypeError("asdict() should be called on dataclass instances")
1037     return _asdict_inner(obj, dict_factory)
1038 
1039 
1040 def _asdict_inner(obj, dict_factory):
1041     if _is_dataclass_instance(obj):
1042         result = []
1043         for f in fields(obj):
1044             value = _asdict_inner(getattr(obj, f.name), dict_factory)
1045             result.append((f.name, value))
1046         return dict_factory(result)
1047     elif isinstance(obj, (list, tuple)):
1048         return type(obj)(_asdict_inner(v, dict_factory) for v in obj)
1049     elif isinstance(obj, dict):
1050         return type(obj)(
1051             (_asdict_inner(k, dict_factory), _asdict_inner(v, dict_factory)) for k, v in obj.items()
1052         )
1053     else:
1054         return copy.deepcopy(obj)

1073         raise TypeError("astuple() should be called on dataclass instances")

1095 def make_dataclass(
1096     cls_name,
1097     fields,
1098     *,
1099     bases=(),
1100     namespace=None,
1101     init=True,
1102     repr=True,
1103     eq=True,
1104     order=False,
1105     unsafe_hash=False,
1106     frozen=False,
1107 ):
1108     """Return a new dynamically created dataclass. 
1109     The dataclass name will be 'cls_name'.  'fields' is an iterable 
1110     of either (name), (name, type) or (name, type, Field) objects. If type is 
1111     omitted, use the string 'typing.Any'.  Field objects are created by 
1112     the equivalent of calling 'field(name, type [, Field-info])'. 
1113       C = make_dataclass('C', ['x', ('y', int), ('z', int, field(init=False))], bases=(Base,)) 
1114     is equivalent to: 
1115       @dataclass 
1116       class C(Base): 
1117           x: 'typing.Any' 
1118           y: int 
1119           z: int = field(init=False) 
1120     For the bases and namespace parameters, see the builtin type() function. 
1121     The parameters init, repr, eq, order, unsafe_hash, and frozen are passed to 
1122     dataclass(). 
1123     """
1124 
1125     if namespace is None:
1126         namespace = {}
1127     else:
1128         # Copy namespace since we're going to mutate it.
1129         namespace = namespace.copy()
1130 
1131     # While we're looking through the field names, validate that they
1132     # are identifiers, are not keywords, and not duplicates.
1133     seen = set()
1134     anns = {}
1135     for item in fields:
1136         if isinstance(item, str):
1137             name = item
1138             tp = "typing.Any"
1139         elif len(item) == 2:
1140             (
1141                 name,
1142                 tp,
1143             ) = item
1144         elif len(item) == 3:
1145             name, tp, spec = item
1146             namespace[name] = spec
1147         else:
1148             raise TypeError(f"Invalid field: {item!r}")
1149 
1150         if not isinstance(name, str) or not name.isidentifier():
1151             raise TypeError(f"Field names must be valid identifers: {name!r}")
1152         if keyword.iskeyword(name):
1153             raise TypeError(f"Field names must not be keywords: {name!r}")
1154         if name in seen:
1155             raise TypeError(f"Field name duplicated: {name!r}")
1156 
1157         seen.add(name)
1158         anns[name] = tp
1159 
1160     namespace["__annotations__"] = anns
1161     # We use `types.new_class()` instead of simply `type()` to allow dynamic creation
1162     # of generic dataclassses.
1163     cls = types.new_class(cls_name, bases, {}, lambda ns: ns.update(namespace))
1164     return dataclass(
1165         cls, init=init, repr=repr, eq=eq, order=order, unsafe_hash=unsafe_hash, frozen=frozen
1166     )

1169 def replace(obj, **changes):
1170     """Return a new object replacing specified fields with new values. 
1171     This is especially useful for frozen classes.  Example usage: 
1172       @dataclass(frozen=True) 
1173       class C: 
1174           x: int 
1175           y: int 
1176       c = C(1, 2) 
1177       c1 = replace(c, x=3) 
1178       assert c1.x == 3 and c1.y == 2 
1179     """
1180 
1181     # We're going to mutate 'changes', but that's okay because it's a
1182     # new dict, even if called with 'replace(obj, **my_changes)'.
1183 
1184     if not _is_dataclass_instance(obj):
1185         raise TypeError("replace() should be called on dataclass instances")
1186 
1187     # It's an error to have init=False fields in 'changes'.
1188     # If a field is not in 'changes', read its value from the provided obj.
1189 
1190     for f in getattr(obj, _FIELDS).values():
1191         if not f.init:
1192             # Error if this field is specified in changes.
1193             if f.name in changes:
1194                 raise ValueError(
1195                     f"field {f.name} is declared with "
1196                     "init=False, it cannot be specified with "
1197                     "replace()"
1198                 )
1199             continue
1200 
1201         if f.name not in changes:
1202             changes[f.name] = getattr(obj, f.name)
1203 
1204     # Create the new object, which calls __init__() and
1205     # __post_init__() (if defined), using all of the init fields we've
1206     # added and/or left in 'changes'.  If there are values supplied in
1207     # changes that aren't fields, this will correctly raise a
1208     # TypeError.
1209     return obj.__class__(**changes)
1210
