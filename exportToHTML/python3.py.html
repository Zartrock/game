56   def _is_old_octal(literal):
57       if _ZERO.match(literal):
58           return False
59       if re.match(r"0\d+", literal):
60           try:
61               int(literal, 8)
62           except ValueError:
63               return False
64           return True
65       return None

68   def _inferred_value_is_dict(value):
69       if isinstance(value, astroid.Dict):
70           return True
71       return isinstance(value, astroid.Instance) and "dict" in value.basenames

109  def _in_iterating_context(node):
110      """Check if the node is being used as an iterator. 
111   
112      Definition is taken from lib2to3.fixer_util.in_special_context(). 
113      """
114      parent = node.parent
115      # Since a call can't be the loop variant we only need to know if the node's
116      # parent is a 'for' loop to know it's being used as the iterator for the
117      # loop.
118      if isinstance(parent, astroid.For):
119          return True
120      # Need to make sure the use of the node is in the iterator part of the
121      # comprehension.
122      if isinstance(parent, astroid.Comprehension):
123          if parent.iter == node:
124              return True
125      # Various built-ins can take in an iterable or list and lead to the same
126      # value.
127      elif isinstance(parent, astroid.Call):
128          if isinstance(parent.func, astroid.Name):
129              if parent.func.name in _ACCEPTS_ITERATOR:
130                  return True
131          elif isinstance(parent.func, astroid.Attribute):
132              if parent.func.attrname in ATTRIBUTES_ACCEPTS_ITERATOR:
133                  return True
134  
135          inferred = utils.safe_infer(parent.func)
136          if inferred:
137              if inferred.qname() in _BUILTIN_METHOD_ACCEPTS_ITERATOR:
138                  return True
139              root = inferred.root()
140              if root and root.name == "itertools":
141                  return True
142      # If the call is in an unpacking, there's no need to warn,
143      # since it can be considered iterating.
144      elif isinstance(parent, astroid.Assign) and isinstance(
145          parent.targets[0], (astroid.List, astroid.Tuple)
146      ):
147          if len(parent.targets[0].elts) > 1:
148              return True
149      # If the call is in a containment check, we consider that to
150      # be an iterating context
151      elif (
152          isinstance(parent, astroid.Compare)
153          and len(parent.ops) == 1
154          and parent.ops[0][0] in ["in", "not in"]
155      ):
156          return True
157      # Also if it's an `yield from`, that's fair
158      elif isinstance(parent, astroid.YieldFrom):
159          return True
160      if isinstance(parent, astroid.Starred):
161          return True
162      return False

165  def _is_conditional_import(node):
166      """Checks if an import node is in the context of a conditional. 
167      """
168      parent = node.parent
169      return isinstance(
170          parent, (astroid.TryExcept, astroid.ExceptHandler, astroid.If, astroid.IfExp)
171      )

1160     def visit_call(self, node):
1161         self._check_cmp_argument(node)

1163         if isinstance(node.func, astroid.Attribute):
1164             inferred_types = set()
1165 
1166             try:
1167                 for inferred_receiver in _infer_if_relevant_attr(
1168                     node.func, self._relevant_call_attrs
1169                 ):
1170                     if inferred_receiver is astroid.Uninferable:
1171                         continue
1172                     inferred_types.add(inferred_receiver)
1173                     if isinstance(inferred_receiver, astroid.Module):
1174                         self._warn_if_deprecated(

1178                             report_on_modules=False,

1181                         _inferred_value_is_dict(inferred_receiver)

1186                             self.add_message(checker, node=node)
1187             except astroid.InferenceError:
1188                 pass
1189             if node.args:
1190                 is_str_confidence = self._could_be_string(inferred_types)
1191                 if is_str_confidence:
1192                     if (
1193                         node.func.attrname in ("encode", "decode")
1194                         and len(node.args) >= 1
1195                         and node.args[0]
1196                     ):
1197                         first_arg = node.args[0]
1198                         self._validate_encoding(first_arg, node)
1199                     if (
1200                         node.func.attrname == "translate"

1203                         and self._is_constant_string_or_name(node.args[1])

1206                         #
1207                         # foo.translate(None, 'abc123')
1208                         #
1209                         # or
1210                         #
1211                         # foo.translate(None, some_variable)
1212                         #
1213                         # This check is somewhat broad and _may_ have some false positives, but
1214                         # after checking several large codebases it did not have any false
1215                         # positives while finding several real issues.  This call pattern seems
1216                         # rare enough that the trade off is worth it.
1217                         self.add_message(
1218                             "deprecated-str-translate-call",
1219                             node=node,
1220                             confidence=is_str_confidence,
1221                         )
1222                 return
1223             if node.keywords:
1224                 return
1225             if node.func.attrname == "next":
1226                 self.add_message("next-method-called", node=node)
1227             elif node.func.attrname in ("iterkeys", "itervalues", "iteritems"):
1228                 self.add_message("dict-iter-method", node=node)
1229             elif node.func.attrname in ("viewkeys", "viewvalues", "viewitems"):
1230                 self.add_message("dict-view-method", node=node)

1232             found_node = node.func.lookup(node.func.name)[0]
1233             if _is_builtin(found_node):
1234                 if node.func.name in ("filter", "map", "range", "zip"):
1235                     if not _in_iterating_context(node):
1236                         checker = "{}-builtin-not-iterating".format(node.func.name)
1237                         self.add_message(checker, node=node)
1238                 elif node.func.name == "open" and node.keywords:
1239                     kwargs = node.keywords
1240                     for kwarg in kwargs or []:

1298     @utils.check_messages("unpacking-in-except", "comprehension-escape")

1300         """Visit an except handler block and check for exception unpacking."""

1302         def _is_used_in_except_block(node, block):
1303             current = node
1304             while current and current is not block:
1305                 current = current.parent
1306             return current is not None
1307 
1308         if isinstance(node.name, (astroid.Tuple, astroid.List)):
1309             self.add_message("unpacking-in-except", node=node)
1310             return

1312         if not node.name:
1313             return

1315         # Find any names
1316         scope = node.parent.scope()
1317         scope_names = scope.nodes_of_class(astroid.Name, skip_klass=astroid.FunctionDef)
1318         scope_names = list(scope_names)
1319         potential_leaked_names = [
1320             scope_name
1321             for scope_name in scope_names
1322             if scope_name.name == node.name.name
1323             and scope_name.lineno > node.lineno
1324             and not _is_used_in_except_block(scope_name, node)
1325         ]
1326         reassignments_for_same_name = {
1327             assign_name.lineno
1328             for assign_name in scope.nodes_of_class(
1329                 astroid.AssignName, skip_klass=astroid.FunctionDef
1330             )
1331             if assign_name.name == node.name.name
1332         }
1333         for leaked_name in potential_leaked_names:
1334             if any(
1335                 node.lineno < elem < leaked_name.lineno
1336                 for elem in reassignments_for_same_name
1337             ):
1338                 continue
1339             self.add_message("exception-escape", node=leaked_name)
1340 
1341     @utils.check_messages("backtick")
1342     def visit_repr(self, node):
1343         self.add_message("backtick", node=node)

1345     @utils.check_messages("raising-string", "old-raise-syntax")
1346     def visit_raise(self, node):
1347         """Visit a raise statement and check for raising 
1348         strings or old-raise-syntax. 
1349         """
1350 
1351         # Ignore empty raise.
1352         if node.exc is None:
1353             return
1354         expr = node.exc
1355         if self._check_raise_value(node, expr):
1356             return
1357         try:
1358             value = next(astroid.unpack_infer(expr))
1359         except astroid.InferenceError:
1360             return
1361         self._check_raise_value(node, value)
1362 
1363     def _check_raise_value(self, node, expr):
1364         if isinstance(expr, astroid.Const):
1365             value = expr.value
1366             if isinstance(value, str):
1367                 self.add_message("raising-string", node=node)
1368                 return True
1369         return None
