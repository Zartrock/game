279      def test_spritecollideany__with_collided_callback(self):
280  
281          # pygame.sprite.spritecollideany(sprite, group) -> sprite
282          # finds any sprites that collide
283  
284          # collided is a callback function used to calculate if
285          # two sprites are colliding. it should take two sprites
286          # as values, and return a bool value indicating if
287          # they are colliding.
288  
289          # This collision test can be faster than pygame.sprite.spritecollide()
290          # since it has less work to do.
291  
292          arg_dict_a = {}
293          arg_dict_b = {}
294          return_container = [True]
295  
296          # This function is configurable using the mutable default arguments!
297          def collided_callback(
298              spr_a,
299              spr_b,
300              arg_dict_a=arg_dict_a,
301              arg_dict_b=arg_dict_b,
302              return_container=return_container,
303          ):
304  
305              count = arg_dict_a.get(spr_a, 0)
306              arg_dict_a[spr_a] = 1 + count
307  
308              count = arg_dict_b.get(spr_b, 0)
309              arg_dict_b[spr_b] = 1 + count
310  
311              return return_container[0]
312  
313          # This should return a sprite from self.ag2 because the callback
314          # function (collided_callback()) currently returns True.
315          expected_sprite_choices = self.ag2.sprites()
316          collided_sprite = sprite.spritecollideany(self.s1, self.ag2, collided_callback)
317  
318          self.assertIn(collided_sprite, expected_sprite_choices)
319  
320          # The callback function should have been called only once, so self.s1
321          # should have only been passed as an argument once
322          self.assertEqual(len(arg_dict_a), 1)
323          self.assertEqual(arg_dict_a[self.s1], 1)
324  
325          # The callback function should have been called only once, so self.s2
326          # exclusive-or self.s3 should have only been passed as an argument
327          # once
328          self.assertEqual(len(arg_dict_b), 1)
329          self.assertEqual(list(arg_dict_b.values())[0], 1)
330          self.assertTrue(self.s2 in arg_dict_b or self.s3 in arg_dict_b)
331  
332          arg_dict_a.clear()
333          arg_dict_b.clear()
334          return_container[0] = False
335  
336          # This should return None because the callback function
337          # (collided_callback()) currently returns False.
338          collided_sprite = sprite.spritecollideany(self.s1, self.ag2, collided_callback)
339  
340          self.assertIsNone(collided_sprite)
341  
342          # The callback function should have been called as many times as
343          # there are sprites in self.ag2
344          self.assertEqual(len(arg_dict_a), 1)
345          self.assertEqual(arg_dict_a[self.s1], len(self.ag2))
346          self.assertEqual(len(arg_dict_b), len(self.ag2))
347  
348          # Each sprite in self.ag2 should be called once.
349          for s in self.ag2:
350              self.assertEqual(arg_dict_b[s], 1)

468      def test_collide_rect(self):
469          # Test colliding - some edges touching
470          self.assertTrue(pygame.sprite.collide_rect(self.s1, self.s2))
471          self.assertTrue(pygame.sprite.collide_rect(self.s2, self.s1))
472  
473          # Test colliding - all edges touching
474          self.s2.rect.center = self.s3.rect.center
475  
476          self.assertTrue(pygame.sprite.collide_rect(self.s2, self.s3))
477          self.assertTrue(pygame.sprite.collide_rect(self.s3, self.s2))
478  
479          # Test colliding - no edges touching
480          self.s2.rect.inflate_ip(10, 10)
481  
482          self.assertTrue(pygame.sprite.collide_rect(self.s2, self.s3))
483          self.assertTrue(pygame.sprite.collide_rect(self.s3, self.s2))
484  
485          # Test colliding - some edges intersecting
486          self.s2.rect.center = (self.s1.rect.right, self.s1.rect.bottom)
487  
488          self.assertTrue(pygame.sprite.collide_rect(self.s1, self.s2))
489          self.assertTrue(pygame.sprite.collide_rect(self.s2, self.s1))
490  
491          # Test not colliding
492          self.assertFalse(pygame.sprite.collide_rect(self.s1, self.s3))
493          self.assertFalse(pygame.sprite.collide_rect(self.s3, self.s1))

1121     def _nondirty_intersections_redrawn(self, use_source_rect=False):
1122         # Helper method to ensure non-dirty sprites are redrawn correctly.
1123         #
1124         # Parameters:
1125         #     use_source_rect - allows non-dirty sprites to be tested
1126         #         with (True) and without (False) a source_rect
1127         #
1128         # This test was written to reproduce the behavior seen in issue #898.
1129         # A non-dirty sprite (using source_rect) was being redrawn incorrectly
1130         # after a dirty sprite intersected with it.
1131         #
1132         # This test does the following.
1133         # 1. Creates a surface filled with white. Also creates an image_source
1134         #    with a default fill color of yellow and adds 2 images to it
1135         #    (red and blue rectangles).
1136         # 2. Creates 2 DirtySprites (red_sprite and blue_sprite) using the
1137         #    image_source and adds them to a LayeredDirty group.
1138         # 3. Moves the red_sprite and calls LayeredDirty.draw(surface) a few
1139         #    times.
1140         # 4. Checks to make sure the sprites were redrawn correctly.
1141         RED = pygame.Color("red")
1142         BLUE = pygame.Color("blue")
1143         WHITE = pygame.Color("white")
1144         YELLOW = pygame.Color("yellow")
1145 
1146         surface = pygame.Surface((60, 80))
1147         surface.fill(WHITE)
1148         start_pos = (10, 10)
1149 
1150         # These rects define each sprite's image area in the image_source.
1151         red_sprite_source = pygame.Rect((45, 0), (5, 4))
1152         blue_sprite_source = pygame.Rect((0, 40), (20, 10))
1153 
1154         # Create a source image/surface.
1155         image_source = pygame.Surface((50, 50))
1156         image_source.fill(YELLOW)
1157         image_source.fill(RED, red_sprite_source)
1158         image_source.fill(BLUE, blue_sprite_source)
1159 
1160         # The blue_sprite is stationary and will not reset its dirty flag. It
1161         # will be the non-dirty sprite in this test. Its values are dependent
1162         # on the use_source_rect flag.
1163         blue_sprite = pygame.sprite.DirtySprite(self.LG)
1164 
1165         if use_source_rect:
1166             blue_sprite.image = image_source
1167             # The rect is a bit smaller than the source_rect to make sure
1168             # LayeredDirty.draw() is using the correct dimensions.
1169             blue_sprite.rect = pygame.Rect(
1170                 start_pos, (blue_sprite_source.w - 7, blue_sprite_source.h - 7)
1171             )
1172             blue_sprite.source_rect = blue_sprite_source
1173             start_x, start_y = blue_sprite.rect.topleft
1174             end_x = start_x + blue_sprite.source_rect.w
1175             end_y = start_y + blue_sprite.source_rect.h
1176         else:
1177             blue_sprite.image = image_source.subsurface(blue_sprite_source)
1178             blue_sprite.rect = pygame.Rect(start_pos, blue_sprite_source.size)
1179             start_x, start_y = blue_sprite.rect.topleft
1180             end_x, end_y = blue_sprite.rect.bottomright
1181 
1182         # The red_sprite is moving and will always be dirty.
1183         red_sprite = pygame.sprite.DirtySprite(self.LG)
1184         red_sprite.image = image_source
1185         red_sprite.rect = pygame.Rect(start_pos, red_sprite_source.size)
1186         red_sprite.source_rect = red_sprite_source
1187         red_sprite.dirty = 2
1188 
1189         # Draw the red_sprite as it moves a few steps.
1190         for _ in range(4):
1191             red_sprite.rect.move_ip(2, 1)
1192 
1193             # This is the method being tested.
