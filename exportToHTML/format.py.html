184  def _last_token_on_line_is(tokens, line_end, token):
185      return (
186          line_end > 0
187          and tokens.token(line_end - 1) == token
188          or line_end > 1
189          and tokens.token(line_end - 2) == token
190          and tokens.type(line_end - 1) == tokenize.COMMENT
191      )

343      def new_line(self, tokens, line_end, line_start):
344          """a new line has been encountered, process it if necessary"""
345          if _last_token_on_line_is(tokens, line_end, ";"):
346              self.add_message("unnecessary-semicolon", line=tokens.start_line(line_end))
347  
348          line_num = tokens.start_line(line_start)
349          line = tokens.line(line_start)
350          if tokens.type(line_start) not in _JUNK_TOKENS:
351              self._lines[line_num] = line.split("\n")[0]
352          self.check_lines(line, line_num)

357      def _check_keyword_parentheses(self, tokens: List[TokenInfo], start: int) -> None:
358          """Check that there are not unnecessary parens after a keyword. 
359   
360          Parens are unnecessary if there is exactly one balanced outer pair on a 
361          line, and it is followed by a colon, and contains no commas (i.e. is not a 
362          tuple). 
363   
364          Args: 
365          tokens: list of Tokens; the entire list of Tokens. 
366          start: int; the position of the keyword in the token list. 
367          """
368          # If the next token is not a paren, we're fine.
369          if self._bracket_stack[-1] == ":" and tokens[start].string == "for":
370              self._bracket_stack.pop()
371          if tokens[start + 1].string != "(":
372              return
373          found_and_or = False
374          contains_walrus_operator = False
375          walrus_operator_depth = 0
376          depth = 0
377          keyword_token = str(tokens[start].string)
378          line_num = tokens[start].start[0]
379          for i in range(start, len(tokens) - 1):
380              token = tokens[i]
381  
382              # If we hit a newline, then assume any parens were for continuation.
383              if token.type == tokenize.NL:
384                  return
385              # Since the walrus operator doesn't exist below python3.8, the tokenizer
386              # generates independent tokens
387              if (
388                  token.string == ":="  #  <-- python3.8+ path
389                  or token.string + tokens[i + 1].string == ":="
390              ):
391                  contains_walrus_operator = True
392                  walrus_operator_depth = depth
393              if token.string == "(":
394                  depth += 1
395              elif token.string == ")":
396                  depth -= 1
397                  if depth:
398                      continue
399                  # ')' can't happen after if (foo), since it would be a syntax error.
400                  if tokens[i + 1].string in (":", ")", "]", "}", "in") or tokens[
401                      i + 1
402                  ].type in (tokenize.NEWLINE, tokenize.ENDMARKER, tokenize.COMMENT):
403                      # The empty tuple () is always accepted.
404                      if contains_walrus_operator and walrus_operator_depth - 1 == depth:
405                          # Reset variable for possible following expressions
406                          contains_walrus_operator = False
407                          continue
408                      if i == start + 2:
409                          return
410                      if keyword_token == "not":
411                          if not found_and_or:
412                              self.add_message(
413                                  "superfluous-parens", line=line_num, args=keyword_token
414                              )
415                      elif keyword_token in ("return", "yield"):
416                          self.add_message(
417                              "superfluous-parens", line=line_num, args=keyword_token
418                          )
419                      elif not found_and_or:
420                          self.add_message(
421                              "superfluous-parens", line=line_num, args=keyword_token
422                          )
423                  return
424              elif depth == 1:
425                  # This is a tuple, which is always acceptable.
426                  if token[1] == ",":
427                      return
428                  # 'and' and 'or' are the only boolean operators with lower precedence
429                  # than 'not', so parens are only required when they are found.
430                  if token[1] in ("and", "or"):
431                      found_and_or = True
432                  # A yield inside an expression must always be in parentheses,
433                  # quit early without error.
434                  elif token[1] == "yield":
435                      return
436                  # A generator expression always has a 'for' token in it, and
437                  # the 'for' token is only legal inside parens when it is in a
438                  # generator expression.  The parens are necessary here, so bail
439                  # without an error.
440                  elif token[1] == "for":
441                      return

443      def _prepare_token_dispatcher(self):
444          dispatch = {}
445          for tokens, handler in [
446              (_KEYWORD_TOKENS, self._check_keyword_parentheses),
447          ]:
448              for token in tokens:
449                  dispatch[token] = handler
450          return dispatch

452      def process_tokens(self, tokens):
453          """process tokens and search for : 
454   
455           _ too long lines (i.e. longer than <max_chars>) 
456           _ optionally bad construct (if given, bad_construct must be a compiled 
457             regular expression). 
458          """
459          self._bracket_stack = [None]
460          indents = [0]
461          check_equal = False
462          line_num = 0
463          self._lines = {}
464          self._visited_lines = {}
465          token_handlers = self._prepare_token_dispatcher()
466          self._last_line_ending = None
467          last_blank_line_num = 0
468          for idx, (tok_type, token, start, _, line) in enumerate(tokens):
469              if start[0] != line_num:
470                  line_num = start[0]
471                  # A tokenizer oddity: if an indented line contains a multi-line
472                  # docstring, the line member of the INDENT token does not contain
473                  # the full line; therefore we check the next token on the line.
474                  if tok_type == tokenize.INDENT:
475                      self.new_line(TokenWrapper(tokens), idx - 1, idx + 1)
476                  else:
477                      self.new_line(TokenWrapper(tokens), idx - 1, idx)
478  
479              if tok_type == tokenize.NEWLINE:
480                  # a program statement, or ENDMARKER, will eventually follow,
481                  # after some (possibly empty) run of tokens of the form
482                  #     (NL | COMMENT)* (INDENT | DEDENT+)?
483                  # If an INDENT appears, setting check_equal is wrong, and will
484                  # be undone when we see the INDENT.
485                  check_equal = True
486                  self._check_line_ending(token, line_num)
487              elif tok_type == tokenize.INDENT:
488                  check_equal = False
489                  self.check_indent_level(token, indents[-1] + 1, line_num)
490                  indents.append(indents[-1] + 1)
491              elif tok_type == tokenize.DEDENT:
492                  # there's nothing we need to check here!  what's important is
493                  # that when the run of DEDENTs ends, the indentation of the
494                  # program statement (or ENDMARKER) that triggered the run is
495                  # equal to what's left at the top of the indents stack
496                  check_equal = True
497                  if len(indents) > 1:
498                      del indents[-1]
499              elif tok_type == tokenize.NL:
500                  if not line.strip("\r\n"):
501                      last_blank_line_num = line_num
502              elif tok_type not in (tokenize.COMMENT, tokenize.ENCODING):
503                  # This is the first concrete token following a NEWLINE, so it
504                  # must be the first token of the next program statement, or an
505                  # ENDMARKER; the "line" argument exposes the leading whitespace
506                  # for this statement; in the case of ENDMARKER, line is an empty
507                  # string, so will properly match the empty string with which the
508                  # "indents" stack was seeded
509                  if check_equal:
510                      check_equal = False
511                      self.check_indent_level(line, indents[-1], line_num)
512  
513              if tok_type == tokenize.NUMBER and token.endswith("l"):
514                  self.add_message("lowercase-l-suffix", line=line_num)
515  
516              try:
517                  handler = token_handlers[token]
518              except KeyError:
519                  pass
520              else:
521                  handler(tokens, idx)
522  
523          line_num -= 1  # to be ok with "wc -l"
524          if line_num > self.config.max_module_lines:
525              # Get the line where the too-many-lines (or its message id)
526              # was disabled or default to 1.
527              message_definition = self.linter.msgs_store.get_message_definitions(
528                  "too-many-lines"
529              )[0]
530              names = (message_definition.msgid, "too-many-lines")
531              line = next(filter(None, map(self.linter._pragma_lineno.get, names)), 1)
532              self.add_message(
533                  "too-many-lines",
534                  args=(line_num, self.config.max_module_lines),
535                  line=line,
536              )
537  
538          # See if there are any trailing lines.  Do not complain about empty
539          # files like __init__.py markers.
540          if line_num == last_blank_line_num and line_num > 0:
541              self.add_message("trailing-newlines", line=line_num)
542  
543      def _check_line_ending(self, line_ending, line_num):
544          # check if line endings are mixed
545          if self._last_line_ending is not None:
546              # line_ending == "" indicates a synthetic newline added at
547              # the end of a file that does not, in fact, end with a
548              # newline.
549              if line_ending and line_ending != self._last_line_ending:
550                  self.add_message("mixed-line-endings", line=line_num)
551  
552          self._last_line_ending = line_ending
553  
554          # check if line ending is as expected
555          expected = self.config.expected_line_ending_format
556          if expected:
557              # reduce multiple \n\n\n\n to one \n
558              line_ending = reduce(lambda x, y: x + y if x != y else x, line_ending, "")
559              line_ending = "LF" if line_ending == "\n" else "CRLF"
560              if line_ending != expected:
561                  self.add_message(
562                      "unexpected-line-ending-format",
563                      args=(line_ending, expected),
564                      line=line_num,
565                  )

567      @check_messages("multiple-statements")
568      def visit_default(self, node):
569          """check the node line number and check it if not yet done"""
570          if not node.is_statement:
571              return
572          if not node.root().pure_python:
573              return
574          prev_sibl = node.previous_sibling()
575          if prev_sibl is not None:
576              prev_line = prev_sibl.fromlineno
577          # The line on which a finally: occurs in a try/finally
578          # is not directly represented in the AST. We infer it
579          # by taking the last line of the body and adding 1, which
580          # should be the line of finally:
581          elif (
582              isinstance(node.parent, nodes.TryFinally) and node in node.parent.finalbody
583          ):
584              prev_line = node.parent.body[0].tolineno + 1
585          else:
586              prev_line = node.parent.statement().fromlineno
587          line = node.fromlineno
588          assert line, node
589          if prev_line == line and self._visited_lines.get(line) != 2:
590              self._check_multi_statement_line(node, line)
591              return
592          if line in self._visited_lines:
593              return
594          try:
595              tolineno = node.blockstart_tolineno
596          except AttributeError:
597              tolineno = node.tolineno
598          assert tolineno, node
599          lines = []
600          for line in range(line, tolineno + 1):
601              self._visited_lines[line] = 1
602              try:
603                  lines.append(self._lines[line].rstrip())
604              except KeyError:
605                  lines.append("")

607      def _check_multi_statement_line(self, node, line):
608          """Check for lines containing multiple statements."""
609          # Do not warn about multiple nested context managers
610          # in with statements.
611          if isinstance(node, nodes.With):
612              return
613          # For try... except... finally..., the two nodes
614          # appear to be on the same line due to how the AST is built.
615          if isinstance(node, nodes.TryExcept) and isinstance(
616              node.parent, nodes.TryFinally
617          ):
618              return
619          if (
620              isinstance(node.parent, nodes.If)
621              and not node.parent.orelse
622              and self.config.single_line_if_stmt
623          ):
624              return
625          if (
626              isinstance(node.parent, nodes.ClassDef)
627              and len(node.parent.body) == 1
628              and self.config.single_line_class_stmt
629          ):
630              return
631  
632          # Function overloads that use ``Ellipsis`` are exempted.
633          if isinstance(node, nodes.Expr) and (
634              isinstance(node.value, nodes.Ellipsis)
635              or (isinstance(node.value, nodes.Const) and node.value.value is Ellipsis)
636          ):
637              frame = node.frame()
638              if is_overload_stub(frame) or is_protocol_class(node_frame_class(frame)):
639                  return
640  
641          self.add_message("multiple-statements", node=node)
642          self._visited_lines[line] = 2

644      def check_line_ending(self, line: str, i: int) -> None:
645          """ 
646          Check that the final newline is not missing and that there is no trailing whitespace. 
647          """
648          if not line.endswith("\n"):
649              self.add_message("missing-final-newline", line=i)
650              return
651          # exclude \f (formfeed) from the rstrip
652          stripped_line = line.rstrip("\t\n\r\v ")
653          if line[len(stripped_line) :] not in ("\n", "\r\n"):
654              self.add_message(
655                  "trailing-whitespace", line=i, col_offset=len(stripped_line)
656              )
657  
658      def check_line_length(self, line: str, i: int) -> None:
659          """ 
660          Check that the line length is less than the authorized value 
661          """
662          max_chars = self.config.max_line_length
663          ignore_long_line = self.config.ignore_long_lines
664          line = line.rstrip()
665          if len(line) > max_chars and not ignore_long_line.search(line):
666              self.add_message("line-too-long", line=i, args=(len(line), max_chars))
667  
668      @staticmethod
669      def remove_pylint_option_from_lines(options_pattern_obj) -> str:
670          """ 
671          Remove the `# pylint ...` pattern from lines 
672          """
673          lines = options_pattern_obj.string
674          purged_lines = (
675              lines[: options_pattern_obj.start(1)].rstrip()
676              + lines[options_pattern_obj.end(1) :]
677          )
678          return purged_lines
679  
680      @staticmethod
681      def is_line_length_check_activated(pylint_pattern_match_object) -> bool:
682          """ 
683          Return true if the line length check is activated 
684          """
685          try:
686              for pragma in parse_pragma(pylint_pattern_match_object.group(2)):
687                  if pragma.action == "disable" and "line-too-long" in pragma.messages:
688                      return False
689          except PragmaParserError:
690              # Printing useful information dealing with this error is done in the lint package
691              pass
692          return True
693  
694      @staticmethod
695      def specific_splitlines(lines: str) -> List[str]:
696          """ 
697          Split lines according to universal newlines except those in a specific sets 
698          """
699          unsplit_ends = {
700              "\v",
701              "\x0b",
702              "\f",
703              "\x0c",
704              "\x1c",
705              "\x1d",
706              "\x1e",
707              "\x85",
708              "\u2028",
709              "\u2029",
710          }
711          res = []
712          buffer = ""
713          for atomic_line in lines.splitlines(True):
714              if atomic_line[-1] not in unsplit_ends:
715                  res.append(buffer + atomic_line)
716                  buffer = ""
717              else:
718                  buffer += atomic_line
719          return res
720  
721      def check_lines(self, lines: str, lineno: int) -> None:
722          """ 
723          Check lines have : 
724              - a final newline 
725              - no trailing whitespace 
726              - less than a maximum number of characters 
727          """
728          #  By default, check the line length
729          check_l_length = True
730  
731          # Line length check may be deactivated through `pylint: disable` comment
732          mobj = OPTION_PO.search(lines)
733          if mobj:
734              check_l_length = self.is_line_length_check_activated(mobj)
735              # The 'pylint: disable whatever' should not be taken into account for line length count
736              lines = self.remove_pylint_option_from_lines(mobj)
737  
738          for line in self.specific_splitlines(lines):
739              if check_l_length:
740                  self.check_line_length(line, lineno)
741              self.check_line_ending(line, lineno)
742              lineno += 1

744      def check_indent_level(self, string, expected, line_num):
745          """return the indent level of the string 
746          """
747          indent = self.config.indent_string
748          if indent == "\\t":  # \t is not interpreted in the configuration file
749              indent = "\t"
750          level = 0
751          unit_size = len(indent)
752          while string[:unit_size] == indent:
753              string = string[unit_size:]
754              level += 1
755          suppl = ""
756          while string and string[0] in " \t":
757              suppl += string[0]
758              string = string[1:]
759          if level != expected or suppl:
760              i_type = "spaces"
761              if indent[0] == "\t":
762                  i_type = "tabs"
763              self.add_message(
764                  "bad-indentation",
765                  line=line_num,
766                  args=(level * unit_size + len(suppl), i_type, expected * unit_size),
767              )
