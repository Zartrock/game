6    """Astroid hooks for the UUID module."""
7    
8    
9    from astroid import MANAGER
10   from astroid import nodes
11   
12   
13   def _patch_uuid_class(node):
14       # The .int member is patched using __dict__
15       node.locals["int"] = [nodes.Const(0, parent=node)]
16   
17   
18   MANAGER.register_transform(
19       nodes.ClassDef, _patch_uuid_class, lambda node: node.qname() == "uuid.UUID"
20   )
