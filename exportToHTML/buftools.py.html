2     
3    Export the Exporter and Importer classes. 
4     
5    Class Exporter has configurable shape and strides. Exporter objects 
6    provide a convient target for unit tests on Pygame objects and functions that 
7    import a new buffer interface. 
8     
9    Class Importer imports a buffer interface with the given PyBUF_* flags. 
10   It returns NULL Py_buffer fields as None. The shape, strides, and suboffsets 
11   arrays are returned as tuples of ints. All Py_buffer field properties are

13   with the actual request. The simular Python builtin memoryview currently 
14   does not support configurable PyBUF_* flags.

81       """An object that exports a multi-dimension new buffer interface 
82    
83          The only array operation this type supports is to export a buffer.

140      def __init__(self, shape, format=None, strides=None, readonly=None, itemsize=None):
141          if format is None:
142              format = "B"
143          if readonly is None:
144              readonly = False
145          prefix = ""
146          typecode = ""
147          i = 0
148          if i < len(format):
149              try:
150                  prefix = self.prefixes[format[i]]
151                  i += 1
152              except LookupError:
153                  pass
154          if i < len(format) and format[i] == "1":
155              i += 1
156          if i == len(format) - 1:
157              typecode = format[i]
158          if itemsize is None:
159              try:
160                  itemsize = ctypes.sizeof(self.types[prefix + typecode])
161              except KeyError:
162                  raise ValueError("Unknown item format '" + format + "'")
163          self.readonly = bool(readonly)
164          self.format = format
165          self._format = ctypes.create_string_buffer(format.encode("latin_1"))
166          self.ndim = len(shape)
167          self.itemsize = itemsize
168          self.len = reduce(operator.mul, shape, 1) * self.itemsize
169          self.shape = tuple(shape)
170          self._shape = (ctypes.c_ssize_t * self.ndim)(*self.shape)
171          if strides is None:
172              self._strides = (ctypes.c_ssize_t * self.ndim)()
173              self._strides[self.ndim - 1] = itemsize
174              for i in range(self.ndim - 1, 0, -1):
175                  self._strides[i - 1] = self.shape[i] * self._strides[i]
176              self.strides = tuple(self._strides)
177          elif len(strides) == self.ndim:
178              self.strides = tuple(strides)
179              self._strides = (ctypes.c_ssize_t * self.ndim)(*self.strides)
180          else:
181              raise ValueError("Mismatch in length of strides and shape")
182          buflen = max(d * abs(s) for d, s in zip(self.shape, self.strides))
183          self.buflen = buflen
184          self._buf = (ctypes.c_ubyte * buflen)()
185          offset = sum(
186              (d - 1) * abs(s) for d, s in zip(self.shape, self.strides) if s < 0
187          )
188          self.buf = ctypes.addressof(self._buf) + offset
189  
190      def buffer_info(self):
191          return (addressof(self.buffer), self.shape[0])
192  
193      def tobytes(self):
194          return cast(self.buffer, POINTER(c_char))[0 : self._len]
195  
196      def __len__(self):
197          return self.shape[0]

199      def _get_buffer(self, view, flags):
200          from ctypes import addressof
201  
202          if (flags & PyBUF_WRITABLE) == PyBUF_WRITABLE and self.readonly:
203              raise BufferError("buffer is read-only")
204          if (
205              flags & PyBUF_C_CONTIGUOUS
206          ) == PyBUF_C_CONTIGUOUS and not self.is_contiguous("C"):
207              raise BufferError("data is not C contiguous")
208          if (
209              flags & PyBUF_F_CONTIGUOUS
210          ) == PyBUF_F_CONTIGUOUS and not self.is_contiguous("F"):
211              raise BufferError("data is not F contiguous")
212          if (
213              flags & PyBUF_ANY_CONTIGUOUS
214          ) == PyBUF_ANY_CONTIGUOUS and not self.is_contiguous("A"):
215              raise BufferError("data is not contiguous")
216          view.buf = self.buf
217          view.readonly = self.readonly
218          view.len = self.len
219          if flags | PyBUF_WRITABLE == PyBUF_WRITABLE:
220              view.ndim = 0
221          else:
222              view.ndim = self.ndim
223          view.itemsize = self.itemsize
224          if (flags & PyBUF_FORMAT) == PyBUF_FORMAT:
225              view.format = addressof(self._format)
226          else:
227              view.format = None
228          if (flags & PyBUF_ND) == PyBUF_ND:
229              view.shape = addressof(self._shape)
230          elif self.is_contiguous("C"):
231              view.shape = None
232          else:
233              raise BufferError(
234                  "shape required for {} dimensional data".format(self.ndim)
235              )
236          if (flags & PyBUF_STRIDES) == PyBUF_STRIDES:
237              view.strides = ctypes.addressof(self._strides)
238          elif view.shape is None or self.is_contiguous("C"):
239              view.strides = None
240          else:
241              raise BufferError("strides required for none C contiguous data")
242          view.suboffsets = None
243          view.internal = None
244          view.obj = self
245  
246      def is_contiguous(self, fortran):
247          if fortran in "CA":
248              if self.strides[-1] == self.itemsize:
249                  for i in range(self.ndim - 1, 0, -1):
250                      if self.strides[i - 1] != self.shape[i] * self.strides[i]:
251                          break
252                  else:
253                      return True
254          if fortran in "FA":
255              if self.strides[0] == self.itemsize:
256                  for i in range(0, self.ndim - 1):
257                      if self.strides[i + 1] != self.shape[i] * self.strides[i]:
258                          break
259                  else:
260                      return True
261          return False

264  class Importer(object):
265      """An object that imports a new buffer interface 
266   
267         The fields of the Py_buffer C struct are exposed by identically 
268         named Importer read-only properties. 
269      """
270  
271      def __init__(self, obj, flags):
272          self._view = pygame.newbuffer.Py_buffer()
273          self._view.get_buffer(obj, flags)
274  
275      @property
276      def obj(self):
277          """return object or None for NULL field"""
278          return self._view.obj
279  
280      @property
281      def buf(self):
282          """return int or None for NULL field"""
283          return self._view.buf
284  
285      @property
286      def len(self):
287          """return int"""
288          return self._view.len
289  
290      @property
291      def readonly(self):
292          """return bool"""
293          return self._view.readonly
294  
295      @property
296      def format(self):
297          """return bytes or None for NULL field"""
298          format_addr = self._view.format
299          if format_addr is None:
300              return None
301          return ctypes.cast(format_addr, ctypes.c_char_p).value.decode("ascii")
302  
303      @property
304      def itemsize(self):
305          """return int"""
306          return self._view.itemsize
307  
308      @property
309      def ndim(self):
310          """return int"""
311          return self._view.ndim
312  
313      @property
314      def shape(self):
315          """return int tuple or None for NULL field"""
316          return self._to_ssize_tuple(self._view.shape)
317  
318      @property
319      def strides(self):
320          """return int tuple or None for NULL field"""
321          return self._to_ssize_tuple(self._view.strides)
322  
323      @property
324      def suboffsets(self):
325          """return int tuple or None for NULL field"""
326          return self._to_ssize_tuple(self._view.suboffsets)
327  
328      @property
329      def internal(self):
330          """return int or None for NULL field"""
331          return self._view.internal
332  
333      def _to_ssize_tuple(self, addr):
334          from ctypes import cast, POINTER, c_ssize_t
335  
336          if addr is None:
337              return None
338          return tuple(cast(addr, POINTER(c_ssize_t))[0 : self._view.ndim])
