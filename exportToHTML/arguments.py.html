18   class CallSite:
19       """Class for understanding arguments passed into a call site 
20    
21       It needs a call context, which contains the arguments and the 
22       keyword arguments that were passed into a given call site. 
23       In order to infer what an argument represents, call :meth:`infer_argument` 
24       with the corresponding function node and the argument name. 
25    
26       :param callcontext: 
27           An instance of :class:`astroid.context.CallContext`, that holds 
28           the arguments for the call site. 
29       :param argument_context_map: 
30           Additional contexts per node, passed in from :attr:`astroid.context.Context.extra_context` 
31       :param context: 
32           An instance of :class:`astroid.context.Context`. 
33       """
34   
35       def __init__(self, callcontext, argument_context_map=None, context=None):
36           if argument_context_map is None:
37               argument_context_map = {}
38           self.argument_context_map = argument_context_map
39           args = callcontext.args
40           keywords = callcontext.keywords
41           self.duplicated_keywords = set()
42           self._unpacked_args = self._unpack_args(args, context=context)
43           self._unpacked_kwargs = self._unpack_keywords(keywords, context=context)
44   
45           self.positional_arguments = [
46               arg for arg in self._unpacked_args if arg is not util.Uninferable
47           ]
48           self.keyword_arguments = {
49               key: value
50               for key, value in self._unpacked_kwargs.items()
51               if value is not util.Uninferable
52           }
53   
54       @classmethod
55       def from_call(cls, call_node, context=None):
56           """Get a CallSite object from the given Call node. 
57    
58           :param context: 
59               An instance of :class:`astroid.context.Context` that will be used 
60               to force a single inference path. 
61           """
62   
63           # Determine the callcontext from the given `context` object if any.
64           context = context or contextmod.InferenceContext()
65           callcontext = contextmod.CallContext(call_node.args, call_node.keywords)
66           return cls(callcontext, context=context)
67   
68       def has_invalid_arguments(self):
69           """Check if in the current CallSite were passed *invalid* arguments 
70    
71           This can mean multiple things. For instance, if an unpacking 
72           of an invalid object was passed, then this method will return True. 
73           Other cases can be when the arguments can't be inferred by astroid, 
74           for example, by passing objects which aren't known statically. 
75           """
76           return len(self.positional_arguments) != len(self._unpacked_args)
77   
78       def has_invalid_keywords(self):
79           """Check if in the current CallSite were passed *invalid* keyword arguments 
80    
81           For instance, unpacking a dictionary with integer keys is invalid 
82           (**{1:2}), because the keys must be strings, which will make this 
83           method to return True. Other cases where this might return True if 
84           objects which can't be inferred were passed. 
85           """
86           return len(self.keyword_arguments) != len(self._unpacked_kwargs)
87   
88       def _unpack_keywords(self, keywords, context=None):
89           values = {}
90           context = context or contextmod.InferenceContext()
91           context.extra_context = self.argument_context_map
92           for name, value in keywords:
93               if name is None:
94                   # Then it's an unpacking operation (**)
95                   try:
96                       inferred = next(value.infer(context=context))
97                   except exceptions.InferenceError:
98                       values[name] = util.Uninferable
99                       continue
100  
101                  if not isinstance(inferred, nodes.Dict):
102                      # Not something we can work with.
103                      values[name] = util.Uninferable
104                      continue
105  
106                  for dict_key, dict_value in inferred.items:
107                      try:
108                          dict_key = next(dict_key.infer(context=context))
109                      except exceptions.InferenceError:
110                          values[name] = util.Uninferable
111                          continue
112                      if not isinstance(dict_key, nodes.Const):
113                          values[name] = util.Uninferable
114                          continue
115                      if not isinstance(dict_key.value, str):
116                          values[name] = util.Uninferable
117                          continue
118                      if dict_key.value in values:
119                          # The name is already in the dictionary
120                          values[dict_key.value] = util.Uninferable
121                          self.duplicated_keywords.add(dict_key.value)
122                          continue
123                      values[dict_key.value] = dict_value
124              else:
125                  values[name] = value
126          return values
127  
128      def _unpack_args(self, args, context=None):
129          values = []
130          context = context or contextmod.InferenceContext()
131          context.extra_context = self.argument_context_map
132          for arg in args:
133              if isinstance(arg, nodes.Starred):
134                  try:
135                      inferred = next(arg.value.infer(context=context))
136                  except exceptions.InferenceError:
137                      values.append(util.Uninferable)
138                      continue
139  
140                  if inferred is util.Uninferable:
141                      values.append(util.Uninferable)
142                      continue
143                  if not hasattr(inferred, "elts"):
144                      values.append(util.Uninferable)
145                      continue
146                  values.extend(inferred.elts)
147              else:
148                  values.append(arg)
149          return values
150  
151      def infer_argument(self, funcnode, name, context):
152          """infer a function argument value according to the call context 
153   
154          Arguments: 
155              funcnode: The function being called. 
156              name: The name of the argument whose value is being inferred. 
157              context: Inference context object 
158          """
159          if name in self.duplicated_keywords:
160              raise exceptions.InferenceError(
161                  "The arguments passed to {func!r} " " have duplicate keywords.",
162                  call_site=self,
163                  func=funcnode,
164                  arg=name,
165                  context=context,
166              )
167  
168          # Look into the keywords first, maybe it's already there.
169          try:
170              return self.keyword_arguments[name].infer(context)
171          except KeyError:
172              pass
173  
174          # Too many arguments given and no variable arguments.
175          if len(self.positional_arguments) > len(funcnode.args.args):
176              if not funcnode.args.vararg:
177                  raise exceptions.InferenceError(
178                      "Too many positional arguments "
179                      "passed to {func!r} that does "
180                      "not have *args.",
181                      call_site=self,
182                      func=funcnode,
183                      arg=name,
184                      context=context,
185                  )
186  
187          positional = self.positional_arguments[: len(funcnode.args.args)]
188          vararg = self.positional_arguments[len(funcnode.args.args) :]
189          argindex = funcnode.args.find_argname(name)[0]
190          kwonlyargs = {arg.name for arg in funcnode.args.kwonlyargs}
191          kwargs = {
192              key: value
193              for key, value in self.keyword_arguments.items()
194              if key not in kwonlyargs
195          }
196          # If there are too few positionals compared to
197          # what the function expects to receive, check to see
198          # if the missing positional arguments were passed
199          # as keyword arguments and if so, place them into the
200          # positional args list.
201          if len(positional) < len(funcnode.args.args):
202              for func_arg in funcnode.args.args:
203                  if func_arg.name in kwargs:
204                      arg = kwargs.pop(func_arg.name)
205                      positional.append(arg)
206  
207          if argindex is not None:
208              # 2. first argument of instance/class method
209              if argindex == 0 and funcnode.type in ("method", "classmethod"):
210                  if context.boundnode is not None:
211                      boundnode = context.boundnode
212                  else:
213                      # XXX can do better ?
214                      boundnode = funcnode.parent.frame()
215  
216                  if isinstance(boundnode, nodes.ClassDef):
217                      # Verify that we're accessing a method
218                      # of the metaclass through a class, as in
219                      # `cls.metaclass_method`. In this case, the
220                      # first argument is always the class.
221                      method_scope = funcnode.parent.scope()
222                      if method_scope is boundnode.metaclass():
223                          return iter((boundnode,))
224  
225                  if funcnode.type == "method":
226                      if not isinstance(boundnode, bases.Instance):
227                          boundnode = boundnode.instantiate_class()
228                      return iter((boundnode,))
229                  if funcnode.type == "classmethod":
230                      return iter((boundnode,))
231              # if we have a method, extract one position
232              # from the index, so we'll take in account
233              # the extra parameter represented by `self` or `cls`
234              if funcnode.type in ("method", "classmethod"):
235                  argindex -= 1
236              # 2. search arg index
237              try:
238                  return self.positional_arguments[argindex].infer(context)
239              except IndexError:
240                  pass
241  
242          if funcnode.args.kwarg == name:
243              # It wants all the keywords that were passed into
244              # the call site.
245              if self.has_invalid_keywords():
246                  raise exceptions.InferenceError(
247                      "Inference failed to find values for all keyword arguments "
248                      "to {func!r}: {unpacked_kwargs!r} doesn't correspond to "
249                      "{keyword_arguments!r}.",
250                      keyword_arguments=self.keyword_arguments,
251                      unpacked_kwargs=self._unpacked_kwargs,
252                      call_site=self,
253                      func=funcnode,
254                      arg=name,
255                      context=context,
256                  )
257              kwarg = nodes.Dict(
258                  lineno=funcnode.args.lineno,
259                  col_offset=funcnode.args.col_offset,
260                  parent=funcnode.args,
261              )
262              kwarg.postinit(
263                  [(nodes.const_factory(key), value) for key, value in kwargs.items()]
264              )
265              return iter((kwarg,))
266          if funcnode.args.vararg == name:
267              # It wants all the args that were passed into
268              # the call site.
269              if self.has_invalid_arguments():
270                  raise exceptions.InferenceError(
271                      "Inference failed to find values for all positional "
272                      "arguments to {func!r}: {unpacked_args!r} doesn't "
273                      "correspond to {positional_arguments!r}.",
274                      positional_arguments=self.positional_arguments,
275                      unpacked_args=self._unpacked_args,
276                      call_site=self,
277                      func=funcnode,
278                      arg=name,
279                      context=context,
280                  )
281              args = nodes.Tuple(
282                  lineno=funcnode.args.lineno,
283                  col_offset=funcnode.args.col_offset,
284                  parent=funcnode.args,
285              )
286              args.postinit(vararg)
287              return iter((args,))
288  
289          # Check if it's a default parameter.
290          try:
291              return funcnode.args.default_value(name).infer(context)
292          except exceptions.NoDefault:
293              pass
294          raise exceptions.InferenceError(
295              "No value found for argument {name} to " "{func!r}",
296              call_site=self,
297              func=funcnode,
298              arg=name,
299              context=context,
300          )
