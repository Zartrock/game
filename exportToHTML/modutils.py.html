143  def _path_from_filename(filename, is_jython=IS_JYTHON):
144      if not is_jython:
145          return filename
146      head, has_pyclass, _ = filename.partition("$py.class")
147      if has_pyclass:
148          return head + ".py"
149      return filename

152  def _handle_blacklist(blacklist, dirnames, filenames):
153      """remove files/directories in the black list 
154   
155      dirnames/filenames are usually from os.walk 
156      """
157      for norecurs in blacklist:
158          if norecurs in dirnames:
159              dirnames.remove(norecurs)
160          elif norecurs in filenames:
161              filenames.remove(norecurs)

167  def _cache_normalize_path(path):
168      """abspath with caching"""
169      # _module_file calls abspath on every path in sys.path every time it's
170      # called; on a larger codebase this easily adds up to half a second just
171      # assembling path components. This cache alleviates that.
172      try:
173          return _NORM_PATH_CACHE[path]
174      except KeyError:
175          if not path:  # don't cache result for ''
176              return _normalize_path(path)
177          result = _NORM_PATH_CACHE[path] = _normalize_path(path)
178          return result

182      """Load a Python module from its name. 
183   
184      :type dotted_name: str 
185      :param dotted_name: python name of a module or package

187      :type path: list or None 
188      :param path: 
189        optional list of path where the module or package should be

193      :param use_sys: 
194        boolean indicating whether the sys.modules dictionary should be 
195        used or not

197   
198      :raise ImportError: if the module or package is not found

201      :return: the loaded module

203      return load_module_from_modpath(dotted_name.split("."), path, use_sys)

206  def load_module_from_modpath(parts, path: Optional[List[str]] = None, use_sys=1):
207      """Load a python module from its split name. 
208   
209      :type parts: list(str) or tuple(str) 
210      :param parts: 
211        python name of a module or package split on '.' 
212   
213      :param path: 
214        Optional list of path where the module or package should be 
215        searched (use sys.path if nothing or None is given) 
216   
217      :type use_sys: bool 
218      :param use_sys: 
219        boolean indicating whether the sys.modules dictionary should be used or not 
220   
221      :raise ImportError: if the module or package is not found 
222   
223      :rtype: module 
224      :return: the loaded module 
225      """
226      if use_sys:
227          try:
228              return sys.modules[".".join(parts)]
229          except KeyError:
230              pass
231      modpath = []
232      prevmodule = None
233      for part in parts:
234          modpath.append(part)
235          curname = ".".join(modpath)
236          module = None
237          if len(modpath) != len(parts):
238              # even with use_sys=False, should try to get outer packages from sys.modules
239              module = sys.modules.get(curname)
240          elif use_sys:
241              # because it may have been indirectly loaded through a parent
242              module = sys.modules.get(curname)
243          if module is None:
244              mp_file, mp_filename, mp_desc = imp.find_module(part, path)
245              module = imp.load_module(curname, mp_file, mp_filename, mp_desc)
246              # mp_file still needs to be closed.
247              if mp_file:
248                  mp_file.close()
249          if prevmodule:
250              setattr(prevmodule, part, module)
251          _file = getattr(module, "__file__", "")
252          prevmodule = module
253          if not _file and util.is_namespace(curname):
254              continue
255          if not _file and len(modpath) != len(parts):
256              raise ImportError("no module in %s" % ".".join(parts[len(modpath) :]))
257          path = [os.path.dirname(_file)]
258      return module

287  def check_modpath_has_init(path, mod_path):
288      """check there are some __init__.py all along the way"""
289      modpath = []
290      for part in mod_path:
291          modpath.append(part)
292          path = os.path.join(path, part)
293          if not _has_init(path):
294              old_namespace = util.is_namespace(".".join(modpath))
295              if not old_namespace:
296                  return False
297      return True

300  def _get_relative_base_path(filename, path_to_check):
301      """Extracts the relative mod path of the file to import from 
302   
303      Check if a file is within the passed in path and if so, returns the 
304      relative mod path from the one passed in. 
305   
306      If the filename is no in path_to_check, returns None 
307   
308      Note this function will look for both abs and realpath of the file, 
309      this allows to find the relative base path even if the file is a 
310      symlink of a file in the passed in path 
311   
312      Examples: 
313          _get_relative_base_path("/a/b/c/d.py", "/a/b") ->  ["c","d"] 
314          _get_relative_base_path("/a/b/c/d.py", "/dev") ->  None 
315      """
316      importable_path = None
317      path_to_check = os.path.normcase(path_to_check)
318      abs_filename = os.path.abspath(filename)
319      if os.path.normcase(abs_filename).startswith(path_to_check):
320          importable_path = abs_filename
321  
322      real_filename = os.path.realpath(filename)
323      if os.path.normcase(real_filename).startswith(path_to_check):
324          importable_path = real_filename
325  
326      if importable_path:
327          base_path = os.path.splitext(importable_path)[0]
328          relative_base_path = base_path[len(path_to_check) :]
329          return [pkg for pkg in relative_base_path.split(os.sep) if pkg]
330  
331      return None

334  def modpath_from_file_with_callback(filename, path=None, is_package_cb=None):
335      filename = os.path.expanduser(_path_from_filename(filename))
336      for pathname in itertools.chain(
337          path or [], map(_canonicalize_path, sys.path), sys.path
338      ):
339          pathname = _cache_normalize_path(pathname)
340          if not pathname:
341              continue
342          modpath = _get_relative_base_path(filename, pathname)
343          if not modpath:
344              continue
345          if is_package_cb(pathname, modpath[:-1]):
346              return modpath
347  
348      raise ImportError(
349          "Unable to find module for %s in %s" % (filename, ", \n".join(sys.path))
350      )

378  def file_info_from_modpath(modpath, path=None, context_file=None):
379      """given a mod path (i.e. split module / package name), return the 
380      corresponding file, giving priority to source file over precompiled 
381      file if it exists 
382   
383      :type modpath: list or tuple 
384      :param modpath: 
385        split module's name (i.e name of a module or package split 
386        on '.') 
387        (this means explicit relative imports that start with dots have 
388        empty strings in this list!) 
389   
390      :type path: list or None 
391      :param path: 
392        optional list of path where the module or package should be 
393        searched (use sys.path if nothing or None is given) 
394   
395      :type context_file: str or None 
396      :param context_file: 
397        context file to consider, necessary if the identifier has been 
398        introduced using a relative import unresolvable in the actual 
399        context (i.e. modutils) 
400   
401      :raise ImportError: if there is no such module in the directory 
402   
403      :rtype: (str or None, import type) 
404      :return: 
405        the path to the module's file or None if it's an integrated 
406        builtin module such as 'sys' 
407      """
408      if context_file is not None:
409          context = os.path.dirname(context_file)
410      else:
411          context = context_file
412      if modpath[0] == "xml":
413          # handle _xmlplus
414          try:
415              return _spec_from_modpath(["_xmlplus"] + modpath[1:], path, context)
416          except ImportError:
417              return _spec_from_modpath(modpath, path, context)
418      elif modpath == ["os", "path"]:
419          # FIXME: currently ignoring search_path...
420          return spec.ModuleSpec(
421              name="os.path", location=os.path.__file__, module_type=imp.PY_SOURCE
422          )
423      return _spec_from_modpath(modpath, path, context)

426  def get_module_part(dotted_name, context_file=None):
427      """given a dotted name return the module part of the name : 
428   
429      >>> get_module_part('astroid.as_string.dump') 
430      'astroid.as_string' 
431   
432      :type dotted_name: str 
433      :param dotted_name: full name of the identifier we are interested in 
434   
435      :type context_file: str or None 
436      :param context_file: 
437        context file to consider, necessary if the identifier has been 
438        introduced using a relative import unresolvable in the actual 
439        context (i.e. modutils) 
440   
441   
442      :raise ImportError: if there is no such module in the directory 
443   
444      :rtype: str or None 
445      :return: 
446        the module part of the name or None if we have not been able at 
447        all to import the given name 
448   
449      XXX: deprecated, since it doesn't handle package precedence over module 
450      (see #10066) 
451      """
452      # os.path trick
453      if dotted_name.startswith("os.path"):
454          return "os.path"
455      parts = dotted_name.split(".")
456      if context_file is not None:
457          # first check for builtin module which won't be considered latter
458          # in that case (path != None)
459          if parts[0] in BUILTIN_MODULES:
460              if len(parts) > 2:
461                  raise ImportError(dotted_name)
462              return parts[0]
463          # don't use += or insert, we want a new list to be created !
464      path = None
465      starti = 0
466      if parts[0] == "":
467          assert (
468              context_file is not None
469          ), "explicit relative import, but no context_file?"
470          path = []  # prevent resolving the import non-relatively
471          starti = 1
472      while parts[starti] == "":  # for all further dots: change context
473          starti += 1
474          context_file = os.path.dirname(context_file)
475      for i in range(starti, len(parts)):
476          try:
477              file_from_modpath(
478                  parts[starti : i + 1], path=path, context_file=context_file
479              )
480          except ImportError:
481              if i < max(1, len(parts) - 2):
482                  raise
483              return ".".join(parts[:i])
484      return dotted_name

487  def get_module_files(src_directory, blacklist, list_all=False):
488      """given a package directory return a list of all available python 
489      module's files in the package and its subpackages 
490   
491      :type src_directory: str 
492      :param src_directory: 
493        path of the directory corresponding to the package 
494   
495      :type blacklist: list or tuple 
496      :param blacklist: iterable 
497        list of files or directories to ignore. 
498   
499      :type list_all: bool 
500      :param list_all: 
501          get files from all paths, including ones without __init__.py 
502   
503      :rtype: list 
504      :return: 
505        the list of all available python module's files in the package and 
506        its subpackages 
507      """
508      files = []
509      for directory, dirnames, filenames in os.walk(src_directory):
510          if directory in blacklist:
511              continue
512          _handle_blacklist(blacklist, dirnames, filenames)
513          # check for __init__.py
514          if not list_all and "__init__.py" not in filenames:
515              dirnames[:] = ()
516              continue
517          for filename in filenames:
518              if _is_python_file(filename):
519                  src = os.path.join(directory, filename)
520                  files.append(src)
521      return files

524  def get_source_file(filename, include_no_ext=False):
525      """given a python module's file name return the matching source file 
526      name (the filename will be returned identically if it's already an 
527      absolute path to a python source file...) 
528   
529      :type filename: str 
530      :param filename: python module's file name 
531   
532   
533      :raise NoSourceFile: if no source file exists on the file system 
534   
535      :rtype: str 
536      :return: the absolute path of the source file if it exists 
537      """
538      filename = os.path.abspath(_path_from_filename(filename))
539      base, orig_ext = os.path.splitext(filename)
540      for ext in PY_SOURCE_EXTS:
541          source_path = "%s.%s" % (base, ext)
542          if os.path.exists(source_path):
543              return source_path
544      if include_no_ext and not orig_ext and os.path.exists(base):
545          return base
546      raise NoSourceFile(filename)

557  def is_standard_module(modname, std_path=None):
558      """try to guess if a module is a standard python module (by default, 
559      see `std_path` parameter's description) 
560   
561      :type modname: str 
562      :param modname: name of the module we are interested in 
563   
564      :type std_path: list(str) or tuple(str) 
565      :param std_path: list of path considered has standard 
566   
567   
568      :rtype: bool 
569      :return: 
570        true if the module: 
571        - is located on the path listed in one of the directory in `std_path` 
572        - is a built-in module 
573      """
574      modname = modname.split(".")[0]
575      try:
576          filename = file_from_modpath([modname])
577      except ImportError:
578          # import failed, i'm probably not so wrong by supposing it's
579          # not standard...
580          return False
581      # modules which are not living in a file are considered standard
582      # (sys and __builtin__ for instance)
583      if filename is None:
584          # we assume there are no namespaces in stdlib
585          return not util.is_namespace(modname)
586      filename = _normalize_path(filename)
587      for path in EXT_LIB_DIRS:
588          if filename.startswith(_cache_normalize_path(path)):
589              return False
590      if std_path is None:
591          std_path = STD_LIB_DIRS
592      for path in std_path:
593          if filename.startswith(_cache_normalize_path(path)):
594              return True
595      return False

598  def is_relative(modname, from_file):
599      """return true if the given module name is relative to the given 
600      file name 
601   
602      :type modname: str 
603      :param modname: name of the module we are interested in 
604   
605      :type from_file: str 
606      :param from_file: 
607        path of the module from which modname has been imported 
608   
609      :rtype: bool 
610      :return: 
611        true if the module has been imported relatively to `from_file` 
612      """
613      if not os.path.isdir(from_file):
614          from_file = os.path.dirname(from_file)
615      if from_file in sys.path:
616          return False
617      try:
618          stream, _, _ = imp.find_module(modname.split(".")[0], [from_file])
619  
620          # Close the stream to avoid ResourceWarnings.
621          if stream:
622              stream.close()
623          return True
624      except ImportError:
625          return False

631  def _spec_from_modpath(modpath, path=None, context=None):
632      """given a mod path (i.e. split module / package name), return the 
633      corresponding spec 
634   
635      this function is used internally, see `file_from_modpath`'s 
636      documentation for more information 
637      """
638      assert modpath
639      location = None
640      if context is not None:
641          try:
642              found_spec = spec.find_spec(modpath, [context])
643              location = found_spec.location
644          except ImportError:
645              found_spec = spec.find_spec(modpath, path)
646              location = found_spec.location
647      else:
648          found_spec = spec.find_spec(modpath, path)
649      if found_spec.type == spec.ModuleType.PY_COMPILED:
650          try:
651              location = get_source_file(found_spec.location)
652              return found_spec._replace(
653                  location=location, type=spec.ModuleType.PY_SOURCE
654              )
655          except NoSourceFile:
656              return found_spec._replace(location=location)
657      elif found_spec.type == spec.ModuleType.C_BUILTIN:
658          # integrated builtin module
659          return found_spec._replace(location=None)
660      elif found_spec.type == spec.ModuleType.PKG_DIRECTORY:
661          location = _has_init(found_spec.location)
662          return found_spec._replace(location=location, type=spec.ModuleType.PY_SOURCE)
663      return found_spec

666  def _is_python_file(filename):
667      """return true if the given filename should be considered as a python file 
668   
669      .pyc and .pyo are ignored 
670      """
671      return filename.endswith((".py", ".so", ".pyd", ".pyw"))

674  def _has_init(directory):
675      """if the given directory has a valid __init__ file, return its path, 
676      else return None 
677      """
678      mod_or_pack = os.path.join(directory, "__init__")
679      for ext in PY_SOURCE_EXTS + ("pyc", "pyo"):
680          if os.path.exists(mod_or_pack + "." + ext):
681              return mod_or_pack + "." + ext
682      return None
