135  def _is_from_future_import(stmt, name):
136      """Check if the name is a future import from another module."""
137      try:
138          module = stmt.do_import_module(stmt.modname)
139      except astroid.AstroidBuildingException:
140          return None
141  
142      for local_node in module.locals.get(name, []):
143          if isinstance(local_node, astroid.ImportFrom) and local_node.modname == FUTURE:
144              return True
145      return None

156  def overridden_method(klass, name):
157      """get overridden method if any"""
158      try:
159          parent = next(klass.local_attr_ancestors(name))
160      except (StopIteration, KeyError):
161          return None
162      try:
163          meth_node = parent[name]
164      except KeyError:
165          # We have found an ancestor defining <name> but it's not in the local
166          # dictionary. This may happen with astroid built from living objects.
167          return None
168      if isinstance(meth_node, astroid.FunctionDef):
169          return meth_node
170      return None

174      """return extra information to add to the message for unpacking-non-sequence 
175      and unbalanced-tuple-unpacking errors 
176      """
177      more = ""
178      inferred_module = inferred.root().name
179      if node.root().name == inferred_module:
180          if node.lineno == inferred.lineno:
181              more = " %s" % inferred.as_string()
182          elif inferred.lineno:
183              more = " defined at line %s" % inferred.lineno
184      elif inferred.lineno:
185          more = " defined at line %s of %s" % (inferred.lineno, inferred_module)
186      return more

189  def _detect_global_scope(node, frame, defframe):
190      """ Detect that the given frames shares a global 
191      scope. 
192   
193      Two frames shares a global scope when neither 
194      of them are hidden under a function scope, as well 
195      as any of parent scope of them, until the root scope. 
196      In this case, depending from something defined later on 
197      will not work, because it is still undefined. 
198   
199      Example: 
200          class A: 
201              # B has the same global scope as `C`, leading to a NameError. 
202              class B(C): ... 
203          class C: ... 
204   
205      """
206      def_scope = scope = None
207      if frame and frame.parent:
208          scope = frame.parent.scope()
209      if defframe and defframe.parent:
210          def_scope = defframe.parent.scope()
211      if isinstance(frame, astroid.FunctionDef):
212          # If the parent of the current node is a
213          # function, then it can be under its scope
214          # (defined in, which doesn't concern us) or
215          # the `->` part of annotations. The same goes
216          # for annotations of function arguments, they'll have
217          # their parent the Arguments node.
218          if not isinstance(node.parent, (astroid.FunctionDef, astroid.Arguments)):
219              return False
220      elif any(
221          not isinstance(f, (astroid.ClassDef, astroid.Module)) for f in (frame, defframe)
222      ):
223          # Not interested in other frames, since they are already
224          # not in a global scope.
225          return False
226  
227      break_scopes = []
228      for current_scope in (scope, def_scope):
229          # Look for parent scopes. If there is anything different
230          # than a module or a class scope, then they frames don't
231          # share a global scope.
232          parent_scope = current_scope
233          while parent_scope:
234              if not isinstance(parent_scope, (astroid.ClassDef, astroid.Module)):
235                  break_scopes.append(parent_scope)
236                  break
237              if parent_scope.parent:
238                  parent_scope = parent_scope.parent.scope()
239              else:
240                  break
241      if break_scopes and len(set(break_scopes)) != 1:
242          # Store different scopes than expected.
243          # If the stored scopes are, in fact, the very same, then it means
244          # that the two frames (frame and defframe) shares the same scope,
245          # and we could apply our lineno analysis over them.
246          # For instance, this works when they are inside a function, the node
247          # that uses a definition and the definition itself.
248          return False
249      # At this point, we are certain that frame and defframe shares a scope
250      # and the definition of the first depends on the second.
251      return frame.lineno < defframe.lineno

260  def _fix_dot_imports(not_consumed):
261      """ Try to fix imports with multiple dots, by returning a dictionary 
262      with the import names expanded. The function unflattens root imports, 
263      like 'xml' (when we have both 'xml.etree' and 'xml.sax'), to 'xml.etree' 
264      and 'xml.sax' respectively. 
265      """
266      names = {}
267      for name, stmts in not_consumed.items():
268          if any(
269              isinstance(stmt, astroid.AssignName)
270              and isinstance(stmt.assign_type(), astroid.AugAssign)
271              for stmt in stmts
272          ):
273              continue
274          for stmt in stmts:
275              if not isinstance(stmt, (astroid.ImportFrom, astroid.Import)):
276                  continue
277              for imports in stmt.names:
278                  second_name = None
279                  import_module_name = imports[0]
280                  if import_module_name == "*":
281                      # In case of wildcard imports,
282                      # pick the name from inside the imported module.
283                      second_name = name
284                  else:
285                      name_matches_dotted_import = False
286                      if (
287                          import_module_name.startswith(name)
288                          and import_module_name.find(".") > -1
289                      ):
290                          name_matches_dotted_import = True
291  
292                      if name_matches_dotted_import or name in imports:
293                          # Most likely something like 'xml.etree',
294                          # which will appear in the .locals as 'xml'.
295                          # Only pick the name if it wasn't consumed.
296                          second_name = import_module_name
297                  if second_name and second_name not in names:
298                      names[second_name] = stmt
299      return sorted(names.items(), key=lambda a: a[1].fromlineno)

302  def _find_frame_imports(name, frame):
303      """ 
304      Detect imports in the frame, with the required 
305      *name*. Such imports can be considered assignments. 
306      Returns True if an import for the given name was found. 
307      """
308      imports = frame.nodes_of_class((astroid.Import, astroid.ImportFrom))
309      for import_node in imports:
310          for import_name, import_alias in import_node.names:
311              # If the import uses an alias, check only that.
312              # Otherwise, check only the import name.
313              if import_alias:
314                  if import_alias == name:
315                      return True
316              elif import_name and import_name == name:
317                  return True
318      return None

322      for import_name, import_alias in stmt.names:

354  def _has_locals_call_after_node(stmt, scope):
355      skip_nodes = (
356          astroid.FunctionDef,
357          astroid.ClassDef,
358          astroid.Import,
359          astroid.ImportFrom,
360      )
361      for call in scope.nodes_of_class(astroid.Call, skip_klass=skip_nodes):
362          inferred = utils.safe_infer(call.func)
363          if (
364              utils.is_builtin_object(inferred)
365              and getattr(inferred, "name", None) == "locals"
366          ):
367              if stmt.lineno < call.lineno:
368                  return True
369      return False

509  class NamesConsumer:
510      """ 
511      A simple class to handle consumed, to consume and scope type info of node locals 
512      """
513  
514      def __init__(self, node, scope_type):
515          self._atomic = ScopeConsumer(copy.copy(node.locals), {}, scope_type)
516          self.node = node
517  
518      def __repr__(self):
519          msg = "\nto_consume : {:s}\n".format(
520              ", ".join(
521                  [
522                      "{}->{}".format(key, val)
523                      for key, val in self._atomic.to_consume.items()
524                  ]
525              )
526          )
527          msg += "consumed : {:s}\n".format(
528              ", ".join(
529                  [
530                      "{}->{}".format(key, val)
531                      for key, val in self._atomic.consumed.items()
532                  ]
533              )
534          )
535          msg += "scope_type : {:s}\n".format(self._atomic.scope_type)
536          return msg
537  
538      def __iter__(self):
539          return iter(self._atomic)
540  
541      @property
542      def to_consume(self):
543          return self._atomic.to_consume
544  
545      @property
546      def consumed(self):
547          return self._atomic.consumed
548  
549      @property
550      def scope_type(self):
551          return self._atomic.scope_type
552  
553      def mark_as_consumed(self, name, new_node):
554          """ 
555          Mark the name as consumed and delete it from 
556          the to_consume dictionary 
557          """
558          self.consumed[name] = new_node
559          del self.to_consume[name]
560  
561      def get_next_to_consume(self, node):
562          # Get the definition of `node` from this scope
563          name = node.name
564          parent_node = node.parent
565          found_node = self.to_consume.get(name)
566          if (
567              found_node
568              and isinstance(parent_node, astroid.Assign)
569              and parent_node == found_node[0].parent
570          ):
571              lhs = found_node[0].parent.targets[0]
572              if lhs.name == name:  # this name is defined in this very statement
573                  found_node = None
574  
575          if (
576              found_node
577              and isinstance(parent_node, astroid.For)
578              and parent_node.iter == node
579              and parent_node.target in found_node
580          ):
581              found_node = None
582          return found_node

953      def visit_name(self, node):
954          """Check that a name is defined in the current scope"""
955          stmt = node.statement()
956          if stmt.fromlineno is None:
957              # name node from an astroid built from live code, skip
958              assert not stmt.root().file.endswith(".py")
959              return
960  
961          name = node.name
962          frame = stmt.scope()
963          start_index = len(self._to_consume) - 1
964  
965          undefined_variable_is_enabled = self.linter.is_message_enabled(
966              "undefined-variable"
967          )
968          used_before_assignment_is_enabled = self.linter.is_message_enabled(
969              "used-before-assignment"
970          )
971  
972          # iterates through parent scopes, from the inner to the outer
973          base_scope_type = self._to_consume[start_index].scope_type
974          # pylint: disable=too-many-nested-blocks; refactoring this block is a pain.
975          for i in range(start_index, -1, -1):
976              current_consumer = self._to_consume[i]
977  
978              # The list of base classes in the class definition is not part
979              # of the class body.
980              # If the current scope is a class scope but it's not the inner
981              # scope, ignore it. This prevents to access this scope instead of
982              # the globals one in function members when there are some common
983              # names.
984              if current_consumer.scope_type == "class" and (
985                  utils.is_ancestor_name(current_consumer.node, node)
986                  or (i != start_index and self._ignore_class_scope(node))
987              ):
988                  continue
989  
990              # if the name node is used as a function default argument's value or as
991              # a decorator, then start from the parent frame of the function instead
992              # of the function frame - and thus open an inner class scope
993              if (
994                  current_consumer.scope_type == "function"
995                  and self._defined_in_function_definition(node, current_consumer.node)
996              ):
997                  # ignore function scope if is an annotation/default/decorator, as not in the body
998                  continue
999  
1000             if current_consumer.scope_type == "lambda" and utils.is_default_argument(
1001                 node, current_consumer.node
1002             ):
1003                 continue
1004 
1005             # the name has already been consumed, only check it's not a loop
1006             # variable used outside the loop
1007             # avoid the case where there are homonyms inside function scope and
1008             #  comprehension current scope (avoid bug #1731)
1009             if name in current_consumer.consumed and not (
1010                 current_consumer.scope_type == "comprehension"
1011                 and self._has_homonym_in_upper_function_scope(node, i)
1012             ):
1013                 defnode = utils.assign_parent(current_consumer.consumed[name][0])
1014                 self._check_late_binding_closure(node, defnode)
1015                 self._loopvar_name(node, name)
1016                 break
1017 
1018             found_node = current_consumer.get_next_to_consume(node)
1019             if found_node is None:
1020                 continue
1021 
1022             # checks for use before assignment
1023             defnode = utils.assign_parent(current_consumer.to_consume[name][0])
1024 
1025             if (
1026                 undefined_variable_is_enabled or used_before_assignment_is_enabled
1027             ) and defnode is not None:
1028                 self._check_late_binding_closure(node, defnode)
1029                 defstmt = defnode.statement()
1030                 defframe = defstmt.frame()
1031                 # The class reuses itself in the class scope.
1032                 recursive_klass = (
1033                     frame is defframe
1034                     and defframe.parent_of(node)
1035                     and isinstance(defframe, astroid.ClassDef)
1036                     and node.name == defframe.name
1037                 )
1038 
1039                 if (
1040                     recursive_klass
1041                     and utils.is_inside_lambda(node)
1042                     and (
1043                         not utils.is_default_argument(node)
1044                         or node.scope().parent.scope() is not defframe
1045                     )
1046                 ):
1047                     # Self-referential class references are fine in lambda's --
1048                     # As long as they are not part of the default argument directly
1049                     # under the scope of the parent self-referring class.
1050                     # Example of valid default argument:
1051                     # class MyName3:
1052                     #     myattr = 1
1053                     #     mylambda3 = lambda: lambda a=MyName3: a
1054                     # Example of invalid default argument:
1055                     # class MyName4:
1056                     #     myattr = 1
1057                     #     mylambda4 = lambda a=MyName4: lambda: a
1058 
1059                     # If the above conditional is True,
1060                     # there is no possibility of undefined-variable
1061                     # Also do not consume class name
1062                     # (since consuming blocks subsequent checks)
1063                     # -- quit
1064                     break
1065 
1066                 (
1067                     maybee0601,
1068                     annotation_return,
1069                     use_outer_definition,
1070                 ) = self._is_variable_violation(
1071                     node,
1072                     name,
1073                     defnode,
1074                     stmt,
1075                     defstmt,
1076                     frame,
1077                     defframe,
1078                     base_scope_type,
1079                     recursive_klass,
1080                 )
1081 
1082                 if use_outer_definition:
1083                     continue
1084 
1085                 if (
1086                     maybee0601
1087                     and not utils.is_defined_before(node)
1088                     and not astroid.are_exclusive(stmt, defstmt, ("NameError",))
1089                 ):
1090 
1091                     # Used and defined in the same place, e.g `x += 1` and `del x`
1092                     defined_by_stmt = defstmt is stmt and isinstance(
1093                         node, (astroid.DelName, astroid.AssignName)
1094                     )
1095                     if (
1096                         recursive_klass
1097                         or defined_by_stmt
1098                         or annotation_return
1099                         or isinstance(defstmt, astroid.Delete)
1100                     ):
1101                         if not utils.node_ignores_exception(node, NameError):
1102 
1103                             # Handle postponed evaluation of annotations
1104                             if not (
1105                                 self._postponed_evaluation_enabled
1106                                 and isinstance(
1107                                     stmt,
1108                                     (
1109                                         astroid.AnnAssign,
1110                                         astroid.FunctionDef,
1111                                         astroid.Arguments,
1112                                     ),
1113                                 )
1114                                 and name in node.root().locals
1115                             ):
1116                                 self.add_message(
1117                                     "undefined-variable", args=name, node=node
1118                                 )
1119                     elif base_scope_type != "lambda":
1120                         # E0601 may *not* occurs in lambda scope.
1121 
1122                         # Handle postponed evaluation of annotations
1123                         if not (
1124                             self._postponed_evaluation_enabled
1125                             and isinstance(
1126                                 stmt, (astroid.AnnAssign, astroid.FunctionDef)
1127                             )
1128                         ):
1129                             self.add_message(
1130                                 "used-before-assignment", args=name, node=node
1131                             )
1132                     elif base_scope_type == "lambda":
1133                         # E0601 can occur in class-level scope in lambdas, as in
1134                         # the following example:
1135                         #   class A:
1136                         #      x = lambda attr: f + attr
1137                         #      f = 42
1138                         if isinstance(frame, astroid.ClassDef) and name in frame.locals:
1139                             if isinstance(node.parent, astroid.Arguments):
1140                                 if stmt.fromlineno <= defstmt.fromlineno:
1141                                     # Doing the following is fine:
1142                                     #   class A:
1143                                     #      x = 42
1144                                     #      y = lambda attr=x: attr
1145                                     self.add_message(
1146                                         "used-before-assignment", args=name, node=node
1147                                     )
1148                             else:
1149                                 self.add_message(
1150                                     "undefined-variable", args=name, node=node
1151                                 )
1152                         elif current_consumer.scope_type == "lambda":
1153                             self.add_message("undefined-variable", node=node, args=name)
1154 
1155             current_consumer.mark_as_consumed(name, found_node)
1156             # check it's not a loop variable used outside the loop
1157             self._loopvar_name(node, name)
1158             break
1159         else:
1160             # we have not found the name, if it isn't a builtin, that's an
1161             # undefined name !
1162             if undefined_variable_is_enabled and not (
1163                 name in astroid.Module.scope_attrs
1164                 or utils.is_builtin(name)
1165                 or name in self.config.additional_builtins
1166                 or (
1167                     name == "__class__"
1168                     and isinstance(frame, astroid.FunctionDef)
1169                     and frame.is_method()
1170                 )
1171             ):
1172                 if not utils.node_ignores_exception(node, NameError):
1173                     self.add_message("undefined-variable", args=name, node=node)

1290     @staticmethod
1291     def _in_lambda_or_comprehension_body(
1292         node: astroid.node_classes.NodeNG, frame: astroid.node_classes.NodeNG
1293     ) -> bool:
1294         """return True if node within a lambda/comprehension body (or similar) and thus should not have access to class attributes in frame"""
1295         child = node
1296         parent = node.parent
1297         while parent is not None:
1298             if parent is frame:
1299                 return False
1300             if isinstance(parent, astroid.Lambda) and not child is parent.args:
1301                 # Body of lambda should not have access to class attributes.
1302                 return True
1303             if (
1304                 isinstance(parent, astroid.node_classes.Comprehension)
1305                 and not child is parent.iter
1306             ):
1307                 # Only iter of list/set/dict/generator comprehension should have access.
1308                 return True
1309             if isinstance(parent, astroid.scoped_nodes.ComprehensionScope) and not (
1310                 parent.generators and child is parent.generators[0]
1311             ):
1312                 # Body of list/set/dict/generator comprehension should not have access to class attributes.
1313                 # Furthermore, only the first generator (if multiple) in comprehension should have access.
1314                 return True

1961     def _check_imports(self, not_consumed):
1962         local_names = _fix_dot_imports(not_consumed)
1963         checked = set()
1964         for name, stmt in local_names:
1965             for imports in stmt.names:
1966                 real_name = imported_name = imports[0]
1967                 if imported_name == "*":
1968                     real_name = name
1969                 as_name = imports[1]
1970                 if real_name in checked:
1971                     continue
1972                 if name not in (real_name, as_name):
1973                     continue
1974                 checked.add(real_name)
1975 
1976                 is_type_annotation_import = (
1977                     imported_name in self._type_annotation_names
1978                     or as_name in self._type_annotation_names
1979                 )
1980                 if isinstance(stmt, astroid.Import) or (
1981                     isinstance(stmt, astroid.ImportFrom) and not stmt.modname
1982                 ):
1983                     if isinstance(stmt, astroid.ImportFrom) and SPECIAL_OBJ.search(
1984                         imported_name
1985                     ):
1986                         # Filter special objects (__doc__, __all__) etc.,
1987                         # because they can be imported for exporting.
1988                         continue
1989 
1990                     if is_type_annotation_import:
1991                         # Most likely a typing import if it wasn't used so far.
1992                         continue
1993 
1994                     if as_name == "_":
1995                         continue
1996                     if as_name is None:
1997                         msg = "import %s" % imported_name
1998                     else:
1999                         msg = "%s imported as %s" % (imported_name, as_name)
2000                     if not _is_type_checking_import(stmt):
2001                         self.add_message("unused-import", args=msg, node=stmt)
2002                 elif isinstance(stmt, astroid.ImportFrom) and stmt.modname != FUTURE:
2003                     if SPECIAL_OBJ.search(imported_name):
2004                         # Filter special objects (__doc__, __all__) etc.,
2005                         # because they can be imported for exporting.
2006                         continue
2007 
2008                     if _is_from_future_import(stmt, name):
2009                         # Check if the name is in fact loaded from a
2010                         # __future__ import in another module.
2011                         continue
2012 
2013                     if is_type_annotation_import:
2014                         # Most likely a typing import if it wasn't used so far.
2015                         continue
2016 
2017                     if imported_name == "*":
2018                         self.add_message("unused-wildcard-import", args=name, node=stmt)
2019                     else:
2020                         if as_name is None:
2021                             msg = "%s imported from %s" % (imported_name, stmt.modname)
2022                         else:
2023                             fields = (imported_name, stmt.modname, as_name)
2024                             msg = "%s imported from %s as %s" % fields
2025                         if not _is_type_checking_import(stmt):
2026                             self.add_message("unused-import", args=msg, node=stmt)
2027         del self._to_consume

2029     def _check_metaclasses(self, node):
2030         """ Update consumption analysis for metaclasses. """
2031         consumed = []  # [(scope_locals, consumed_key)]
2032 
2033         for child_node in node.get_children():
2034             if isinstance(child_node, astroid.ClassDef):
2035                 consumed.extend(self._check_classdef_metaclasses(child_node, node))
2036 
2037         # Pop the consumed items, in order to avoid having
2038         # unused-import and unused-variable false positives
2039         for scope_locals, name in consumed:
2040             scope_locals.pop(name, None)

2042     def _check_classdef_metaclasses(self, klass, parent_node):
2043         if not klass._metaclass:
2044             # Skip if this class doesn't use explicitly a metaclass, but inherits it from ancestors
2045             return []
2046 
2047         consumed = []  # [(scope_locals, consumed_key)]
2048         metaclass = klass.metaclass()
2049 
2050         name = None
2051         if isinstance(klass._metaclass, astroid.Name):
2052             name = klass._metaclass.name
2053         elif isinstance(klass._metaclass, astroid.Attribute) and klass._metaclass.expr:
2054             name = klass._metaclass.expr.name
2055         elif metaclass:
2056             name = metaclass.root().name
2057 
2058         found = None
2059         name = METACLASS_NAME_TRANSFORMS.get(name, name)
2060         if name:
2061             # check enclosing scopes starting from most local
2062             for scope_locals, _, _ in self._to_consume[::-1]:
2063                 found = scope_locals.get(name)
2064                 if found:
2065                     consumed.append((scope_locals, name))
2066                     break
2067 
2068         if found is None and not metaclass:
2069             name = None
2070             if isinstance(klass._metaclass, astroid.Name):
2071                 name = klass._metaclass.name
2072             elif (
2073                 isinstance(klass._metaclass, astroid.Attribute)
2074                 and klass._metaclass.expr
2075             ):
2076                 name = klass._metaclass.expr.name
2077 
2078             if name is not None:
2079                 if not (
2080                     name in astroid.Module.scope_attrs
2081                     or utils.is_builtin(name)
2082                     or name in self.config.additional_builtins
2083                     or name in parent_node.locals
2084                 ):
2085                     self.add_message("undefined-variable", node=klass, args=(name,))
2086 
2087         return consumed
