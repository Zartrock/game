94   SPECIAL_OBJ = re.compile("^_{2}[a-z]+_{2}$")
95   DATACLASSES_DECORATORS = frozenset({"dataclass", "attrs"})
96   DATACLASS_IMPORT = "dataclasses"
97   TYPING_NAMEDTUPLE = "typing.NamedTuple"

100  def _is_exempt_from_public_methods(node: astroid.ClassDef) -> bool:
101      """Check if a class is exempt from too-few-public-methods"""
102  
103      # If it's a typing.Namedtuple or an Enum
104      for ancestor in node.ancestors():
105          if ancestor.name == "Enum" and ancestor.root().name == "enum":
106              return True
107          if ancestor.qname() == TYPING_NAMEDTUPLE:
108              return True
109  
110      # Or if it's a dataclass
111      if not node.decorators:
112          return False
113  
114      root_locals = set(node.root().locals)
115      for decorator in node.decorators.nodes:
116          if isinstance(decorator, astroid.Call):
117              decorator = decorator.func
118          if not isinstance(decorator, (astroid.Name, astroid.Attribute)):
119              continue
120          if isinstance(decorator, astroid.Name):
121              name = decorator.name
122          else:
123              name = decorator.attrname
124          if name in DATACLASSES_DECORATORS and (
125              root_locals.intersection(DATACLASSES_DECORATORS)
126              or DATACLASS_IMPORT in root_locals
127          ):
128              return True
129      return False

132  def _count_boolean_expressions(bool_op):
133      """Counts the number of boolean expressions in BoolOp `bool_op` (recursive) 
134   
135      example: a and (b or c or (d and e)) ==> 5 boolean expressions 
136      """
137      nb_bool_expr = 0
138      for bool_expr in bool_op.get_children():
139          if isinstance(bool_expr, BoolOp):
140              nb_bool_expr += _count_boolean_expressions(bool_expr)
141          else:
142              nb_bool_expr += 1
143      return nb_bool_expr

146  def _count_methods_in_class(node):
147      all_methods = sum(1 for method in node.methods() if not method.name.startswith("_"))
148      # Special methods count towards the number of public methods,
149      # but don't count towards there being too many methods.
150      for method in node.mymethods():
151          if SPECIAL_OBJ.search(method.name) and method.name != "__init__":
152              all_methods += 1
153      return all_methods

268      def __init__(self, linter=None):
269          BaseChecker.__init__(self, linter)
270          self.stats = None
271          self._returns = None
272          self._branches = None
273          self._stmts = None
274  
275      def open(self):
276          """initialize visit variables"""
277          self.stats = self.linter.add_stats()
278          self._returns = []
279          self._branches = defaultdict(int)
280          self._stmts = []
281  
282      def _inc_all_stmts(self, amount):
283          for i in range(len(self._stmts)):
284              self._stmts[i] += amount
285  
286      @decorators.cachedproperty
287      def _ignored_argument_names(self):
288          return utils.get_global_option(self, "ignored-argument-names", default=None)

290      @check_messages(
291          "too-many-ancestors",
292          "too-many-instance-attributes",
293          "too-few-public-methods",
294          "too-many-public-methods",
295      )
296      def visit_classdef(self, node):
297          """check size of inheritance hierarchy and number of instance attributes 
298          """
299          nb_parents = len(list(node.ancestors()))
300          if nb_parents > self.config.max_parents:
301              self.add_message(
302                  "too-many-ancestors",
303                  node=node,
304                  args=(nb_parents, self.config.max_parents),
305              )
306  
307          if len(node.instance_attrs) > self.config.max_attributes:
308              self.add_message(
309                  "too-many-instance-attributes",
310                  node=node,
311                  args=(len(node.instance_attrs), self.config.max_attributes),
312              )
313  
314      @check_messages("too-few-public-methods", "too-many-public-methods")
315      def leave_classdef(self, node):
316          """check number of public methods"""
317          my_methods = sum(
318              1 for method in node.mymethods() if not method.name.startswith("_")
319          )
320  
321          # Does the class contain less than n public methods ?
322          # This checks only the methods defined in the current class,
323          # since the user might not have control over the classes
324          # from the ancestors. It avoids some false positives
325          # for classes such as unittest.TestCase, which provides
326          # a lot of assert methods. It doesn't make sense to warn
327          # when the user subclasses TestCase to add his own tests.
328          if my_methods > self.config.max_public_methods:
329              self.add_message(
330                  "too-many-public-methods",
331                  node=node,
332                  args=(my_methods, self.config.max_public_methods),
333              )
334  
335          # Stop here for exception, metaclass, interface classes and other
336          # classes for which we don't need to count the methods.
337          if node.type != "class" or _is_exempt_from_public_methods(node):
338              return
339  
340          # Does the class contain more than n public methods ?
341          # This checks all the methods defined by ancestors and
342          # by the current class.
343          all_methods = _count_methods_in_class(node)
344          if all_methods < self.config.min_public_methods:
345              self.add_message(
346                  "too-few-public-methods",
347                  node=node,
348                  args=(all_methods, self.config.min_public_methods),
349              )
350  
351      @check_messages(
352          "too-many-return-statements",
353          "too-many-branches",
354          "too-many-arguments",
355          "too-many-locals",
356          "too-many-statements",
357          "keyword-arg-before-vararg",
358      )
359      def visit_functiondef(self, node):
360          """check function name, docstring, arguments, redefinition, 
361          variable names, max locals 
362          """
363          # init branch and returns counters
364          self._returns.append(0)
365          # check number of arguments
366          args = node.args.args
367          ignored_argument_names = self._ignored_argument_names
368          if args is not None:
369              ignored_args_num = 0
370              if ignored_argument_names:
371                  ignored_args_num = sum(
372                      1 for arg in args if ignored_argument_names.match(arg.name)
373                  )
374  
375              argnum = len(args) - ignored_args_num
376              if argnum > self.config.max_args:
377                  self.add_message(
378                      "too-many-arguments",
379                      node=node,
380                      args=(len(args), self.config.max_args),
381                  )
382          else:
383              ignored_args_num = 0
384          # check number of local variables
385          locnum = len(node.locals) - ignored_args_num
386          if locnum > self.config.max_locals:
387              self.add_message(
388                  "too-many-locals", node=node, args=(locnum, self.config.max_locals)
389              )
390          # init new statements counter
391          self._stmts.append(1)
392  
393      visit_asyncfunctiondef = visit_functiondef
394  
395      @check_messages(
396          "too-many-return-statements",
397          "too-many-branches",
398          "too-many-arguments",
399          "too-many-locals",
400          "too-many-statements",
401      )
402      def leave_functiondef(self, node):
403          """most of the work is done here on close: 
404          checks for max returns, branch, return in __init__ 
405          """
406          returns = self._returns.pop()
407          if returns > self.config.max_returns:
408              self.add_message(
409                  "too-many-return-statements",
410                  node=node,
411                  args=(returns, self.config.max_returns),
412              )
413          branches = self._branches[node]
414          if branches > self.config.max_branches:
415              self.add_message(
416                  "too-many-branches",
417                  node=node,
418                  args=(branches, self.config.max_branches),
419              )
420          # check number of statements
421          stmts = self._stmts.pop()
422          if stmts > self.config.max_statements:
423              self.add_message(
424                  "too-many-statements",
425                  node=node,
426                  args=(stmts, self.config.max_statements),
427              )
428  
429      leave_asyncfunctiondef = leave_functiondef
430  
431      def visit_return(self, _):
432          """count number of returns"""
433          if not self._returns:
434              return  # return outside function, reported by the base checker
435          self._returns[-1] += 1
436  
437      def visit_default(self, node):
438          """default visit method -> increments the statements counter if 
439          necessary 
440          """
441          if node.is_statement:
442              self._inc_all_stmts(1)
443  
444      def visit_tryexcept(self, node):
445          """increments the branches counter"""
446          branches = len(node.handlers)
447          if node.orelse:
448              branches += 1
449          self._inc_branch(node, branches)
450          self._inc_all_stmts(branches)
451  
452      def visit_tryfinally(self, node):
453          """increments the branches counter"""
454          self._inc_branch(node, 2)
455          self._inc_all_stmts(2)
456  
457      @check_messages("too-many-boolean-expressions")
458      def visit_if(self, node):
459          """increments the branches counter and checks boolean expressions"""
460          self._check_boolean_expressions(node)
461          branches = 1
462          # don't double count If nodes coming from some 'elif'
463          if node.orelse and (len(node.orelse) > 1 or not isinstance(node.orelse[0], If)):
464              branches += 1
465          self._inc_branch(node, branches)
466          self._inc_all_stmts(branches)
467  
468      def _check_boolean_expressions(self, node):
469          """Go through "if" node `node` and counts its boolean expressions 
470   
471          if the "if" node test is a BoolOp node 
472          """
473          condition = node.test
474          if not isinstance(condition, BoolOp):
475              return
476          nb_bool_expr = _count_boolean_expressions(condition)
477          if nb_bool_expr > self.config.max_bool_expr:
478              self.add_message(
479                  "too-many-boolean-expressions",
480                  node=condition,
481                  args=(nb_bool_expr, self.config.max_bool_expr),
482              )
483  
484      def visit_while(self, node):
485          """increments the branches counter"""
486          branches = 1
487          if node.orelse:
488              branches += 1
489          self._inc_branch(node, branches)
490  
491      visit_for = visit_while
492  
493      def _inc_branch(self, node, branchesnum=1):
494          """increments the branches counter"""
495          self._branches[node.scope()] += branchesnum
