111  def is_method_call(func, types=(), methods=()):
112      """Determines if a BoundMethod node represents a method call. 
113   
114      Args: 
115        func (astroid.BoundMethod): The BoundMethod AST node to check. 
116        types (Optional[String]): Optional sequence of caller type names to restrict check. 
117        methods (Optional[String]): Optional sequence of method names to restrict check. 
118   
119      Returns: 
120        bool: true if the node represents a method call for the given type and 
121        method names, False otherwise. 
122      """
123      return (
124          isinstance(func, astroid.BoundMethod)
125          and isinstance(func.bound, astroid.Instance)
126          and (func.bound.name in types if types else True)
127          and (func.name in methods if methods else True)
128      )

131  class LoggingChecker(checkers.BaseChecker):
132      """Checks use of the logging module."""
133  
134      __implements__ = interfaces.IAstroidChecker
135      name = "logging"
136      msgs = MSGS
137  
138      options = (
139          (
140              "logging-modules",
141              {
142                  "default": ("logging",),
143                  "type": "csv",
144                  "metavar": "<comma separated list>",
145                  "help": "Logging modules to check that the string format "
146                  "arguments are in logging function parameter format.",
147              },
148          ),
149          (
150              "logging-format-style",
151              {
152                  "default": "old",
153                  "type": "choice",
154                  "metavar": "<old (%) or new ({)>",
155                  "choices": ["old", "new"],
156                  "help": "The type of string formatting that logging methods do. "
157                  "`old` means using % formatting, `new` is for `{}` formatting.",
158              },
159          ),
160      )
161  
162      def visit_module(self, node):  # pylint: disable=unused-argument
163          """Clears any state left in this checker from last module checked."""
164          # The code being checked can just as easily "import logging as foo",
165          # so it is necessary to process the imports and store in this field
166          # what name the logging module is actually given.
167          self._logging_names = set()
168          logging_mods = self.config.logging_modules
169  
170          self._format_style = self.config.logging_format_style
171  
172          self._logging_modules = set(logging_mods)
173          self._from_imports = {}
174          for logging_mod in logging_mods:
175              parts = logging_mod.rsplit(".", 1)
176              if len(parts) > 1:
177                  self._from_imports[parts[0]] = parts[1]
178  
179      def visit_importfrom(self, node):
180          """Checks to see if a module uses a non-Python logging module."""
181          try:
182              logging_name = self._from_imports[node.modname]
183              for module, as_name in node.names:
184                  if module == logging_name:
185                      self._logging_names.add(as_name or module)
186          except KeyError:
187              pass
188  
189      def visit_import(self, node):
190          """Checks to see if this module uses Python's built-in logging."""
191          for module, as_name in node.names:
192              if module in self._logging_modules:
193                  self._logging_names.add(as_name or module)
194  
195      @check_messages(*MSGS)
196      def visit_call(self, node):
197          """Checks calls to logging methods."""
198  
199          def is_logging_name():
200              return (
201                  isinstance(node.func, astroid.Attribute)
202                  and isinstance(node.func.expr, astroid.Name)
203                  and node.func.expr.name in self._logging_names
204              )
205  
206          def is_logger_class():
207              try:
208                  for inferred in node.func.infer():
209                      if isinstance(inferred, astroid.BoundMethod):
210                          parent = inferred._proxied.parent
211                          if isinstance(parent, astroid.ClassDef) and (
212                              parent.qname() == "logging.Logger"
213                              or any(
214                                  ancestor.qname() == "logging.Logger"
215                                  for ancestor in parent.ancestors()
216                              )
217                          ):
218                              return True, inferred._proxied.name
219              except astroid.exceptions.InferenceError:
220                  pass
221              return False, None
222  
223          if is_logging_name():
224              name = node.func.attrname
225          else:
226              result, name = is_logger_class()
227              if not result:
228                  return
229          self._check_log_method(node, name)
230  
231      def _check_log_method(self, node, name):
232          """Checks calls to logging.log(level, format, *format_args)."""
233          if name == "log":
234              if node.starargs or node.kwargs or len(node.args) < 2:
235                  # Either a malformed call, star args, or double-star args. Beyond
236                  # the scope of this checker.
237                  return
238              format_pos = 1
239          elif name in CHECKED_CONVENIENCE_FUNCTIONS:
240              if node.starargs or node.kwargs or not node.args:
241                  # Either no args, star args, or double-star args. Beyond the
242                  # scope of this checker.
243                  return
244              format_pos = 0
245          else:
246              return
247  
248          if isinstance(node.args[format_pos], astroid.BinOp):
249              binop = node.args[format_pos]
250              emit = binop.op == "%"
251              if binop.op == "+":
252                  total_number_of_strings = sum(
253                      1
254                      for operand in (binop.left, binop.right)
255                      if self._is_operand_literal_str(utils.safe_infer(operand))
256                  )
257                  emit = total_number_of_strings > 0
258              if emit:
259                  self.add_message(
260                      "logging-not-lazy", node=node, args=(self._helper_string(node),),
261                  )
262          elif isinstance(node.args[format_pos], astroid.Call):
263              self._check_call_func(node.args[format_pos])
264          elif isinstance(node.args[format_pos], astroid.Const):
265              self._check_format_string(node, format_pos)
266          elif isinstance(node.args[format_pos], astroid.JoinedStr):
267              self.add_message(
268                  "logging-fstring-interpolation",
269                  node=node,
270                  args=(self._helper_string(node),),
271              )
272  
273      def _helper_string(self, node):
274          """Create a string that lists the valid types of formatting for this node."""
275          valid_types = ["lazy %"]
276  
277          if not self.linter.is_message_enabled(
278              "logging-fstring-formatting", node.fromlineno
279          ):
280              valid_types.append("fstring")
281          if not self.linter.is_message_enabled(
282              "logging-format-interpolation", node.fromlineno
283          ):
284              valid_types.append(".format()")
285          if not self.linter.is_message_enabled("logging-not-lazy", node.fromlineno):
286              valid_types.append("%")
287  
288          return " or ".join(valid_types)
289  
290      @staticmethod
291      def _is_operand_literal_str(operand):
292          """ 
293          Return True if the operand in argument is a literal string 
294          """
295          return isinstance(operand, astroid.Const) and operand.name == "str"
296  
297      def _check_call_func(self, node):
298          """Checks that function call is not format_string.format(). 
299   
300          Args: 
301            node (astroid.node_classes.Call): 
302              Call AST node to be checked. 
303          """
304          func = utils.safe_infer(node.func)
305          types = ("str", "unicode")
306          methods = ("format",)
307          if is_method_call(func, types, methods) and not is_complex_format_str(
308              func.bound
309          ):
310              self.add_message(
311                  "logging-format-interpolation",
312                  node=node,
313                  args=(self._helper_string(node),),
314              )
315  
316      def _check_format_string(self, node, format_arg):
317          """Checks that format string tokens match the supplied arguments. 
318   
319          Args: 
320            node (astroid.node_classes.NodeNG): AST node to be checked. 
321            format_arg (int): Index of the format string in the node arguments. 
322          """
323          num_args = _count_supplied_tokens(node.args[format_arg + 1 :])
324          if not num_args:
325              # If no args were supplied the string is not interpolated and can contain
326              # formatting characters - it's used verbatim. Don't check any further.
327              return
328  
329          format_string = node.args[format_arg].value
330          required_num_args = 0
331          if isinstance(format_string, bytes):
332              format_string = format_string.decode()
333          if isinstance(format_string, str):
334              try:
335                  if self._format_style == "old":
336                      keyword_args, required_num_args, _, _ = utils.parse_format_string(
337                          format_string
338                      )
339                      if keyword_args:
340                          # Keyword checking on logging strings is complicated by
341                          # special keywords - out of scope.
342                          return
343                  elif self._format_style == "new":
344                      (
345                          keyword_arguments,
346                          implicit_pos_args,
347                          explicit_pos_args,
348                      ) = utils.parse_format_method_string(format_string)
349  
350                      keyword_args_cnt = len(
351                          {k for k, l in keyword_arguments if not isinstance(k, int)}
352                      )
353                      required_num_args = (
354                          keyword_args_cnt + implicit_pos_args + explicit_pos_args
355                      )
356              except utils.UnsupportedFormatCharacter as ex:
357                  char = format_string[ex.index]
358                  self.add_message(
359                      "logging-unsupported-format",
360                      node=node,
361                      args=(char, ord(char), ex.index),
362                  )
363                  return
364              except utils.IncompleteFormatString:
365                  self.add_message("logging-format-truncated", node=node)
366                  return
367          if num_args > required_num_args:
368              self.add_message("logging-too-many-args", node=node)
369          elif num_args < required_num_args:
370              self.add_message("logging-too-few-args", node=node)

373  def is_complex_format_str(node):
374      """Checks if node represents a string with complex formatting specs. 
375   
376      Args: 
377          node (astroid.node_classes.NodeNG): AST node to check 
378      Returns: 
379          bool: True if inferred string uses complex formatting, False otherwise 
380      """
381      inferred = utils.safe_infer(node)
382      if inferred is None or not (
383          isinstance(inferred, astroid.Const) and isinstance(inferred.value, str)
384      ):
385          return True
386      try:
387          parsed = list(string.Formatter().parse(inferred.value))
388      except ValueError:
389          # This format string is invalid
390          return False
391      for _, _, format_spec, _ in parsed:
392          if format_spec:
393              return True
394      return False

397  def _count_supplied_tokens(args):
398      """Counts the number of tokens in an args list. 
399   
400      The Python log functions allow for special keyword arguments: func, 
401      exc_info and extra. To handle these cases correctly, we only count 
402      arguments that aren't keywords. 
403   
404      Args: 
405        args (list): AST nodes that are arguments for a log format string. 
406   
407      Returns: 
408        int: Number of AST nodes that aren't keywords. 
409      """
410      return sum(1 for arg in args if not isinstance(arg, astroid.Keyword))
