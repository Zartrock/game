116  class ArrayInterface(object):
117      def __init__(self, arr):
118          try:
119              self._cobj = arr.__array_struct__
120          except AttributeError:
121              raise TypeError("The array object lacks an array structure")
122          if not self._cobj:
123              raise TypeError("The array object has a NULL array structure value")
124          try:
125              vp = PyCObject_AsVoidPtr(self._cobj)
126          except TypeError:
127              if PyCapsule_IsValid(self._cobj, None):
128                  vp = PyCapsule_GetPointer(self._cobj, None)
129              else:
130                  raise TypeError("The array object has an invalid array structure")
131              self.desc = PyCapsule_GetContext(self._cobj)
132          else:
133              self.desc = PyCObject_GetDesc(self._cobj)
134          self._inter = cast(vp, PAI_Ptr)[0]
135  
136      def __getattr__(self, name):
137          if name == "typekind":
138              return self._inter.typekind.decode("latin-1")
139          return getattr(self._inter, name)
140  
141      def __str__(self):
142          if isinstance(self.desc, tuple):
143              ver = self.desc[0]
144          else:
145              ver = "N/A"
146          return (
147              "nd: %i\n"
148              "typekind: %s\n"
149              "itemsize: %i\n"
150              "flags: %s\n"
151              "shape: %s\n"
152              "strides: %s\n"
153              "ver: %s\n"
154              % (
155                  self.nd,
156                  self.typekind,
157                  self.itemsize,
158                  format_flags(self.flags),
159                  format_shape(self.nd, self.shape),
160                  format_strides(self.nd, self.strides),
161                  ver,
162              )
163          )

166  def format_flags(flags):
167      names = []
168      for flag, name in [
169          (PAI_CONTIGUOUS, "CONTIGUOUS"),
170          (PAI_FORTRAN, "FORTRAN"),
171          (PAI_ALIGNED, "ALIGNED"),
172          (PAI_NOTSWAPPED, "NOTSWAPPED"),
173          (PAI_WRITEABLE, "WRITEABLE"),
174          (PAI_ARR_HAS_DESCR, "ARR_HAS_DESCR"),
175      ]:
176          if flag & flags:
177              names.append(name)
178      return ", ".join(names)

182      return ", ".join([str(shape[i]) for i in range(nd)])

186      return ", ".join([str(strides[i]) for i in range(nd)])

189  class Exporter(object):
190      def __init__(
191          self, shape, typekind=None, itemsize=None, strides=None, descr=None, flags=None
192      ):
193          if typekind is None:
194              typekind = "u"
195          if itemsize is None:
196              itemsize = 1
197          if flags is None:
198              flags = PAI_WRITEABLE | PAI_ALIGNED | PAI_NOTSWAPPED
199          if descr is not None:
200              flags |= PAI_ARR_HAS_DESCR
201          if len(typekind) != 1:
202              raise ValueError("Argument 'typekind' must be length 1 string")
203          nd = len(shape)
204          self.typekind = typekind
205          self.itemsize = itemsize
206          self.nd = nd
207          self.shape = tuple(shape)
208          self._shape = (c_ssize_t * self.nd)(*self.shape)
209          if strides is None:
210              self._strides = (c_ssize_t * self.nd)()
211              self._strides[self.nd - 1] = self.itemsize
212              for i in range(self.nd - 1, 0, -1):
213                  self._strides[i - 1] = self.shape[i] * self._strides[i]
214              strides = tuple(self._strides)
215              self.strides = strides
216          elif len(strides) == nd:
217              self.strides = tuple(strides)
218              self._strides = (c_ssize_t * self.nd)(*self.strides)
219          else:
220              raise ValueError("Mismatch in length of strides and shape")
221          self.descr = descr
222          if self.is_contiguous("C"):
223              flags |= PAI_CONTIGUOUS
224          if self.is_contiguous("F"):
225              flags |= PAI_FORTRAN
226          self.flags = flags
227          sz = max(shape[i] * strides[i] for i in range(nd))
228          self._data = (c_ubyte * sz)()
229          self.data = addressof(self._data)
230          self._inter = PyArrayInterface(
231              2,
232              nd,
233              typekind.encode("latin_1"),
234              itemsize,
235              flags,
236              self._shape,
237              self._strides,
238              self.data,
239              descr,
240          )
241          self.len = itemsize
242          for i in range(nd):
243              self.len *= self.shape[i]
244  
245      __array_struct__ = property(lambda self: capsule_new(self._inter))
246  
247      def is_contiguous(self, fortran):
248          if fortran in "CA":
249              if self.strides[-1] == self.itemsize:
250                  for i in range(self.nd - 1, 0, -1):
251                      if self.strides[i - 1] != self.shape[i] * self.strides[i]:
252                          break
253                  else:
254                      return True
255          if fortran in "FA":
256              if self.strides[0] == self.itemsize:
257                  for i in range(0, self.nd - 1):
258                      if self.strides[i + 1] != self.shape[i] * self.strides[i]:
259                          break
260                  else:
261                      return True
262          return False

265  class Array(Exporter):
266      _ctypes = {
267          ("u", 1): c_uint8,
268          ("u", 2): c_uint16,
269          ("u", 4): c_uint32,
270          ("u", 8): c_uint64,
271          ("i", 1): c_int8,
272          ("i", 2): c_int16,
273          ("i", 4): c_int32,
274          ("i", 8): c_int64,
275      }
276  
277      def __init__(self, *args, **kwds):
278          super(Array, self).__init__(*args, **kwds)
279          try:
280              if self.flags & PAI_NOTSWAPPED:
281                  ct = self._ctypes[self.typekind, self.itemsize]
282              elif c_int.__ctype_le__ is c_int:
283                  ct = self._ctypes[self.typekind, self.itemsize].__ctype_be__
284              else:
285                  ct = self._ctypes[self.typekind, self.itemsize].__ctype_le__
286          except KeyError:
287              ct = c_uint8 * self.itemsize
288          self._ctype = ct
289          self._ctype_p = POINTER(ct)
290  
291      def __getitem__(self, key):
292          return cast(self._addr_at(key), self._ctype_p)[0]
293  
294      def __setitem__(self, key, value):
295          cast(self._addr_at(key), self._ctype_p)[0] = value
296  
297      def _addr_at(self, key):
298          if not isinstance(key, tuple):
299              key = (key,)
300          if len(key) != self.nd:
301              raise ValueError("wrong number of indexes")
302          for i in range(self.nd):
303              if not (0 <= key[i] < self.shape[i]):
304                  raise IndexError("index {} out of range".format(i))
305          return self.data + sum(i * s for i, s in zip(key, self.strides))

321      def test_is_contiguous(self):
322          a = Exporter((10,), itemsize=2)
323          self.assertTrue(a.is_contiguous("C"))
324          self.assertTrue(a.is_contiguous("F"))
325          self.assertTrue(a.is_contiguous("A"))
326          a = Exporter((10, 4), itemsize=2)
327          self.assertTrue(a.is_contiguous("C"))
328          self.assertTrue(a.is_contiguous("A"))
329          self.assertFalse(a.is_contiguous("F"))
330          a = Exporter((13, 5, 11, 3), itemsize=2, strides=(330, 66, 6, 2))
331          self.assertTrue(a.is_contiguous("C"))
332          self.assertTrue(a.is_contiguous("A"))
333          self.assertFalse(a.is_contiguous("F"))
334          a = Exporter((10, 4), itemsize=2, strides=(2, 20))
335          self.assertTrue(a.is_contiguous("F"))
336          self.assertTrue(a.is_contiguous("A"))
337          self.assertFalse(a.is_contiguous("C"))
338          a = Exporter((13, 5, 11, 3), itemsize=2, strides=(2, 26, 130, 1430))
339          self.assertTrue(a.is_contiguous("F"))
340          self.assertTrue(a.is_contiguous("A"))
341          self.assertFalse(a.is_contiguous("C"))
342          a = Exporter((2, 11, 6, 4), itemsize=2, strides=(576, 48, 8, 2))
343          self.assertFalse(a.is_contiguous("A"))
344          a = Exporter((2, 11, 6, 4), itemsize=2, strides=(2, 4, 48, 288))
345          self.assertFalse(a.is_contiguous("A"))
346          a = Exporter((3, 2, 2), itemsize=2, strides=(16, 8, 4))
347          self.assertFalse(a.is_contiguous("A"))
348          a = Exporter((3, 2, 2), itemsize=2, strides=(4, 12, 24))
349          self.assertFalse(a.is_contiguous("A"))

351      def check_args(
352          self, call_flags, shape, typekind, strides, length, bufsize, itemsize, offset=0
353      ):
354          if call_flags & 1:
355              typekind_arg = typekind
356          else:
357              typekind_arg = None
358          if call_flags & 2:
359              strides_arg = strides
360          else:
361              strides_arg = None
362          a = Exporter(shape, itemsize=itemsize, strides=strides_arg)
363          self.assertEqual(sizeof(a._data), bufsize)
364          self.assertEqual(a.data, ctypes.addressof(a._data) + offset)
365          m = ArrayInterface(a)
366          self.assertEqual(m.data, a.data)
367          self.assertEqual(m.itemsize, itemsize)
368          self.assertEqual(tuple(m.shape[0 : m.nd]), shape)
369          self.assertEqual(tuple(m.strides[0 : m.nd]), strides)

436          a = Array((1,), "u", 4, flags=(PAI_ALIGNED | PAI_WRITEABLE))
437          ct = a._ctype
