30   def space_indentation(s):
31       """The number of leading spaces in a string 
32    
33       :param str s: input string 
34    
35       :rtype: int 
36       :return: number of leading spaces 
37       """
38       return len(s) - len(s.lstrip(" "))

41   def get_setters_property_name(node):
42       """Get the name of the property that the given node is a setter for. 
43    
44       :param node: The node to get the property name for. 
45       :type node: str 
46    
47       :rtype: str or None 
48       :returns: The name of the property that the node is a setter for, 
49           or None if one could not be found. 
50       """
51       decorators = node.decorators.nodes if node.decorators else []
52       for decorator in decorators:
53           if (
54               isinstance(decorator, astroid.Attribute)
55               and decorator.attrname == "setter"
56               and isinstance(decorator.expr, astroid.Name)
57           ):
58               return decorator.expr.name
59       return None

62   def get_setters_property(node):
63       """Get the property node for the given setter node. 
64    
65       :param node: The node to get the property for. 
66       :type node: astroid.FunctionDef 
67    
68       :rtype: astroid.FunctionDef or None 
69       :returns: The node relating to the property of the given setter node, 
70           or None if one could not be found. 
71       """
72       property_ = None
73   
74       property_name = get_setters_property_name(node)
75       class_node = utils.node_frame_class(node)
76       if property_name and class_node:
77           class_attrs = class_node.getattr(node.name)
78           for attr in class_attrs:
79               if utils.decorated_with_property(attr):
80                   property_ = attr
81                   break
82   
83       return property_

86   def returns_something(return_node):
87       """Check if a return node returns a value other than None. 
88    
89       :param return_node: The return node to check. 
90       :type return_node: astroid.Return 
91    
92       :rtype: bool 
93       :return: True if the return node returns a value other than None, 
94           False otherwise. 
95       """
96       returns = return_node.value
97   
98       if returns is None:
99           return False
100  
101      return not (isinstance(returns, astroid.Const) and returns.value is None)

104  def _get_raise_target(node):
105      if isinstance(node.exc, astroid.Call):
106          func = node.exc.func
107          if isinstance(func, (astroid.Name, astroid.Attribute)):
108              return utils.safe_infer(func)
109      return None

112  def _split_multiple_exc_types(target: str) -> List[str]:
113      delimiters = r"(\s*,(?:\s*or\s)?\s*|\s+or\s+)"
114      return re.split(delimiters, target)

117  def possible_exc_types(node):
118      """ 
119      Gets all of the possible raised exception types for the given raise node. 
120   
121      .. note:: 
122   
123          Caught exception types are ignored. 
124   
125   
126      :param node: The raise node to find exception types for. 
127      :type node: astroid.node_classes.NodeNG 
128   
129      :returns: A list of exception types possibly raised by :param:`node`. 
130      :rtype: set(str) 
131      """
132      excs = []
133      if isinstance(node.exc, astroid.Name):
134          inferred = utils.safe_infer(node.exc)
135          if inferred:
136              excs = [inferred.name]
137      elif node.exc is None:
138          handler = node.parent
139          while handler and not isinstance(handler, astroid.ExceptHandler):
140              handler = handler.parent
141  
142          if handler and handler.type:
143              inferred_excs = astroid.unpack_infer(handler.type)
144              excs = (exc.name for exc in inferred_excs if exc is not astroid.Uninferable)
145      else:
146          target = _get_raise_target(node)
147          if isinstance(target, astroid.ClassDef):
148              excs = [target.name]
149          elif isinstance(target, astroid.FunctionDef):
150              for ret in target.nodes_of_class(astroid.Return):
151                  if ret.frame() != target:
152                      # return from inner function - ignore it
153                      continue
154  
155                  val = utils.safe_infer(ret.value)
156                  if (
157                      val
158                      and isinstance(val, (astroid.Instance, astroid.ClassDef))
159                      and utils.inherit_from_std_ex(val)
160                  ):
161                      excs.append(val.name)
162  
163      try:
164          return {exc for exc in excs if not utils.node_ignores_exception(node, exc)}
165      except astroid.InferenceError:
166          return set()

169  def docstringify(docstring, default_type="default"):
170      for docstring_type in [
171          SphinxDocstring,
172          EpytextDocstring,
173          GoogleDocstring,
174          NumpyDocstring,
175      ]:
176          instance = docstring_type(docstring)
177          if instance.is_valid():
178              return instance
179  
180      docstring_type = DOCSTRING_TYPES.get(default_type, Docstring)
181      return docstring_type(docstring)

184  class Docstring:
185      re_for_parameters_see = re.compile(
186          r""" 
187          For\s+the\s+(other)?\s*parameters\s*,\s+see 
188          """,
189          re.X | re.S,
190      )
191  
192      supports_yields = None
193      """True if the docstring supports a "yield" section. 
194   
195      False if the docstring uses the returns section to document generators. 
196      """
197  
198      # These methods are designed to be overridden
199      # pylint: disable=no-self-use
200      def __init__(self, doc):
201          doc = doc or ""
202          self.doc = doc.expandtabs()
203  
204      def is_valid(self):
205          return False
206  
207      def exceptions(self):
208          return set()
209  
210      def has_params(self):
211          return False
212  
213      def has_returns(self):
214          return False
215  
216      def has_rtype(self):
217          return False
218  
219      def has_property_returns(self):
220          return False
221  
222      def has_property_type(self):
223          return False
224  
225      def has_yields(self):
226          return False
227  
228      def has_yields_type(self):
229          return False
230  
231      def match_param_docs(self):
232          return set(), set()
233  
234      def params_documented_elsewhere(self):
235          return self.re_for_parameters_see.search(self.doc) is not None

238  class SphinxDocstring(Docstring):
239      re_type = r""" 
240          [~!.]?               # Optional link style prefix 
241          \w(?:\w|\.[^\.])*    # Valid python name 
242          """
243  
244      re_simple_container_type = r""" 
245          {type}                        # a container type 
246          [\(\[] [^\n\s]+ [\)\]]        # with the contents of the container 
247      """.format(
248          type=re_type
249      )
250  
251      re_multiple_simple_type = r""" 
252          (?:{container_type}|{type}) 
253          (?:(?:\s+(?:of|or)\s+|\s*,\s*)(?:{container_type}|{type}))* 
254      """.format(
255          type=re_type, container_type=re_simple_container_type
256      )
257  
258      re_xref = r""" 
259          (?::\w+:)?                    # optional tag 
260          `{}`                         # what to reference 
261          """.format(
262          re_type
263      )
264  
265      re_param_raw = r""" 
266          :                       # initial colon 
267          (?:                     # Sphinx keywords 
268          param|parameter| 
269          arg|argument| 
270          key|keyword 
271          ) 
272          \s+                     # whitespace 
273   
274          (?:                     # optional type declaration 
275          ({type}|{container_type}) 
276          \s+ 
277          )? 
278   
279          (\w+)                   # Parameter name 
280          \s*                     # whitespace 
281          :                       # final colon 
282          """.format(
283          type=re_type, container_type=re_simple_container_type
284      )
285      re_param_in_docstring = re.compile(re_param_raw, re.X | re.S)
286  
287      re_type_raw = r""" 
288          :type                   # Sphinx keyword 
289          \s+                     # whitespace 
290          ({type})                # Parameter name 
291          \s*                     # whitespace 
292          :                       # final colon 
293          """.format(
294          type=re_multiple_simple_type
295      )
296      re_type_in_docstring = re.compile(re_type_raw, re.X | re.S)
297  
298      re_property_type_raw = r""" 
299          :type:                  # Sphinx keyword 
300          \s+                     # whitespace 
301          {type}                  # type declaration 
302          """.format(
303          type=re_multiple_simple_type
304      )
305      re_property_type_in_docstring = re.compile(re_property_type_raw, re.X | re.S)
306  
307      re_raise_raw = r""" 
308          :                       # initial colon 
309          (?:                     # Sphinx keyword 
310          raises?| 
311          except|exception 
312          ) 
313          \s+                     # whitespace 
314          ({type})                # exception type 
315          \s*                     # whitespace 
316          :                       # final colon 
317          """.format(
318          type=re_multiple_simple_type
319      )
320      re_raise_in_docstring = re.compile(re_raise_raw, re.X | re.S)
321  
322      re_rtype_in_docstring = re.compile(r":rtype:")
323  
324      re_returns_in_docstring = re.compile(r":returns?:")
325  
326      supports_yields = False
327  
328      def is_valid(self):
329          return bool(
330              self.re_param_in_docstring.search(self.doc)
331              or self.re_raise_in_docstring.search(self.doc)
332              or self.re_rtype_in_docstring.search(self.doc)
333              or self.re_returns_in_docstring.search(self.doc)
334              or self.re_property_type_in_docstring.search(self.doc)
335          )
336  
337      def exceptions(self):
338          types = set()
339  
340          for match in re.finditer(self.re_raise_in_docstring, self.doc):
341              raise_type = match.group(1)
342              types.update(_split_multiple_exc_types(raise_type))
343  
344          return types
345  
346      def has_params(self):
347          if not self.doc:
348              return False
349  
350          return self.re_param_in_docstring.search(self.doc) is not None
351  
352      def has_returns(self):
353          if not self.doc:
354              return False
355  
356          return bool(self.re_returns_in_docstring.search(self.doc))
357  
358      def has_rtype(self):
359          if not self.doc:
360              return False
361  
362          return bool(self.re_rtype_in_docstring.search(self.doc))
363  
364      def has_property_returns(self):
365          if not self.doc:
366              return False
367  
368          # The summary line is the return doc,
369          # so the first line must not be a known directive.
370          return not self.doc.lstrip().startswith(":")
371  
372      def has_property_type(self):
373          if not self.doc:
374              return False
375  
376          return bool(self.re_property_type_in_docstring.search(self.doc))
377  
378      def match_param_docs(self):
379          params_with_doc = set()
380          params_with_type = set()
381  
382          for match in re.finditer(self.re_param_in_docstring, self.doc):
383              name = match.group(2)
384              params_with_doc.add(name)
385              param_type = match.group(1)
386              if param_type is not None:
387                  params_with_type.add(name)
388  
389          params_with_type.update(re.findall(self.re_type_in_docstring, self.doc))
390          return params_with_doc, params_with_type

393  class EpytextDocstring(SphinxDocstring):
394      """ 
395      Epytext is similar to Sphinx. See the docs: 
396          http://epydoc.sourceforge.net/epytext.html 
397          http://epydoc.sourceforge.net/fields.html#fields 
398   
399      It's used in PyCharm: 
400          https://www.jetbrains.com/help/pycharm/2016.1/creating-documentation-comments.html#d848203e314 
401          https://www.jetbrains.com/help/pycharm/2016.1/using-docstrings-to-specify-types.html 
402      """
403  
404      re_param_in_docstring = re.compile(
405          SphinxDocstring.re_param_raw.replace(":", "@", 1), re.X | re.S
406      )
407  
408      re_type_in_docstring = re.compile(
409          SphinxDocstring.re_type_raw.replace(":", "@", 1), re.X | re.S
410      )
411  
412      re_property_type_in_docstring = re.compile(
413          SphinxDocstring.re_property_type_raw.replace(":", "@", 1), re.X | re.S
414      )
415  
416      re_raise_in_docstring = re.compile(
417          SphinxDocstring.re_raise_raw.replace(":", "@", 1), re.X | re.S
418      )
419  
420      re_rtype_in_docstring = re.compile(
421          r""" 
422          @                       # initial "at" symbol 
423          (?:                     # Epytext keyword 
424          rtype|returntype 
425          ) 
426          :                       # final colon 
427          """,
428          re.X | re.S,
429      )
430  
431      re_returns_in_docstring = re.compile(r"@returns?:")
432  
433      def has_property_returns(self):
434          if not self.doc:
435              return False
436  
437          # If this is a property docstring, the summary is the return doc.
438          if self.has_property_type():
439              # The summary line is the return doc,
440              # so the first line must not be a known directive.
441              return not self.doc.lstrip().startswith("@")
442  
443          return False

446  class GoogleDocstring(Docstring):
447      re_type = SphinxDocstring.re_type
448  
449      re_xref = SphinxDocstring.re_xref
450  
451      re_container_type = r""" 
452          (?:{type}|{xref})             # a container type 
453          [\(\[] [^\n]+ [\)\]]          # with the contents of the container 
454      """.format(
455          type=re_type, xref=re_xref
456      )
457  
458      re_multiple_type = r""" 
459          (?:{container_type}|{type}|{xref}) 
460          (?:(?:\s+(?:of|or)\s+|\s*,\s*)(?:{container_type}|{type}|{xref}))* 
461      """.format(
462          type=re_type, xref=re_xref, container_type=re_container_type
463      )
464  
465      _re_section_template = r""" 
466          ^([ ]*)   {0} \s*:   \s*$     # Google parameter header 
467          (  .* )                       # section 
468          """
469  
470      re_param_section = re.compile(
471          _re_section_template.format(r"(?:Args|Arguments|Parameters)"),
472          re.X | re.S | re.M,
473      )
474  
475      re_keyword_param_section = re.compile(
476          _re_section_template.format(r"Keyword\s(?:Args|Arguments|Parameters)"),
477          re.X | re.S | re.M,
478      )
479  
480      re_param_line = re.compile(
481          r""" 
482          \s*  \*{{0,2}}(\w+)             # identifier potentially with asterisks 
483          \s*  ( [(] 
484              {type} 
485              (?:,\s+optional)? 
486              [)] )? \s* :                # optional type declaration 
487          \s*  (.*)                       # beginning of optional description 
488      """.format(
489              type=re_multiple_type
490          ),
491          re.X | re.S | re.M,
492      )
493  
494      re_raise_section = re.compile(
495          _re_section_template.format(r"Raises"), re.X | re.S | re.M
496      )
497  
498      re_raise_line = re.compile(
499          r""" 
500          \s*  ({type}) \s* :              # identifier 
501          \s*  (.*)                        # beginning of optional description 
502      """.format(
503              type=re_multiple_type
504          ),
505          re.X | re.S | re.M,
506      )
507  
508      re_returns_section = re.compile(
509          _re_section_template.format(r"Returns?"), re.X | re.S | re.M
510      )
511  
512      re_returns_line = re.compile(
513          r""" 
514          \s* ({type}:)?                    # identifier 
515          \s* (.*)                          # beginning of description 
516      """.format(
517              type=re_multiple_type
518          ),
519          re.X | re.S | re.M,
520      )
521  
522      re_property_returns_line = re.compile(
523          r""" 
524          ^{type}:                       # indentifier 
525          \s* (.*)                       # Summary line / description 
526      """.format(
527              type=re_multiple_type
528          ),
529          re.X | re.S | re.M,
530      )
531  
532      re_yields_section = re.compile(
533          _re_section_template.format(r"Yields?"), re.X | re.S | re.M
534      )
535  
536      re_yields_line = re_returns_line
537  
538      supports_yields = True
539  
540      def is_valid(self):
541          return bool(
542              self.re_param_section.search(self.doc)
543              or self.re_raise_section.search(self.doc)
544              or self.re_returns_section.search(self.doc)
545              or self.re_yields_section.search(self.doc)
546              or self.re_property_returns_line.search(self._first_line())
547          )
548  
549      def has_params(self):
550          if not self.doc:
551              return False
552  
553          return self.re_param_section.search(self.doc) is not None
554  
555      def has_returns(self):
556          if not self.doc:
557              return False
558  
559          entries = self._parse_section(self.re_returns_section)
560          for entry in entries:
561              match = self.re_returns_line.match(entry)
562              if not match:
563                  continue
564  
565              return_desc = match.group(2)
566              if return_desc:
567                  return True
568  
569          return False
570  
571      def has_rtype(self):
572          if not self.doc:
573              return False
574  
575          entries = self._parse_section(self.re_returns_section)
576          for entry in entries:
577              match = self.re_returns_line.match(entry)
578              if not match:
579                  continue
580  
581              return_type = match.group(1)
582              if return_type:
583                  return True
584  
585          return False
586  
587      def has_property_returns(self):
588          # The summary line is the return doc,
589          # so the first line must not be a known directive.
590          first_line = self._first_line()
591          return not bool(
592              self.re_param_section.search(first_line)
593              or self.re_raise_section.search(first_line)
594              or self.re_returns_section.search(first_line)
595              or self.re_yields_section.search(first_line)
596          )
597  
598      def has_property_type(self):
599          if not self.doc:
600              return False
601  
602          return bool(self.re_property_returns_line.match(self._first_line()))
603  
604      def has_yields(self):
605          if not self.doc:
606              return False
607  
608          entries = self._parse_section(self.re_yields_section)
609          for entry in entries:
610              match = self.re_yields_line.match(entry)
611              if not match:
612                  continue
613  
614              yield_desc = match.group(2)
615              if yield_desc:
616                  return True
617  
618          return False
619  
620      def has_yields_type(self):
621          if not self.doc:
622              return False
623  
624          entries = self._parse_section(self.re_yields_section)
625          for entry in entries:
626              match = self.re_yields_line.match(entry)
627              if not match:
628                  continue
629  
630              yield_type = match.group(1)
631              if yield_type:
632                  return True
633  
634          return False
635  
636      def exceptions(self):
637          types = set()
638  
639          entries = self._parse_section(self.re_raise_section)
640          for entry in entries:
641              match = self.re_raise_line.match(entry)
642              if not match:
643                  continue
644  
645              exc_type = match.group(1)
646              exc_desc = match.group(2)
647              if exc_desc:
648                  types.update(_split_multiple_exc_types(exc_type))
649  
650          return types
651  
652      def match_param_docs(self):
653          params_with_doc = set()
654          params_with_type = set()
655  
656          entries = self._parse_section(self.re_param_section)
657          entries.extend(self._parse_section(self.re_keyword_param_section))
658          for entry in entries:
659              match = self.re_param_line.match(entry)
660              if not match:
661                  continue
662  
663              param_name = match.group(1)
664              param_type = match.group(2)
665              param_desc = match.group(3)
666              if param_type:
667                  params_with_type.add(param_name)
668  
669              if param_desc:
670                  params_with_doc.add(param_name)
671  
672          return params_with_doc, params_with_type
673  
674      def _first_line(self):
675          return self.doc.lstrip().split("\n", 1)[0]
676  
677      @staticmethod
678      def min_section_indent(section_match):
679          return len(section_match.group(1)) + 1
680  
681      @staticmethod
682      def _is_section_header(_):
683          # Google parsing does not need to detect section headers,
684          # because it works off of indentation level only
685          return False
686  
687      def _parse_section(self, section_re):
688          section_match = section_re.search(self.doc)
689          if section_match is None:
690              return []
691  
692          min_indentation = self.min_section_indent(section_match)
693  
694          entries = []
695          entry = []
696          is_first = True
697          for line in section_match.group(2).splitlines():
698              if not line.strip():
699                  continue
700              indentation = space_indentation(line)
701              if indentation < min_indentation:
702                  break
703  
704              # The first line after the header defines the minimum
705              # indentation.
706              if is_first:
707                  min_indentation = indentation
708                  is_first = False
709  
710              if indentation == min_indentation:
711                  if self._is_section_header(line):
712                      break
713                  # Lines with minimum indentation must contain the beginning
714                  # of a new parameter documentation.
715                  if entry:
716                      entries.append("\n".join(entry))
717                      entry = []
718  
719              entry.append(line)
720  
721          if entry:
722              entries.append("\n".join(entry))
723  
724          return entries

727  class NumpyDocstring(GoogleDocstring):
728      _re_section_template = r""" 
729          ^([ ]*)   {0}   \s*?$          # Numpy parameters header 
730          \s*     [-=]+   \s*?$          # underline 
731          (  .* )                        # section 
732      """
