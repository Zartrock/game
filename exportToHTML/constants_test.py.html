149  def create_overlap_set(constant_names):
150      """Helper function to find overlapping constant values/names. 
151   
152      Returns a set of fronzensets: 
153          set(frozenset(names of overlapping constants), ...) 
154      """
155      # Create an overlap dict.
156      overlap_dict = {}
157  
158      for name in constant_names:
159          value = getattr(pygame.constants, name)
160          overlap_dict.setdefault(value, set()).add(name)
161  
162      # Get all entries with more than 1 value.
163      overlaps = set()
164  
165      for overlap_names in overlap_dict.values():
166          if len(overlap_names) > 1:
167              overlaps.add(frozenset(overlap_names))
168  
169      return overlaps

407              "KMOD_NONE",

409              "KMOD_RSHIFT",
410              "KMOD_LCTRL",

421          kmods = 0

423          for name in NO_BITWISE_OVERLAP:
424              value = getattr(pygame.constants, name)

426              self.assertFalse(kmods & value)

428              kmods |= value

430      def test_kmod__bitwise_overlap(self):
431          """Ensures certain KMOD constants have overlapping bits."""
432          # KMODS that are comprised of other KMODs.
433          KMOD_COMPRISED_DICT = {
434              "KMOD_SHIFT": ("KMOD_LSHIFT", "KMOD_RSHIFT"),
435              "KMOD_CTRL": ("KMOD_LCTRL", "KMOD_RCTRL"),
436              "KMOD_ALT": ("KMOD_LALT", "KMOD_RALT"),
437              "KMOD_META": ("KMOD_LMETA", "KMOD_RMETA"),
438          }
439  
440          if SDL2:
441              KMOD_COMPRISED_DICT.update({"KMOD_GUI": ("KMOD_LGUI", "KMOD_RGUI")})
442  
443          for base_name, seq_names in KMOD_COMPRISED_DICT.items():
444              expected_value = 0  # Reset.
445  
446              for name in seq_names:
447                  expected_value |= getattr(pygame.constants, name)
448  
449              value = getattr(pygame.constants, base_name)
450  
451              self.assertEqual(value, expected_value)
