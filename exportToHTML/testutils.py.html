60   def _get_tests_info(input_dir, msg_dir, prefix, suffix):
61       """get python input examples and output messages 
62    
63       We use following conventions for input files and messages: 
64       for different inputs: 
65           test for python  >= x.y    ->  input   =  <name>_pyxy.py 
66           test for python  <  x.y    ->  input   =  <name>_py_xy.py 
67       for one input and different messages: 
68           message for python >=  x.y ->  message =  <name>_pyxy.txt 
69           lower versions             ->  message with highest num 
70       """
71       result = []
72       for fname in glob(join(input_dir, prefix + "*" + suffix)):
73           infile = basename(fname)
74           fbase = splitext(infile)[0]
75           # filter input files :
76           pyrestr = fbase.rsplit("_py", 1)[-1]  # like _26 or 26
77           if pyrestr.isdigit():  # '24', '25'...
78               if SYS_VERS_STR < pyrestr:
79                   continue
80           if pyrestr.startswith("_") and pyrestr[1:].isdigit():
81               # skip test for higher python versions
82               if SYS_VERS_STR >= pyrestr[1:]:
83                   continue
84           messages = glob(join(msg_dir, fbase + "*.txt"))
85           # the last one will be without ext, i.e. for all or upper versions:
86           if messages:
87               for outfile in sorted(messages, reverse=True):
88                   py_rest = outfile.rsplit("_py", 1)[-1][:-4]
89                   if py_rest.isdigit() and SYS_VERS_STR >= py_rest:
90                       break
91           else:
92               # This will provide an error message indicating the missing filename.
93               outfile = join(msg_dir, fbase + ".txt")
94           result.append((infile, outfile))
95       return result

98   class TestReporter(BaseReporter):
99       """reporter storing plain text messages"""
100  
101      __implements__ = interfaces.IReporter
102  
103      def __init__(self):  # pylint: disable=super-init-not-called
104  
105          self.message_ids = {}
106          self.reset()
107          self.path_strip_prefix = getcwd() + sep
108  
109      def reset(self):
110          self.out = StringIO()
111          self.messages = []
112  
113      def handle_message(self, msg):
114          """manage message of different type and in the context of path """
115          obj = msg.obj
116          line = msg.line
117          msg_id = msg.msg_id
118          msg = msg.msg
119          self.message_ids[msg_id] = 1
120          if obj:
121              obj = ":%s" % obj
122          sigle = msg_id[0]
123          if linesep != "\n":
124              # 2to3 writes os.linesep instead of using
125              # the previosly used line separators
126              msg = msg.replace("\r\n", "\n")
127          self.messages.append("%s:%3s%s: %s" % (sigle, line, obj, msg))
128  
129      def finalize(self):
130          self.messages.sort()
131          for msg in self.messages:
132              print(msg, file=self.out)
133          result = self.out.getvalue()
134          self.reset()
135          return result
136

138      def on_set_current_module(self, module, filepath):
139          pass
140  
141      # pylint: enable=unused-argument
142  
143      def display_reports(self, layout):
144          """ignore layouts"""

160      collections.namedtuple("Message", ["msg_id", "line", "node", "args", "confidence"])
161  ):
162      def __new__(cls, msg_id, line=None, node=None, args=None, confidence=None):
163          return tuple.__new__(cls, (msg_id, line, node, args, confidence))

166          if isinstance(other, Message):
167              if self.confidence and other.confidence:
168                  return super().__eq__(other)
169              return self[:-1] == other[:-1]
170          return NotImplemented  # pragma: no cover

209  def set_config(**kwargs):
210      """Decorator for setting config values on a checker."""
211  
212      def _wrapper(fun):
213          @functools.wraps(fun)
214          def _forward(self):
215              for key, value in kwargs.items():
216                  setattr(self.checker.config, key, value)
217              if isinstance(self, CheckerTestCase):
218                  # reopen checker in case, it may be interested in configuration change
219                  self.checker.open()
220              fun(self)
221  
222          return _forward
223  
224      return _wrapper

227  class CheckerTestCase:
228      """A base testcase class for unit testing individual checker classes."""
229  
230      CHECKER_CLASS = None
231      CONFIG = {}
232  
233      def setup_method(self):
234          self.linter = UnittestLinter()
235          self.checker = self.CHECKER_CLASS(self.linter)  # pylint: disable=not-callable
236          for key, value in self.CONFIG.items():
237              setattr(self.checker.config, key, value)
238          self.checker.open()
239  
240      @contextlib.contextmanager
241      def assertNoMessages(self):
242          """Assert that no messages are added by the given method."""
243          with self.assertAddsMessages():
244              yield
245  
246      @contextlib.contextmanager
247      def assertAddsMessages(self, *messages):
248          """Assert that exactly the given method adds the given messages. 
249   
250          The list of messages must exactly match *all* the messages added by the 
251          method. Additionally, we check to see whether the args in each message can 
252          actually be substituted into the message string. 
253          """
254          yield
255          got = self.linter.release_messages()
256          msg = "Expected messages did not match actual.\n" "Expected:\n%s\nGot:\n%s" % (
257              "\n".join(repr(m) for m in messages),
258              "\n".join(repr(m) for m in got),
259          )
260          assert list(messages) == got, msg
261  
262      def walk(self, node):
263          """recursive walk on the given node"""
264          walker = ASTWalker(linter)
265          walker.add_checker(self.checker)
266          walker.walk(node)

281  @contextlib.contextmanager
282  def _create_tempfile(content=None):
283      """Create a new temporary file. 
284   
285      If *content* parameter is given, then it will be written 
286      in the temporary file, before passing it back. 
287      This is a context manager and should be used with a *with* statement. 
288      """
289      # Can't use tempfile.NamedTemporaryFile here
290      # because on Windows the file must be closed before writing to it,
291      # see https://bugs.python.org/issue14243
292      file_handle, tmp = tempfile.mkstemp()
293      if content:
294          # erff
295          write(file_handle, bytes(content, "ascii"))
296      try:
297          yield tmp
298      finally:
299          close(file_handle)
300          remove(tmp)

304  def _create_file_backed_module(code):
305      """Create an astroid module for the given code, backed by a real file."""
306      with _create_tempfile() as temp:
307          module = astroid.parse(code)
308          module.file = temp
309          yield module

316  class OutputLine(
317      collections.namedtuple(
318          "OutputLine", ["symbol", "lineno", "object", "msg", "confidence"]
319      )
320  ):
321      @classmethod
322      def from_msg(cls, msg):
323          return cls(
324              msg.symbol,
325              msg.line,
326              msg.obj or "",
327              msg.msg.replace("\r\n", "\n"),
328              msg.confidence.name
329              if msg.confidence != interfaces.UNDEFINED
330              else interfaces.HIGH.name,
331          )
332  
333      @classmethod
334      def from_csv(cls, row):
335          confidence = row[4] if len(row) == 5 else interfaces.HIGH.name
336          return cls(row[0], int(row[1]), row[2], row[3], confidence)
337  
338      def to_csv(self):
339          if self.confidence == interfaces.HIGH.name:
340              return self[:-1]
341  
342          return self

345  # Common sub-expressions.
346  _MESSAGE = {"msg": r"[a-z][a-z\-]+"}
347  # Matches a #,
348  #  - followed by a comparison operator and a Python version (optional),
349  #  - followed by a line number with a +/- (optional),
350  #  - followed by a list of bracketed message symbols.
351  # Used to extract expected messages from testdata files.
352  _EXPECTED_RE = re.compile(
353      r"\s*#\s*(?:(?P<line>[+-]?[0-9]+):)?"
354      r"(?:(?P<op>[><=]+) *(?P<version>[0-9.]+):)?"
355      r"\s*\[(?P<msgs>%(msg)s(?:,\s*%(msg)s)*)\]" % _MESSAGE
356  )

377      _CONVERTERS = {
378          "min_pyver": parse_python_version,
379          "max_pyver": parse_python_version,

384          self._directory = directory
385          self.base = filename.replace(".py", "")

401          try:
402              cp.read(self.option_file)
403          except NoFileError:

406          for name, value in cp.items("testoptions"):
407              conv = self._CONVERTERS.get(name, lambda v: v)

441  def get_expected_messages(stream):
442      """Parses a file and get expected messages. 
443   
444      :param stream: File-like input stream. 
445      :type stream: enumerable 
446      :returns: A dict mapping line,msg-symbol tuples to the count on this line. 
447      :rtype: dict 
448      """
449      messages = collections.Counter()
450      for i, line in enumerate(stream):
451          match = _EXPECTED_RE.search(line)
452          if match is None:
453              continue
454          line = match.group("line")
455          if line is None:
456              line = i + 1
457          elif line.startswith("+") or line.startswith("-"):
458              line = i + 1 + int(line)
459          else:
460              line = int(line)
461  
462          version = match.group("version")
463          op = match.group("op")
464          if version:
465              required = parse_python_version(version)
466              if not _OPERATORS[op](sys.version_info, required):
467                  continue
468  
469          for msg_id in match.group("msgs").split(","):
470              messages[line, msg_id.strip()] += 1
471      return messages

474  def multiset_difference(left_op, right_op):
475      """Takes two multisets and compares them. 
476   
477      A multiset is a dict with the cardinality of the key as the value. 
478   
479      :param left_op: The expected entries. 
480      :type left_op: set 
481      :param right_op: Actual entries. 
482      :type right_op: set 
483   
484      :returns: The two multisets of missing and unexpected messages. 
485      :rtype: tuple 
486      """
487      missing = left_op.copy()
488      missing.subtract(right_op)
489      unexpected = {}
490      for key, value in list(missing.items()):
491          if value <= 0:
492              missing.pop(key)
493              if value < 0:
494                  unexpected[key] = -value
495      return missing, unexpected

498  class LintModuleTest:
499      maxDiff = None
500  
501      def __init__(self, test_file):
502          _test_reporter = FunctionalTestReporter()
503          self._linter = PyLinter()
504          self._linter.set_reporter(_test_reporter)
505          self._linter.config.persistent = 0
506          checkers.initialize(self._linter)
507          self._linter.disable("I")
508          try:
509              self._linter.read_config_file(test_file.option_file)
510              self._linter.load_config_file()
511          except NoFileError:
512              pass
513          self._test_file = test_file
514  
515      def setUp(self):
516          if self._should_be_skipped_due_to_version():
517              pytest.skip(
518                  "Test cannot run with Python %s." % (sys.version.split(" ")[0],)
519              )
520          missing = []
521          for req in self._test_file.options["requires"]:
522              try:
523                  __import__(req)
524              except ImportError:
525                  missing.append(req)
526          if missing:
527              pytest.skip("Requires %s to be present." % (",".join(missing),))
528          if self._test_file.options["except_implementations"]:
529              implementations = [
530                  item.strip()
531                  for item in self._test_file.options["except_implementations"].split(",")
532              ]
533              implementation = platform.python_implementation()
534              if implementation in implementations:
535                  pytest.skip(
536                      "Test cannot run with Python implementation %r" % (implementation,)
537                  )
538          if self._test_file.options["exclude_platforms"]:
539              platforms = [
540                  item.strip()
541                  for item in self._test_file.options["exclude_platforms"].split(",")
542              ]
543              if sys.platform.lower() in platforms:
544                  pytest.skip("Test cannot run on platform %r" % (sys.platform,))
545  
546      def _should_be_skipped_due_to_version(self):
547          return (
548              sys.version_info < self._test_file.options["min_pyver"]
549              or sys.version_info > self._test_file.options["max_pyver"]
550          )
551  
552      def __str__(self):
553          return "%s (%s.%s)" % (
554              self._test_file.base,
555              self.__class__.__module__,
556              self.__class__.__name__,
557          )
558  
559      def _open_expected_file(self):
560          return open(self._test_file.expected_output)
561  
562      def _open_source_file(self):
563          if self._test_file.base == "invalid_encoded_data":
564              return open(self._test_file.source)
565          if "latin1" in self._test_file.base:
566              return open(self._test_file.source, encoding="latin1")
567          return open(self._test_file.source, encoding="utf8")
568  
569      def _get_expected(self):
570          with self._open_source_file() as fobj:
571              expected_msgs = get_expected_messages(fobj)
572  
573          if expected_msgs:
574              with self._open_expected_file() as fobj:
575                  expected_output_lines = parse_expected_output(fobj)
576          else:
577              expected_output_lines = []
578          return expected_msgs, expected_output_lines
579  
580      def _get_received(self):
581          messages = self._linter.reporter.messages
582          messages.sort(key=lambda m: (m.line, m.symbol, m.msg))
583          received_msgs = collections.Counter()
584          received_output_lines = []
585          for msg in messages:
586              assert (
587                  msg.symbol != "fatal"
588              ), "Pylint analysis failed because of '{}'".format(msg.msg)
589              received_msgs[msg.line, msg.symbol] += 1
590              received_output_lines.append(OutputLine.from_msg(msg))
591          return received_msgs, received_output_lines
592  
593      def _runTest(self):
594          modules_to_check = [self._test_file.source]
595          self._linter.check(modules_to_check)
596          expected_messages, expected_text = self._get_expected()
597          received_messages, received_text = self._get_received()
598  
599          if expected_messages != received_messages:
600              msg = ['Wrong results for file "%s":' % (self._test_file.base)]
601              missing, unexpected = multiset_difference(
602                  expected_messages, received_messages
603              )
604              if missing:
605                  msg.append("\nExpected in testdata:")
606                  msg.extend(" %3d: %s" % msg for msg in sorted(missing))
607              if unexpected:
608                  msg.append("\nUnexpected in testdata:")
609                  msg.extend(" %3d: %s" % msg for msg in sorted(unexpected))
610              pytest.fail("\n".join(msg))
611          self._check_output_text(expected_messages, expected_text, received_text)
612  
613      @classmethod
614      def _split_lines(cls, expected_messages, lines):
615          emitted, omitted = [], []
616          for msg in lines:
617              if (msg[1], msg[0]) in expected_messages:
618                  emitted.append(msg)
619              else:
620                  omitted.append(msg)
621          return emitted, omitted
622  
623      def _check_output_text(self, expected_messages, expected_lines, received_lines):
624          expected_lines = self._split_lines(expected_messages, expected_lines)[0]
625          assert (
626              expected_lines == received_lines
627          ), "Expected test lines did not match for test: {}".format(self._test_file.base)
628
