42   PROPERTIES = {BUILTINS + ".property", "abc.abstractproperty"}
43   # List of possible property names. We use this list in order
44   # to see if a method is a property or not. This should be
45   # pretty reliable and fast, the alternative being to check each
46   # decorator to see if its a real property-like descriptor, which
47   # can be too complicated.
48   # Also, these aren't qualified, because each project can
49   # define them, we shouldn't expect to know every possible
50   # property-like decorator!
51   POSSIBLE_PROPERTIES = {
52       "cached_property",
53       "cachedproperty",
54       "lazyproperty",
55       "lazy_property",
56       "reify",
57       "lazyattribute",
58       "lazy_attribute",
59       "LazyProperty",
60       "lazy",
61       "cache_readonly",
62   }

65   def _is_property(meth, context=None):
66       decoratornames = meth.decoratornames(context=context)
67       if PROPERTIES.intersection(decoratornames):
68           return True
69       stripped = {
70           name.split(".")[-1] for name in decoratornames if name is not util.Uninferable
71       }
72       if any(name in stripped for name in POSSIBLE_PROPERTIES):
73           return True
74   
75       # Lookup for subclasses of *property*
76       if not meth.decorators:
77           return False
78       for decorator in meth.decorators.nodes or ():
79           inferred = helpers.safe_infer(decorator, context=context)
80           if inferred is None or inferred is util.Uninferable:
81               continue
82           if inferred.__class__.__name__ == "ClassDef":
83               for base_class in inferred.bases:
84                   if base_class.__class__.__name__ != "Name":
85                       continue
86                   module, _ = base_class.lookup(base_class.name)
87                   if module.name == BUILTINS and base_class.name == "property":
88                       return True
89   
90       return False

93   class Proxy:
94       """a simple proxy object 
95    
96       Note: 
97    
98       Subclasses of this object will need a custom __getattr__ 
99       if new instance attributes are created. See the Const class 
100      """
101  
102      _proxied = None  # proxied object may be set by class or by instance
103  
104      def __init__(self, proxied=None):
105          if proxied is not None:
106              self._proxied = proxied
107  
108      def __getattr__(self, name):
109          if name == "_proxied":
110              return getattr(self.__class__, "_proxied")
111          if name in self.__dict__:
112              return self.__dict__[name]
113          return getattr(self._proxied, name)
114  
115      def infer(self, context=None):
116          yield self
117  
118  
119  def _infer_stmts(stmts, context, frame=None):
120      """Return an iterator on statements inferred by each statement in *stmts*."""
121      inferred = False
122      if context is not None:
123          name = context.lookupname
124          context = context.clone()
125      else:
126          name = None
127          context = contextmod.InferenceContext()
128  
129      for stmt in stmts:
130          if stmt is util.Uninferable:
131              yield stmt
132              inferred = True
133              continue
134          context.lookupname = stmt._infer_name(frame, name)
135          try:
136              for inferred in stmt.infer(context=context):
137                  yield inferred
138                  inferred = True
139          except exceptions.NameInferenceError:
140              continue
141          except exceptions.InferenceError:
142              yield util.Uninferable
143              inferred = True
144      if not inferred:
145          raise exceptions.InferenceError(
146              "Inference failed for all members of {stmts!r}.",
147              stmts=stmts,
148              frame=frame,
149              context=context,
150          )

153  def _infer_method_result_truth(instance, method_name, context):
154      # Get the method from the instance and try to infer
155      # its return's truth value.
156      meth = next(instance.igetattr(method_name, context=context), None)
157      if meth and hasattr(meth, "infer_call_result"):
158          if not meth.callable():
159              return util.Uninferable
160          try:
161              for value in meth.infer_call_result(instance, context=context):
162                  if value is util.Uninferable:
163                      return value
164  
165                  inferred = next(value.infer(context=context))
166                  return inferred.bool_value()
167          except exceptions.InferenceError:
168              pass
169      return util.Uninferable

172  class BaseInstance(Proxy):
173      """An instance base class, which provides lookup methods for potential instances."""
174  
175      special_attributes = None
176  
177      def display_type(self):
178          return "Instance of"
179  
180      def getattr(self, name, context=None, lookupclass=True):
181          try:
182              values = self._proxied.instance_attr(name, context)
183          except exceptions.AttributeInferenceError as exc:
184              if self.special_attributes and name in self.special_attributes:
185                  return [self.special_attributes.lookup(name)]
186  
187              if lookupclass:
188                  # Class attributes not available through the instance
189                  # unless they are explicitly defined.
190                  return self._proxied.getattr(name, context, class_context=False)
191  
192              raise exceptions.AttributeInferenceError(
193                  target=self, attribute=name, context=context
194              ) from exc
195          # since we've no context information, return matching class members as
196          # well
197          if lookupclass:
198              try:
199                  return values + self._proxied.getattr(
200                      name, context, class_context=False
201                  )
202              except exceptions.AttributeInferenceError:
203                  pass
204          return values
205  
206      def igetattr(self, name, context=None):
207          """inferred getattr"""
208          if not context:
209              context = contextmod.InferenceContext()
210          try:
211              # avoid recursively inferring the same attr on the same class
212              if context.push((self._proxied, name)):
213                  raise exceptions.InferenceError(
214                      message="Cannot infer the same attribute again",
215                      node=self,
216                      context=context,
217                  )
218  
219              # XXX frame should be self._proxied, or not ?
220              get_attr = self.getattr(name, context, lookupclass=False)
221              yield from _infer_stmts(
222                  self._wrap_attr(get_attr, context), context, frame=self
223              )
224          except exceptions.AttributeInferenceError as error:
225              try:
226                  # fallback to class.igetattr since it has some logic to handle
227                  # descriptors
228                  # But only if the _proxied is the Class.
229                  if self._proxied.__class__.__name__ != "ClassDef":
230                      raise
231                  attrs = self._proxied.igetattr(name, context, class_context=False)
232                  yield from self._wrap_attr(attrs, context)
233              except exceptions.AttributeInferenceError as error:
234                  raise exceptions.InferenceError(**vars(error)) from error
235  
236      def _wrap_attr(self, attrs, context=None):
237          """wrap bound methods of attrs in a InstanceMethod proxies"""
238          for attr in attrs:
239              if isinstance(attr, UnboundMethod):
240                  if _is_property(attr):
241                      yield from attr.infer_call_result(self, context)
242                  else:
243                      yield BoundMethod(attr, self)
244              elif hasattr(attr, "name") and attr.name == "<lambda>":
245                  if attr.args.arguments and attr.args.arguments[0].name == "self":
246                      yield BoundMethod(attr, self)
247                      continue
248                  yield attr
249              else:
250                  yield attr
251  
252      def infer_call_result(self, caller, context=None):
253          """infer what a class instance is returning when called"""
254          context = contextmod.bind_context_to_node(context, self)
255          inferred = False
256          for node in self._proxied.igetattr("__call__", context):
257              if node is util.Uninferable or not node.callable():
258                  continue
259              for res in node.infer_call_result(caller, context):
260                  inferred = True
261                  yield res
262          if not inferred:
263              raise exceptions.InferenceError(node=self, caller=caller, context=context)

266  class Instance(BaseInstance):
267      """A special node representing a class instance."""
268  
269      # pylint: disable=unnecessary-lambda
270      special_attributes = util.lazy_descriptor(lambda: objectmodel.InstanceModel())
271  
272      def __repr__(self):
273          return "<Instance of %s.%s at 0x%s>" % (
274              self._proxied.root().name,
275              self._proxied.name,
276              id(self),
277          )
278  
279      def __str__(self):
280          return "Instance of %s.%s" % (self._proxied.root().name, self._proxied.name)
281  
282      def callable(self):
283          try:
284              self._proxied.getattr("__call__", class_context=False)
285              return True
286          except exceptions.AttributeInferenceError:
287              return False
288  
289      def pytype(self):
290          return self._proxied.qname()
291  
292      def display_type(self):
293          return "Instance of"
294  
295      def bool_value(self, context=None):
296          """Infer the truth value for an Instance 
297   
298          The truth value of an instance is determined by these conditions: 
299   
300             * if it implements __bool__ on Python 3 or __nonzero__ 
301               on Python 2, then its bool value will be determined by 
302               calling this special method and checking its result. 
303             * when this method is not defined, __len__() is called, if it 
304               is defined, and the object is considered true if its result is 
305               nonzero. If a class defines neither __len__() nor __bool__(), 
306               all its instances are considered true. 
307          """
308          context = context or contextmod.InferenceContext()
309          context.callcontext = contextmod.CallContext(args=[])
310          context.boundnode = self
311  
312          try:
313              result = _infer_method_result_truth(self, BOOL_SPECIAL_METHOD, context)
314          except (exceptions.InferenceError, exceptions.AttributeInferenceError):
315              # Fallback to __len__.
316              try:
317                  result = _infer_method_result_truth(self, "__len__", context)
318              except (exceptions.AttributeInferenceError, exceptions.InferenceError):
319                  return True
320          return result
321  
322      # This is set in inference.py.
323      def getitem(self, index, context=None):
324          pass

327  class UnboundMethod(Proxy):
328      """a special node representing a method not bound to an instance"""
329  
330      # pylint: disable=unnecessary-lambda
331      special_attributes = util.lazy_descriptor(lambda: objectmodel.UnboundMethodModel())
332  
333      def __repr__(self):
334          frame = self._proxied.parent.frame()
335          return "<%s %s of %s at 0x%s" % (
336              self.__class__.__name__,
337              self._proxied.name,
338              frame.qname(),
339              id(self),
340          )
341  
342      def implicit_parameters(self):
343          return 0
344  
345      def is_bound(self):
346          return False
347  
348      def getattr(self, name, context=None):
349          if name in self.special_attributes:
350              return [self.special_attributes.lookup(name)]
351          return self._proxied.getattr(name, context)
352  
353      def igetattr(self, name, context=None):
354          if name in self.special_attributes:
355              return iter((self.special_attributes.lookup(name),))
356          return self._proxied.igetattr(name, context)
357  
358      def infer_call_result(self, caller, context):
359          """ 
360          The boundnode of the regular context with a function called 
361          on ``object.__new__`` will be of type ``object``, 
362          which is incorrect for the argument in general. 
363          If no context is given the ``object.__new__`` call argument will 
364          correctly inferred except when inside a call that requires 
365          the additional context (such as a classmethod) of the boundnode 
366          to determine which class the method was called from 
367          """
368  
369          # If we're unbound method __new__ of builtin object, the result is an
370          # instance of the class given as first argument.
371          if (
372              self._proxied.name == "__new__"
373              and self._proxied.parent.frame().qname() == "%s.object" % BUILTINS
374          ):
375              if caller.args:
376                  node_context = context.extra_context.get(caller.args[0])
377                  infer = caller.args[0].infer(context=node_context)
378              else:
379                  infer = []
380              return (Instance(x) if x is not util.Uninferable else x for x in infer)
381          return self._proxied.infer_call_result(caller, context)
382  
383      def bool_value(self, context=None):
384          return True

387  class BoundMethod(UnboundMethod):
388      """a special node representing a method bound to an instance"""
389  
390      # pylint: disable=unnecessary-lambda
391      special_attributes = util.lazy_descriptor(lambda: objectmodel.BoundMethodModel())
392  
393      def __init__(self, proxy, bound):
394          UnboundMethod.__init__(self, proxy)
395          self.bound = bound
396  
397      def implicit_parameters(self):
398          if self.name == "__new__":
399              # __new__ acts as a classmethod but the class argument is not implicit.
400              return 0
401          return 1
402  
403      def is_bound(self):
404          return True

406      def _infer_type_new_call(self, caller, context):
407          """Try to infer what type.__new__(mcs, name, bases, attrs) returns. 
408   
409          In order for such call to be valid, the metaclass needs to be 
410          a subtype of ``type``, the name needs to be a string, the bases 
411          needs to be a tuple of classes 
412          """
413          # pylint: disable=import-outside-toplevel; circular import
414          from astroid import node_classes
415  
416          # Verify the metaclass
417          mcs = next(caller.args[0].infer(context=context))
418          if mcs.__class__.__name__ != "ClassDef":
419              # Not a valid first argument.
420              return None
421          if not mcs.is_subtype_of("%s.type" % BUILTINS):
422              # Not a valid metaclass.
423              return None
424  
425          # Verify the name
426          name = next(caller.args[1].infer(context=context))
427          if name.__class__.__name__ != "Const":
428              # Not a valid name, needs to be a const.
429              return None
430          if not isinstance(name.value, str):
431              # Needs to be a string.
432              return None
433  
434          # Verify the bases
435          bases = next(caller.args[2].infer(context=context))
436          if bases.__class__.__name__ != "Tuple":
437              # Needs to be a tuple.
438              return None
439          inferred_bases = [next(elt.infer(context=context)) for elt in bases.elts]
440          if any(base.__class__.__name__ != "ClassDef" for base in inferred_bases):
441              # All the bases needs to be Classes
442              return None
443  
444          # Verify the attributes.
445          attrs = next(caller.args[3].infer(context=context))
446          if attrs.__class__.__name__ != "Dict":
447              # Needs to be a dictionary.
448              return None
449          cls_locals = collections.defaultdict(list)
450          for key, value in attrs.items:
451              key = next(key.infer(context=context))
452              value = next(value.infer(context=context))
453              # Ignore non string keys
454              if key.__class__.__name__ == "Const" and isinstance(key.value, str):
455                  cls_locals[key.value].append(value)
456  
457          # Build the class from now.
458          cls = mcs.__class__(
459              name=name.value,
460              lineno=caller.lineno,
461              col_offset=caller.col_offset,
462              parent=caller,
463          )
464          empty = node_classes.Pass()
465          cls.postinit(
466              bases=bases.elts,
467              body=[empty],
468              decorators=[],
469              newstyle=True,
470              metaclass=mcs,
471              keywords=[],
472          )
473          cls.locals = cls_locals
474          return cls
475  
476      def infer_call_result(self, caller, context=None):
477          context = contextmod.bind_context_to_node(context, self.bound)
478          if (
479              self.bound.__class__.__name__ == "ClassDef"
480              and self.bound.name == "type"
481              and self.name == "__new__"
482              and len(caller.args) == 4
483          ):
484              # Check if we have a ``type.__new__(mcs, name, bases, attrs)`` call.
485              new_cls = self._infer_type_new_call(caller, context)
486              if new_cls:
487                  return iter((new_cls,))
488  
489          return super().infer_call_result(caller, context)
490  
491      def bool_value(self, context=None):
492          return True

495  class Generator(BaseInstance):
496      """a special node representing a generator. 
497   
498      Proxied class is set once for all in raw_building. 
499      """
500  
501      # pylint: disable=unnecessary-lambda
502      special_attributes = util.lazy_descriptor(lambda: objectmodel.GeneratorModel())
503  
504      # pylint: disable=super-init-not-called
505      def __init__(self, parent=None):
506          self.parent = parent
507  
508      def callable(self):
509          return False
510  
511      def pytype(self):
512          return "%s.generator" % BUILTINS
513  
514      def display_type(self):
515          return "Generator"
516  
517      def bool_value(self, context=None):
518          return True
519  
520      def __repr__(self):
521          return "<Generator(%s) l.%s at 0x%s>" % (
522              self._proxied.name,
523              self.lineno,
524              id(self),
525          )
526  
527      def __str__(self):
528          return "Generator(%s)" % (self._proxied.name)
