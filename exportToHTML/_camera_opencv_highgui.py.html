61       def get_surface(self, dest_surf=None):
62           camera = self.camera
63   
64           im = highgui.cvQueryFrame(camera)
65           #convert Ipl image to PIL image
66           #print type(im)
67           if im:
68               xx = opencv.adaptors.Ipl2NumPy(im)
69               #print type(xx)
70               #print xx.iscontiguous()
71               #print dir(xx)
72               #print xx.shape
73               xxx = numpy.reshape(xx, (numpy.product(xx.shape),))
74   
75               if xx.shape[2] != 3:
76                   raise ValueError("not sure what to do about this size")
77   
78               pg_img = pygame.image.frombuffer(xxx, (xx.shape[1], xx.shape[0]), "RGB")
79   
80               # if there is a destination surface given, we blit onto that.
81               if dest_surf:
82                   dest_surf.blit(pg_img, (0, 0))
83               return dest_surf
84               #return pg_img
