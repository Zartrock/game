124  @decorators.yes_if_nothing_inferred
125  def const_infer_binary_op(self, opnode, operator, other, context, _):
126      not_implemented = nodes.Const(NotImplemented)
127      if isinstance(other, nodes.Const):
128          try:
129              impl = BIN_OP_IMPL[operator]
130              try:
131                  yield nodes.const_factory(impl(self.value, other.value))
132              except TypeError:
133                  # ArithmeticError is not enough: float >> float is a TypeError
134                  yield not_implemented
135              except Exception:  # pylint: disable=broad-except
136                  yield util.Uninferable
137          except TypeError:
138              yield not_implemented
139      elif isinstance(self.value, str) and operator == "%":
140          # TODO(cpopa): implement string interpolation later on.
141          yield util.Uninferable
142      else:
143          yield not_implemented

149  def _multiply_seq_by_int(self, opnode, other, context):
150      node = self.__class__(parent=opnode)
151      filtered_elts = (
152          helpers.safe_infer(elt, context) or util.Uninferable
153          for elt in self.elts
154          if elt is not util.Uninferable
155      )
156      node.elts = list(filtered_elts) * other.value
157      return node

160  def _filter_uninferable_nodes(elts, context):
161      for elt in elts:
162          if elt is util.Uninferable:
163              yield nodes.Unknown()
164          else:
165              for inferred in elt.infer(context):
166                  if inferred is not util.Uninferable:
167                      yield inferred
168                  else:
169                      yield nodes.Unknown()

172  @decorators.yes_if_nothing_inferred
173  def tl_infer_binary_op(self, opnode, operator, other, context, method):
174      not_implemented = nodes.Const(NotImplemented)
175      if isinstance(other, self.__class__) and operator == "+":
176          node = self.__class__(parent=opnode)
177          node.elts = list(
178              itertools.chain(
179                  _filter_uninferable_nodes(self.elts, context),
180                  _filter_uninferable_nodes(other.elts, context),
181              )
182          )
183          yield node
184      elif isinstance(other, nodes.Const) and operator == "*":
185          if not isinstance(other.value, int):
186              yield not_implemented
187              return
188          yield _multiply_seq_by_int(self, opnode, other, context)
189      elif isinstance(other, bases.Instance) and operator == "*":
190          # Verify if the instance supports __index__.
191          as_index = helpers.class_instance_as_index(other)
192          if not as_index:
193              yield util.Uninferable
194          else:
195              yield _multiply_seq_by_int(self, opnode, as_index, context)
196      else:
197          yield not_implemented

215  """the assigned_stmts method is responsible to return the assigned statement 
216  (e.g. not inferred) according to the assignment type. 
217   
218  The `assign_path` argument is used to record the lhs path of the original node. 
219  For instance if we want assigned statements for 'c' in 'a, (b,c)', assign_path 
220  will be [1, 1] once arrived to the Assign node. 
221   
222  The `context` argument is the current inference context which should be given 
223  to any intermediary inference necessary. 
224  """
225

227  def _resolve_looppart(parts, assign_path, context):
228      """recursive function to resolve multiple assignments on loops"""
229      assign_path = assign_path[:]
230      index = assign_path.pop(0)
231      for part in parts:
232          if part is util.Uninferable:
233              continue
234          if not hasattr(part, "itered"):
235              continue
236          try:
237              itered = part.itered()
238          except TypeError:
239              continue
240          for stmt in itered:
241              index_node = nodes.Const(index)
242              try:
243                  assigned = stmt.getitem(index_node, context)
244              except (
245                  AttributeError,
246                  exceptions.AstroidTypeError,
247                  exceptions.AstroidIndexError,
248              ):
249                  continue
250              if not assign_path:
251                  # we achieved to resolved the assignment path,
252                  # don't infer the last part
253                  yield assigned
254              elif assigned is util.Uninferable:
255                  break
256              else:
257                  # we are not yet on the last part of the path
258                  # search on each possibly inferred value
259                  try:
260                      yield from _resolve_looppart(
261                          assigned.infer(context), assign_path, context
262                      )
263                  except exceptions.InferenceError:
264                      break

267  @decorators.raise_if_nothing_inferred
268  def for_assigned_stmts(self, node=None, context=None, assign_path=None):
269      if isinstance(self, nodes.AsyncFor) or getattr(self, "is_async", False):
270          # Skip inferring of async code for now
271          return dict(node=self, unknown=node, assign_path=assign_path, context=context)
272      if assign_path is None:
273          for lst in self.iter.infer(context):
274              if isinstance(lst, (nodes.Tuple, nodes.List)):
275                  yield from lst.elts
276      else:
277          yield from _resolve_looppart(self.iter.infer(context), assign_path, context)
278      return dict(node=self, unknown=node, assign_path=assign_path, context=context)

285  def sequence_assigned_stmts(self, node=None, context=None, assign_path=None):
286      if assign_path is None:
287          assign_path = []
288      try:
289          index = self.elts.index(node)
290      except ValueError as exc:
291          raise exceptions.InferenceError(
292              "Tried to retrieve a node {node!r} which does not exist",
293              node=self,
294              assign_path=assign_path,
295              context=context,
296          ) from exc
297  
298      assign_path.insert(0, index)
299      return self.parent.assigned_stmts(
300          node=self, context=context, assign_path=assign_path
301      )

316  def _arguments_infer_argname(self, name, context):
317      # arguments information may be missing, in which case we can't do anything
318      # more
319      if not (self.arguments or self.vararg or self.kwarg):
320          yield util.Uninferable
321          return
322  
323      functype = self.parent.type
324      # first argument of instance/class method
325      if (
326          self.arguments
327          and getattr(self.arguments[0], "name", None) == name
328          and functype != "staticmethod"
329      ):
330          cls = self.parent.parent.scope()
331          is_metaclass = isinstance(cls, nodes.ClassDef) and cls.type == "metaclass"
332          # If this is a metaclass, then the first argument will always
333          # be the class, not an instance.
334          if context.boundnode and isinstance(context.boundnode, bases.Instance):
335              cls = context.boundnode._proxied
336          if is_metaclass or functype == "classmethod":
337              yield cls
338              return
339          if functype == "method":
340              yield cls.instantiate_class()
341              return
342  
343      if context and context.callcontext:
344          call_site = arguments.CallSite(context.callcontext, context.extra_context)
345          yield from call_site.infer_argument(self.parent, name, context)
346          return
347  
348      if name == self.vararg:
349          vararg = nodes.const_factory(())
350          vararg.parent = self
351          if not self.arguments and self.parent.name == "__init__":
352              cls = self.parent.parent.scope()
353              vararg.elts = [cls.instantiate_class()]
354          yield vararg
355          return
356      if name == self.kwarg:
357          kwarg = nodes.const_factory({})
358          kwarg.parent = self
359          yield kwarg
360          return
361      # if there is a default value, yield it. And then yield Uninferable to reflect
362      # we can't guess given argument value
363      try:
364          context = contextmod.copy_context(context)
365          yield from self.default_value(name).infer(context)
366          yield util.Uninferable
367      except exceptions.NoDefault:
368          yield util.Uninferable

371  def arguments_assigned_stmts(self, node=None, context=None, assign_path=None):
372      if context.callcontext:
373          # reset call context/name
374          callcontext = context.callcontext
375          context = contextmod.copy_context(context)
376          context.callcontext = None
377          args = arguments.CallSite(callcontext, context=context)
378          return args.infer_argument(self.parent, node.name, context)
379      return _arguments_infer_argname(self, node.name, context)

410  def _resolve_assignment_parts(parts, assign_path, context):
411      """recursive function to resolve multiple assignments"""
412      assign_path = assign_path[:]
413      index = assign_path.pop(0)
414      for part in parts:
415          assigned = None
416          if isinstance(part, nodes.Dict):
417              # A dictionary in an iterating context
418              try:
419                  assigned, _ = part.items[index]
420              except IndexError:
421                  return
422  
423          elif hasattr(part, "getitem"):
424              index_node = nodes.Const(index)
425              try:
426                  assigned = part.getitem(index_node, context)
427              except (exceptions.AstroidTypeError, exceptions.AstroidIndexError):
428                  return
429  
430          if not assigned:
431              return
432  
433          if not assign_path:
434              # we achieved to resolved the assignment path, don't infer the
435              # last part
436              yield assigned
437          elif assigned is util.Uninferable:
438              return
439          else:
440              # we are not yet on the last part of the path search on each
441              # possibly inferred value
442              try:
443                  yield from _resolve_assignment_parts(
444                      assigned.infer(context), assign_path, context
445                  )
446              except exceptions.InferenceError:
447                  return

463  def _infer_context_manager(self, mgr, context):
464      inferred = next(mgr.infer(context=context))
465      if isinstance(inferred, bases.Generator):
466          # Check if it is decorated with contextlib.contextmanager.
467          func = inferred.parent
468          if not func.decorators:
469              raise exceptions.InferenceError(
470                  "No decorators found on inferred generator %s", node=func
471              )
472  
473          for decorator_node in func.decorators.nodes:
474              decorator = next(decorator_node.infer(context=context))
475              if isinstance(decorator, nodes.FunctionDef):
476                  if decorator.qname() == _CONTEXTLIB_MGR:
477                      break
478          else:
479              # It doesn't interest us.
480              raise exceptions.InferenceError(node=func)
481  
482          # Get the first yield point. If it has multiple yields,
483          # then a RuntimeError will be raised.
484  
485          possible_yield_points = func.nodes_of_class(nodes.Yield)
486          # Ignore yields in nested functions
487          yield_point = next(
488              (node for node in possible_yield_points if node.scope() == func), None
489          )
490          if yield_point:
491              if not yield_point.value:
492                  const = nodes.Const(None)
493                  const.parent = yield_point
494                  const.lineno = yield_point.lineno
495                  yield const
496              else:
497                  yield from yield_point.value.infer(context=context)
498      elif isinstance(inferred, bases.Instance):
499          try:
500              enter = next(inferred.igetattr("__enter__", context=context))
501          except (exceptions.InferenceError, exceptions.AttributeInferenceError):
502              raise exceptions.InferenceError(node=inferred)
503          if not isinstance(enter, bases.BoundMethod):
504              raise exceptions.InferenceError(node=enter)
505          yield from enter.infer_call_result(self, context)
506      else:
507          raise exceptions.InferenceError(node=mgr)

510  @decorators.raise_if_nothing_inferred
511  def with_assigned_stmts(self, node=None, context=None, assign_path=None):
512      """Infer names and other nodes from a *with* statement. 
513   
514      This enables only inference for name binding in a *with* statement. 
515      For instance, in the following code, inferring `func` will return 
516      the `ContextManager` class, not whatever ``__enter__`` returns. 
517      We are doing this intentionally, because we consider that the context 
518      manager result is whatever __enter__ returns and what it is binded 
519      using the ``as`` keyword. 
520   
521          class ContextManager(object): 
522              def __enter__(self): 
523                  return 42 
524          with ContextManager() as f: 
525              pass 
526   
527          # ContextManager().infer() will return ContextManager 
528          # f.infer() will return 42. 
529   
530      Arguments: 
531          self: nodes.With 
532          node: The target of the assignment, `as (a, b)` in `with foo as (a, b)`. 
533          context: Inference context used for caching already inferred objects 
534          assign_path: 
535              A list of indices, where each index specifies what item to fetch from 
536              the inference results. 
537      """
538      try:
539          mgr = next(mgr for (mgr, vars) in self.items if vars == node)
540      except StopIteration:
541          return None
542      if assign_path is None:
543          yield from _infer_context_manager(self, mgr, context)
544      else:
545          for result in _infer_context_manager(self, mgr, context):
546              # Walk the assign_path and get the item at the final index.
547              obj = result
548              for index in assign_path:
549                  if not hasattr(obj, "elts"):
550                      raise exceptions.InferenceError(
551                          "Wrong type ({targets!r}) for {node!r} assignment",
552                          node=self,
553                          targets=node,
554                          assign_path=assign_path,
555                          context=context,
556                      )
557                  try:
558                      obj = obj.elts[index]
559                  except IndexError as exc:
560                      raise exceptions.InferenceError(
561                          "Tried to infer a nonexistent target with index {index} "
562                          "in {node!r}.",
563                          node=self,
564                          targets=node,
565                          assign_path=assign_path,
566                          context=context,
567                      ) from exc
568                  except TypeError as exc:
569                      raise exceptions.InferenceError(
570                          "Tried to unpack a non-iterable value " "in {node!r}.",
571                          node=self,
572                          targets=node,
573                          assign_path=assign_path,
574                          context=context,
575                      ) from exc
576              yield obj
577      return dict(node=self, unknown=node, assign_path=assign_path, context=context)

600  @decorators.yes_if_nothing_inferred
601  def starred_assigned_stmts(self, node=None, context=None, assign_path=None):
602      """ 
603      Arguments: 
604          self: nodes.Starred 
605          node: a node related to the current underlying Node. 
606          context: Inference context used for caching already inferred objects 
607          assign_path: 
608              A list of indices, where each index specifies what item to fetch from 
609              the inference results. 
610      """
611      # pylint: disable=too-many-locals,too-many-branches,too-many-statements
612      def _determine_starred_iteration_lookups(starred, target, lookups):
613          # Determine the lookups for the rhs of the iteration
614          itered = target.itered()
615          for index, element in enumerate(itered):
616              if (
617                  isinstance(element, nodes.Starred)
618                  and element.value.name == starred.value.name
619              ):
620                  lookups.append((index, len(itered)))
621                  break
622              if isinstance(element, nodes.Tuple):
623                  lookups.append((index, len(element.itered())))
624                  _determine_starred_iteration_lookups(starred, element, lookups)
625  
626      stmt = self.statement()
627      if not isinstance(stmt, (nodes.Assign, nodes.For)):
628          raise exceptions.InferenceError(
629              "Statement {stmt!r} enclosing {node!r} " "must be an Assign or For node.",
630              node=self,
631              stmt=stmt,
632              unknown=node,
633              context=context,
634          )
635  
636      if context is None:
637          context = contextmod.InferenceContext()
638  
639      if isinstance(stmt, nodes.Assign):
640          value = stmt.value
641          lhs = stmt.targets[0]
642  
643          if sum(1 for _ in lhs.nodes_of_class(nodes.Starred)) > 1:
644              raise exceptions.InferenceError(
645                  "Too many starred arguments in the " " assignment targets {lhs!r}.",
646                  node=self,
647                  targets=lhs,
648                  unknown=node,
649                  context=context,
650              )
651  
652          try:
653              rhs = next(value.infer(context))
654          except exceptions.InferenceError:
655              yield util.Uninferable
656              return
657          if rhs is util.Uninferable or not hasattr(rhs, "itered"):
658              yield util.Uninferable
659              return
660  
661          try:
662              elts = collections.deque(rhs.itered())
663          except TypeError:
664              yield util.Uninferable
665              return
666  
667          # Unpack iteratively the values from the rhs of the assignment,
668          # until the find the starred node. What will remain will
669          # be the list of values which the Starred node will represent
670          # This is done in two steps, from left to right to remove
671          # anything before the starred node and from right to left
672          # to remove anything after the starred node.
673  
674          for index, left_node in enumerate(lhs.elts):
675              if not isinstance(left_node, nodes.Starred):
676                  if not elts:
677                      break
678                  elts.popleft()
679                  continue
680              lhs_elts = collections.deque(reversed(lhs.elts[index:]))
681              for right_node in lhs_elts:
682                  if not isinstance(right_node, nodes.Starred):
683                      if not elts:
684                          break
685                      elts.pop()
686                      continue
687  
688                  # We're done unpacking.
689                  elts = list(elts)
690                  packed = nodes.List(
691                      ctx=Store, parent=self, lineno=lhs.lineno, col_offset=lhs.col_offset
692                  )
693                  packed.postinit(elts=elts)
694                  yield packed
695                  break
696  
697      if isinstance(stmt, nodes.For):
698          try:
699              inferred_iterable = next(stmt.iter.infer(context=context))
700          except exceptions.InferenceError:
701              yield util.Uninferable
702              return
703          if inferred_iterable is util.Uninferable or not hasattr(
704              inferred_iterable, "itered"
705          ):
706              yield util.Uninferable
707              return
708          try:
709              itered = inferred_iterable.itered()
710          except TypeError:
711              yield util.Uninferable
712              return
713  
714          target = stmt.target
715  
716          if not isinstance(target, nodes.Tuple):
717              raise exceptions.InferenceError(
718                  "Could not make sense of this, the target must be a tuple",
719                  context=context,
720              )
721  
722          lookups = []
723          _determine_starred_iteration_lookups(self, target, lookups)
724          if not lookups:
725              raise exceptions.InferenceError(
726                  "Could not make sense of this, needs at least a lookup", context=context
727              )
728  
729          # Make the last lookup a slice, since that what we want for a Starred node
730          last_element_index, last_element_length = lookups[-1]
731          is_starred_last = last_element_index == (last_element_length - 1)
732  
733          lookup_slice = slice(
734              last_element_index,
735              None if is_starred_last else (last_element_length - last_element_index),
736          )
737          lookups[-1] = lookup_slice
738  
739          for element in itered:
740  
741              # We probably want to infer the potential values *for each* element in an
742              # iterable, but we can't infer a list of all values, when only a list of
743              # step values are expected:
744              #
745              # for a, *b in [...]:
746              #   b
747              #
748              # *b* should now point to just the elements at that particular iteration step,
749              # which astroid can't know about.
750  
751              found_element = None
752              for lookup in lookups:
753                  if not hasattr(element, "itered"):
754                      break
755                  if not isinstance(lookup, slice):
756                      # Grab just the index, not the whole length
757                      lookup = lookup[0]
758                  try:
759                      itered_inner_element = element.itered()
760                      element = itered_inner_element[lookup]
761                  except IndexError:
762                      break
763                  except TypeError:
764                      # Most likely the itered() call failed, cannot make sense of this
765                      yield util.Uninferable
766                      return
767                  else:
768                      found_element = element
769  
770              unpacked = nodes.List(
771                  ctx=Store, parent=self, lineno=self.lineno, col_offset=self.col_offset
772              )
773              unpacked.postinit(elts=found_element or [])
774              yield unpacked
775              return
776  
777          yield util.Uninferable
778  
779  
780  nodes.Starred.assigned_stmts = starred_assigned_stmts
