44   def _extend_str(class_node, rvalue):
45       """function to extend builtin str/unicode class"""
46       code = dedent(
47           """ 
48       class whatever(object): 
49           def join(self, iterable): 
50               return {rvalue} 
51           def replace(self, old, new, count=None): 
52               return {rvalue} 
53           def format(self, *args, **kwargs): 
54               return {rvalue} 
55           def encode(self, encoding='ascii', errors=None): 
56               return '' 
57           def decode(self, encoding='ascii', errors=None): 
58               return u'' 
59           def capitalize(self): 
60               return {rvalue} 
61           def title(self): 
62               return {rvalue} 
63           def lower(self): 
64               return {rvalue} 
65           def upper(self): 
66               return {rvalue} 
67           def swapcase(self): 
68               return {rvalue} 
69           def index(self, sub, start=None, end=None): 
70               return 0 
71           def find(self, sub, start=None, end=None): 
72               return 0 
73           def count(self, sub, start=None, end=None): 
74               return 0 
75           def strip(self, chars=None): 
76               return {rvalue} 
77           def lstrip(self, chars=None): 
78               return {rvalue} 
79           def rstrip(self, chars=None): 
80               return {rvalue} 
81           def rjust(self, width, fillchar=None): 
82               return {rvalue} 
83           def center(self, width, fillchar=None): 
84               return {rvalue} 
85           def ljust(self, width, fillchar=None): 
86               return {rvalue} 
87       """
88       )
89       code = code.format(rvalue=rvalue)
90       fake = AstroidBuilder(MANAGER).string_build(code)["whatever"]
91       for method in fake.mymethods():
92           method.parent = class_node
93           method.lineno = None
94           method.col_offset = None
95           if "__class__" in method.locals:
96               method.locals["__class__"] = [class_node]
97           class_node.locals[method.name] = [method]
98           method.parent = class_node

115  def _builtin_filter_predicate(node, builtin_name):
116      if isinstance(node.func, nodes.Name) and node.func.name == builtin_name:
117          return True
118      if isinstance(node.func, nodes.Attribute):
119          return (
120              node.func.attrname == "fromkeys"
121              and isinstance(node.func.expr, nodes.Name)
122              and node.func.expr.name == "dict"
123          )
124      return False

127  def register_builtin_transform(transform, builtin_name):
128      """Register a new transform function for the given *builtin_name*. 
129   
130      The transform function must accept two parameters, a node and 
131      an optional context. 
132      """
133  
134      def _transform_wrapper(node, context=None):
135          result = transform(node, context=context)
136          if result:
137              if not result.parent:
138                  # Let the transformation function determine
139                  # the parent for its result. Otherwise,
140                  # we set it to be the node we transformed from.
141                  result.parent = node
142  
143              if result.lineno is None:
144                  result.lineno = node.lineno
145              if result.col_offset is None:
146                  result.col_offset = node.col_offset
147          return iter([result])
148  
149      MANAGER.register_transform(
150          nodes.Call,
151          inference_tip(_transform_wrapper),
152          partial(_builtin_filter_predicate, builtin_name=builtin_name),
153      )

156  def _container_generic_inference(node, context, node_type, transform):
157      args = node.args
158      if not args:
159          return node_type()
160      if len(node.args) > 1:
161          raise UseInferenceDefault()
162  
163      (arg,) = args
164      transformed = transform(arg)
165      if not transformed:
166          try:
167              inferred = next(arg.infer(context=context))
168          except (InferenceError, StopIteration):
169              raise UseInferenceDefault()
170          if inferred is util.Uninferable:
171              raise UseInferenceDefault()
172          transformed = transform(inferred)
173      if not transformed or transformed is util.Uninferable:
174          raise UseInferenceDefault()
175      return transformed

178  def _container_generic_transform(arg, context, klass, iterables, build_elts):
179      if isinstance(arg, klass):
180          return arg
181      elif isinstance(arg, iterables):
182          if all(isinstance(elt, nodes.Const) for elt in arg.elts):
183              elts = [elt.value for elt in arg.elts]
184          else:
185              # TODO: Does not handle deduplication for sets.
186              elts = []
187              for element in arg.elts:
188                  inferred = helpers.safe_infer(element, context=context)
189                  if inferred:
190                      evaluated_object = nodes.EvaluatedObject(
191                          original=element, value=inferred
192                      )
193                      elts.append(evaluated_object)
194      elif isinstance(arg, nodes.Dict):
195          # Dicts need to have consts as strings already.
196          if not all(isinstance(elt[0], nodes.Const) for elt in arg.items):
197              raise UseInferenceDefault()
198          elts = [item[0].value for item in arg.items]
199      elif isinstance(arg, nodes.Const) and isinstance(
200          arg.value, (six.string_types, six.binary_type)
201      ):
202          elts = arg.value
203      else:
204          return
205      return klass.from_elements(elts=build_elts(elts))

266  def _get_elts(arg, context):
267      is_iterable = lambda n: isinstance(n, (nodes.List, nodes.Tuple, nodes.Set))
268      try:
269          inferred = next(arg.infer(context))
270      except (InferenceError, NameInferenceError):
271          raise UseInferenceDefault()
272      if isinstance(inferred, nodes.Dict):
273          items = inferred.items
274      elif is_iterable(inferred):
275          items = []
276          for elt in inferred.elts:
277              # If an item is not a pair of two items,
278              # then fallback to the default inference.
279              # Also, take in consideration only hashable items,
280              # tuples and consts. We are choosing Names as well.
281              if not is_iterable(elt):
282                  raise UseInferenceDefault()
283              if len(elt.elts) != 2:
284                  raise UseInferenceDefault()
285              if not isinstance(elt.elts[0], (nodes.Tuple, nodes.Const, nodes.Name)):
286                  raise UseInferenceDefault()
287              items.append(tuple(elt.elts))
288      else:
289          raise UseInferenceDefault()
290      return items

293  def infer_dict(node, context=None):
294      """Try to infer a dict call to a Dict node. 
295   
296      The function treats the following cases: 
297   
298          * dict() 
299          * dict(mapping) 
300          * dict(iterable) 
301          * dict(iterable, **kwargs) 
302          * dict(mapping, **kwargs) 
303          * dict(**kwargs) 
304   
305      If a case can't be inferred, we'll fallback to default inference. 
306      """
307      call = arguments.CallSite.from_call(node, context=context)
308      if call.has_invalid_arguments() or call.has_invalid_keywords():
309          raise UseInferenceDefault
310  
311      args = call.positional_arguments
312      kwargs = list(call.keyword_arguments.items())
313  
314      if not args and not kwargs:
315          # dict()
316          return nodes.Dict()
317      elif kwargs and not args:
318          # dict(a=1, b=2, c=4)
319          items = [(nodes.Const(key), value) for key, value in kwargs]
320      elif len(args) == 1 and kwargs:
321          # dict(some_iterable, b=2, c=4)
322          elts = _get_elts(args[0], context)
323          keys = [(nodes.Const(key), value) for key, value in kwargs]
324          items = elts + keys
325      elif len(args) == 1:
326          items = _get_elts(args[0], context)
327      else:
328          raise UseInferenceDefault()
329  
330      value = nodes.Dict(
331          col_offset=node.col_offset, lineno=node.lineno, parent=node.parent
332      )
333      value.postinit(items)
334      return value

337  def infer_super(node, context=None):
338      """Understand super calls. 
339   
340      There are some restrictions for what can be understood: 
341   
342          * unbounded super (one argument form) is not understood. 
343   
344          * if the super call is not inside a function (classmethod or method), 
345            then the default inference will be used. 
346   
347          * if the super arguments can't be inferred, the default inference 
348            will be used. 
349      """
350      if len(node.args) == 1:
351          # Ignore unbounded super.
352          raise UseInferenceDefault
353  
354      scope = node.scope()
355      if not isinstance(scope, nodes.FunctionDef):
356          # Ignore non-method uses of super.
357          raise UseInferenceDefault
358      if scope.type not in ("classmethod", "method"):
359          # Not interested in staticmethods.
360          raise UseInferenceDefault
361  
362      cls = scoped_nodes.get_wrapping_class(scope)
363      if not len(node.args):
364          mro_pointer = cls
365          # In we are in a classmethod, the interpreter will fill
366          # automatically the class as the second argument, not an instance.
367          if scope.type == "classmethod":
368              mro_type = cls
369          else:
370              mro_type = cls.instantiate_class()
371      else:
372          try:
373              mro_pointer = next(node.args[0].infer(context=context))
374          except InferenceError:
375              raise UseInferenceDefault
376          try:
377              mro_type = next(node.args[1].infer(context=context))
378          except InferenceError:
379              raise UseInferenceDefault
380  
381      if mro_pointer is util.Uninferable or mro_type is util.Uninferable:
382          # No way we could understand this.
383          raise UseInferenceDefault
384  
385      super_obj = objects.Super(
386          mro_pointer=mro_pointer, mro_type=mro_type, self_class=cls, scope=scope
387      )
388      super_obj.parent = node
389      return super_obj

392  def _infer_getattr_args(node, context):
393      if len(node.args) not in (2, 3):
394          # Not a valid getattr call.
395          raise UseInferenceDefault
396  
397      try:
398          obj = next(node.args[0].infer(context=context))
399          attr = next(node.args[1].infer(context=context))
400      except InferenceError:
401          raise UseInferenceDefault
402  
403      if obj is util.Uninferable or attr is util.Uninferable:
404          # If one of the arguments is something we can't infer,
405          # then also make the result of the getattr call something
406          # which is unknown.
407          return util.Uninferable, util.Uninferable
408  
409      is_string = isinstance(attr, nodes.Const) and isinstance(
410          attr.value, six.string_types
411      )
412      if not is_string:
413          raise UseInferenceDefault
414  
415      return obj, attr.value

418  def infer_getattr(node, context=None):
419      """Understand getattr calls 
420   
421      If one of the arguments is an Uninferable object, then the 
422      result will be an Uninferable object. Otherwise, the normal attribute 
423      lookup will be done. 
424      """
425      obj, attr = _infer_getattr_args(node, context)
426      if (
427          obj is util.Uninferable
428          or attr is util.Uninferable
429          or not hasattr(obj, "igetattr")
430      ):
431          return util.Uninferable
432  
433      try:
434          return next(obj.igetattr(attr, context=context))
435      except (StopIteration, InferenceError, AttributeInferenceError):
436          if len(node.args) == 3:
437              # Try to infer the default and return it instead.
438              try:
439                  return next(node.args[2].infer(context=context))
440              except InferenceError:
441                  raise UseInferenceDefault
442  
443      raise UseInferenceDefault

446  def infer_hasattr(node, context=None):
447      """Understand hasattr calls 
448   
449      This always guarantees three possible outcomes for calling 
450      hasattr: Const(False) when we are sure that the object 
451      doesn't have the intended attribute, Const(True) when 
452      we know that the object has the attribute and Uninferable 
453      when we are unsure of the outcome of the function call. 
454      """
455      try:
456          obj, attr = _infer_getattr_args(node, context)
457          if (
458              obj is util.Uninferable
459              or attr is util.Uninferable
460              or not hasattr(obj, "getattr")
461          ):
462              return util.Uninferable
463          obj.getattr(attr, context=context)
464      except UseInferenceDefault:
465          # Can't infer something from this function call.
466          return util.Uninferable
467      except AttributeInferenceError:
468          # Doesn't have it.
469          return nodes.Const(False)
470      return nodes.Const(True)

473  def infer_callable(node, context=None):
474      """Understand callable calls 
475   
476      This follows Python's semantics, where an object 
477      is callable if it provides an attribute __call__, 
478      even though that attribute is something which can't be 
479      called. 
480      """
481      if len(node.args) != 1:
482          # Invalid callable call.
483          raise UseInferenceDefault
484  
485      argument = node.args[0]
486      try:
487          inferred = next(argument.infer(context=context))
488      except InferenceError:
489          return util.Uninferable
490      if inferred is util.Uninferable:
491          return util.Uninferable
492      return nodes.Const(inferred.callable())

495  def infer_property(node, context=None):
496      """Understand `property` class 
497   
498      This only infers the output of `property` 
499      call, not the arguments themselves. 
500      """
501      if len(node.args) < 1:
502          # Invalid property call.
503          raise UseInferenceDefault
504  
505      getter = node.args[0]
506      try:
507          inferred = next(getter.infer(context=context))
508      except InferenceError:
509          raise UseInferenceDefault
510  
511      if not isinstance(inferred, (nodes.FunctionDef, nodes.Lambda)):
512          raise UseInferenceDefault
513  
514      return objects.Property(
515          function=inferred,
516          name=inferred.name,
517          doc=getattr(inferred, "doc", None),
518          lineno=node.lineno,
519          parent=node,
520          col_offset=node.col_offset,
521      )

524  def infer_bool(node, context=None):
525      """Understand bool calls."""
526      if len(node.args) > 1:
527          # Invalid bool call.
528          raise UseInferenceDefault
529  
530      if not node.args:
531          return nodes.Const(False)
532  
533      argument = node.args[0]
534      try:
535          inferred = next(argument.infer(context=context))
536      except InferenceError:
537          return util.Uninferable
538      if inferred is util.Uninferable:
539          return util.Uninferable
540  
541      bool_value = inferred.bool_value(context=context)
542      if bool_value is util.Uninferable:
543          return util.Uninferable
544      return nodes.Const(bool_value)

555  def infer_slice(node, context=None):
556      """Understand `slice` calls."""
557      args = node.args
558      if not 0 < len(args) <= 3:
559          raise UseInferenceDefault
560  
561      infer_func = partial(helpers.safe_infer, context=context)
562      args = [infer_func(arg) for arg in args]
563      for arg in args:
564          if not arg or arg is util.Uninferable:
565              raise UseInferenceDefault
566          if not isinstance(arg, nodes.Const):
567              raise UseInferenceDefault
568          if not isinstance(arg.value, (type(None), int)):
569              raise UseInferenceDefault
570  
571      if len(args) < 3:
572          # Make sure we have 3 arguments.
573          args.extend([None] * (3 - len(args)))
574  
575      slice_node = nodes.Slice(
576          lineno=node.lineno, col_offset=node.col_offset, parent=node.parent
577      )
578      slice_node.postinit(*args)
579      return slice_node

588  def _infer_object__new__decorator_check(node):
589      """Predicate before inference_tip 
590   
591      Check if the given ClassDef has an @object.__new__ decorator 
592      """
593      if not node.decorators:
594          return False
595  
596      for decorator in node.decorators.nodes:
597          if isinstance(decorator, nodes.Attribute):
598              if decorator.as_string() == OBJECT_DUNDER_NEW:
599                  return True
600      return False

603  def infer_issubclass(callnode, context=None):
604      """Infer issubclass() calls 
605   
606      :param nodes.Call callnode: an `issubclass` call 
607      :param InferenceContext: the context for the inference 
608      :rtype nodes.Const: Boolean Const value of the `issubclass` call 
609      :raises UseInferenceDefault: If the node cannot be inferred 
610      """
611      call = arguments.CallSite.from_call(callnode, context=context)
612      if call.keyword_arguments:
613          # issubclass doesn't support keyword arguments
614          raise UseInferenceDefault("TypeError: issubclass() takes no keyword arguments")
615      if len(call.positional_arguments) != 2:
616          raise UseInferenceDefault(
617              "Expected two arguments, got {count}".format(
618                  count=len(call.positional_arguments)
619              )
620          )
621      # The left hand argument is the obj to be checked
622      obj_node, class_or_tuple_node = call.positional_arguments
623  
624      try:
625          obj_type = next(obj_node.infer(context=context))
626      except InferenceError as exc:
627          raise UseInferenceDefault from exc
628      if not isinstance(obj_type, nodes.ClassDef):
629          raise UseInferenceDefault("TypeError: arg 1 must be class")
630  
631      # The right hand argument is the class(es) that the given
632      # object is to be checked against.
633      try:
634          class_container = _class_or_tuple_to_container(
635              class_or_tuple_node, context=context
636          )
637      except InferenceError as exc:
638          raise UseInferenceDefault from exc
639      try:
640          issubclass_bool = helpers.object_issubclass(obj_type, class_container, context)
641      except AstroidTypeError as exc:
642          raise UseInferenceDefault("TypeError: " + str(exc)) from exc
643      except MroError as exc:
644          raise UseInferenceDefault from exc
645      return nodes.Const(issubclass_bool)

648  def infer_isinstance(callnode, context=None):
649      """Infer isinstance calls 
650   
651      :param nodes.Call callnode: an isinstance call 
652      :param InferenceContext: context for call 
653          (currently unused but is a common interface for inference) 
654      :rtype nodes.Const: Boolean Const value of isinstance call 
655   
656      :raises UseInferenceDefault: If the node cannot be inferred 
657      """
658      call = arguments.CallSite.from_call(callnode, context=context)
659      if call.keyword_arguments:
660          # isinstance doesn't support keyword arguments
661          raise UseInferenceDefault("TypeError: isinstance() takes no keyword arguments")
662      if len(call.positional_arguments) != 2:
663          raise UseInferenceDefault(
664              "Expected two arguments, got {count}".format(
665                  count=len(call.positional_arguments)
666              )
667          )
668      # The left hand argument is the obj to be checked
669      obj_node, class_or_tuple_node = call.positional_arguments
670      # The right hand argument is the class(es) that the given
671      # obj is to be check is an instance of
672      try:
673          class_container = _class_or_tuple_to_container(
674              class_or_tuple_node, context=context
675          )
676      except InferenceError:
677          raise UseInferenceDefault
678      try:
679          isinstance_bool = helpers.object_isinstance(obj_node, class_container, context)
680      except AstroidTypeError as exc:
681          raise UseInferenceDefault("TypeError: " + str(exc))
682      except MroError as exc:
683          raise UseInferenceDefault from exc
684      if isinstance_bool is util.Uninferable:
685          raise UseInferenceDefault
686      return nodes.Const(isinstance_bool)

689  def _class_or_tuple_to_container(node, context=None):
690      # Move inferences results into container
691      # to simplify later logic
692      # raises InferenceError if any of the inferences fall through
693      node_infer = next(node.infer(context=context))
694      # arg2 MUST be a type or a TUPLE of types
695      # for isinstance
696      if isinstance(node_infer, nodes.Tuple):
697          class_container = [
698              next(node.infer(context=context)) for node in node_infer.elts
699          ]
700          class_container = [
701              klass_node for klass_node in class_container if klass_node is not None
702          ]
703      else:
704          class_container = [node_infer]
705      return class_container

708  def infer_len(node, context=None):
709      """Infer length calls 
710   
711      :param nodes.Call node: len call to infer 
712      :param context.InferenceContext: node context 
713      :rtype nodes.Const: a Const node with the inferred length, if possible 
714      """
715      call = arguments.CallSite.from_call(node, context=context)
716      if call.keyword_arguments:
717          raise UseInferenceDefault("TypeError: len() must take no keyword arguments")
718      if len(call.positional_arguments) != 1:
719          raise UseInferenceDefault(
720              "TypeError: len() must take exactly one argument "
721              "({len}) given".format(len=len(call.positional_arguments))
722          )
723      [argument_node] = call.positional_arguments
724      try:
725          return nodes.Const(helpers.object_len(argument_node, context=context))
726      except (AstroidTypeError, InferenceError) as exc:
727          raise UseInferenceDefault(str(exc)) from exc

746  def infer_int(node, context=None):
747      """Infer int() calls 
748   
749      :param nodes.Call node: int() call to infer 
750      :param context.InferenceContext: node context 
751      :rtype nodes.Const: a Const containing the integer value of the int() call 
752      """
753      call = arguments.CallSite.from_call(node, context=context)
754      if call.keyword_arguments:
755          raise UseInferenceDefault("TypeError: int() must take no keyword arguments")
756  
757      if call.positional_arguments:
758          try:
759              first_value = next(call.positional_arguments[0].infer(context=context))
760          except (InferenceError, StopIteration) as exc:
761              raise UseInferenceDefault(str(exc)) from exc
762  
763          if first_value is util.Uninferable:
764              raise UseInferenceDefault
765  
766          if isinstance(first_value, nodes.Const) and isinstance(
767              first_value.value, (int, str)
768          ):
769              try:
770                  actual_value = int(first_value.value)
771              except ValueError:
772                  return nodes.Const(0)
773              return nodes.Const(actual_value)
774  
775      return nodes.Const(0)

778  def infer_dict_fromkeys(node, context=None):
779      """Infer dict.fromkeys 
780   
781      :param nodes.Call node: dict.fromkeys() call to infer 
782      :param context.InferenceContext: node context 
783      :rtype nodes.Dict: 
784          a Dictionary containing the values that astroid was able to infer. 
785          In case the inference failed for any reason, an empty dictionary 
786          will be inferred instead. 
787      """
788  
789      def _build_dict_with_elements(elements):
790          new_node = nodes.Dict(
791              col_offset=node.col_offset, lineno=node.lineno, parent=node.parent
792          )
793          new_node.postinit(elements)
794          return new_node
795  
796      call = arguments.CallSite.from_call(node, context=context)
797      if call.keyword_arguments:
798          raise UseInferenceDefault("TypeError: int() must take no keyword arguments")
799      if len(call.positional_arguments) not in {1, 2}:
800          raise UseInferenceDefault(
801              "TypeError: Needs between 1 and 2 positional arguments"
802          )
803  
804      default = nodes.Const(None)
805      values = call.positional_arguments[0]
806      try:
807          inferred_values = next(values.infer(context=context))
808      except InferenceError:
809          return _build_dict_with_elements([])
810      if inferred_values is util.Uninferable:
811          return _build_dict_with_elements([])
812  
813      # Limit to a couple of potential values, as this can become pretty complicated
814      accepted_iterable_elements = (nodes.Const,)
815      if isinstance(inferred_values, (nodes.List, nodes.Set, nodes.Tuple)):
816          elements = inferred_values.elts
817          for element in elements:
818              if not isinstance(element, accepted_iterable_elements):
819                  # Fallback to an empty dict
820                  return _build_dict_with_elements([])
821  
822          elements_with_value = [(element, default) for element in elements]
823          return _build_dict_with_elements(elements_with_value)
824  
825      elif isinstance(inferred_values, nodes.Const) and isinstance(
826          inferred_values.value, (str, bytes)
827      ):
828          elements = [
829              (nodes.Const(element), default) for element in inferred_values.value
830          ]
831          return _build_dict_with_elements(elements)
832      elif isinstance(inferred_values, nodes.Dict):
833          keys = inferred_values.itered()
834          for key in keys:
835              if not isinstance(key, accepted_iterable_elements):
836                  # Fallback to an empty dict
837                  return _build_dict_with_elements([])
838  
839          elements_with_value = [(element, default) for element in keys]
840          return _build_dict_with_elements(elements_with_value)
841  
842      # Fallback to an empty dictionary
843      return _build_dict_with_elements([])
