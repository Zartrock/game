10   def _thread_transform():
11       return astroid.parse(
12           """ 
13       class lock(object): 
14           def acquire(self, blocking=True, timeout=-1): 
15               return False 
16           def release(self): 
17               pass 
18           def __enter__(self): 
19               return True 
20           def __exit__(self, *args): 
21               pass 
22           def locked(self): 
23               return False 
24    
25       def Lock(): 
26           return lock() 
27       """
28       )
29   
30   
31   astroid.register_module_extender(astroid.MANAGER, "threading", _thread_transform)
