222  def get_access_path(key, parts):
223      """ Given a list of format specifiers, returns 
224      the final access path (e.g. a.b.c[0][1]). 
225      """
226      path = []
227      for is_attribute, specifier in parts:
228          if is_attribute:
229              path.append(".{}".format(specifier))
230          else:
231              path.append("[{!r}]".format(specifier))
232      return str(key) + "".join(path)

235  def arg_matches_format_type(arg_type, format_type):
236      if format_type in "sr":
237          # All types can be printed with %s and %r
238          return True
239      if isinstance(arg_type, astroid.Instance):
240          arg_type = arg_type.pytype()
241          if arg_type == BUILTINS_STR:
242              return format_type == "c"
243          if arg_type == BUILTINS_FLOAT:
244              return format_type in "deEfFgGn%"
245          if arg_type == BUILTINS_INT:
246              # Integers allow all types
247              return True
248          return False
249      return True

252  class StringFormatChecker(BaseChecker):
253      """Checks string formatting operations to ensure that the format string 
254      is valid and the arguments match the format string. 
255      """
256  
257      __implements__ = (IAstroidChecker,)
258      name = "string"
259      msgs = MSGS
260  
261      # pylint: disable=too-many-branches
262      @check_messages(
263          "bad-format-character",
264          "truncated-format-string",
265          "mixed-format-string",
266          "bad-format-string-key",
267          "missing-format-string-key",
268          "unused-format-string-key",
269          "bad-string-format-type",
270          "format-needs-mapping",
271          "too-many-format-args",
272          "too-few-format-args",
273          "bad-string-format-type",
274      )
275      def visit_binop(self, node):
276          if node.op != "%":
277              return
278          left = node.left
279          args = node.right
280  
281          if not (isinstance(left, astroid.Const) and isinstance(left.value, str)):
282              return
283          format_string = left.value
284          try:
285              (
286                  required_keys,
287                  required_num_args,
288                  required_key_types,
289                  required_arg_types,
290              ) = utils.parse_format_string(format_string)
291          except utils.UnsupportedFormatCharacter as exc:
292              formatted = format_string[exc.index]
293              self.add_message(
294                  "bad-format-character",
295                  node=node,
296                  args=(formatted, ord(formatted), exc.index),
297              )
298              return
299          except utils.IncompleteFormatString:
300              self.add_message("truncated-format-string", node=node)
301              return
302          if required_keys and required_num_args:
303              # The format string uses both named and unnamed format
304              # specifiers.
305              self.add_message("mixed-format-string", node=node)
306          elif required_keys:
307              # The format string uses only named format specifiers.
308              # Check that the RHS of the % operator is a mapping object
309              # that contains precisely the set of keys required by the
310              # format string.
311              if isinstance(args, astroid.Dict):
312                  keys = set()
313                  unknown_keys = False
314                  for k, _ in args.items:
315                      if isinstance(k, astroid.Const):
316                          key = k.value
317                          if isinstance(key, str):
318                              keys.add(key)
319                          else:
320                              self.add_message(
321                                  "bad-format-string-key", node=node, args=key
322                              )
323                      else:
324                          # One of the keys was something other than a
325                          # constant.  Since we can't tell what it is,
326                          # suppress checks for missing keys in the
327                          # dictionary.
328                          unknown_keys = True
329                  if not unknown_keys:
330                      for key in required_keys:
331                          if key not in keys:
332                              self.add_message(
333                                  "missing-format-string-key", node=node, args=key
334                              )
335                  for key in keys:
336                      if key not in required_keys:
337                          self.add_message(
338                              "unused-format-string-key", node=node, args=key
339                          )
340                  for key, arg in args.items:
341                      if not isinstance(key, astroid.Const):
342                          continue
343                      format_type = required_key_types.get(key.value, None)
344                      arg_type = utils.safe_infer(arg)
345                      if (
346                          format_type is not None
347                          and arg_type not in (None, astroid.Uninferable)
348                          and not arg_matches_format_type(arg_type, format_type)
349                      ):
350                          self.add_message(
351                              "bad-string-format-type",
352                              node=node,
353                              args=(arg_type.pytype(), format_type),
354                          )
355              elif isinstance(args, (OTHER_NODES, astroid.Tuple)):
356                  type_name = type(args).__name__
357                  self.add_message("format-needs-mapping", node=node, args=type_name)
358              # else:
359              # The RHS of the format specifier is a name or
360              # expression.  It may be a mapping object, so
361              # there's nothing we can check.
362          else:
363              # The format string uses only unnamed format specifiers.
364              # Check that the number of arguments passed to the RHS of
365              # the % operator matches the number required by the format
366              # string.
367              args_elts = ()
368              if isinstance(args, astroid.Tuple):
369                  rhs_tuple = utils.safe_infer(args)
370                  num_args = None
371                  if hasattr(rhs_tuple, "elts"):
372                      args_elts = rhs_tuple.elts
373                      num_args = len(args_elts)
374              elif isinstance(args, (OTHER_NODES, (astroid.Dict, astroid.DictComp))):
375                  args_elts = [args]
376                  num_args = 1
377              else:
378                  # The RHS of the format specifier is a name or
379                  # expression.  It could be a tuple of unknown size, so
380                  # there's nothing we can check.
381                  num_args = None
382              if num_args is not None:
383                  if num_args > required_num_args:
384                      self.add_message("too-many-format-args", node=node)
385                  elif num_args < required_num_args:
386                      self.add_message("too-few-format-args", node=node)
387                  for arg, format_type in zip(args_elts, required_arg_types):
388                      if not arg:
389                          continue
390                      arg_type = utils.safe_infer(arg)
391                      if arg_type not in (
392                          None,
393                          astroid.Uninferable,
394                      ) and not arg_matches_format_type(arg_type, format_type):
395                          self.add_message(
396                              "bad-string-format-type",
397                              node=node,
398                              args=(arg_type.pytype(), format_type),
399                          )

401      @check_messages("f-string-without-interpolation")
402      def visit_joinedstr(self, node):
403          if isinstance(node.parent, astroid.FormattedValue):
404              return
405          for value in node.values:
406              if isinstance(value, astroid.FormattedValue):
407                  return
408          self.add_message("f-string-without-interpolation", node=node)

410      @check_messages(*MSGS)
411      def visit_call(self, node):
412          func = utils.safe_infer(node.func)
413          if (
414              isinstance(func, astroid.BoundMethod)
415              and isinstance(func.bound, astroid.Instance)
416              and func.bound.name in ("str", "unicode", "bytes")
417          ):
418              if func.name in ("strip", "lstrip", "rstrip") and node.args:
419                  arg = utils.safe_infer(node.args[0])
420                  if not isinstance(arg, astroid.Const) or not isinstance(arg.value, str):
421                      return
422                  if len(arg.value) != len(set(arg.value)):
423                      self.add_message(
424                          "bad-str-strip-call",
425                          node=node,
426                          args=(func.bound.name, func.name),
427                      )
428              elif func.name == "format":
429                  self._check_new_format(node, func)

431      def _detect_vacuous_formatting(self, node, positional_arguments):
432          counter = collections.Counter(
433              arg.name for arg in positional_arguments if isinstance(arg, astroid.Name)
434          )
435          for name, count in counter.items():
436              if count == 1:
437                  continue
438              self.add_message(
439                  "duplicate-string-formatting-argument", node=node, args=(name,)
440              )

442      def _check_new_format(self, node, func):
443          """Check the new string formatting. """
444          # Skip ormat nodes which don't have an explicit string on the
445          # left side of the format operation.
446          # We do this because our inference engine can't properly handle
447          # redefinitions of the original string.
448          # Note that there may not be any left side at all, if the format method
449          # has been assigned to another variable. See issue 351. For example:
450          #
451          #    fmt = 'some string {}'.format
452          #    fmt('arg')
453          if isinstance(node.func, astroid.Attribute) and not isinstance(
454              node.func.expr, astroid.Const
455          ):
456              return
457          if node.starargs or node.kwargs:
458              return
459          try:
460              strnode = next(func.bound.infer())
461          except astroid.InferenceError:
462              return
463          if not (isinstance(strnode, astroid.Const) and isinstance(strnode.value, str)):
464              return
465          try:
466              call_site = CallSite.from_call(node)
467          except astroid.InferenceError:
468              return
469  
470          try:
471              fields, num_args, manual_pos = utils.parse_format_method_string(
472                  strnode.value
473              )
474          except utils.IncompleteFormatString:
475              self.add_message("bad-format-string", node=node)
476              return
477  
478          positional_arguments = call_site.positional_arguments
479          named_arguments = call_site.keyword_arguments
480          named_fields = {field[0] for field in fields if isinstance(field[0], str)}
481          if num_args and manual_pos:
482              self.add_message("format-combined-specification", node=node)
483              return
484  
485          check_args = False
486          # Consider "{[0]} {[1]}" as num_args.
487          num_args += sum(1 for field in named_fields if field == "")
488          if named_fields:
489              for field in named_fields:
490                  if field and field not in named_arguments:
491                      self.add_message(
492                          "missing-format-argument-key", node=node, args=(field,)
493                      )
494              for field in named_arguments:
495                  if field not in named_fields:
496                      self.add_message(
497                          "unused-format-string-argument", node=node, args=(field,)
498                      )
499              # num_args can be 0 if manual_pos is not.
500              num_args = num_args or manual_pos
501              if positional_arguments or num_args:
502                  empty = any(True for field in named_fields if field == "")
503                  if named_arguments or empty:
504                      # Verify the required number of positional arguments
505                      # only if the .format got at least one keyword argument.
506                      # This means that the format strings accepts both
507                      # positional and named fields and we should warn
508                      # when one of the them is missing or is extra.
509                      check_args = True
510          else:
511              check_args = True
512          if check_args:
513              # num_args can be 0 if manual_pos is not.
514              num_args = num_args or manual_pos
515              if len(positional_arguments) > num_args:
516                  self.add_message("too-many-format-args", node=node)
517              elif len(positional_arguments) < num_args:
518                  self.add_message("too-few-format-args", node=node)
519  
520          self._detect_vacuous_formatting(node, positional_arguments)
521          self._check_new_format_specifiers(node, fields, named_arguments)
522  
523      def _check_new_format_specifiers(self, node, fields, named):
524          """ 
525          Check attribute and index access in the format 
526          string ("{0.a}" and "{0[a]}"). 
527          """
528          for key, specifiers in fields:
529              # Obtain the argument. If it can't be obtained
530              # or inferred, skip this check.
531              if key == "":
532                  # {[0]} will have an unnamed argument, defaulting
533                  # to 0. It will not be present in `named`, so use the value
534                  # 0 for it.
535                  key = 0
536              if isinstance(key, numbers.Number):
537                  try:
538                      argname = utils.get_argument_from_call(node, key)
539                  except utils.NoSuchArgumentError:
540                      continue
541              else:
542                  if key not in named:
543                      continue
544                  argname = named[key]
545              if argname in (astroid.Uninferable, None):
546                  continue
547              try:
548                  argument = utils.safe_infer(argname)
549              except astroid.InferenceError:
550                  continue
551              if not specifiers or not argument:
552                  # No need to check this key if it doesn't
553                  # use attribute / item access
554                  continue
555              if argument.parent and isinstance(argument.parent, astroid.Arguments):
556                  # Ignore any object coming from an argument,
557                  # because we can't infer its value properly.
558                  continue
559              previous = argument
560              parsed = []
561              for is_attribute, specifier in specifiers:
562                  if previous is astroid.Uninferable:
563                      break
564                  parsed.append((is_attribute, specifier))
565                  if is_attribute:
566                      try:
567                          previous = previous.getattr(specifier)[0]
568                      except astroid.NotFoundError:
569                          if (
570                              hasattr(previous, "has_dynamic_getattr")
571                              and previous.has_dynamic_getattr()
572                          ):
573                              # Don't warn if the object has a custom __getattr__
574                              break
575                          path = get_access_path(key, parsed)
576                          self.add_message(
577                              "missing-format-attribute",
578                              args=(specifier, path),
579                              node=node,
580                          )
581                          break
582                  else:
583                      warn_error = False
584                      if hasattr(previous, "getitem"):
585                          try:
586                              previous = previous.getitem(astroid.Const(specifier))
587                          except (
588                              astroid.AstroidIndexError,
589                              astroid.AstroidTypeError,
590                              astroid.AttributeInferenceError,
591                          ):
592                              warn_error = True
593                          except astroid.InferenceError:
594                              break
595                          if previous is astroid.Uninferable:
596                              break
597                      else:
598                          try:
599                              # Lookup __getitem__ in the current node,
600                              # but skip further checks, because we can't
601                              # retrieve the looked object
602                              previous.getattr("__getitem__")
603                              break
604                          except astroid.NotFoundError:
605                              warn_error = True
606                      if warn_error:
607                          path = get_access_path(key, parsed)
608                          self.add_message(
609                              "invalid-format-index", args=(specifier, path), node=node
610                          )
611                          break
612  
613                  try:
614                      previous = next(previous.infer())
615                  except astroid.InferenceError:
616                      # can't check further if we can't infer it
617                      break

647          "W1405": (
648              "Quote delimiter %s is inconsistent with the rest of the file",
649              "inconsistent-quotes",
650              "Quote delimiters are not used consistently throughout a module "
651              "(with allowances made for avoiding unnecessary escaping).",
652          ),
653      }
654      options = (
655          (
656              "check-str-concat-over-line-jumps",
657              {
658                  "default": False,
659                  "type": "yn",
660                  "metavar": "<y_or_n>",
661                  "help": "This flag controls whether the "
662                  "implicit-str-concat should generate a warning "
663                  "on implicit string concatenation in sequences defined over "
664                  "several lines.",
665              },
666          ),
667          (
668              "check-quote-consistency",
669              {
670                  "default": False,
671                  "type": "yn",
672                  "metavar": "<y_or_n>",
673                  "help": "This flag controls whether inconsistent-quotes generates a "
674                  "warning when the character used as a quote delimiter is used "
675                  "inconsistently within a module.",
676              },
677          ),
678      )
679  
680      # Characters that have a special meaning after a backslash in either
681      # Unicode or byte strings.
682      ESCAPE_CHARACTERS = "abfnrtvx\n\r\t\\'\"01234567"
683  
684      # Characters that have a special meaning after a backslash but only in
685      # Unicode strings.
686      UNICODE_ESCAPE_CHARACTERS = "uUN"

688      def __init__(self, *args, **kwargs):
689          super().__init__(*args, **kwargs)
690          self.string_tokens = {}  # token position -> (token value, next token)
691  
692      def process_module(self, module):
693          self._unicode_literals = "unicode_literals" in module.future_imports
694  
695      def process_tokens(self, tokens):
696          encoding = "ascii"
697          for i, (tok_type, token, start, _, line) in enumerate(tokens):
698              if tok_type == tokenize.ENCODING:
699                  # this is always the first token processed
700                  encoding = token
701              elif tok_type == tokenize.STRING:
702                  # 'token' is the whole un-parsed token; we can look at the start
703                  # of it to see whether it's a raw or unicode string etc.
704                  self.process_string_token(token, start[0])
705                  # We figure the next token, ignoring comments & newlines:
706                  j = i + 1
707                  while j < len(tokens) and tokens[j].type in (
708                      tokenize.NEWLINE,
709                      tokenize.NL,
710                      tokenize.COMMENT,
711                  ):
712                      j += 1
713                  next_token = tokens[j] if j < len(tokens) else None
714                  if encoding != "ascii":
715                      # We convert `tokenize` character count into a byte count,
716                      # to match with astroid `.col_offset`
717                      start = (start[0], len(line[: start[1]].encode(encoding)))
718                  self.string_tokens[start] = (str_eval(token), next_token)
719  
720          if self.config.check_quote_consistency:
721              self.check_for_consistent_string_delimiters(tokens)
722

739      def check_for_consistent_string_delimiters(
740          self, tokens: Iterable[tokenize.TokenInfo]
741      ) -> None:
742          """Adds a message for each string using inconsistent quote delimiters. 
743   
744          Quote delimiters are used inconsistently if " and ' are mixed in a module's 
745          shortstrings without having done so to avoid escaping an internal quote 
746          character. 
747   
748          Args: 
749            tokens: The tokens to be checked against for consistent usage. 
750          """
751          string_delimiters = collections.Counter()  # type: typing.Counter[str]
752  
753          # First, figure out which quote character predominates in the module
754          for tok_type, token, _, _, _ in tokens:
755              if tok_type == tokenize.STRING and _is_quote_delimiter_chosen_freely(token):
756                  string_delimiters[_get_quote_delimiter(token)] += 1
757  
758          if len(string_delimiters) > 1:
759              # Ties are broken arbitrarily
760              most_common_delimiter = string_delimiters.most_common(1)[0][0]
761              for tok_type, token, start, _, _ in tokens:
762                  if tok_type != tokenize.STRING:
763                      continue
764                  quote_delimiter = _get_quote_delimiter(token)
765                  if (
766                      _is_quote_delimiter_chosen_freely(token)
767                      and quote_delimiter != most_common_delimiter
768                  ):
769                      self.add_message(
770                          "inconsistent-quotes", line=start[0], args=(quote_delimiter,)
771                      )
772  
773      def check_for_concatenated_strings(self, elements, iterable_type):
774          for elt in elements:
775              if not (isinstance(elt, Const) and elt.pytype() in _AST_NODE_STR_TYPES):
776                  continue
777              if elt.col_offset < 0:
778                  # This can happen in case of escaped newlines
779                  continue
780              if (elt.lineno, elt.col_offset) not in self.string_tokens:
781                  # This may happen with Latin1 encoding
782                  # cf. https://github.com/PyCQA/pylint/issues/2610
783                  continue
784              matching_token, next_token = self.string_tokens[
785                  (elt.lineno, elt.col_offset)
786              ]
787              # We detect string concatenation: the AST Const is the
788              # combination of 2 string tokens
789              if matching_token != elt.value and next_token is not None:
790                  if next_token.type == tokenize.STRING and (
791                      next_token.start[0] == elt.lineno
792                      or self.config.check_str_concat_over_line_jumps
793                  ):
794                      self.add_message(
795                          "implicit-str-concat", line=elt.lineno, args=(iterable_type,)
796                      )
797  
798      def process_string_token(self, token, start_row):
799          quote_char = None
800          index = None
801          for index, char in enumerate(token):
802              if char in "'\"":
803                  quote_char = char
804                  break
805          if quote_char is None:
806              return
807  
808          prefix = token[:index].lower()  # markers like u, b, r.
809          after_prefix = token[index:]
810          if after_prefix[:3] == after_prefix[-3:] == 3 * quote_char:
811              string_body = after_prefix[3:-3]
812          else:
813              string_body = after_prefix[1:-1]  # Chop off quotes
814          # No special checks on raw strings at the moment.
815          if "r" not in prefix:
816              self.process_non_raw_string_token(prefix, string_body, start_row)
817  
818      def process_non_raw_string_token(self, prefix, string_body, start_row):
819          """check for bad escapes in a non-raw string. 
820   
821          prefix: lowercase string of eg 'ur' string prefix markers. 
822          string_body: the un-parsed body of the string, not including the quote 
823          marks. 
824          start_row: integer line number in the source. 
825          """
826          # Walk through the string; if we see a backslash then escape the next
827          # character, and skip over it.  If we see a non-escaped character,
828          # alert, and continue.
829          #
830          # Accept a backslash when it escapes a backslash, or a quote, or
831          # end-of-line, or one of the letters that introduce a special escape
832          # sequence <https://docs.python.org/reference/lexical_analysis.html>
833          #
834          index = 0
835          while True:
836              index = string_body.find("\\", index)
837              if index == -1:
838                  break
839              # There must be a next character; having a backslash at the end
840              # of the string would be a SyntaxError.
841              next_char = string_body[index + 1]
842              match = string_body[index : index + 2]
843              if next_char in self.UNICODE_ESCAPE_CHARACTERS:
844                  if "u" in prefix:
845                      pass
846                  elif "b" not in prefix:
847                      pass  # unicode by default
848                  else:
849                      self.add_message(
850                          "anomalous-unicode-escape-in-string",
851                          line=start_row,
852                          args=(match,),
853                          col_offset=index,
854                      )
855              elif next_char not in self.ESCAPE_CHARACTERS:
856                  self.add_message(
857                      "anomalous-backslash-in-string",
858                      line=start_row,
859                      args=(match,),
860                      col_offset=index,
861                  )
862              # Whether it was a valid escape or not, backslash followed by
863              # another character can always be consumed whole: the second
864              # character can never be the start of a new backslash escape.
865              index += 2

874  def str_eval(token):
875      """ 
876      Mostly replicate `ast.literal_eval(token)` manually to avoid any performance hit. 
877      This supports f-strings, contrary to `ast.literal_eval`. 
878      We have to support all string literal notations: 
879      https://docs.python.org/3/reference/lexical_analysis.html#string-and-bytes-literals 
880      """
881      if token[0:2].lower() in ("fr", "rf"):
882          token = token[2:]
883      elif token[0].lower() in ("r", "u", "f"):
884          token = token[1:]
885      if token[0:3] in ('"""', "'''"):
886          return token[3:-3]
887      return token[1:-1]

890  def _is_long_string(string_token: str) -> bool:
891      """Is this string token a "longstring" (is it triple-quoted)? 
892   
893      Long strings are triple-quoted as defined in 
894      https://docs.python.org/3/reference/lexical_analysis.html#string-and-bytes-literals 
895   
896      This function only checks characters up through the open quotes.  Because it's meant 
897      to be applied only to tokens that represent string literals, it doesn't bother to 
898      check for close-quotes (demonstrating that the literal is a well-formed string). 
899   
900      Args: 
901          string_token: The string token to be parsed. 
902   
903      Returns: 
904          A boolean representing whether or not this token matches a longstring 
905          regex. 
906      """
907      return bool(
908          SINGLE_QUOTED_REGEX.match(string_token)
909          or DOUBLE_QUOTED_REGEX.match(string_token)
910      )

913  def _get_quote_delimiter(string_token: str) -> str:
914      """Returns the quote character used to delimit this token string. 
915   
916      This function does little checking for whether the token is a well-formed 
917      string. 
918   
919      Args: 
920          string_token: The token to be parsed. 
921   
922      Returns: 
923          A string containing solely the first quote delimiter character in the passed 
924          string. 
925   
926      Raises: 
927        ValueError: No quote delimiter characters are present. 
928      """
929      match = QUOTE_DELIMITER_REGEX.match(string_token)
930      if not match:
931          raise ValueError("string token %s is not a well-formed string" % string_token)
932      return match.group(2)

935  def _is_quote_delimiter_chosen_freely(string_token: str) -> bool:
936      """Was there a non-awkward option for the quote delimiter? 
937   
938      Args: 
939          string_token: The quoted string whose delimiters are to be checked. 
940   
941      Returns: 
942          Whether there was a choice in this token's quote character that would 
943          not have involved backslash-escaping an interior quote character.  Long 
944          strings are excepted from this analysis under the assumption that their 
945          quote characters are set by policy. 
946      """
947      quote_delimiter = _get_quote_delimiter(string_token)
948      unchosen_delimiter = '"' if quote_delimiter == "'" else "'"
949      return bool(
950          quote_delimiter
951          and not _is_long_string(string_token)
952          and unchosen_delimiter not in str_eval(string_token)
953      )
