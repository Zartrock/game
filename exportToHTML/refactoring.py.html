80   def _is_trailing_comma(tokens, index):
81       """Check if the given token is a trailing comma 
82    
83       :param tokens: Sequence of modules tokens 
84       :type tokens: list[tokenize.TokenInfo] 
85       :param int index: Index of token under check in tokens 
86       :returns: True if the token is a comma which trails an expression 
87       :rtype: bool 
88       """
89       token = tokens[index]
90       if token.exact_type != tokenize.COMMA:
91           return False
92       # Must have remaining tokens on the same line such as NEWLINE
93       left_tokens = itertools.islice(tokens, index + 1, None)
94       same_line_remaining_tokens = list(
95           itertools.takewhile(
96               lambda other_token, _token=token: other_token.start[0] == _token.start[0],
97               left_tokens,
98           )
99       )
100      # Note: If the newline is tokenize.NEWLINE and not tokenize.NL
101      # then the newline denotes the end of expression
102      is_last_element = all(
103          other_token.type in (tokenize.NEWLINE, tokenize.COMMENT)
104          for other_token in same_line_remaining_tokens
105      )
106      if not same_line_remaining_tokens or not is_last_element:
107          return False
108  
109      def get_curline_index_start():
110          """Get the index denoting the start of the current line"""
111          for subindex, token in enumerate(reversed(tokens[:index])):
112              # See Lib/tokenize.py and Lib/token.py in cpython for more info
113              if token.type == tokenize.NEWLINE:
114                  return index - subindex
115          return 0
116  
117      curline_start = get_curline_index_start()
118      expected_tokens = {"return", "yield"}
119      for prevtoken in tokens[curline_start:index]:
120          if "=" in prevtoken.string or prevtoken.string in expected_tokens:
121              return True
122      return False

372      def _is_actual_elif(self, node):
373          """Check if the given node is an actual elif 
374   
375          This is a problem we're having with the builtin ast module, 
376          which splits `elif` branches into a separate if statement. 
377          Unfortunately we need to know the exact type in certain 
378          cases. 
379          """
380          if isinstance(node.parent, astroid.If):
381              orelse = node.parent.orelse
382              # current if node must directly follow an "else"
383              if orelse and orelse == [node]:
384                  if (node.lineno, node.col_offset) in self._elifs:
385                      return True
386          return False

388      def _check_simplifiable_if(self, node):
389          """Check if the given if node can be simplified. 
390   
391          The if statement can be reduced to a boolean expression 
392          in some cases. For instance, if there are two branches 
393          and both of them return a boolean value that depends on 
394          the result of the statement's test, then this can be reduced 
395          to `bool(test)` without losing any functionality. 
396          """
397  
398          if self._is_actual_elif(node):
399              # Not interested in if statements with multiple branches.
400              return
401          if len(node.orelse) != 1 or len(node.body) != 1:
402              return
403  
404          # Check if both branches can be reduced.
405          first_branch = node.body[0]
406          else_branch = node.orelse[0]
407          if isinstance(first_branch, astroid.Return):
408              if not isinstance(else_branch, astroid.Return):
409                  return
410              first_branch_is_bool = self._is_bool_const(first_branch)
411              else_branch_is_bool = self._is_bool_const(else_branch)
412              reduced_to = "'return bool(test)'"
413          elif isinstance(first_branch, astroid.Assign):
414              if not isinstance(else_branch, astroid.Assign):
415                  return
416  
417              # Check if we assign to the same value
418              first_branch_targets = [
419                  target.name
420                  for target in first_branch.targets
421                  if isinstance(target, astroid.AssignName)
422              ]
423              else_branch_targets = [
424                  target.name
425                  for target in else_branch.targets
426                  if isinstance(target, astroid.AssignName)
427              ]
428              if not first_branch_targets or not else_branch_targets:
429                  return
430              if sorted(first_branch_targets) != sorted(else_branch_targets):
431                  return
432  
433              first_branch_is_bool = self._is_bool_const(first_branch)
434              else_branch_is_bool = self._is_bool_const(else_branch)
435              reduced_to = "'var = bool(test)'"
436          else:
437              return
438  
439          if not first_branch_is_bool or not else_branch_is_bool:
440              return
441          if not first_branch.value.value:
442              # This is a case that can't be easily simplified and
443              # if it can be simplified, it will usually result in a
444              # code that's harder to understand and comprehend.
445              # Let's take for instance `arg and arg <= 3`. This could theoretically be
446              # reduced to `not arg or arg > 3`, but the net result is that now the
447              # condition is harder to understand, because it requires understanding of
448              # an extra clause:
449              #   * first, there is the negation of truthness with `not arg`
450              #   * the second clause is `arg > 3`, which occurs when arg has a
451              #     a truth value, but it implies that `arg > 3` is equivalent
452              #     with `arg and arg > 3`, which means that the user must
453              #     think about this assumption when evaluating `arg > 3`.
454              #     The original form is easier to grasp.
455              return
456  
457          self.add_message("simplifiable-if-statement", node=node, args=(reduced_to,))

459      def process_tokens(self, tokens):
460          # Process tokens and look for 'if' or 'elif'
461          for index, token in enumerate(tokens):
462              token_string = token[1]
463              if token_string == "elif":
464                  # AST exists by the time process_tokens is called, so
465                  # it's safe to assume tokens[index+1]
466                  # exists. tokens[index+1][2] is the elif's position as
467                  # reported by CPython and PyPy,
468                  # tokens[index][2] is the actual position and also is
469                  # reported by IronPython.
470                  self._elifs.extend([tokens[index][2], tokens[index + 1][2]])
471              elif _is_trailing_comma(tokens, index):
472                  if self.linter.is_message_enabled("trailing-comma-tuple"):
473                      self.add_message("trailing-comma-tuple", line=token.start[0])

485      def _check_redefined_argument_from_local(self, name_node):
486          if self._dummy_rgx and self._dummy_rgx.match(name_node.name):
487              return
488          if not name_node.lineno:
489              # Unknown position, maybe it is a manually built AST?
490              return
491  
492          scope = name_node.scope()
493          if not isinstance(scope, astroid.FunctionDef):
494              return
495  
496          for defined_argument in scope.args.nodes_of_class(
497              astroid.AssignName, skip_klass=(astroid.Lambda,)
498          ):
499              if defined_argument.name == name_node.name:
500                  self.add_message(
501                      "redefined-argument-from-local",
502                      node=name_node,
503                      args=(name_node.name,),
504                  )

562      @staticmethod
563      def _type_and_name_are_equal(node_a, node_b):
564          for _type in [astroid.Name, astroid.AssignName]:
565              if all(isinstance(_node, _type) for _node in [node_a, node_b]):
566                  return node_a.name == node_b.name
567          if all(isinstance(_node, astroid.Const) for _node in [node_a, node_b]):
568              return node_a.value == node_b.value
569          return False

571      def _is_dict_get_block(self, node):
572  
573          # "if <compare node>"
574          if not isinstance(node.test, astroid.Compare):
575              return False
576  
577          # Does not have a single statement in the guard's body
578          if len(node.body) != 1:
579              return False
580  
581          # Look for a single variable assignment on the LHS and a subscript on RHS
582          stmt = node.body[0]
583          if not (
584              isinstance(stmt, astroid.Assign)
585              and len(node.body[0].targets) == 1
586              and isinstance(node.body[0].targets[0], astroid.AssignName)
587              and isinstance(stmt.value, astroid.Subscript)
588          ):
589              return False
590  
591          # The subscript's slice needs to be the same as the test variable.
592          # Python 3.9 we no longer have the `Index` node.
593          slice_value = stmt.value.slice
594          if isinstance(slice_value, astroid.Index):
595              slice_value = slice_value.value
596          if not (
597              self._type_and_name_are_equal(stmt.value.value, node.test.ops[0][1])
598              and self._type_and_name_are_equal(slice_value, node.test.left)
599          ):
600              return False
601  
602          # The object needs to be a dictionary instance

803      def _check_nested_blocks(self, node):
804          """Update and check the number of nested blocks 
805          """
806          # only check block levels inside functions or methods
807          if not isinstance(node.scope(), astroid.FunctionDef):
808              return
809          # messages are triggered on leaving the nested block. Here we save the
810          # stack in case the current node isn't nested in the previous one
811          nested_blocks = self._nested_blocks[:]
812          if node.parent == node.scope():
813              self._nested_blocks = [node]
814          else:
815              # go through ancestors from the most nested to the less
816              for ancestor_node in reversed(self._nested_blocks):
817                  if ancestor_node == node.parent:
818                      break
819                  self._nested_blocks.pop()
820              # if the node is an elif, this should not be another nesting level
821              if isinstance(node, astroid.If) and self._is_actual_elif(node):
822                  if self._nested_blocks:
823                      self._nested_blocks.pop()
824              self._nested_blocks.append(node)
825  
826          # send message only once per group of nested blocks
827          if len(nested_blocks) > len(self._nested_blocks):
828              self._emit_nested_blocks_message_if_needed(nested_blocks)
829  
830      def _emit_nested_blocks_message_if_needed(self, nested_blocks):
831          if len(nested_blocks) > self.config.max_nested_blocks:
832              self.add_message(
833                  "too-many-nested-blocks",
834                  node=nested_blocks[0],
835                  args=(len(nested_blocks), self.config.max_nested_blocks),
836              )

838      @staticmethod
839      def _duplicated_isinstance_types(node):
840          """Get the duplicated types from the underlying isinstance calls. 
841   
842          :param astroid.BoolOp node: Node which should contain a bunch of isinstance calls. 
843          :returns: Dictionary of the comparison objects from the isinstance calls, 
844                    to duplicate values from consecutive calls. 
845          :rtype: dict 
846          """
847          duplicated_objects = set()
848          all_types = collections.defaultdict(set)
849  
850          for call in node.values:
851              if not isinstance(call, astroid.Call) or len(call.args) != 2:
852                  continue
853  
854              inferred = utils.safe_infer(call.func)
855              if not inferred or not utils.is_builtin_object(inferred):
856                  continue
857  
858              if inferred.name != "isinstance":
859                  continue
860  
861              isinstance_object = call.args[0].as_string()
862              isinstance_types = call.args[1]
863  
864              if isinstance_object in all_types:
865                  duplicated_objects.add(isinstance_object)
866  
867              if isinstance(isinstance_types, astroid.Tuple):
868                  elems = [
869                      class_type.as_string() for class_type in isinstance_types.itered()
870                  ]
871              else:
872                  elems = [isinstance_types.as_string()]
873              all_types[isinstance_object].update(elems)
874  
875          # Remove all keys which not duplicated
876          return {
877              key: value for key, value in all_types.items() if key in duplicated_objects
878          }

880      def _check_consider_merging_isinstance(self, node):
881          """Check isinstance calls which can be merged together."""
882          if node.op != "or":
883              return
884  
885          first_args = self._duplicated_isinstance_types(node)
886          for duplicated_name, class_names in first_args.items():
887              names = sorted(name for name in class_names)

890                  node=node,

894      def _check_consider_using_in(self, node):
895          allowed_ops = {"or": "==", "and": "!="}
896  
897          if node.op not in allowed_ops or len(node.values) < 2:
898              return
899  
900          for value in node.values:
901              if (
902                  not isinstance(value, astroid.Compare)
903                  or len(value.ops) != 1
904                  or value.ops[0][0] not in allowed_ops[node.op]
905              ):
906                  return
907              for comparable in value.left, value.ops[0][1]:
908                  if isinstance(comparable, astroid.Call):
909                      return
910  
911          # Gather variables and values from comparisons
912          variables, values = [], []
913          for value in node.values:
914              variable_set = set()
915              for comparable in value.left, value.ops[0][1]:
916                  if isinstance(comparable, astroid.Name):
917                      variable_set.add(comparable.as_string())
918                  values.append(comparable.as_string())
919              variables.append(variable_set)
920  
921          # Look for (common-)variables that occur in all comparisons
922          common_variables = reduce(lambda a, b: a.intersection(b), variables)
923  
924          if not common_variables:
925              return
926  
927          # Gather information for the suggestion
928          common_variable = sorted(list(common_variables))[0]
929          comprehension = "in" if node.op == "or" else "not in"
930          values = list(collections.OrderedDict.fromkeys(values))
931          values.remove(common_variable)
932          values_string = ", ".join(values) if len(values) != 1 else values[0] + ","
933          suggestion = "%s %s (%s)" % (common_variable, comprehension, values_string)
934  
935          self.add_message("consider-using-in", node=node, args=(suggestion,))

937      def _check_chained_comparison(self, node):
938          """Check if there is any chained comparison in the expression. 
939   
940          Add a refactoring message if a boolOp contains comparison like a < b and b < c, 
941          which can be chained as a < b < c. 
942   
943          Care is taken to avoid simplifying a < b < c and b < d. 
944          """
945          if node.op != "and" or len(node.values) < 2:
946              return
947  
948          def _find_lower_upper_bounds(comparison_node, uses):
949              left_operand = comparison_node.left
950              for operator, right_operand in comparison_node.ops:
951                  for operand in (left_operand, right_operand):
952                      value = None
953                      if isinstance(operand, astroid.Name):
954                          value = operand.name
955                      elif isinstance(operand, astroid.Const):
956                          value = operand.value
957  
958                      if value is None:
959                          continue
960  
961                      if operator in ("<", "<="):
962                          if operand is left_operand:
963                              uses[value]["lower_bound"].add(comparison_node)
964                          elif operand is right_operand:
965                              uses[value]["upper_bound"].add(comparison_node)
966                      elif operator in (">", ">="):
967                          if operand is left_operand:
968                              uses[value]["upper_bound"].add(comparison_node)
969                          elif operand is right_operand:
970                              uses[value]["lower_bound"].add(comparison_node)
971                  left_operand = right_operand
972  
973          uses = collections.defaultdict(
974              lambda: {"lower_bound": set(), "upper_bound": set()}
975          )
976          for comparison_node in node.values:
977              if isinstance(comparison_node, astroid.Compare):
978                  _find_lower_upper_bounds(comparison_node, uses)
979  
980          for _, bounds in uses.items():
981              num_shared = len(bounds["lower_bound"].intersection(bounds["upper_bound"]))
982              num_lower_bounds = len(bounds["lower_bound"])
983              num_upper_bounds = len(bounds["upper_bound"])
984              if num_shared < num_lower_bounds and num_shared < num_upper_bounds:
985                  self.add_message("chained-comparison", node=node)
986                  break

1198     def _is_node_return_ended(self, node):
1199         """Check if the node ends with an explicit return statement. 
1200  
1201         Args: 
1202             node (astroid.NodeNG): node to be checked. 
1203  
1204         Returns: 
1205             bool: True if the node ends with an explicit statement, False otherwise. 
1206  
1207         """
1208         #  Recursion base case
1209         if isinstance(node, astroid.Return):
1210             return True
1211         if isinstance(node, astroid.Call):
1212             try:
1213                 funcdef_node = node.func.inferred()[0]
1214                 if self._is_function_def_never_returning(funcdef_node):
1215                     return True
1216             except astroid.InferenceError:
1217                 pass
1218         # Avoid the check inside while loop as we don't know
1219         #  if they will be completed
1220         if isinstance(node, astroid.While):
1221             return True
1222         if isinstance(node, astroid.Raise):
1223             # a Raise statement doesn't need to end with a return statement
1224             # but if the exception raised is handled, then the handler has to
1225             # ends with a return statement
1226             if not node.exc:
1227                 # Ignore bare raises
1228                 return True
1229             if not utils.is_node_inside_try_except(node):
1230                 # If the raise statement is not inside a try/except statement
1231                 #  then the exception is raised and cannot be caught. No need
1232                 #  to infer it.
1233                 return True
1234             exc = utils.safe_infer(node.exc)
1235             if exc is None or exc is astroid.Uninferable or not hasattr(exc, "pytype"):
1236                 return False
1237             exc_name = exc.pytype().split(".")[-1]
1238             handlers = utils.get_exception_handlers(node, exc_name)
1239             handlers = list(handlers) if handlers is not None else []
1240             if handlers:
1241                 # among all the handlers handling the exception at least one
1242                 # must end with a return statement
1243                 return any(
1244                     self._is_node_return_ended(_handler) for _handler in handlers
1245                 )
1246             # if no handlers handle the exception then it's ok
1247             return True
1248         if isinstance(node, astroid.If):
1249             # if statement is returning if there are exactly two return statements in its
1250             #  children : one for the body part, the other for the orelse part
1251             # Do not check if inner function definition are return ended.
1252             is_orelse_returning = any(
1253                 self._is_node_return_ended(_ore)
1254                 for _ore in node.orelse
1255                 if not isinstance(_ore, astroid.FunctionDef)
1256             )
1257             is_if_returning = any(
1258                 self._is_node_return_ended(_ifn)
1259                 for _ifn in node.body
1260                 if not isinstance(_ifn, astroid.FunctionDef)
1261             )
1262             return is_if_returning and is_orelse_returning
1263         #  recurses on the children of the node except for those which are except handler
1264         # because one cannot be sure that the handler will really be used
1265         return any(
1266             self._is_node_return_ended(_child)
1267             for _child in node.get_children()
1268             if not isinstance(_child, astroid.ExceptHandler)
1269         )
1270 
1271     def _is_function_def_never_returning(self, node):
1272         """Return True if the function never returns. False otherwise. 
1273  
1274         Args: 
1275             node (astroid.FunctionDef): function definition node to be analyzed. 
1276  
1277         Returns: 
1278             bool: True if the function never returns, False otherwise. 
1279         """
1280         try:
1281             return node.qname() in self._never_returning_functions
1282         except TypeError:
1283             return False
1284 
1285     def _check_return_at_the_end(self, node):
1286         """Check for presence of a *single* return statement at the end of a 
1287         function. "return" or "return None" are useless because None is the 
1288         default return type if they are missing. 
1289  
1290         NOTE: produces a message only if there is a single return statement 
1291         in the function body. Otherwise _check_consistent_returns() is called! 
1292         Per its implementation and PEP8 we can have a "return None" at the end 
1293         of the function body if there are other return statements before that! 
1294         """
1295         if len(self._return_nodes[node.name]) > 1:
1296             return
1297         if len(node.body) <= 1:
1298             return
1299 
1300         last = node.body[-1]
1301         if isinstance(last, astroid.Return):
1302             # e.g. "return"
1303             if last.value is None:
1304                 self.add_message("useless-return", node=node)
1305             # return None"
1306             elif isinstance(last.value, astroid.Const) and (last.value.value is None):
1307                 self.add_message("useless-return", node=node)

1310 class RecommandationChecker(checkers.BaseChecker):
1311     __implements__ = (interfaces.IAstroidChecker,)
1312     name = "refactoring"
1313     msgs = {
1314         "C0200": (
1315             "Consider using enumerate instead of iterating with range and len",
1316             "consider-using-enumerate",
1317             "Emitted when code that iterates with range and len is "
1318             "encountered. Such code can be simplified by using the "
1319             "enumerate builtin.",
1320         ),
1321         "C0201": (
1322             "Consider iterating the dictionary directly instead of calling .keys()",
1323             "consider-iterating-dictionary",
1324             "Emitted when the keys of a dictionary are iterated through the .keys() "
1325             "method. It is enough to just iterate through the dictionary itself, as "
1326             'in "for key in dictionary".',
1327         ),
1328     }
1329 
1330     @staticmethod
1331     def _is_builtin(node, function):
1332         inferred = utils.safe_infer(node)
1333         if not inferred:
1334             return False
1335         return utils.is_builtin_object(inferred) and inferred.name == function
1336 
1337     @utils.check_messages("consider-iterating-dictionary")
1338     def visit_call(self, node):
1339         if not isinstance(node.func, astroid.Attribute):
1340             return
1341         if node.func.attrname != "keys":
1342             return
1343         if not isinstance(node.parent, (astroid.For, astroid.Comprehension)):
1344             return
1345 
1346         inferred = utils.safe_infer(node.func)
1347         if not isinstance(inferred, astroid.BoundMethod) or not isinstance(
1348             inferred.bound, astroid.Dict
1349         ):
1350             return
1351 
1352         if isinstance(node.parent, (astroid.For, astroid.Comprehension)):
1353             self.add_message("consider-iterating-dictionary", node=node)
1354 
1355     @utils.check_messages("consider-using-enumerate")
1356     def visit_for(self, node):
1357         """Emit a convention whenever range and len are used for indexing."""
1358         # Verify that we have a `range([start], len(...), [stop])` call and
1359         # that the object which is iterated is used as a subscript in the
1360         # body of the for.
1361 
1362         # Is it a proper range call?
1363         if not isinstance(node.iter, astroid.Call):
1364             return
1365         if not self._is_builtin(node.iter.func, "range"):
1366             return
1367         if len(node.iter.args) == 2 and not _is_constant_zero(node.iter.args[0]):
1368             return
1369         if len(node.iter.args) > 2:
1370             return
1371 
1372         # Is it a proper len call?
1373         if not isinstance(node.iter.args[-1], astroid.Call):
1374             return
1375         second_func = node.iter.args[-1].func
1376         if not self._is_builtin(second_func, "len"):
1377             return
1378         len_args = node.iter.args[-1].args
1379         if not len_args or len(len_args) != 1:
1380             return
1381         iterating_object = len_args[0]
1382         if not isinstance(iterating_object, astroid.Name):
1383             return
1384         # If we're defining __iter__ on self, enumerate won't work
1385         scope = node.scope()
1386         if iterating_object.name == "self" and scope.name == "__iter__":
1387             return
1388 
1389         # Verify that the body of the for loop uses a subscript
1390         # with the object that was iterated. This uses some heuristics
1391         # in order to make sure that the same object is used in the
1392         # for body.
1393         for child in node.body:
1394             for subscript in child.nodes_of_class(astroid.Subscript):
1395                 if not isinstance(subscript.value, astroid.Name):
1396                     continue
1397 
1398                 value = subscript.slice
1399                 if isinstance(value, astroid.Index):
1400                     value = value.value
1401                 if not isinstance(value, astroid.Name):
1402                     continue
1403                 if value.name != node.target.name:
1404                     continue
1405                 if iterating_object.name != subscript.value.name:
1406                     continue
1407                 if subscript.value.scope() != node.scope():
1408                     # Ignore this subscript if it's not in the same
1409                     # scope. This means that in the body of the for
1410                     # loop, another scope was created, where the same
1411                     # name for the iterating object was used.
1412                     continue
1413                 self.add_message("consider-using-enumerate", node=node)
1414                 return

1417 class NotChecker(checkers.BaseChecker):
1418     """checks for too many not in comparison expressions 
1419  
1420     - "not not" should trigger a warning 
1421     - "not" followed by a comparison should trigger a warning 
1422     """
1423 
1424     __implements__ = (interfaces.IAstroidChecker,)
1425     msgs = {
1426         "C0113": (
1427             'Consider changing "%s" to "%s"',
1428             "unneeded-not",
1429             "Used when a boolean expression contains an unneeded negation.",
1430         )
1431     }
1432     name = "refactoring"
1433     reverse_op = {
1434         "<": ">=",
1435         "<=": ">",
1436         ">": "<=",
1437         ">=": "<",
1438         "==": "!=",
1439         "!=": "==",
1440         "in": "not in",
1441         "is": "is not",
1442     }
1443     # sets are not ordered, so for example "not set(LEFT_VALS) <= set(RIGHT_VALS)" is
1444     # not equivalent to "set(LEFT_VALS) > set(RIGHT_VALS)"
1445     skipped_nodes = (astroid.Set,)
1446     # 'builtins' py3, '__builtin__' py2
1447     skipped_classnames = [
1448         "%s.%s" % (builtins.__name__, qname) for qname in ("set", "frozenset")
1449     ]
1450 
1451     @utils.check_messages("unneeded-not")
1452     def visit_unaryop(self, node):
1453         if node.op != "not":
1454             return
1455         operand = node.operand
1456 
1457         if isinstance(operand, astroid.UnaryOp) and operand.op == "not":
1458             self.add_message(
1459                 "unneeded-not",
1460                 node=node,
1461                 args=(node.as_string(), operand.operand.as_string()),
1462             )
1463         elif isinstance(operand, astroid.Compare):
1464             left = operand.left
1465             # ignore multiple comparisons
1466             if len(operand.ops) > 1:
1467                 return
1468             operator, right = operand.ops[0]
1469             if operator not in self.reverse_op:
1470                 return
1471             # Ignore __ne__ as function of __eq__
1472             frame = node.frame()
1473             if frame.name == "__ne__" and operator == "==":
1474                 return
1475             for _type in (utils.node_type(left), utils.node_type(right)):
1476                 if not _type:
1477                     return
1478                 if isinstance(_type, self.skipped_nodes):
1479                     return
1480                 if (
1481                     isinstance(_type, astroid.Instance)
1482                     and _type.qname() in self.skipped_classnames
1483                 ):
1484                     return
1485             suggestion = "%s %s %s" % (
1486                 left.as_string(),
1487                 self.reverse_op[operator],
1488                 right.as_string(),
1489             )
1490             self.add_message(
1491                 "unneeded-not", node=node, args=(node.as_string(), suggestion)
1492             )

1495 class LenChecker(checkers.BaseChecker):
1496     """Checks for incorrect usage of len() inside conditions. 
1497     Pep8 states: 
1498     For sequences, (strings, lists, tuples), use the fact that empty sequences are false. 
1499  
1500         Yes: if not seq: 
1501              if seq: 
1502  
1503         No: if len(seq): 
1504             if not len(seq): 
1505  
1506     Problems detected: 
1507     * if len(sequence): 
1508     * if not len(sequence): 
1509     * elif len(sequence): 
1510     * elif not len(sequence): 
1511     * while len(sequence): 
1512     * while not len(sequence): 
1513     * assert len(sequence): 
1514     * assert not len(sequence): 
1515     """
1516 
1517     __implements__ = (interfaces.IAstroidChecker,)
1518 
1519     # configuration section name
1520     name = "refactoring"
1521     msgs = {
1522         "C1801": (
1523             "Do not use `len(SEQUENCE)` without comparison to determine if a sequence is empty",
1524             "len-as-condition",
1525             "Used when Pylint detects that len(sequence) is being used "
1526             "without explicit comparison inside a condition to determine if a sequence is empty. "
1527             "Instead of coercing the length to a boolean, either "
1528             "rely on the fact that empty sequences are false or "
1529             "compare the length against a scalar.",
1530         )
1531     }
1532 
1533     priority = -2
1534     options = ()
1535 
1536     @utils.check_messages("len-as-condition")
1537     def visit_call(self, node):
1538         # a len(S) call is used inside a test condition
1539         # could be if, while, assert or if expression statement
1540         # e.g. `if len(S):`
1541         if _is_len_call(node):
1542             # the len() call could also be nested together with other
1543             # boolean operations, e.g. `if z or len(x):`
1544             parent = node.parent
1545             while isinstance(parent, astroid.BoolOp):
1546                 parent = parent.parent
1547 
1548             # we're finally out of any nested boolean operations so check if
1549             # this len() call is part of a test condition
1550             if not _node_is_test_condition(parent):
1551                 return
1552             if not (node is parent.test or parent.test.parent_of(node)):
1553                 return
1554             self.add_message("len-as-condition", node=node)
1555 
1556     @utils.check_messages("len-as-condition")
1557     def visit_unaryop(self, node):
1558         """`not len(S)` must become `not S` regardless if the parent block 
1559         is a test condition or something else (boolean expression) 
1560         e.g. `if not len(S):`"""
1561         if (
1562             isinstance(node, astroid.UnaryOp)
1563             and node.op == "not"
1564             and _is_len_call(node.operand)
1565         ):
1566             self.add_message("len-as-condition", node=node)
