19   class _ObjectProxyMethods(object):
20   
21       # We use properties to override the values of __module__ and
22       # __doc__. If we add these in ObjectProxy, the derived class
23       # __dict__ will still be setup to have string variants of these
24       # attributes and the rules of descriptors means that they appear to
25       # take precedence over the properties in the base class. To avoid
26       # that, we copy the properties into the derived class type itself
27       # via a meta class. In that way the properties will always take
28       # precedence.
29   
30       @property
31       def __module__(self):
32           return self.__wrapped__.__module__
33   
34       @__module__.setter
35       def __module__(self, value):
36           self.__wrapped__.__module__ = value
37   
38       @property
39       def __doc__(self):
40           return self.__wrapped__.__doc__
41   
42       @__doc__.setter
43       def __doc__(self, value):
44           self.__wrapped__.__doc__ = value
45   
46       # We similar use a property for __dict__. We need __dict__ to be
47       # explicit to ensure that vars() works as expected.
48   
49       @property
50       def __dict__(self):
51           return self.__wrapped__.__dict__
52   
53       # Need to also propagate the special __weakref__ attribute for case
54       # where decorating classes which will define this. If do not define
55       # it and use a function like inspect.getmembers() on a decorator
56       # class it will fail. This can't be in the derived classes.
57   
58       @property
59       def __weakref__(self):
60           return self.__wrapped__.__weakref__

62   class _ObjectProxyMetaType(type):
63       def __new__(cls, name, bases, dictionary):
64           # Copy our special properties into the class so that they
65           # always take precedence over attributes of the same name added
66           # during construction of a derived class. This is to save
67           # duplicating the implementation for them in all derived classes.
68   
69           dictionary.update(vars(_ObjectProxyMethods))
70   
71           return type.__new__(cls, name, bases, dictionary)

439  class PartialCallableObjectProxy(ObjectProxy):
440  
441      def __init__(self, *args, **kwargs):
442          if len(args) < 1:
443              raise TypeError('partial type takes at least one argument')
444  
445          wrapped, args = args[0], args[1:]
446  
447          if not callable(wrapped):
448              raise TypeError('the first argument must be callable')
449  
450          super(PartialCallableObjectProxy, self).__init__(wrapped)
451  
452          self._self_args = args
453          self._self_kwargs = kwargs
454  
455      def __call__(self, *args, **kwargs):
456          _args = self._self_args + args
457  
458          _kwargs = dict(self._self_kwargs)
459          _kwargs.update(kwargs)
460  
461          return self.__wrapped__(*_args, **_kwargs)

463  class _FunctionWrapperBase(ObjectProxy):
464  
465      __slots__ = ('_self_instance', '_self_wrapper', '_self_enabled',
466              '_self_binding', '_self_parent')
467  
468      def __init__(self, wrapped, instance, wrapper, enabled=None,
469              binding='function', parent=None):
470  
471          super(_FunctionWrapperBase, self).__init__(wrapped)
472  
473          object.__setattr__(self, '_self_instance', instance)
474          object.__setattr__(self, '_self_wrapper', wrapper)
475          object.__setattr__(self, '_self_enabled', enabled)
476          object.__setattr__(self, '_self_binding', binding)
477          object.__setattr__(self, '_self_parent', parent)
478  
479      def __get__(self, instance, owner):
480          # This method is actually doing double duty for both unbound and
481          # bound derived wrapper classes. It should possibly be broken up
482          # and the distinct functionality moved into the derived classes.
483          # Can't do that straight away due to some legacy code which is
484          # relying on it being here in this base class.
485          #
486          # The distinguishing attribute which determines whether we are
487          # being called in an unbound or bound wrapper is the parent
488          # attribute. If binding has never occurred, then the parent will
489          # be None.
490          #
491          # First therefore, is if we are called in an unbound wrapper. In
492          # this case we perform the binding.
493          #
494          # We have one special case to worry about here. This is where we
495          # are decorating a nested class. In this case the wrapped class
496          # would not have a __get__() method to call. In that case we
497          # simply return self.
498          #
499          # Note that we otherwise still do binding even if instance is
500          # None and accessing an unbound instance method from a class.
501          # This is because we need to be able to later detect that
502          # specific case as we will need to extract the instance from the
503          # first argument of those passed in.
504  
505          if self._self_parent is None:
506              if not inspect.isclass(self.__wrapped__):
507                  descriptor = self.__wrapped__.__get__(instance, owner)
508  
509                  return self.__bound_function_wrapper__(descriptor, instance,
510                          self._self_wrapper, self._self_enabled,
511                          self._self_binding, self)
512  
513              return self
514  
515          # Now we have the case of binding occurring a second time on what
516          # was already a bound function. In this case we would usually
517          # return ourselves again. This mirrors what Python does.
518          #
519          # The special case this time is where we were originally bound
520          # with an instance of None and we were likely an instance
521          # method. In that case we rebind against the original wrapped
522          # function from the parent again.
523  
524          if self._self_instance is None and self._self_binding == 'function':
525              descriptor = self._self_parent.__wrapped__.__get__(
526                      instance, owner)
527  
528              return self._self_parent.__bound_function_wrapper__(
529                      descriptor, instance, self._self_wrapper,
530                      self._self_enabled, self._self_binding,
531                      self._self_parent)
532  
533          return self
534  
535      def __call__(self, *args, **kwargs):
536          # If enabled has been specified, then evaluate it at this point
537          # and if the wrapper is not to be executed, then simply return
538          # the bound function rather than a bound wrapper for the bound
539          # function. When evaluating enabled, if it is callable we call
540          # it, otherwise we evaluate it as a boolean.
541  
542          if self._self_enabled is not None:
543              if callable(self._self_enabled):
544                  if not self._self_enabled():
545                      return self.__wrapped__(*args, **kwargs)
546              elif not self._self_enabled:
547                  return self.__wrapped__(*args, **kwargs)
548  
549          # This can occur where initial function wrapper was applied to
550          # a function that was already bound to an instance. In that case
551          # we want to extract the instance from the function and use it.
552  
553          if self._self_binding == 'function':
554              if self._self_instance is None:
555                  instance = getattr(self.__wrapped__, '__self__', None)
556                  if instance is not None:
557                      return self._self_wrapper(self.__wrapped__, instance,
558                              args, kwargs)
559  
560          # This is generally invoked when the wrapped function is being
561          # called as a normal function and is not bound to a class as an
562          # instance method. This is also invoked in the case where the
563          # wrapped function was a method, but this wrapper was in turn
564          # wrapped using the staticmethod decorator.
565  
566          return self._self_wrapper(self.__wrapped__, self._self_instance,
567                  args, kwargs)

569  class BoundFunctionWrapper(_FunctionWrapperBase):
570  
571      def __call__(self, *args, **kwargs):
572          # If enabled has been specified, then evaluate it at this point
573          # and if the wrapper is not to be executed, then simply return
574          # the bound function rather than a bound wrapper for the bound
575          # function. When evaluating enabled, if it is callable we call
576          # it, otherwise we evaluate it as a boolean.
577  
578          if self._self_enabled is not None:
579              if callable(self._self_enabled):
580                  if not self._self_enabled():
581                      return self.__wrapped__(*args, **kwargs)
582              elif not self._self_enabled:
583                  return self.__wrapped__(*args, **kwargs)
584  
585          # We need to do things different depending on whether we are
586          # likely wrapping an instance method vs a static method or class
587          # method.
588  
589          if self._self_binding == 'function':
590              if self._self_instance is None:
591                  # This situation can occur where someone is calling the
592                  # instancemethod via the class type and passing the instance
593                  # as the first argument. We need to shift the args before
594                  # making the call to the wrapper and effectively bind the
595                  # instance to the wrapped function using a partial so the
596                  # wrapper doesn't see anything as being different.
597  
598                  if not args:
599                      raise TypeError('missing 1 required positional argument')
600  
601                  instance, args = args[0], args[1:]
602                  wrapped = PartialCallableObjectProxy(self.__wrapped__, instance)
603                  return self._self_wrapper(wrapped, instance, args, kwargs)
604  
605              return self._self_wrapper(self.__wrapped__, self._self_instance,
606                      args, kwargs)
607  
608          else:
609              # As in this case we would be dealing with a classmethod or
610              # staticmethod, then _self_instance will only tell us whether
611              # when calling the classmethod or staticmethod they did it via an
612              # instance of the class it is bound to and not the case where
613              # done by the class type itself. We thus ignore _self_instance
614              # and use the __self__ attribute of the bound function instead.
615              # For a classmethod, this means instance will be the class type
616              # and for a staticmethod it will be None. This is probably the
617              # more useful thing we can pass through even though we loose
618              # knowledge of whether they were called on the instance vs the
619              # class type, as it reflects what they have available in the
620              # decoratored function.
621  
622              instance = getattr(self.__wrapped__, '__self__', None)
623  
624              return self._self_wrapper(self.__wrapped__, instance, args,
625                      kwargs)

627  class FunctionWrapper(_FunctionWrapperBase):
628  
629      __bound_function_wrapper__ = BoundFunctionWrapper
630  
631      def __init__(self, wrapped, wrapper, enabled=None):
632          # What it is we are wrapping here could be anything. We need to
633          # try and detect specific cases though. In particular, we need
634          # to detect when we are given something that is a method of a
635          # class. Further, we need to know when it is likely an instance
636          # method, as opposed to a class or static method. This can
637          # become problematic though as there isn't strictly a fool proof
638          # method of knowing.
639          #
640          # The situations we could encounter when wrapping a method are:
641          #
642          # 1. The wrapper is being applied as part of a decorator which
643          # is a part of the class definition. In this case what we are
644          # given is the raw unbound function, classmethod or staticmethod
645          # wrapper objects.
646          #
647          # The problem here is that we will not know we are being applied
648          # in the context of the class being set up. This becomes
649          # important later for the case of an instance method, because in
650          # that case we just see it as a raw function and can't
651          # distinguish it from wrapping a normal function outside of
652          # a class context.
653          #
654          # 2. The wrapper is being applied when performing monkey
655          # patching of the class type afterwards and the method to be
656          # wrapped was retrieved direct from the __dict__ of the class
657          # type. This is effectively the same as (1) above.
658          #
659          # 3. The wrapper is being applied when performing monkey
660          # patching of the class type afterwards and the method to be
661          # wrapped was retrieved from the class type. In this case
662          # binding will have been performed where the instance against
663          # which the method is bound will be None at that point.
664          #
665          # This case is a problem because we can no longer tell if the
666          # method was a static method, plus if using Python3, we cannot
667          # tell if it was an instance method as the concept of an
668          # unnbound method no longer exists.
669          #
670          # 4. The wrapper is being applied when performing monkey
671          # patching of an instance of a class. In this case binding will
672          # have been perfomed where the instance was not None.
673          #
674          # This case is a problem because we can no longer tell if the
675          # method was a static method.
676          #
677          # Overall, the best we can do is look at the original type of the
678          # object which was wrapped prior to any binding being done and
679          # see if it is an instance of classmethod or staticmethod. In
680          # the case where other decorators are between us and them, if
681          # they do not propagate the __class__  attribute so that the
682          # isinstance() checks works, then likely this will do the wrong
683          # thing where classmethod and staticmethod are used.
684          #
685          # Since it is likely to be very rare that anyone even puts
686          # decorators around classmethod and staticmethod, likelihood of
687          # that being an issue is very small, so we accept it and suggest
688          # that those other decorators be fixed. It is also only an issue
689          # if a decorator wants to actually do things with the arguments.
690          #
691          # As to not being able to identify static methods properly, we
692          # just hope that that isn't something people are going to want
693          # to wrap, or if they do suggest they do it the correct way by
694          # ensuring that it is decorated in the class definition itself,
695          # or patch it in the __dict__ of the class type.
696          #
697          # So to get the best outcome we can, whenever we aren't sure what
698          # it is, we label it as a 'function'. If it was already bound and
699          # that is rebound later, we assume that it will be an instance
700          # method and try an cope with the possibility that the 'self'
701          # argument it being passed as an explicit argument and shuffle
702          # the arguments around to extract 'self' for use as the instance.
703  
704          if isinstance(wrapped, classmethod):
705              binding = 'classmethod'
706  
707          elif isinstance(wrapped, staticmethod):
708              binding = 'staticmethod'
709  
710          elif hasattr(wrapped, '__self__'):
711              if inspect.isclass(wrapped.__self__):
712                  binding = 'classmethod'
713              else:
714                  binding = 'function'
715  
716          else:
717              binding = 'function'
718  
719          super(FunctionWrapper, self).__init__(wrapped, None, wrapper,
720                  enabled, binding)

730  # Helper functions for applying wrappers to existing functions.
731  
732  def resolve_path(module, name):
733      if isinstance(module, string_types):
734          __import__(module)
735          module = sys.modules[module]
736  
737      parent = module
738  
739      path = name.split('.')
740      attribute = path[0]
741  
742      # We can't just always use getattr() because in doing
743      # that on a class it will cause binding to occur which
744      # will complicate things later and cause some things not
745      # to work. For the case of a class we therefore access
746      # the __dict__ directly. To cope though with the wrong
747      # class being given to us, or a method being moved into
748      # a base class, we need to walk the class hierarchy to
749      # work out exactly which __dict__ the method was defined
750      # in, as accessing it from __dict__ will fail if it was
751      # not actually on the class given. Fallback to using
752      # getattr() if we can't find it. If it truly doesn't
753      # exist, then that will fail.
754  
755      def lookup_attribute(parent, attribute):
756          if inspect.isclass(parent):
757              for cls in inspect.getmro(parent):
758                  if attribute in vars(cls):
759                      return vars(cls)[attribute]
760              else:
761                  return getattr(parent, attribute)
762          else:
763              return getattr(parent, attribute)
764  
765      original = lookup_attribute(parent, attribute)
766  
767      for attribute in path[1:]:
768          parent = original
769          original = lookup_attribute(parent, attribute)
770  
771      return (parent, attribute, original)
772  
773  def apply_patch(parent, attribute, replacement):
774      setattr(parent, attribute, replacement)
775  
776  def wrap_object(module, name, factory, args=(), kwargs={}):
777      (parent, attribute, original) = resolve_path(module, name)
778      wrapper = factory(original, *args, **kwargs)
779      apply_patch(parent, attribute, wrapper)
780      return wrapper

782  # Function for applying a proxy object to an attribute of a class
783  # instance. The wrapper works by defining an attribute of the same name
784  # on the class which is a descriptor and which intercepts access to the
785  # instance attribute. Note that this cannot be used on attributes which
786  # are themselves defined by a property object.
787  
788  class AttributeWrapper(object):
789  
790      def __init__(self, attribute, factory, args, kwargs):
791          self.attribute = attribute
792          self.factory = factory
793          self.args = args
794          self.kwargs = kwargs
795  
796      def __get__(self, instance, owner):
797          value = instance.__dict__[self.attribute]
798          return self.factory(value, *self.args, **self.kwargs)
799  
800      def __set__(self, instance, value):
801          instance.__dict__[self.attribute] = value
802  
803      def __delete__(self, instance):
804          del instance.__dict__[self.attribute]
805  
806  def wrap_object_attribute(module, name, factory, args=(), kwargs={}):
807      path, attribute = name.rsplit('.', 1)
808      parent = resolve_path(module, path)[2]
809      wrapper = AttributeWrapper(attribute, factory, args, kwargs)
810      apply_patch(parent, attribute, wrapper)
811      return wrapper

813  # Functions for creating a simple decorator using a FunctionWrapper,

815  # for use when doing monkey patching. For a more featured way of
816  # creating decorators see the decorator decorator instead.

818  def function_wrapper(wrapper):
819      def _wrapper(wrapped, instance, args, kwargs):
820          target_wrapped = args[0]
821          if instance is None:
822              target_wrapper = wrapper
823          elif inspect.isclass(instance):
824              target_wrapper = wrapper.__get__(None, instance)
825          else:
826              target_wrapper = wrapper.__get__(instance, type(instance))
827          return FunctionWrapper(target_wrapped, target_wrapper)
828      return FunctionWrapper(wrapper, _wrapper)

838  def transient_function_wrapper(module, name):
839      def _decorator(wrapper):
840          def _wrapper(wrapped, instance, args, kwargs):
841              target_wrapped = args[0]
842              if instance is None:
843                  target_wrapper = wrapper
844              elif inspect.isclass(instance):
845                  target_wrapper = wrapper.__get__(None, instance)
846              else:
847                  target_wrapper = wrapper.__get__(instance, type(instance))
848              def _execute(wrapped, instance, args, kwargs):
849                  (parent, attribute, original) = resolve_path(module, name)
850                  replacement = FunctionWrapper(original, target_wrapper)
851                  setattr(parent, attribute, replacement)
852                  try:
853                      return wrapped(*args, **kwargs)
854                  finally:
855                      setattr(parent, attribute, original)
856              return FunctionWrapper(target_wrapped, _execute)
857          return FunctionWrapper(wrapper, _wrapper)
858      return _decorator

860  # A weak function proxy. This will work on instance methods, class
861  # methods, static methods and regular functions. Special treatment is
862  # needed for the method types because the bound method is effectively a
863  # transient object and applying a weak reference to one will immediately
864  # result in it being destroyed and the weakref callback called. The weak
865  # reference is therefore applied to the instance the method is bound to
866  # and the original function. The function is then rebound at the point
867  # of a call via the weak function proxy.
868  
869  def _weak_function_proxy_callback(ref, proxy, callback):
870      if proxy._self_expired:
871          return
872  
873      proxy._self_expired = True
874  
875      # This could raise an exception. We let it propagate back and let
876      # the weakref.proxy() deal with it, at which point it generally
877      # prints out a short error message direct to stderr and keeps going.
878  
879      if callback is not None:
880          callback(proxy)
881  
882  class WeakFunctionProxy(ObjectProxy):
883  
884      __slots__ = ('_self_expired', '_self_instance')
885  
886      def __init__(self, wrapped, callback=None):
887          # We need to determine if the wrapped function is actually a
888          # bound method. In the case of a bound method, we need to keep a
889          # reference to the original unbound function and the instance.
890          # This is necessary because if we hold a reference to the bound
891          # function, it will be the only reference and given it is a
892          # temporary object, it will almost immediately expire and
893          # the weakref callback triggered. So what is done is that we
894          # hold a reference to the instance and unbound function and
895          # when called bind the function to the instance once again and
896          # then call it. Note that we avoid using a nested function for
897          # the callback here so as not to cause any odd reference cycles.
898  
899          _callback = callback and functools.partial(
900                  _weak_function_proxy_callback, proxy=self,
901                  callback=callback)
902  
903          self._self_expired = False
904  
905          if isinstance(wrapped, _FunctionWrapperBase):
906              self._self_instance = weakref.ref(wrapped._self_instance,
907                      _callback)
908  
909              if wrapped._self_parent is not None:
910                  super(WeakFunctionProxy, self).__init__(
911                          weakref.proxy(wrapped._self_parent, _callback))
912  
913              else:
914                  super(WeakFunctionProxy, self).__init__(
915                          weakref.proxy(wrapped, _callback))
916  
917              return
918  
919          try:
920              self._self_instance = weakref.ref(wrapped.__self__, _callback)
921  
922              super(WeakFunctionProxy, self).__init__(
923                      weakref.proxy(wrapped.__func__, _callback))
924  
925          except AttributeError:
926              self._self_instance = None
927  
928              super(WeakFunctionProxy, self).__init__(
929                      weakref.proxy(wrapped, _callback))
930  
931      def __call__(self, *args, **kwargs):
932          # We perform a boolean check here on the instance and wrapped
933          # function as that will trigger the reference error prior to
934          # calling if the reference had expired.
935  
936          instance = self._self_instance and self._self_instance()
937          function = self.__wrapped__ and self.__wrapped__
938  
939          # If the wrapped function was originally a bound function, for
940          # which we retained a reference to the instance and the unbound
941          # function we need to rebind the function and then call it. If
942          # not just called the wrapped function.
943  
944          if instance is None:
945              return self.__wrapped__(*args, **kwargs)
946  
947          return function.__get__(instance, type(instance))(*args, **kwargs)
